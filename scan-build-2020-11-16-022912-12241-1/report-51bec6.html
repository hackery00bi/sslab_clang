616 goto cleanup;
617 }
618 }
619#endif
620
621 /* If a link points to a directory then we need to treat it as a directory. */
622 if (!opts.follow_symlinks && is_symlink(path, dir)) {
623 log_debug("File %s ignored becaused it's a symlink", dir->d_name);
624 goto cleanup;
625 }
626
627 if (!is_directory(path, dir)) {
628 if (opts.file_search_regex) {
629 rc = pcre_exec(opts.file_search_regex, NULL((void*)0), dir_full_path, strlen(dir_full_path),
630 0, 0, offset_vector, 3);
631 if (rc < 0) { /* no match */
632 log_debug("Skipping %s due to file_search_regex.", dir_full_path);
633 goto cleanup;
634 } else if (opts.match_files) {
635 log_debug("match_files: file_search_regex matched for %s.", dir_full_path);
636 pthread_mutex_lock(&print_mtx);
637 print_path(dir_full_path, opts.path_sep);
638 pthread_mutex_unlock(&print_mtx);
639 opts.match_found = 1;
640 goto cleanup;
641 }
642 }
643
644 queue_item = ag_malloc(sizeof(work_queue_t));
645 queue_item->path = dir_full_path;
646 queue_item->next = NULL((void*)0);
647 pthread_mutex_lock(&work_queue_mtx);
648 if (work_queue_tail == NULL((void*)0)) {
649 work_queue = queue_item;
650 } else {
651 work_queue_tail->next = queue_item;
652 }
653 work_queue_tail = queue_item;
654 pthread_cond_signal(&files_ready);
655 pthread_mutex_unlock(&work_queue_mtx);
656 log_debug("%s added to work queue", dir_full_path);
657 } else if (opts.recurse_dirs) {
658 if (depth < opts.max_search_depth || opts.max_search_depth == -1) {
659 log_debug("Searching dir %s", dir_full_path);
660 ignores *child_ig;
661#ifdef HAVE_DIRENT_DNAMLEN
662 child_ig = init_ignore(ig, dir->d_name, dir->d_namlen);
663#else
664 child_ig = init_ignore(ig, dir->d_name, strlen(dir->d_name));
665#endif
666 search_dir(child_ig, base_path, dir_full_path, depth + 1,
667 original_dev);
668 cleanup_ignore(child_ig);
669 } else {
670 if (opts.max_search_depth == DEFAULT_MAX_SEARCH_DEPTH25) {
671 /*
672 * If the user didn't intentionally specify a particular depth,
673 * this is a warning...
674 */
675 log_err("Skipping %s. Use the --depth option to search deeper.", dir_full_path);
676 } else {
677 /* ... if they did, let's settle for debug. */
678 log_debug("Skipping %s. Use the --depth option to search deeper.", dir_full_path);
679 }
680 }
681 }
682
683 cleanup:
684 free(dir);
685 dir = NULL((void*)0);
686 if (queue_item == NULL((void*)0)) {
687 free(dir_full_path);
688 dir_full_path = NULL((void*)0);
689 }
690 }
691
692search_dir_cleanup:
693 check_symloop_leave(&current_dirkey);
694 free(dir_list);
695 dir_list = NULL((void*)0);
696}
