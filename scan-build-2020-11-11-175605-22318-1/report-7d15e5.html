82/* Plain unlink() can block for quite some time in order to actually apply
83 * the file deletion to the filesystem. This call removes the file in a
84 * background thread instead. We actually just do close() in the thread,
85 * by using the fact that if there is another instance of the same file open,
86 * the foreground unlink() will only remove the fs name, and deleting the
87 * file's storage space will only happen once the last reference is lost. */
88int bg_unlink(const char *filename) {
89 int fd = open(filename,O_RDONLY00|O_NONBLOCK04000);
90 if (fd == -1) {

157/* Add data to the replication backlog.
158 * This function also increments the global replication offset stored at
159 * server.master_repl_offset, because there is no case where we want to feed
160 * the backlog without incrementing the offset. */
161void feedReplicationBacklog(void *ptr, size_t len) {
162 unsigned char *p = ptr;
163
164 server.master_repl_offset += len;
165
166 /* This is a circular buffer, so write as much data we can at every
167 * iteration and rewind the "idx" index if we reach the limit. */
168 while(len) {
169 size_t thislen = server.repl_backlog_size - server.repl_backlog_idx;
170 if (thislen > len) thislen = len;
171 memcpy(server.repl_backlog+server.repl_backlog_idx,p,thislen);
172 server.repl_backlog_idx += thislen;
173 if (server.repl_backlog_idx == server.repl_backlog_size)
174 server.repl_backlog_idx = 0;
175 len -= thislen;
176 p += thislen;
177 server.repl_backlog_histlen += thislen;
178 }
179 if (server.repl_backlog_histlen > server.repl_backlog_size)
180 server.repl_backlog_histlen = server.repl_backlog_size;
181 /* Set the offset of the first byte we have in the backlog. */
182 server.repl_backlog_off = server.master_repl_offset -
183 server.repl_backlog_histlen + 1;
184}

411/* Feed the slave 'c' with the replication backlog starting from the
412 * specified 'offset' up to the end of the backlog. */
413long long addReplyReplicationBacklog(client *c, long long offset) {
414 long long j, skip, len;
415
416 serverLog(LL_DEBUG0, "[PSYNC] Replica request offset: %lld", offset);
417
418 if (server.repl_backlog_histlen == 0) {
419 serverLog(LL_DEBUG0, "[PSYNC] Backlog history len is zero");
420 return 0;
421 }
422
423 serverLog(LL_DEBUG0, "[PSYNC] Backlog size: %lld",
424 server.repl_backlog_size);
425 serverLog(LL_DEBUG0, "[PSYNC] First byte: %lld",
426 server.repl_backlog_off);
427 serverLog(LL_DEBUG0, "[PSYNC] History len: %lld",
428 server.repl_backlog_histlen);
429 serverLog(LL_DEBUG0, "[PSYNC] Current index: %lld",
430 server.repl_backlog_idx);
431
432 /* Compute the amount of bytes we need to discard. */
433 skip = offset - server.repl_backlog_off;
434 serverLog(LL_DEBUG0, "[PSYNC] Skipping: %lld", skip);
435
436 /* Point j to the oldest byte, that is actually our
437 * server.repl_backlog_off byte. */
438 j = (server.repl_backlog_idx +
439 (server.repl_backlog_size-server.repl_backlog_histlen)) %
440 server.repl_backlog_size;
441 serverLog(LL_DEBUG0, "[PSYNC] Index of first byte: %lld", j);
442
443 /* Discard the amount of data to seek to the specified 'offset'. */
444 j = (j + skip) % server.repl_backlog_size;
445
446 /* Feed slave with data. Since it is a circular buffer we have to
447 * split the reply in two parts if we are cross-boundary. */
448 len = server.repl_backlog_histlen - skip;
449 serverLog(LL_DEBUG0, "[PSYNC] Reply total length: %lld", len);
450 while(len) {
451 long long thislen =
452 ((server.repl_backlog_size - j) < len) ?
453 (server.repl_backlog_size - j) : len;
454
455 serverLog(LL_DEBUG0, "[PSYNC] addReply() length: %lld", thislen);
456 addReplySds(c,sdsnewlen(server.repl_backlog + j, thislen));
457 len -= thislen;
458 j = 0;
459 }
460 return server.repl_backlog_histlen - skip;
461}

539 if (master_replid[0] != '?') {
540 if (strcasecmp(master_replid, server.replid) &&
541 strcasecmp(master_replid, server.replid2))
542 {
543 serverLog(LL_NOTICE2,"Partial resynchronization not accepted: "
544 "Replication ID mismatch (Replica asked for '%s', my "
545 "replication IDs are '%s' and '%s')",
546 master_replid, server.replid, server.replid2);
547 } else {
548 serverLog(LL_NOTICE2,"Partial resynchronization not accepted: "
549 "Requested offset for second ID was %lld, but I can reply "
550 "up to %lld", psync_offset, server.second_replid_offset);
551 }
552 } else {
553 serverLog(LL_NOTICE2,"Full resync requested by replica %s",
554 replicationGetSlaveName(c));
555 }
556 goto need_full_resync;
557 }
558
559 /* We still have the data our slave is asking for? */
560 if (!server.repl_backlog ||
561 psync_offset < server.repl_backlog_off ||
562 psync_offset > (server.repl_backlog_off + server.repl_backlog_histlen))
563 {
564 serverLog(LL_NOTICE2,
565 "Unable to partial resync with replica %s for lack of backlog (Replica request was: %lld).", replicationGetSlaveName(c), psync_offset);
566 if (psync_offset > server.master_repl_offset) {
567 serverLog(LL_WARNING3,
568 "Warning: replica %s tried to PSYNC with an offset that is greater than the master replication offset.", replicationGetSlaveName(c));
569 }
570 goto need_full_resync;
571 }
572
573 /* If we reached this point, we are able to perform a partial resync:
574 * 1) Set client state to make it a slave.
575 * 2) Inform the client we can continue with +CONTINUE
576 * 3) Send the backlog data (from the offset to the end) to the slave. */
577 c->flags |= CLIENT_SLAVE(1<<0);
578 c->replstate = SLAVE_STATE_ONLINE9;
579 c->repl_ack_time = server.unixtime;
580 c->repl_put_online_on_ack = 0;
581 listAddNodeTail(server.slaves,c);
582 /* We can't use the connection buffers since they are used to accumulate
583 * new commands at this stage. But we are sure the socket send buffer is
584 * empty so this write will never fail actually. */
585 if (c->slave_capa & SLAVE_CAPA_PSYNC2(1<<1)) {
586 buflen = snprintf(buf,sizeof(buf),"+CONTINUE %s\r\n", server.replid);
587 } else {
588 buflen = snprintf(buf,sizeof(buf),"+CONTINUE\r\n");
589 }
590 if (connWrite(c->conn,buf,buflen) != buflen) {
591 freeClientAsync(c);
592 return C_OK0;
593 }
594 psync_len = addReplyReplicationBacklog(c,psync_offset);
595 serverLog(LL_NOTICE2,
596 "Partial resynchronization request from %s accepted. Sending %lld bytes of backlog starting from offset %lld.",
597 replicationGetSlaveName(c),
598 psync_len, psync_offset);
599 /* Note that we don't need to set the selected DB at server.slaveseldb
600 * to -1 to force the master to emit SELECT, since the slave already
601 * has this state from the previous connection with the master. */
602
603 refreshGoodSlavesCount();
604
605 /* Fire the replica change modules event. */
606 moduleFireServerEvent(REDISMODULE_EVENT_REPLICA_CHANGE6,
607 REDISMODULE_SUBEVENT_REPLICA_CHANGE_ONLINE0,
608 NULL((void*)0));
609
610 return C_OK0; /* The caller can return, no full resync needed. */
611
612need_full_resync:
613 /* We need a full resync for some reason... Note that we can't
614 * reply to PSYNC right now if a full SYNC is needed. The reply
615 * must include the master offset at the time the RDB file we transfer
616 * is generated, so we need to delay the reply to that moment. */
617 return C_ERR-1;
618}

620/* Start a BGSAVE for replication goals, which is, selecting the disk or
621 * socket target depending on the configuration, and making sure that
622 * the script cache is flushed before to start.
623 *
624 * The mincapa argument is the bitwise AND among all the slaves capabilities
625 * of the slaves waiting for this BGSAVE, so represents the slave capabilities
626 * all the slaves support. Can be tested via SLAVE_CAPA_* macros.
627 *
628 * Side effects, other than starting a BGSAVE:
629 *
630 * 1) Handle the slaves in WAIT_START state, by preparing them for a full
631 * sync if the BGSAVE was successfully started, or sending them an error
632 * and dropping them from the list of slaves.
633 *
634 * 2) Flush the Lua scripting script cache if the BGSAVE was actually
635 * started.
636 *
637 * Returns C_OK on success or C_ERR otherwise. */
638int startBgsaveForReplication(int mincapa) {
639 int retval;
640 int socket_target = server.repl_diskless_sync && (mincapa & SLAVE_CAPA_EOF(1<<0));
641 listIter li;
642 listNode *ln;
643
644 serverLog(LL_NOTICE2,"Starting BGSAVE for SYNC with target: %s",
645 socket_target ? "replicas sockets" : "disk");
646
647 rdbSaveInfo rsi, *rsiptr;
648 rsiptr = rdbPopulateSaveInfo(&rsi);
649 /* Only do rdbSave* when rsiptr is not NULL,
650 * otherwise slave will miss repl-stream-db. */
651 if (rsiptr) {
652 if (socket_target)
653 retval = rdbSaveToSlavesSockets(rsiptr);
654 else
655 retval = rdbSaveBackground(server.rdb_filename,rsiptr);
656 } else {
657 serverLog(LL_WARNING3,"BGSAVE for replication: replication information not available, can't generate the RDB file right now. Try later.");
658 retval = C_ERR-1;
659 }
660
661 /* If we succeeded to start a BGSAVE with disk target, let's remember
662 * this fact, so that we can later delete the file if needed. Note
663 * that we don't set the flag to 1 if the feature is disabled, otherwise
664 * it would never be cleared: the file is not deleted. This way if
665 * the user enables it later with CONFIG SET, we are fine. */
666 if (retval == C_OK0 && !socket_target && server.rdb_del_sync_files)
667 RDBGeneratedByReplication = 1;
668
669 /* If we failed to BGSAVE, remove the slaves waiting for a full
670 * resynchronization from the list of slaves, inform them with
671 * an error about what happened, close the connection ASAP. */
672 if (retval == C_ERR-1) {
673 serverLog(LL_WARNING3,"BGSAVE for replication failed");
674 listRewind(server.slaves,&li);
675 while((ln = listNext(&li))) {
676 client *slave = ln->value;
677
678 if (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_START6) {
679 slave->replstate = REPL_STATE_NONE0;
680 slave->flags &= ~CLIENT_SLAVE(1<<0);
681 listDelNode(server.slaves,ln);
682 addReplyError(slave,
683 "BGSAVE failed, replication can't continue");
684 slave->flags |= CLIENT_CLOSE_AFTER_REPLY(1<<6);
685 }
686 }
687 return retval;
688 }
689
690 /* If the target is socket, rdbSaveToSlavesSockets() already setup
691 * the slaves for a full resync. Otherwise for disk target do it now.*/
692 if (!socket_target) {
693 listRewind(server.slaves,&li);
694 while((ln = listNext(&li))) {
695 client *slave = ln->value;
696
697 if (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_START6) {
698 replicationSetupSlaveForFullResync(slave,
699 getPsyncInitialOffset());
700 }
701 }
702 }
703
704 /* Flush the script cache, since we need that slave differences are
705 * accumulated without requiring slaves to match our cached scripts. */
706 if (retval == C_OK0) replicationScriptCacheFlush();
707 return retval;
708}

710/* SYNC and PSYNC command implementation. */
711void syncCommand(client *c) {
712 /* ignore SYNC if already slave or in monitor mode */
713 if (c->flags & CLIENT_SLAVE(1<<0)) return;
714
715 /* Refuse SYNC requests if we are a slave but the link with our master
716 * is not ok... */
717 if (server.masterhost && server.repl_state != REPL_STATE_CONNECTED15) {
718 addReplySds(c,sdsnew("-NOMASTERLINK Can't SYNC while not connected with my master\r\n"));
719 return;
720 }
721
722 /* SYNC can't be issued when the server has pending data to send to
723 * the client about already issued commands. We need a fresh reply
724 * buffer registering the differences between the BGSAVE and the current
725 * dataset, so that we can copy to other slaves if needed. */
726 if (clientHasPendingReplies(c)) {
727 addReplyError(c,"SYNC and PSYNC are invalid with pending output");
728 return;
729 }
730
731 serverLog(LL_NOTICE2,"Replica %s asks for synchronization",
732 replicationGetSlaveName(c));
733
734 /* Try a partial resynchronization if this is a PSYNC command.
735 * If it fails, we continue with usual full resynchronization, however
736 * when this happens masterTryPartialResynchronization() already
737 * replied with:
738 *
739 * +FULLRESYNC <replid> <offset>
740 *
741 * So the slave knows the new replid and offset to try a PSYNC later
742 * if the connection with the master is lost. */
743 if (!strcasecmp(c->argv[0]->ptr,"psync")) {
744 if (masterTryPartialResynchronization(c) == C_OK0) {
745 server.stat_sync_partial_ok++;
746 return; /* No full resync needed, return. */
747 } else {
748 char *master_replid = c->argv[1]->ptr;
749
750 /* Increment stats for failed PSYNCs, but only if the
751 * replid is not "?", as this is used by slaves to force a full
752 * resync on purpose when they are not albe to partially
753 * resync. */
754 if (master_replid[0] != '?') server.stat_sync_partial_err++;
755 }
756 } else {
757 /* If a slave uses SYNC, we are dealing with an old implementation
758 * of the replication protocol (like redis-cli --slave). Flag the client
759 * so that we don't expect to receive REPLCONF ACK feedbacks. */
760 c->flags |= CLIENT_PRE_PSYNC(1<<16);
761 }
762
763 /* Full resynchronization. */
764 server.stat_sync_full++;
765
766 /* Setup the slave as one waiting for BGSAVE to start. The following code
767 * paths will change the state if we handle the slave differently. */
768 c->replstate = SLAVE_STATE_WAIT_BGSAVE_START6;
769 if (server.repl_disable_tcp_nodelay)
770 connDisableTcpNoDelay(c->conn); /* Non critical if it fails. */
771 c->repldbfd = -1;
772 c->flags |= CLIENT_SLAVE(1<<0);
773 listAddNodeTail(server.slaves,c);
774
775 /* Create the replication backlog if needed. */
776 if (listLength(server.slaves)((server.slaves)->len) == 1 && server.repl_backlog == NULL((void*)0)) {
777 /* When we create the backlog from scratch, we always use a new
778 * replication ID and clear the ID2, since there is no valid
779 * past history. */
780 changeReplicationId();
781 clearReplicationId2();
782 createReplicationBacklog();
783 serverLog(LL_NOTICE2,"Replication backlog created, my new "
784 "replication IDs are '%s' and '%s'",
785 server.replid, server.replid2);
786 }
787
788 /* CASE 1: BGSAVE is in progress, with disk target. */
789 if (server.rdb_child_pid != -1 &&
790 server.rdb_child_type == RDB_CHILD_TYPE_DISK1)
791 {
792 /* Ok a background save is in progress. Let's check if it is a good
793 * one for replication, i.e. if there is another slave that is
794 * registering differences since the server forked to save. */
795 client *slave;
796 listNode *ln;
797 listIter li;
798
799 listRewind(server.slaves,&li);
800 while((ln = listNext(&li))) {
801 slave = ln->value;
802 if (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_END7) break;
803 }
804 /* To attach this slave, we check that it has at least all the
805 * capabilities of the slave that triggered the current BGSAVE. */
806 if (ln && ((c->slave_capa & slave->slave_capa) == slave->slave_capa)) {
807 /* Perfect, the server is already registering differences for
808 * another slave. Set the right state, and copy the buffer. */
809 copyClientOutputBuffer(c,slave);
810 replicationSetupSlaveForFullResync(c,slave->psync_initial_offset);
811 serverLog(LL_NOTICE2,"Waiting for end of BGSAVE for SYNC");
812 } else {
813 /* No way, we need to wait for the next BGSAVE in order to
814 * register differences. */
815 serverLog(LL_NOTICE2,"Can't attach the replica to the current BGSAVE. Waiting for next BGSAVE for SYNC");
816 }
817
818 /* CASE 2: BGSAVE is in progress, with socket target. */
819 } else if (server.rdb_child_pid != -1 &&
820 server.rdb_child_type == RDB_CHILD_TYPE_SOCKET2)
821 {
822 /* There is an RDB child process but it is writing directly to
823 * children sockets. We need to wait for the next BGSAVE
824 * in order to synchronize. */
825 serverLog(LL_NOTICE2,"Current BGSAVE has socket target. Waiting for next BGSAVE for SYNC");
826
827 /* CASE 3: There is no BGSAVE is progress. */
828 } else {
829 if (server.repl_diskless_sync && (c->slave_capa & SLAVE_CAPA_EOF(1<<0))) {
830 /* Diskless replication RDB child is created inside
831 * replicationCron() since we want to delay its start a
832 * few seconds to wait for more slaves to arrive. */
833 if (server.repl_diskless_sync_delay)
834 serverLog(LL_NOTICE2,"Delay next BGSAVE for diskless SYNC");
835 else
836 startBgsaveForReplication(c->slave_capa);
837 } else {
838 /* Target is disk (or the slave is not capable of supporting
839 * diskless replication) and we don't have a BGSAVE in progress,
840 * let's start one. */
841 if (!hasActiveChildProcess()) {
842 startBgsaveForReplication(c->slave_capa);
843 } else {
844 serverLog(LL_NOTICE2,
845 "No BGSAVE in progress, but another BG operation is active. "
846 "BGSAVE for replication delayed");
847 }
848 }
849 }
850 return;
851}

853/* REPLCONF <option> <value> <option> <value> ...
854 * This command is used by a slave in order to configure the replication
855 * process before starting it with the SYNC command.
856 *
857 * Currently the only use of this command is to communicate to the master
858 * what is the listening port of the Slave redis instance, so that the
859 * master can accurately list slaves and their listening ports in
860 * the INFO output.
861 *
862 * In the future the same command can be used in order to configure
863 * the replication to initiate an incremental replication instead of a
864 * full resync. */
865void replconfCommand(client *c) {
866 int j;
867
868 if ((c->argc % 2) == 0) {
869 /* Number of arguments must be odd to make sure that every
870 * option has a corresponding value. */
871 addReply(c,shared.syntaxerr);
872 return;
873 }
874
875 /* Process every option-value pair. */
876 for (j = 1; j < c->argc; j+=2) {
877 if (!strcasecmp(c->argv[j]->ptr,"listening-port")) {
878 long port;
879
880 if ((getLongFromObjectOrReply(c,c->argv[j+1],
881 &port,NULL((void*)0)) != C_OK0))
882 return;
883 c->slave_listening_port = port;
884 } else if (!strcasecmp(c->argv[j]->ptr,"ip-address")) {
885 sds ip = c->argv[j+1]->ptr;
886 if (sdslen(ip) < sizeof(c->slave_ip)) {
887 memcpy(c->slave_ip,ip,sdslen(ip)+1);
888 } else {
889 addReplyErrorFormat(c,"REPLCONF ip-address provided by "
890 "replica instance is too long: %zd bytes", sdslen(ip));
891 return;
892 }
893 } else if (!strcasecmp(c->argv[j]->ptr,"capa")) {
894 /* Ignore capabilities not understood by this master. */
895 if (!strcasecmp(c->argv[j+1]->ptr,"eof"))
896 c->slave_capa |= SLAVE_CAPA_EOF(1<<0);
897 else if (!strcasecmp(c->argv[j+1]->ptr,"psync2"))
898 c->slave_capa |= SLAVE_CAPA_PSYNC2(1<<1);
899 } else if (!strcasecmp(c->argv[j]->ptr,"ack")) {
900 /* REPLCONF ACK is used by slave to inform the master the amount
901 * of replication stream that it processed so far. It is an
902 * internal only command that normal clients should never use. */
903 long long offset;
904
905 if (!(c->flags & CLIENT_SLAVE(1<<0))) return;
906 if ((getLongLongFromObject(c->argv[j+1], &offset) != C_OK0))
907 return;
908 if (offset > c->repl_ack_off)
909 c->repl_ack_off = offset;
910 c->repl_ack_time = server.unixtime;
911 /* If this was a diskless replication, we need to really put
912 * the slave online when the first ACK is received (which
913 * confirms slave is online and ready to get more data). This
914 * allows for simpler and less CPU intensive EOF detection
915 * when streaming RDB files.
916 * There's a chance the ACK got to us before we detected that the
917 * bgsave is done (since that depends on cron ticks), so run a
918 * quick check first (instead of waiting for the next ACK. */
919 if (server.rdb_child_pid != -1 && c->replstate == SLAVE_STATE_WAIT_BGSAVE_END7)
920 checkChildrenDone();
921 if (c->repl_put_online_on_ack && c->replstate == SLAVE_STATE_ONLINE9)
922 putSlaveOnline(c);
923 /* Note: this command does not reply anything! */
924 return;
925 } else if (!strcasecmp(c->argv[j]->ptr,"getack")) {
926 /* REPLCONF GETACK is used in order to request an ACK ASAP
927 * to the slave. */
928 if (server.masterhost && server.master) replicationSendAck();
929 return;
930 } else {
931 addReplyErrorFormat(c,"Unrecognized REPLCONF option: %s",
932 (char*)c->argv[j]->ptr);
933 return;
934 }
935 }
936 addReply(c,shared.ok);
937}

939/* This function puts a replica in the online state, and should be called just
940 * after a replica received the RDB file for the initial synchronization, and
941 * we are finally ready to send the incremental stream of commands.
942 *
943 * It does a few things:
944 *
945 * 1) Put the slave in ONLINE state. Note that the function may also be called
946 * for a replicas that are already in ONLINE state, but having the flag
947 * repl_put_online_on_ack set to true: we still have to install the write
948 * handler in that case. This function will take care of that.
949 * 2) Make sure the writable event is re-installed, since calling the SYNC
950 * command disables it, so that we can accumulate output buffer without
951 * sending it to the replica.
952 * 3) Update the count of "good replicas". */
953void putSlaveOnline(client *slave) {
954 slave->replstate = SLAVE_STATE_ONLINE9;
955 slave->repl_put_online_on_ack = 0;
956 slave->repl_ack_time = server.unixtime; /* Prevent false timeout. */
957 if (connSetWriteHandler(slave->conn, sendReplyToClient) == C_ERR-1) {
958 serverLog(LL_WARNING3,"Unable to register writable event for replica bulk transfer: %s", strerror(errno(*__errno_location ())));
959 freeClient(slave);
960 return;
961 }
962 refreshGoodSlavesCount();
963 /* Fire the replica change modules event. */
964 moduleFireServerEvent(REDISMODULE_EVENT_REPLICA_CHANGE6,
965 REDISMODULE_SUBEVENT_REPLICA_CHANGE_ONLINE0,
966 NULL((void*)0));
967 serverLog(LL_NOTICE2,"Synchronization with replica %s succeeded",
968 replicationGetSlaveName(slave));
969}

972 * generated because of replication, in an instance that is otherwise
973 * without any persistence. We don't want instances without persistence
974 * to take RDB files around, this violates certain policies in certain
975 * environments. */
976void removeRDBUsedToSyncReplicas(void) {

978 * RDBGeneratedByReplication flag in case it was set. Otherwise if the
979 * feature was enabled, but gets disabled later with CONFIG SET, the
980 * flag may remain set to one: then next time the feature is re-enabled
981 * via CONFIG SET we have have it set even if no RDB was generated
982 * because of replication recently. */
983 if (!server.rdb_del_sync_files) {
984 RDBGeneratedByReplication = 0;
985 return;
986 }
987
988 if (allPersistenceDisabled() && RDBGeneratedByReplication) {
989 client *slave;
990 listNode *ln;
991 listIter li;
992
993 int delrdb = 1;
994 listRewind(server.slaves,&li);
995 while((ln = listNext(&li))) {
996 slave = ln->value;
997 if (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_START6 ||
998 slave->replstate == SLAVE_STATE_WAIT_BGSAVE_END7 ||
999 slave->replstate == SLAVE_STATE_SEND_BULK8)
1000 {
1001 delrdb = 0;
1002 break; /* No need to check the other replicas. */
1003 }
1004 }
1005 if (delrdb) {
1006 struct stat sb;
1007 if (lstat(server.rdb_filename,&sb) != -1) {
1008 RDBGeneratedByReplication = 0;
1009 serverLog(LL_NOTICE2,
1010 "Removing the RDB file used to feed replicas "
1011 "in a persistence-less instance");
1012 bg_unlink(server.rdb_filename);
1013 }
1014 }
1015 }

2536void replicationUnsetMaster(void) {

2915/* ----------------------- REPLICATION SCRIPT CACHE --------------------------
2916 * The goal of this code is to keep track of scripts already sent to every
2917 * connected slave, in order to be able to replicate EVALSHA as it is without
2918 * translating it to EVAL every time it is possible.
2919 *
2920 * We use a capped collection implemented by a hash table for fast lookup
2921 * of scripts we can send as EVALSHA, plus a linked list that is used for
2922 * eviction of the oldest entry when the max number of items is reached.
2923 *
2924 * We don't care about taking a different cache for every different slave
2925 * since to fill the cache again is not very costly, the goal of this code
2926 * is to avoid that the same big script is transmitted a big number of times
2927 * per second wasting bandwidth and processor speed, but it is not a problem
2928 * if we need to rebuild the cache from scratch from time to time, every used
2929 * script will need to be transmitted a single time to reappear in the cache.
2930 *
2931 * This is how the system works:
2932 *
2933 * 1) Every time a new slave connects, we flush the whole script cache.
2934 * 2) We only send as EVALSHA what was sent to the master as EVALSHA, without
2935 * trying to convert EVAL into EVALSHA specifically for slaves.
2936 * 3) Every time we transmit a script as EVAL to the slaves, we also add the
2937 * corresponding SHA1 of the script into the cache as we are sure every
2938 * slave knows about the script starting from now.
2939 * 4) On SCRIPT FLUSH command, we replicate the command to all the slaves
2940 * and at the same time flush the script cache.
2941 * 5) When the last slave disconnects, flush the cache.
2942 * 6) We handle SCRIPT LOAD as well since that's how scripts are loaded
2943 * in the master sometimes.
2944 */
2945
2946/* Initialize the script cache, only called at startup. */
2947void replicationScriptCacheInit(void) {
2948 server.repl_scriptcache_size = 10000;
2949 server.repl_scriptcache_dict = dictCreate(&replScriptCacheDictType,NULL((void*)0));
2950 server.repl_scriptcache_fifo = listCreate();
2951}
2952
2953/* Empty the script cache. Should be called every time we are no longer sure
2954 * that every slave knows about all the scripts in our set, or when the
2955 * current AOF "context" is no longer aware of the script. In general we
2956 * should flush the cache:
2957 *
2958 * 1) Every time a new slave reconnects to this master and performs a
2959 * full SYNC (PSYNC does not require flushing).
2960 * 2) Every time an AOF rewrite is performed.
2961 * 3) Every time we are left without slaves at all, and AOF is off, in order
2962 * to reclaim otherwise unused memory.
2963 */
2964void replicationScriptCacheFlush(void) {
2965 dictEmpty(server.repl_scriptcache_dict,NULL((void*)0));
2966 listRelease(server.repl_scriptcache_fifo);
2967 server.repl_scriptcache_fifo = listCreate();
2968}

2999/* ----------------------- SYNCHRONOUS REPLICATION --------------------------
3000 * Redis synchronous replication design can be summarized in points:
3001 *
3002 * - Redis masters have a global replication offset, used by PSYNC.
3003 * - Master increment the offset every time new commands are sent to slaves.
3004 * - Slaves ping back masters with the offset processed so far.
3005 *
3006 * So synchronous replication adds a new WAIT command in the form:
3007 *
3008 * WAIT <num_replicas> <milliseconds_timeout>
3009 *
3010 * That returns the number of replicas that processed the query when
3011 * we finally have at least num_replicas, or when the timeout was
3012 * reached.
3013 *
3014 * The command is implemented in this way:
3015 *
3016 * - Every time a client processes a command, we remember the replication
3017 * offset after sending that command to the slaves.
3018 * - When WAIT is called, we ask slaves to send an acknowledgement ASAP.
3019 * The client is blocked at the same time (see blocked.c).
3020 * - Once we receive enough ACKs for a given offset or when the timeout
3021 * is reached, the WAIT command is unblocked and the reply sent to the
3022 * client.
3023 */
3024
3025/* This just set a flag so that we broadcast a REPLCONF GETACK command
3026 * to all the slaves in the beforeSleep() function. Note that this way
3027 * we "group" all the clients that want to wait for synchronous replication
3028 * in a given event loop iteration, and send a single GETACK for them all. */
3029void replicationRequestAckFromSlaves(void) {
3030 server.get_ack_from_slaves = 1;
3031}

3033/* Return the number of slaves that already acknowledged the specified
3034 * replication offset. */
3035int replicationCountAcksByOffset(long long offset) {
3036 listIter li;
3037 listNode *ln;
3038 int count = 0;
3039
3040 listRewind(server.slaves,&li);
3041 while((ln = listNext(&li))) {
3042 client *slave = ln->value;
3043
3044 if (slave->replstate != SLAVE_STATE_ONLINE9) continue;
3045 if (slave->repl_ack_off >= offset) count++;
3046 }
3047 return count;
3048}

3050/* WAIT for N replicas to acknowledge the processing of our latest
3051 * write command (and all the previous commands). */
3052void waitCommand(client *c) {
3053 mstime_t timeout;
3054 long numreplicas, ackreplicas;
3055 long long offset = c->woff;
3056
3057 if (server.masterhost) {
3058 addReplyError(c,"WAIT cannot be used with replica instances. Please also note that since Redis 4.0 if a replica is configured to be writable (which is not the default) writes to replicas are just local and are not propagated.");
3059 return;
3060 }
3061
3062 /* Argument parsing. */
3063 if (getLongFromObjectOrReply(c,c->argv[1],&numreplicas,NULL((void*)0)) != C_OK0)
3064 return;
3065 if (getTimeoutFromObjectOrReply(c,c->argv[2],&timeout,UNIT_MILLISECONDS1)
3066 != C_OK0) return;
3067
3068 /* First try without blocking at all. */
3069 ackreplicas = replicationCountAcksByOffset(c->woff);
3070 if (ackreplicas >= numreplicas || c->flags & CLIENT_MULTI(1<<3)) {
3071 addReplyLongLong(c,ackreplicas);
3072 return;
3073 }
3074
3075 /* Otherwise block the client and put it into our list of clients
3076 * waiting for ack from slaves. */
3077 c->bpop.timeout = timeout;
3078 c->bpop.reploffset = offset;
3079 c->bpop.numreplicas = numreplicas;
3080 listAddNodeHead(server.clients_waiting_acks,c);
3081 blockClient(c,BLOCKED_WAIT2);
3082
3083 /* Make sure that the server will send an ACK request to all the slaves
3084 * before returning to the event loop. */
3085 replicationRequestAckFromSlaves();
3086}

3089 * specific cleanup. We just remove the client from the list of clients
3090 * waiting for replica acks. Never call it directly, call unblockClient()
3091 * instead. */

3098/* Check if there are clients blocked in WAIT that can be unblocked since
3099 * we received enough ACKs from slaves. */
3100void processClientsWaitingReplicas(void) {
3101 long long last_offset = 0;
3102 int last_numreplicas = 0;
3103
3104 listIter li;
3105 listNode *ln;
3106
3107 listRewind(server.clients_waiting_acks,&li);
3108 while((ln = listNext(&li))) {

3110
3111 /* Every time we find a client that is satisfied for a given
3112 * offset and number of replicas, we remember it so the next client
3113 * may be unblocked without calling replicationCountAcksByOffset()
3114 * if the requested offset / replicas were equal or less. */
3115 if (last_offset && last_offset >= c->bpop.reploffset &&
3116 last_numreplicas >= c->bpop.numreplicas)

3158
3159 /* Non blocking connection timeout? */
3160 if (server.masterhost &&
3161 (server.repl_state == REPL_STATE_CONNECTING2 ||
3162 slaveIsInHandshakeState()) &&
3163 (time(NULL((void*)0))-server.repl_transfer_lastio) > server.repl_timeout)
3164 {
3165 serverLog(LL_WARNING3,"Timeout connecting to the MASTER...");
3166 cancelReplicationHandshake(1);
3167 }
3168

3170 if (server.masterhost && server.repl_state == REPL_STATE_TRANSFER14 &&

3174 cancelReplicationHandshake(1);
3175 }
3176
3177 /* Timed out master when we are an already connected slave? */
3178 if (server.masterhost && server.repl_state == REPL_STATE_CONNECTED15 &&
3179 (time(NULL((void*)0))-server.master->lastinteraction) > server.repl_timeout)
3180 {
3181 serverLog(LL_WARNING3,"MASTER timeout: no data nor PING received...");
3182 freeClient(server.master);
3183 }
3184
3185 /* Check if we should connect to a MASTER */
3186 if (server.repl_state == REPL_STATE_CONNECT1) {
3187 serverLog(LL_NOTICE2,"Connecting to MASTER %s:%d",
3188 server.masterhost, server.masterport);
3189 connectWithMaster();
3190 }
3191

3193 * Note that we do not send periodic acks to masters that don't
3194 * support PSYNC and replication offsets. */
3195 if (server.masterhost && server.master &&
3196 !(server.master->flags & CLIENT_PRE_PSYNC(1<<16)))
3197 replicationSendAck();
3198
3199 /* If we have attached slaves, PING them from time to time.
3200 * So slaves can implement an explicit timeout to masters, and will
3201 * be able to detect a link disconnection even if the TCP connection
3202 * will not actually go down. */
3203 listIter li;
3204 listNode *ln;
3205 robj *ping_argv[1];
3206

3208 if ((replication_cron_loops % server.repl_ping_slave_period) == 0 &&
3209 listLength(server.slaves)((server.slaves)->len))
3210 {

3212 * a Redis Cluster manual failover: the PING we send will otherwise
3213 * alter the replication offsets of master and slave, and will no longer
3214 * match the one stored into 'mf_master_offset' state. */
3215 int manual_failover_in_progress =
3216 server.cluster_enabled &&
3217 server.cluster->mf_end &&
3218 clientsArePaused();

3225 }
3226 }

3229 * stage, that is, slaves waiting for the master to create the RDB file.
3230 *
3231 * Also send the a newline to all the chained slaves we have, if we lost
3232 * connection from our master, to keep the slaves aware that their
3233 * master is online. This is needed since sub-slaves only receive proxied
3234 * data from top-level masters, so there is no explicit pinging in order
3235 * to avoid altering the replication offsets. This special out of band
3236 * pings (newlines) can be sent, they will have no effect in the offset.
3237 *
3238 * The newline will be ignored by the slave but will refresh the
3239 * last interaction timer preventing a timeout. In this case we ignore the
3240 * ping period and refresh the connection once per second since certain
3241 * timeouts are set at a few seconds (example: PSYNC response). */

3243 while((ln = listNext(&li))) {
3244 client *slave = ln->value;

3246 int is_presync =
3247 (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_START6 ||

3253 }
3254 }
3255

3257 if (listLength(server.slaves)((server.slaves)->len)) {
3258 listIter li;
3259 listNode *ln;
3260
3261 listRewind(server.slaves,&li);
3262 while((ln = listNext(&li))) {
3263 client *slave = ln->value;
3264
3265 if (slave->replstate != SLAVE_STATE_ONLINE9) continue;
3266 if (slave->flags & CLIENT_PRE_PSYNC(1<<16)) continue;
3267 if ((server.unixtime - slave->repl_ack_time) > server.repl_timeout)
3268 {

3274 }

3276 /* If this is a master without attached slaves and there is a replication
3277 * backlog active, in order to reclaim memory we can free it after some
3278 * (configured) time. Note that this cannot be done for slaves: slaves
3279 * without sub-slaves attached should still accumulate data into the
3280 * backlog, in order to reply to PSYNC queries if they are turned into
3281 * masters after a failover. */
3282 if (listLength(server.slaves)((server.slaves)->len) == 0 && server.repl_backlog_time_limit &&
3283 server.repl_backlog && server.masterhost == NULL((void*)0))
3284 {
3285 time_t idle = server.unixtime - server.repl_no_slaves_since;
3286
3287 if (idle > server.repl_backlog_time_limit) {
3288 /* When we free the backlog, we always use a new
3289 * replication ID and clear the ID2. This is needed
3290 * because when there is no backlog, the master_repl_offset
3291 * is not updated, but we would still retain our replication
3292 * ID, leading to the following problem:
3293 *
3294 * 1. We are a master instance.
3295 * 2. Our slave is promoted to master. It's repl-id-2 will
3296 * be the same as our repl-id.
3297 * 3. We, yet as master, receive some updates, that will not
3298 * increment the master_repl_offset.
3299 * 4. Later we are turned into a slave, connect to the new
3300 * master that will accept our PSYNC request by second
3301 * replication ID, but there will be data inconsistency
3302 * because we received writes. */
3303 changeReplicationId();
3304 clearReplicationId2();
3305 freeReplicationBacklog();
3306 serverLog(LL_NOTICE2,
3307 "Replication backlog freed after %d seconds "
3308 "without connected replicas.",
3309 (int) server.repl_backlog_time_limit);
3310 }
3311 }
3312
3313 /* If AOF is disabled and we no longer have attached slaves, we can
3314 * free our Replication Script Cache as there is no need to propagate
3315 * EVALSHA at all. */
3316 if (listLength(server.slaves)((server.slaves)->len) == 0 &&
3317 server.aof_state == AOF_OFF0 &&
3318 listLength(server.repl_scriptcache_fifo)((server.repl_scriptcache_fifo)->len) != 0)
3319 {
3320 replicationScriptCacheFlush();
3321 }
3322
3323 replicationStartPendingFork();
3324
3325 /* Remove the RDB file used for replication if Redis is not running
3326 * with any persistence. */
3327 removeRDBUsedToSyncReplicas();
3328
3329 /* Refresh the number of slaves with lag <= min-slaves-max-lag. */
3330 refreshGoodSlavesCount();
3331 replication_cron_loops++; /* Incremented with frequency 1 HZ. */
3332}

3334void replicationStartPendingFork(void) {
3335 /* Start a BGSAVE good for replication if we have slaves in
3336 * WAIT_BGSAVE_START state.
3337 *
3338 * In case of diskless replication, we make sure to wait the specified
3339 * number of seconds (according to configuration) so that other slaves
3340 * have the time to arrive before we start streaming. */
3341 if (!hasActiveChildProcess()) {
3342 time_t idle, max_idle = 0;
3343 int slaves_waiting = 0;
3344 int mincapa = -1;
3345 listNode *ln;
3346 listIter li;
3347
3348 listRewind(server.slaves,&li);
3349 while((ln = listNext(&li))) {
3350 client *slave = ln->value;
3351 if (slave->replstate == SLAVE_STATE_WAIT_BGSAVE_START6) {
3352 idle = server.unixtime - slave->lastinteraction;
3353 if (idle > max_idle) max_idle = idle;
3354 slaves_waiting++;
3355 mincapa = (mincapa == -1) ? slave->slave_capa :
3356 (mincapa & slave->slave_capa);
3357 }
3358 }
3359
3360 if (slaves_waiting &&
3361 (!server.repl_diskless_sync ||
3362 max_idle >= server.repl_diskless_sync_delay))
3363 {
3364 /* Start the BGSAVE. The called function may start a
3365 * BGSAVE with socket target or disk target depending on the
3366 * configuration and slaves capabilities. */
3367 startBgsaveForReplication(mincapa);
3368 }
