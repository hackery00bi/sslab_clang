166int BN_from_montgomery(BIGNUM *ret, const BIGNUM *a, BN_MONT_CTX *mont,
167 BN_CTX *ctx)
168{
169 int retn = 0;
170#ifdef MONT_WORD
171 BIGNUM *t;
172
173 BN_CTX_start(ctx);
174 if ((t = BN_CTX_get(ctx)) && BN_copy(t, a))
175 retn = BN_from_montgomery_word(ret, t, mont);
176 BN_CTX_end(ctx);
177#else /* !MONT_WORD */
178 BIGNUM *t1, *t2;
179
180 BN_CTX_start(ctx);
181 t1 = BN_CTX_get(ctx);
182 t2 = BN_CTX_get(ctx);
183 if (t1 == NULL((void*)0) || t2 == NULL((void*)0))
184 goto err;
185
186 if (!BN_copy(t1, a))
187 goto err;
188 BN_mask_bits(t1, mont->ri);
189
190 if (!BN_mul(t2, t1, &mont->Ni, ctx))
191 goto err;
192 BN_mask_bits(t2, mont->ri);
193
194 if (!BN_mul(t1, t2, &mont->N, ctx))
195 goto err;
196 if (!BN_add(t2, a, t1))
197 goto err;
198 if (!BN_rshift(ret, t2, mont->ri))
199 goto err;
200
201 if (BN_ucmp(ret, &(mont->N)) >= 0) {
202 if (!BN_usub(ret, ret, &(mont->N)))
203 goto err;
204 }
205 retn = 1;
206 bn_check_top(ret);
207 err:
208 BN_CTX_end(ctx);
209#endif /* MONT_WORD */
210 return (retn);
211}

401BN_MONT_CTX *BN_MONT_CTX_set_locked(BN_MONT_CTX **pmont, CRYPTO_RWLOCK *lock,
402 const BIGNUM *mod, BN_CTX *ctx)
403{
404 BN_MONT_CTX *ret;
405
406 CRYPTO_THREAD_read_lock(lock);
407 ret = *pmont;
408 CRYPTO_THREAD_unlock(lock);
409 if (ret)
410 return ret;
411
412 /*
413 * We don't want to serialise globally while doing our lazy-init math in
414 * BN_MONT_CTX_set. That punishes threads that are doing independent
415 * things. Instead, punish the case where more than one thread tries to
416 * lazy-init the same 'pmont', by having each do the lazy-init math work
417 * independently and only use the one from the thread that wins the race
418 * (the losers throw away the work they've done).
419 */
420 ret = BN_MONT_CTX_new();
421 if (ret == NULL((void*)0))
422 return NULL((void*)0);
423 if (!BN_MONT_CTX_set(ret, mod, ctx)) {
424 BN_MONT_CTX_free(ret);
425 return NULL((void*)0);
426 }
427
428 /* The locked compare-and-set, after the local work is done. */
429 CRYPTO_THREAD_write_lock(lock);
430 if (*pmont) {
431 BN_MONT_CTX_free(ret);
432 ret = *pmont;
433 } else
434 *pmont = ret;
435 CRYPTO_THREAD_unlock(lock);
436 return ret;
