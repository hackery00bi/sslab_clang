121 return 0;
122 }

139 * If r < 3*range, use r := r MOD range (which is either r, r -

141 * 3*range = 11..._2, each iteration succeeds with probability >=
142 * .75.
143 */
144 if (BN_cmp(r, range) >= 0) {
145 if (!BN_sub(r, r, range))
146 return 0;
147 if (BN_cmp(r, range) >= 0)
148 if (!BN_sub(r, r, range))
149 return 0;
150 }

153 BNerr(BN_F_BN_RAND_RANGE, BN_R_TOO_MANY_ITERATIONS)ERR_put_error(3,(122),(113),"crypto/bn/bn_rand.c",153);
154 return 0;
155 }

157 }

162 if (!bn_rand(r, n, BN_RAND_TOP_ANY-1, BN_RAND_BOTTOM_ANY0))
