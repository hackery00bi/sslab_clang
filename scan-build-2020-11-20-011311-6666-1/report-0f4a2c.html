500/*
501 * Note that we could do the handshake transparently under an SSL_write,
502 * but separating the steps is more helpful for debugging test failures.
503 */
504static void do_handshake_step(PEER *peer)
505{
506 int ret;
507
508 TEST_check(peer->status == PEER_RETRY)do { if (!(peer->status == PEER_RETRY)) { ERR_print_errors_fp
(stderr); (void)((!"peer->status == PEER_RETRY") ? 0 : (OPENSSL_die

, 508), 1)); } } while (0);;
509 ret = SSL_do_handshake(peer->ssl);
510
511 if (ret == 1) {
512 peer->status = PEER_SUCCESS;
513 } else if (ret == 0) {

515 } else {
516 int error = SSL_get_error(peer->ssl, ret);

523/*-
524 * Send/receive some application data. The read-write sequence is
525 * Peer A: (R) W - first read will yield no data
526 * Peer B: R W
527 * ...
528 * Peer A: R W
529 * Peer B: R W
530 * Peer A: R
531 */
532static void do_app_data_step(PEER *peer)
533{
534 int ret = 1, write_bytes;
535
536 TEST_check(peer->status == PEER_RETRY)do { if (!(peer->status == PEER_RETRY)) { ERR_print_errors_fp
(stderr); (void)((!"peer->status == PEER_RETRY") ? 0 : (OPENSSL_die
("assertion failed: " "!\"peer->status == PEER_RETRY\"", "test/handshake_helper.c"
, 536), 1)); } } while (0);;
537
538 /* We read everything available... */
539 while (ret > 0 && peer->bytes_to_read) {
540 ret = SSL_read(peer->ssl, peer->read_buf, peer->read_buf_len);
541 if (ret > 0) {
542 TEST_check(ret <= peer->bytes_to_read)do { if (!(ret <= peer->bytes_to_read)) { ERR_print_errors_fp
(stderr); (void)((!"ret <= peer->bytes_to_read") ? 0 : (
OPENSSL_die("assertion failed: " "!\"ret <= peer->bytes_to_read\""
, "test/handshake_helper.c", 542), 1)); } } while (0);;
543 peer->bytes_to_read -= ret;
544 } else if (ret == 0) {
545 peer->status = PEER_ERROR;
546 return;
547 } else {
548 int error = SSL_get_error(peer->ssl, ret);
549 if (error != SSL_ERROR_WANT_READ2) {
550 peer->status = PEER_ERROR;
551 return;
552 } /* Else continue with write. */
553 }
554 }
555
556 /* ... but we only write one write-buffer-full of data. */
557 write_bytes = peer->bytes_to_write < peer->write_buf_len ? peer->bytes_to_write :
558 peer->write_buf_len;
559 if (write_bytes) {
560 ret = SSL_write(peer->ssl, peer->write_buf, write_bytes);
561 if (ret > 0) {
562 /* SSL_write will only succeed with a complete write. */
563 TEST_check(ret == write_bytes)do { if (!(ret == write_bytes)) { ERR_print_errors_fp(stderr)
; (void)((!"ret == write_bytes") ? 0 : (OPENSSL_die("assertion failed: "
"!\"ret == write_bytes\"", "test/handshake_helper.c", 563), 1
)); } } while (0);;
564 peer->bytes_to_write -= ret;
565 } else {
566 /*
567 * We should perhaps check for SSL_ERROR_WANT_READ/WRITE here
568 * but this doesn't yet occur with current app data sizes.
569 */
570 peer->status = PEER_ERROR;
571 return;
572 }
573 }
574
575 /*
576 * We could simply finish when there was nothing to read, and we have
577 * nothing left to write. But keeping track of the expected number of bytes
578 * to read gives us somewhat better guarantees that all data sent is in fact
579 * received.
580 */
581 if (!peer->bytes_to_write && !peer->bytes_to_read) {
582 peer->status = PEER_SUCCESS;
583 }
584}

768typedef enum {
769 /* Both parties succeeded. */
770 HANDSHAKE_SUCCESS,
771 /* Client errored. */
772 CLIENT_ERROR,
773 /* Server errored. */
774 SERVER_ERROR,
775 /* Peers are in inconsistent state. */
776 INTERNAL_ERROR,
777 /* One or both peers not done. */
778 HANDSHAKE_RETRY
779} handshake_status_t;
780
781/*
782 * Determine the handshake outcome.
783 * last_status: the status of the peer to have acted last.
784 * previous_status: the status of the peer that didn't act last.
785 * client_spoke_last: 1 if the client went last.
786 */
787static handshake_status_t handshake_status(peer_status_t last_status,
788 peer_status_t previous_status,
789 int client_spoke_last)
790{
791 switch (last_status) {
792 case PEER_SUCCESS:
793 switch (previous_status) {
794 case PEER_SUCCESS:
795 /* Both succeeded. */
796 return HANDSHAKE_SUCCESS;
797 case PEER_RETRY:
798 /* Let the first peer finish. */
799 return HANDSHAKE_RETRY;
800 case PEER_ERROR:
801 /*
802 * Second peer succeeded despite the fact that the first peer
803 * already errored. This shouldn't happen.
804 */
805 return INTERNAL_ERROR;
806 }
807
808 case PEER_RETRY:
809 if (previous_status == PEER_RETRY) {
810 /* Neither peer is done. */
811 return HANDSHAKE_RETRY;
812 } else {
813 /*
814 * Deadlock: second peer is waiting for more input while first
815 * peer thinks they're done (no more input is coming).
816 */
817 return INTERNAL_ERROR;
818 }
819 case PEER_ERROR:
820 switch (previous_status) {
821 case PEER_SUCCESS:
822 /*
823 * First peer succeeded but second peer errored.
824 * TODO(emilia): we should be able to continue here (with some
825 * application data?) to ensure the first peer receives the
826 * alert / close_notify.
827 * (No tests currently exercise this branch.)
828 */
829 return client_spoke_last ? CLIENT_ERROR : SERVER_ERROR;
830 case PEER_RETRY:
831 /* We errored; let the peer finish. */
832 return HANDSHAKE_RETRY;
833 case PEER_ERROR:
834 /* Both peers errored. Return the one that errored first. */
835 return client_spoke_last ? SERVER_ERROR : CLIENT_ERROR;
836 }
837 }
838 /* Control should never reach here. */
839 return INTERNAL_ERROR;
