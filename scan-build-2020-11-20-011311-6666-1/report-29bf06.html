68/*
69 * Is a CertificateRequest message allowed at the moment or not?
70 *
71 * Return values are:
72 * 1: Yes
73 * 0: No
74 */
75static ossl_inlineinline int cert_req_allowed(SSL *s)
76{
77 /* TLS does not like anon-DH with client cert */
78 if ((s->version > SSL3_VERSION0x0300
79 && (s->s3->tmp.new_cipher->algorithm_auth & SSL_aNULL0x00000004U))
80 || (s->s3->tmp.new_cipher->algorithm_auth & (SSL_aSRP0x00000040U | SSL_aPSK0x00000010U)))
81 return 0;
82
83 return 1;
84}

104 }

110 * ossl_statem_client_read_transition() encapsulates the logic for the allowed
111 * handshake state transitions when the client is reading messages from the
112 * server. The message type that the server has sent is provided in |mt|. The
113 * current state is in |s->statem.hand_state|.
114 *
115 * Return values are:
116 * 1: Success (transition allowed)
117 * 0: Error (transition not allowed)
118 */
119int ossl_statem_client_read_transition(SSL *s, int mt)
120{
121 OSSL_STATEM *st = &s->statem;
122 int ske_expected;
123
124 switch (st->hand_state) {
125 case TLS_ST_CW_CLNT_HELLO:
126 if (mt == SSL3_MT_SERVER_HELLO2) {
127 st->hand_state = TLS_ST_CR_SRVR_HELLO;
128 return 1;
129 }
130
131 if (SSL_IS_DTLS(s)(s->method->ssl3_enc->enc_flags & 0x8)) {
132 if (mt == DTLS1_MT_HELLO_VERIFY_REQUEST3) {
133 st->hand_state = DTLS_ST_CR_HELLO_VERIFY_REQUEST;
134 return 1;
135 }
136 }
137 break;
138
139 case TLS_ST_CR_SRVR_HELLO:

143 st->hand_state = TLS_ST_CR_SESSION_TICKET;
144 return 1;
145 }
146 } else if (mt == SSL3_MT_CHANGE_CIPHER_SPEC0x0101) {
147 st->hand_state = TLS_ST_CR_CHANGE;
148 return 1;
149 }

170 st->hand_state = TLS_ST_CR_CERT;

208 /* SKE is optional for some PSK ciphersuites */

243 st->hand_state = TLS_ST_CR_CHANGE;

273/*
274 * client_write_transition() works out what handshake state to move to next
275 * when the client is writing messages to be sent to the server.
276 */
277WRITE_TRAN ossl_statem_client_write_transition(SSL *s)
278{
279 OSSL_STATEM *st = &s->statem;
280
281 switch (st->hand_state) {
282 case TLS_ST_OK:
283 /* Renegotiation - fall through */
284 case TLS_ST_BEFORE:
285 st->hand_state = TLS_ST_CW_CLNT_HELLO;
286 return WRITE_TRAN_CONTINUE;
287
288 case TLS_ST_CW_CLNT_HELLO:
289 /*
290 * No transition at the end of writing because we don't know what
291 * we will be sent
292 */
293 return WRITE_TRAN_FINISHED;
294
295 case DTLS_ST_CR_HELLO_VERIFY_REQUEST:
296 st->hand_state = TLS_ST_CW_CLNT_HELLO;
297 return WRITE_TRAN_CONTINUE;
298
299 case TLS_ST_CR_SRVR_DONE:
300 if (s->s3->tmp.cert_req)
301 st->hand_state = TLS_ST_CW_CERT;
302 else
303 st->hand_state = TLS_ST_CW_KEY_EXCH;
304 return WRITE_TRAN_CONTINUE;
305
306 case TLS_ST_CW_CERT:
307 st->hand_state = TLS_ST_CW_KEY_EXCH;
308 return WRITE_TRAN_CONTINUE;
309
310 case TLS_ST_CW_KEY_EXCH:
311 /*
312 * For TLS, cert_req is set to 2, so a cert chain of nothing is
313 * sent, but no verify packet is sent
314 */
315 /*
316 * XXX: For now, we do not support client authentication in ECDH
317 * cipher suites with ECDH (rather than ECDSA) certificates. We
318 * need to skip the certificate verify message when client's
319 * ECDH public key is sent inside the client certificate.
320 */
321 if (s->s3->tmp.cert_req == 1) {
322 st->hand_state = TLS_ST_CW_CERT_VRFY;
323 } else {
324 st->hand_state = TLS_ST_CW_CHANGE;
325 }
326 if (s->s3->flags & TLS1_FLAGS_SKIP_CERT_VERIFY0x0010) {
327 st->hand_state = TLS_ST_CW_CHANGE;
328 }
329 return WRITE_TRAN_CONTINUE;
330
331 case TLS_ST_CW_CERT_VRFY:
332 st->hand_state = TLS_ST_CW_CHANGE;
333 return WRITE_TRAN_CONTINUE;
334
335 case TLS_ST_CW_CHANGE:
336#if defined(OPENSSL_NO_NEXTPROTONEG)
337 st->hand_state = TLS_ST_CW_FINISHED;
338#else
339 if (!SSL_IS_DTLS(s)(s->method->ssl3_enc->enc_flags & 0x8) && s->s3->next_proto_neg_seen)
340 st->hand_state = TLS_ST_CW_NEXT_PROTO;
341 else
342 st->hand_state = TLS_ST_CW_FINISHED;
343#endif
344 return WRITE_TRAN_CONTINUE;
345
346#if !defined(OPENSSL_NO_NEXTPROTONEG)
347 case TLS_ST_CW_NEXT_PROTO:
348 st->hand_state = TLS_ST_CW_FINISHED;
349 return WRITE_TRAN_CONTINUE;
350#endif
351
352 case TLS_ST_CW_FINISHED:
353 if (s->hit) {
354 st->hand_state = TLS_ST_OK;
355 ossl_statem_set_in_init(s, 0);
356 return WRITE_TRAN_CONTINUE;
357 } else {
358 return WRITE_TRAN_FINISHED;
359 }
360
361 case TLS_ST_CR_FINISHED:
362 if (s->hit) {
363 st->hand_state = TLS_ST_CW_CHANGE;
364 return WRITE_TRAN_CONTINUE;
365 } else {
366 st->hand_state = TLS_ST_OK;
367 ossl_statem_set_in_init(s, 0);
368 return WRITE_TRAN_CONTINUE;
369 }
370
371 default:
372 /* Shouldn't happen */
373 return WRITE_TRAN_ERROR;
374 }
375}

377/*
378 * Perform any pre work that needs to be done prior to sending a message from
379 * the client to the server.
380 */
381WORK_STATE ossl_statem_client_pre_work(SSL *s, WORK_STATE wst)
382{
383 OSSL_STATEM *st = &s->statem;
384
385 switch (st->hand_state) {
386 case TLS_ST_CW_CLNT_HELLO:
387 s->shutdown = 0;
388 if (SSL_IS_DTLS(s)(s->method->ssl3_enc->enc_flags & 0x8)) {
389 /* every DTLS ClientHello resets Finished MAC */
390 if (!ssl3_init_finished_mac(s)) {
391 ossl_statem_set_error(s);
392 return WORK_ERROR;
393 }
394 }
395 break;
396
397 case TLS_ST_CW_CHANGE:
398 if (SSL_IS_DTLS(s)(s->method->ssl3_enc->enc_flags & 0x8)) {
399 if (s->hit) {
400 /*
401 * We're into the last flight so we don't retransmit these
402 * messages unless we need to.
403 */
404 st->use_timer = 0;
405 }
406#ifndef OPENSSL_NO_SCTP
407 if (BIO_dgram_is_sctp(SSL_get_wbio(s)))
408 return dtls_wait_for_dry(s);
409#endif
410 }
411 return WORK_FINISHED_CONTINUE;
412
413 case TLS_ST_OK:
414 return tls_finish_handshake(s, wst);
415
416 default:
417 /* No pre work to be done */
418 break;
419 }
420
421 return WORK_FINISHED_CONTINUE;
422}

424/*
425 * Perform any work that needs to be done after sending a message from the
426 * client to the server.
427 */
428WORK_STATE ossl_statem_client_post_work(SSL *s, WORK_STATE wst)
429{
430 OSSL_STATEM *st = &s->statem;
431
432 s->init_num = 0;
433
434 switch (st->hand_state) {
435 case TLS_ST_CW_CLNT_HELLO:
436 if (wst == WORK_MORE_A && statem_flush(s) != 1)
437 return WORK_MORE_A;
438
439 if (SSL_IS_DTLS(s)(s->method->ssl3_enc->enc_flags & 0x8)) {
440 /* Treat the next message as the first packet */
441 s->first_packet = 1;
442 }
443 break;
444
445 case TLS_ST_CW_KEY_EXCH:
446 if (tls_client_key_exchange_post_work(s) == 0)
447 return WORK_ERROR;
448 break;
449
450 case TLS_ST_CW_CHANGE:
451 s->session->cipher = s->s3->tmp.new_cipher;
452#ifdef OPENSSL_NO_COMP
453 s->session->compress_meth = 0;
454#else
455 if (s->s3->tmp.new_compression == NULL((void*)0))
456 s->session->compress_meth = 0;
457 else
458 s->session->compress_meth = s->s3->tmp.new_compression->id;
459#endif
460 if (!s->method->ssl3_enc->setup_key_block(s))
461 return WORK_ERROR;
462
463 if (!s->method->ssl3_enc->change_cipher_state(s,
464 SSL3_CHANGE_CIPHER_CLIENT_WRITE(0x10|0x02)))
465 return WORK_ERROR;
466
467 if (SSL_IS_DTLS(s)(s->method->ssl3_enc->enc_flags & 0x8)) {
468#ifndef OPENSSL_NO_SCTP
469 if (s->hit) {
470 /*
471 * Change to new shared key of SCTP-Auth, will be ignored if
472 * no SCTP used.
473 */
474 BIO_ctrl(SSL_get_wbio(s), BIO_CTRL_DGRAM_SCTP_NEXT_AUTH_KEY,
475 0, NULL((void*)0));
476 }
477#endif
478
479 dtls1_reset_seq_numbers(s, SSL3_CC_WRITE0x02);
480 }
481 break;
482
483 case TLS_ST_CW_FINISHED:
484#ifndef OPENSSL_NO_SCTP
485 if (wst == WORK_MORE_A && SSL_IS_DTLS(s)(s->method->ssl3_enc->enc_flags & 0x8) && s->hit == 0) {
486 /*
487 * Change to new shared key of SCTP-Auth, will be ignored if
488 * no SCTP used.
489 */
490 BIO_ctrl(SSL_get_wbio(s), BIO_CTRL_DGRAM_SCTP_NEXT_AUTH_KEY,
491 0, NULL((void*)0));
492 }
493#endif
494 if (statem_flush(s) != 1)
495 return WORK_MORE_B;
496 break;
497
498 default:
499 /* No post work to be done */
500 break;
501 }
502
503 return WORK_FINISHED_CONTINUE;
504}

1592{

1597
1598 alg_k = s->s3->tmp.new_cipher->algorithm_mkey;
1599

1601
1602#if !defined(OPENSSL_NO_EC) || !defined(OPENSSL_NO_DH)
1603 EVP_PKEY_free(s->s3->peer_tmp);
1604 s->s3->peer_tmp = NULL((void*)0);
1605#endif
1606
1607 if (alg_k & SSL_PSK(0x00000008U | 0x00000040U | 0x00000080U | 0x00000100U)) {
1608 if (!tls_process_ske_psk_preamble(s, pkt, &al))
1609 goto err;
1610 }
1611
1612 /* Nothing else to do for plain PSK or RSAPSK */
1613 if (alg_k & (SSL_kPSK0x00000008U | SSL_kRSAPSK0x00000040U)) {
1614 } else if (alg_k & SSL_kSRP0x00000020U) {
1615 if (!tls_process_ske_srp(s, pkt, &pkey, &al))
1616 goto err;
1617 } else if (alg_k & (SSL_kDHE0x00000002U | SSL_kDHEPSK0x00000100U)) {
1618 if (!tls_process_ske_dhe(s, pkt, &pkey, &al))
1619 goto err;
1620 } else if (alg_k & (SSL_kECDHE0x00000004U | SSL_kECDHEPSK0x00000080U)) {
1621 if (!tls_process_ske_ecdhe(s, pkt, &pkey, &al))
1622 goto err;
1623 } else if (alg_k) {
1624 al = SSL_AD_UNEXPECTED_MESSAGE10;
1625 SSLerr(SSL_F_TLS_PROCESS_KEY_EXCHANGE, SSL_R_UNEXPECTED_MESSAGE)ERR_put_error(20,(365),(244),"ssl/statem/statem_clnt.c",1625);
1626 goto err;
1627 }
1628
1629 /* if it was signed, check the signature */
1630 if (pkey != NULL((void*)0)) {
1631 PACKET params;
1632 int maxsig;
1633 const EVP_MD *md = NULL((void*)0);
1634 EVP_MD_CTX *md_ctx;
1635
1636 /*
1637 * |pkt| now points to the beginning of the signature, so the difference
1638 * equals the length of the parameters.
1639 */
1640 if (!PACKET_get_sub_packet(&save_param_start, &params,
1641 PACKET_remaining(&save_param_start) -
1642 PACKET_remaining(pkt))) {
1643 al = SSL_AD_INTERNAL_ERROR80;
1644 SSLerr(SSL_F_TLS_PROCESS_KEY_EXCHANGE, ERR_R_INTERNAL_ERROR)ERR_put_error(20,(365),((4|64)),"ssl/statem/statem_clnt.c",1644
);
1645 goto err;
1646 }
1647

1649 const unsigned char *sigalgs;

1652 al = SSL_AD_DECODE_ERROR50;

1654 goto err;
1655 }
1656 rv = tls12_check_peer_sigalg(&md, s, sigalgs, pkey);
1657 if (rv == -1) {
1658 al = SSL_AD_INTERNAL_ERROR80;
1659 goto err;
1660 } else if (rv == 0) {
1661 al = SSL_AD_DECODE_ERROR50;
1662 goto err;
1663 }
1664#ifdef SSL_DEBUG
1665 fprintf(stderrstderr, "USING TLSv1.2 HASH %s\n", EVP_MD_name(md)OBJ_nid2sn(EVP_MD_type(md)));
1666#endif
1667 } else if (EVP_PKEY_id(pkey) == EVP_PKEY_RSA6) {
1668 md = EVP_md5_sha1();
1669 } else {
1670 md = EVP_sha1();
1671 }
1672
1673 if (!PACKET_get_length_prefixed_2(pkt, &signature)
1674 || PACKET_remaining(pkt) != 0) {
1675 al = SSL_AD_DECODE_ERROR50;

1677 goto err;
1678 }
1679 maxsig = EVP_PKEY_size(pkey);
1680 if (maxsig < 0) {
1681 al = SSL_AD_INTERNAL_ERROR80;
1682 SSLerr(SSL_F_TLS_PROCESS_KEY_EXCHANGE, ERR_R_INTERNAL_ERROR)ERR_put_error(20,(365),((4|64)),"ssl/statem/statem_clnt.c",1682
);
1683 goto err;
1684 }
1685
1686 /*
1687 * Check signature length
1688 */
1689 if (PACKET_remaining(&signature) > (size_t)maxsig) {
1690 /* wrong packet length */
1691 al = SSL_AD_DECODE_ERROR50;
1692 SSLerr(SSL_F_TLS_PROCESS_KEY_EXCHANGE,ERR_put_error(20,(365),(264),"ssl/statem/statem_clnt.c",1693)
1693 SSL_R_WRONG_SIGNATURE_LENGTH)ERR_put_error(20,(365),(264),"ssl/statem/statem_clnt.c",1693);
1694 goto err;
1695 }
1696
1697 md_ctx = EVP_MD_CTX_new();
1698 if (md_ctx == NULL((void*)0)) {
1699 al = SSL_AD_INTERNAL_ERROR80;
1700 SSLerr(SSL_F_TLS_PROCESS_KEY_EXCHANGE, ERR_R_MALLOC_FAILURE)ERR_put_error(20,(365),((1|64)),"ssl/statem/statem_clnt.c",1700
);
1701 goto err;
1702 }
1703
1704 if (EVP_VerifyInit_ex(md_ctx, md, NULL)EVP_DigestInit_ex(md_ctx,md,((void*)0)) <= 0
1705 || EVP_VerifyUpdate(md_ctx, &(s->s3->client_random[0]),EVP_DigestUpdate(md_ctx,&(s->s3->client_random[0]),
32)
1706 SSL3_RANDOM_SIZE)EVP_DigestUpdate(md_ctx,&(s->s3->client_random[0]),
32) <= 0
1707 || EVP_VerifyUpdate(md_ctx, &(s->s3->server_random[0]),EVP_DigestUpdate(md_ctx,&(s->s3->server_random[0]),
32)
1708 SSL3_RANDOM_SIZE)EVP_DigestUpdate(md_ctx,&(s->s3->server_random[0]),
32) <= 0
1709 || EVP_VerifyUpdate(md_ctx, PACKET_data(&params),EVP_DigestUpdate(md_ctx,PACKET_data(&params),PACKET_remaining
(&params))
1710 PACKET_remaining(&params))EVP_DigestUpdate(md_ctx,PACKET_data(&params),PACKET_remaining
(&params)) <= 0) {
1711 EVP_MD_CTX_free(md_ctx);
1712 al = SSL_AD_INTERNAL_ERROR80;
1713 SSLerr(SSL_F_TLS_PROCESS_KEY_EXCHANGE, ERR_R_EVP_LIB)ERR_put_error(20,(365),(6),"ssl/statem/statem_clnt.c",1713);
1714 goto err;
1715 }
1716 if (EVP_VerifyFinal(md_ctx, PACKET_data(&signature),
1717 PACKET_remaining(&signature), pkey) <= 0) {
1718 /* bad signature */
1719 EVP_MD_CTX_free(md_ctx);
1720 al = SSL_AD_DECRYPT_ERROR51;
1721 SSLerr(SSL_F_TLS_PROCESS_KEY_EXCHANGE, SSL_R_BAD_SIGNATURE)ERR_put_error(20,(365),(123),"ssl/statem/statem_clnt.c",1721);
1722 goto err;
1723 }
1724 EVP_MD_CTX_free(md_ctx);
1725 } else {
1726 /* aNULL, aSRP or PSK do not need public keys */
1727 if (!(s->s3->tmp.new_cipher->algorithm_auth & (SSL_aNULL0x00000004U | SSL_aSRP0x00000040U))
1728 && !(alg_k & SSL_PSK(0x00000008U | 0x00000040U | 0x00000080U | 0x00000100U))) {
1729 /* Might be wrong key type, check it */
1730 if (ssl3_check_cert_and_algorithm(s)) {
1731 /* Otherwise this shouldn't happen */
1732 al = SSL_AD_INTERNAL_ERROR80;
1733 SSLerr(SSL_F_TLS_PROCESS_KEY_EXCHANGE, ERR_R_INTERNAL_ERROR)ERR_put_error(20,(365),((4|64)),"ssl/statem/statem_clnt.c",1733
);
1734 } else {
1735 al = SSL_AD_DECODE_ERROR50;
1736 }
1737 goto err;
1738 }
1739 /* still data left over */
1740 if (PACKET_remaining(pkt) != 0) {
1741 al = SSL_AD_DECODE_ERROR50;
1742 SSLerr(SSL_F_TLS_PROCESS_KEY_EXCHANGE, SSL_R_EXTRA_DATA_IN_MESSAGE)ERR_put_error(20,(365),(153),"ssl/statem/statem_clnt.c",1742);
1743 goto err;
1744 }
1745 }
1746
1747 return MSG_PROCESS_CONTINUE_READING;
1748 err:
1749 if (al != -1)
1750 ssl3_send_alert(s, SSL3_AL_FATAL2, al);
1751 ossl_statem_set_error(s);
1752 return MSG_PROCESS_ERROR;
1753}

1883{

1889 || !PACKET_get_net_2(pkt, &ticklen)

1893 goto f_err;
1894 }
1895
