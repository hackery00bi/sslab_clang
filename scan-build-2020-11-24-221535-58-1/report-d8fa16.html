344void V8InspectorSessionImpl::dispatchProtocolMessage(StringView message) {
345 using v8_crdtp::span;
346 using v8_crdtp::SpanFrom;
347 span<uint8_t> cbor;
348 std::vector<uint8_t> converted_cbor;
349 if (IsCBORMessage(message)) {
350 use_binary_protocol_ = true;
351 m_state->setBoolean("use_binary_protocol", true);
352 cbor = span<uint8_t>(message.characters8(), message.length());
353 } else {
354 // We're ignoring the return value of the conversion function
355 // intentionally. It means the |parsed_message| below will be nullptr.
356 auto status = ConvertToCBOR(message, &converted_cbor);
357 if (!status.ok()) {
358 m_channel->sendNotification(
359 serializeForFrontend(v8_crdtp::CreateErrorNotification(
360 v8_crdtp::DispatchResponse::ParseError(status.ToASCIIString()))));
361 return;
362 }
363 cbor = SpanFrom(converted_cbor);
364 }
365 v8_crdtp::Dispatchable dispatchable(cbor);
366 if (!dispatchable.ok()) {
367 if (dispatchable.HasCallId()) {
368 m_channel->sendNotification(serializeForFrontend(
369 v8_crdtp::CreateErrorNotification(dispatchable.DispatchError())));
370 } else {
371 m_channel->sendResponse(
372 dispatchable.CallId(),
373 serializeForFrontend(v8_crdtp::CreateErrorResponse(
374 dispatchable.CallId(), dispatchable.DispatchError())));
375 }
376 return;
377 }
378 m_dispatcher.Dispatch(dispatchable).Run();
379}
