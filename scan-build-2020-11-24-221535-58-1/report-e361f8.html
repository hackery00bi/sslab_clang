542void InitializeContextRuntime(Local<Context> context) {
543 Isolate* isolate = context->GetIsolate();
544 HandleScope handle_scope(isolate);
545
546 // Delete `Intl.v8BreakIterator`
547 // https://github.com/nodejs/node/issues/14909
548 Local<String> intl_string = FIXED_ONE_BYTE_STRING(isolate, "Intl");
549 Local<String> break_iter_string =
550 FIXED_ONE_BYTE_STRING(isolate, "v8BreakIterator");
551 Local<Value> intl_v;
552 if (context->Global()->Get(context, intl_string).ToLocal(&intl_v) &&
553 intl_v->IsObject()) {
554 Local<Object> intl = intl_v.As<Object>();
555 intl->Delete(context, break_iter_string).Check();
556 }
557
558 // Delete `Atomics.wake`
559 // https://github.com/nodejs/node/issues/21219
560 Local<String> atomics_string = FIXED_ONE_BYTE_STRING(isolate, "Atomics");
561 Local<String> wake_string = FIXED_ONE_BYTE_STRING(isolate, "wake");
562 Local<Value> atomics_v;
563 if (context->Global()->Get(context, atomics_string).ToLocal(&atomics_v) &&
564 atomics_v->IsObject()) {

566 atomics->Delete(context, wake_string).Check();
567 }
568
569 // Remove __proto__
570 // https://github.com/nodejs/node/issues/31951
571 Local<String> object_string = FIXED_ONE_BYTE_STRING(isolate, "Object");
572 Local<String> prototype_string = FIXED_ONE_BYTE_STRING(isolate, "prototype");
573 Local<Object> prototype = context->Global()
574 ->Get(context, object_string)
575 .ToLocalChecked()
576 .As<Object>()
577 ->Get(context, prototype_string)
578 .ToLocalChecked()

605bool InitializePrimordials(Local<Context> context) {
606 // Run per-context JS files.
607 Isolate* isolate = context->GetIsolate();
608 Context::Scope context_scope(context);
609 Local<Object> exports;
610
611 Local<String> primordials_string =
612 FIXED_ONE_BYTE_STRING(isolate, "primordials");
613 Local<String> global_string = FIXED_ONE_BYTE_STRING(isolate, "global");
614 Local<String> exports_string = FIXED_ONE_BYTE_STRING(isolate, "exports");
615
616 // Create primordials first and make it available to per-context scripts.
617 Local<Object> primordials = Object::New(isolate);
618 if (!primordials->SetPrototype(context, Null(isolate)).FromJust() ||
619 !GetPerContextExports(context).ToLocal(&exports) ||
620 !exports->Set(context, primordials_string, primordials).FromJust()) {
621 return false;
622 }

624 static const char* context_files[] = {"internal/per_context/primordials",
625 "internal/per_context/domexception",
626 "internal/per_context/messageport",
627 nullptr};
628
629 for (const char** module = context_files; *module != nullptr; module++) {
630 std::vector<Local<String>> parameters = {
631 global_string, exports_string, primordials_string};
632 Local<Value> arguments[] = {context->Global(), exports, primordials};
633 MaybeLocal<Function> maybe_fn =
634 native_module::NativeModuleEnv::LookupAndCompile(

642 // Execution failed during context creation.
643 // TODO(joyeecheung): deprecate this signature and return a MaybeLocal.
644 if (result.IsEmpty()) {
645 return false;
646 }
647 }
648
649 return true;
650}

413template <typename T>
414v8::MaybeLocal<v8::Value> ToV8Value(v8::Local<v8::Context> context,
415 const std::vector<T>& vec,
416 v8::Isolate* isolate) {
417 if (isolate == nullptr) isolate = context->GetIsolate();
418 v8::EscapableHandleScope handle_scope(isolate);
419
420 MaybeStackBuffer<v8::Local<v8::Value>, 128> arr(vec.size());
421 arr.SetLength(vec.size());
422 for (size_t i = 0; i < vec.size(); ++i) {
423 if (!ToV8Value(context, vec[i], isolate).ToLocal(&arr[i]))
424 return v8::MaybeLocal<v8::Value>();
425 }
426
427 return handle_scope.Escape(v8::Array::New(isolate, arr.out(), arr.length()));
428}

430template <typename T, typename U>
431v8::MaybeLocal<v8::Value> ToV8Value(v8::Local<v8::Context> context,
432 const std::unordered_map<T, U>& map,
433 v8::Isolate* isolate) {
434 if (isolate == nullptr) isolate = context->GetIsolate();
435 v8::EscapableHandleScope handle_scope(isolate);
436
437 v8::Local<v8::Map> ret = v8::Map::New(isolate);
438 for (const auto& item : map) {
439 v8::Local<v8::Value> first, second;
440 if (!ToV8Value(context, item.first, isolate).ToLocal(&first) ||
441 !ToV8Value(context, item.second, isolate).ToLocal(&second) ||
442 ret->Set(context, first, second).IsEmpty()) {
443 return v8::MaybeLocal<v8::Value>();
444 }
445 }
446
447 return handle_scope.Escape(ret);
448}

450template <typename T, typename >
451v8::MaybeLocal<v8::Value> ToV8Value(v8::Local<v8::Context> context,
452 const T& number,
453 v8::Isolate* isolate) {
454 if (isolate == nullptr) isolate = context->GetIsolate();
455
456 using Limits = std::numeric_limits<T>;
457 // Choose Uint32, Int32, or Double depending on range checks.
458 // These checks should all collapse at compile time.
459 if (static_cast<uint32_t>(Limits::max()) <=
460 std::numeric_limits<uint32_t>::max() &&
461 static_cast<uint32_t>(Limits::min()) >=
462 std::numeric_limits<uint32_t>::min() && Limits::is_exact) {
463 return v8::Integer::NewFromUnsigned(isolate, static_cast<uint32_t>(number));
464 }
465
466 if (static_cast<int32_t>(Limits::max()) <=
467 std::numeric_limits<int32_t>::max() &&
468 static_cast<int32_t>(Limits::min()) >=
469 std::numeric_limits<int32_t>::min() && Limits::is_exact) {
470 return v8::Integer::New(isolate, static_cast<int32_t>(number));
471 }
472
473 return v8::Number::New(isolate, static_cast<double>(number));
474}

478 const size_t length = static_cast<size_t>(args.Length());

483 for (size_t i = 0; i < size; ++i)
484 (*this)[i] = args[i + start];

507template <typename T, size_t S>
508void ArrayBufferViewContents<T, S>::Read(v8::Local<v8::ArrayBufferView> abv) {
509 static_assert(sizeof(T) == 1, "Only supports one-byte data at the moment");
510 length_ = abv->ByteLength();
511 if (length_ > sizeof(stack_storage_) || abv->HasBuffer()) {
512 data_ = static_cast<T*>(abv->Buffer()->GetBackingStore()->Data()) +
513 abv->ByteOffset();
514 } else {
515 abv->CopyContents(stack_storage_, sizeof(stack_storage_));
516 data_ = stack_storage_;
517 }
518}

520// ECMA262 20.1.2.5
521inline bool IsSafeJsInt(v8::Local<v8::Value> v) {
522 if (!v->IsNumber()) return false;
523 double v_d = v.As<v8::Number>()->Value();
524 if (std::isnan(v_d)) return false;
525 if (std::isinf(v_d)) return false;
526 if (std::trunc(v_d) != v_d) return false; // not int
527 if (std::abs(v_d) <= static_cast<double>(kMaxSafeJsInteger)) return true;
528 return false;
529}

531constexpr size_t FastStringKey::HashImpl(const char* str) {
532 // Low-quality hash (djb2), but just fine for current use cases.
533 size_t h = 5381;
534 while (*str != '\0') {
535 h = h * 33 + *(str++); // NOLINT(readability/pointer_notation)
536 }
537 return h;
538}

545constexpr bool FastStringKey::operator==(const FastStringKey& other) const {
546 const char* p1 = name_;
547 const char* p2 = other.name_;
548 if (p1 == p2) return true;
549 do {
550 if (*(p1++) != *(p2++)) return false;
551 } while (*p1 != '\0');
552 return *p2 == '\0';
553}
