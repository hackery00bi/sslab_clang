292Handle<BigInt> MutableBigInt::NewFromDouble(Isolate* isolate, double value) {
293 DCHECK_EQ(value, std::floor(value))((void) 0);
294 if (value == 0) return Zero(isolate);
295
296 bool sign = value < 0; // -0 was already handled above.
297 uint64_t double_bits = bit_cast<uint64_t>(value);
298 int raw_exponent =
299 static_cast<int>(double_bits >> Double::kPhysicalSignificandSize) & 0x7FF;
300 DCHECK_NE(raw_exponent, 0x7FF)((void) 0);
301 DCHECK_GE(raw_exponent, 0x3FF)((void) 0);
302 int exponent = raw_exponent - 0x3FF;
303 int digits = exponent / kDigitBits + 1;
304 Handle<MutableBigInt> result = Cast(isolate->factory()->NewBigInt(digits));
305 result->initialize_bitfield(sign, digits);
306
307 // We construct a BigInt from the double {value} by shifting its mantissa
308 // according to its exponent and mapping the bit pattern onto digits.
309 //
310 // <----------- bitlength = exponent + 1 ----------->
311 // <----- 52 ------> <------ trailing zeroes ------>
312 // mantissa: 1yyyyyyyyyyyyyyyyy 0000000000000000000000000000000
313 // digits: 0001xxxx xxxxxxxx xxxxxxxx xxxxxxxx xxxxxxxx xxxxxxxx
314 // <--> <------>
315 // msd_topbit kDigitBits
316 //
317 uint64_t mantissa =
318 (double_bits & Double::kSignificandMask) | Double::kHiddenBit;
319 const int kMantissaTopBit = Double::kSignificandSize - 1; // 0-indexed.
320 // 0-indexed position of most significant bit in the most significant digit.
321 int msd_topbit = exponent % kDigitBits;
322 // Number of unused bits in {mantissa}. We'll keep them shifted to the
323 // left (i.e. most significant part) of the underlying uint64_t.
324 int remaining_mantissa_bits = 0;
325 // Next digit under construction.
326 digit_t digit;
327
328 // First, build the MSD by shifting the mantissa appropriately.
329 if (msd_topbit < kMantissaTopBit) {
330 remaining_mantissa_bits = kMantissaTopBit - msd_topbit;
331 digit = mantissa >> remaining_mantissa_bits;
332 mantissa = mantissa << (64 - remaining_mantissa_bits);
333 } else {
334 DCHECK_GE(msd_topbit, kMantissaTopBit)((void) 0);
335 digit = mantissa << (msd_topbit - kMantissaTopBit);
336 mantissa = 0;
337 }
338 result->set_digit(digits - 1, digit);
339 // Then fill in the rest of the digits.
340 for (int digit_index = digits - 2; digit_index >= 0; digit_index--) {
341 if (remaining_mantissa_bits > 0) {
342 remaining_mantissa_bits -= kDigitBits;
343 if (sizeof(digit) == 4) {
344 digit = mantissa >> 32;
345 mantissa = mantissa << 32;
346 } else {
347 DCHECK_EQ(sizeof(digit), 8)((void) 0);
348 digit = mantissa;
349 mantissa = 0;
350 }
351 } else {
352 digit = 0;
353 }
354 result->set_digit(digit_index, digit);
355 }
356 return MakeImmutable(result);
357}

388void MutableBigInt::Canonicalize(MutableBigInt result) {
389 // Check if we need to right-trim any leading zero-digits.
390 int old_length = result.length();
391 int new_length = old_length;
392 while (new_length > 0 && result.digit(new_length - 1) == 0) new_length--;
393 int to_trim = old_length - new_length;
394 if (to_trim != 0) {
395 int size_delta = to_trim * MutableBigInt::kDigitSize;
396 Address new_end = result.address() + BigInt::SizeFor(new_length);
397 Heap* heap = result.GetHeap();
398 if (!heap->IsLargeObject(result)) {
399 // We do not create a filler for objects in large object space.
400 // TODO(hpayer): We should shrink the large object page if the size
401 // of the object changed significantly.
402 heap->CreateFillerObjectAt(new_end, size_delta, ClearRecordedSlots::kNo);
403 }
404 result.synchronized_set_length(new_length);
405
406 // Canonicalize -0n.
407 if (new_length == 0) {
408 result.set_sign(false);
409 // TODO(jkummerow): If we cache a canonical 0n, return that here.
410 }
411 }
412 DCHECK_IMPLIES(result.length() > 0,((void) 0)
413 result.digit(result.length() - 1) != 0)((void) 0); // MSD is non-zero.

467 // (-1) ** odd_number == -1; 1 ** anything == 1.
468 return base;
469 }
470 // For all bases >= 2, very large exponents would lead to unrepresentable
471 // results.
472 STATIC_ASSERT(kMaxLengthBits < std::numeric_limits<digit_t>::max())static_assert(kMaxLengthBits < std::numeric_limits<digit_t
>::max(), "kMaxLengthBits < std::numeric_limits<digit_t>::max()"
);
473 if (exponent->length() > 1) {
474 return ThrowBigIntTooBig<BigInt>(isolate);
475 }
476 digit_t exp_value = exponent->digit(0);
477 if (exp_value == 1) return base;
478 if (exp_value >= kMaxLengthBits) {
479 return ThrowBigIntTooBig<BigInt>(isolate);
480 }
481 STATIC_ASSERT(kMaxLengthBits <= kMaxInt)static_assert(kMaxLengthBits <= kMaxInt, "kMaxLengthBits <= kMaxInt"
);
482 int n = static_cast<int>(exp_value);
483 if (base->length() == 1 && base->digit(0) == 2) {
484 // Fast path for 2^n.
485 int needed_digits = 1 + (n / kDigitBits);
486 Handle<MutableBigInt> result;
487 if (!MutableBigInt::New(isolate, needed_digits).ToHandle(&result)) {
488 return MaybeHandle<BigInt>();
489 }
490 result->InitializeDigits(needed_digits);
491 // All bits are zero. Now set the n-th bit.
492 digit_t msd = static_cast<digit_t>(1) << (n % kDigitBits);
493 result->set_digit(needed_digits - 1, msd);
494 // Result is negative for odd powers of -2n.
495 if (base->sign()) result->set_sign((n & 1) != 0);
496 return MutableBigInt::MakeImmutable(result);
497 }
498 Handle<BigInt> result;
499 Handle<BigInt> running_square = base;
500 // This implicitly sets the result's sign correctly.
501 if (n & 1) result = base;
502 n >>= 1;
503 for (; n != 0; n >>= 1) {
504 MaybeHandle<BigInt> maybe_result =
505 Multiply(isolate, running_square, running_square);
506 if (!maybe_result.ToHandle(&running_square)) return maybe_result;
507 if (n & 1) {
508 if (result.is_null()) {
509 result = running_square;
510 } else {
511 maybe_result = Multiply(isolate, result, running_square);
512 if (!maybe_result.ToHandle(&result)) return maybe_result;
513 }
514 }
515 }
516 return result;
517}

900ComparisonResult BigInt::CompareToDouble(Handle<BigInt> x, double y) {
901 if (std::isnan(y)) return ComparisonResult::kUndefined;
902 if (y == V8_INFINITYstd::numeric_limits<double>::infinity()) return ComparisonResult::kLessThan;
903 if (y == -V8_INFINITYstd::numeric_limits<double>::infinity()) return ComparisonResult::kGreaterThan;
904 bool x_sign = x->sign();
905 // Note that this is different from the double's sign bit for -0. That's
906 // intentional because -0 must be treated like 0.
907 bool y_sign = (y < 0);
908 if (x_sign != y_sign) return UnequalSign(x_sign);
909 if (y == 0) {
910 DCHECK(!x_sign)((void) 0);
911 return x->is_zero() ? ComparisonResult::kEqual
912 : ComparisonResult::kGreaterThan;
913 }
914 if (x->is_zero()) {
915 DCHECK(!y_sign)((void) 0);
916 return ComparisonResult::kLessThan;
917 }
918 uint64_t double_bits = bit_cast<uint64_t>(y);
919 int raw_exponent =
920 static_cast<int>(double_bits >> Double::kPhysicalSignificandSize) & 0x7FF;
921 uint64_t mantissa = double_bits & Double::kSignificandMask;
922 // Non-finite doubles are handled above.
923 DCHECK_NE(raw_exponent, 0x7FF)((void) 0);
924 int exponent = raw_exponent - 0x3FF;
925 if (exponent < 0) {
926 // The absolute value of the double is less than 1. Only 0n has an
927 // absolute value smaller than that, but we've already covered that case.
928 DCHECK(!x->is_zero())((void) 0);
929 return AbsoluteGreater(x_sign);
930 }
931 int x_length = x->length();
932 digit_t x_msd = x->digit(x_length - 1);
933 int msd_leading_zeros = base::bits::CountLeadingZeros(x_msd);
934 int x_bitlength = x_length * kDigitBits - msd_leading_zeros;
935 int y_bitlength = exponent + 1;
936 if (x_bitlength < y_bitlength) return AbsoluteLess(x_sign);
937 if (x_bitlength > y_bitlength) return AbsoluteGreater(x_sign);
938
939 // At this point, we know that signs and bit lengths (i.e. position of
940 // the most significant bit in exponent-free representation) are identical.
941 // {x} is not zero, {y} is finite and not denormal.
942 // Now we virtually convert the double to an integer by shifting its
943 // mantissa according to its exponent, so it will align with the BigInt {x},
944 // and then we compare them bit for bit until we find a difference or the
945 // least significant bit.
946 // <----- 52 ------> <-- virtual trailing zeroes -->
947 // y / mantissa: 1yyyyyyyyyyyyyyyyy 0000000000000000000000000000000
948 // x / digits: 0001xxxx xxxxxxxx xxxxxxxx ...
949 // <--> <------>
950 // msd_topbit kDigitBits
951 //
952 mantissa |= Double::kHiddenBit;
953 const int kMantissaTopBit = 52; // 0-indexed.
954 // 0-indexed position of {x}'s most significant bit within the {msd}.
955 int msd_topbit = kDigitBits - 1 - msd_leading_zeros;
956 DCHECK_EQ(msd_topbit, (x_bitlength - 1) % kDigitBits)((void) 0);
957 // Shifted chunk of {mantissa} for comparing with {digit}.
958 digit_t compare_mantissa;
959 // Number of unprocessed bits in {mantissa}. We'll keep them shifted to
960 // the left (i.e. most significant part) of the underlying uint64_t.
961 int remaining_mantissa_bits = 0;
962
963 // First, compare the most significant digit against the beginning of
964 // the mantissa.
965 if (msd_topbit < kMantissaTopBit) {
966 remaining_mantissa_bits = (kMantissaTopBit - msd_topbit);
967 compare_mantissa = mantissa >> remaining_mantissa_bits;
968 mantissa = mantissa << (64 - remaining_mantissa_bits);
969 } else {
970 DCHECK_GE(msd_topbit, kMantissaTopBit)((void) 0);
971 compare_mantissa = mantissa << (msd_topbit - kMantissaTopBit);
972 mantissa = 0;
973 }
974 if (x_msd > compare_mantissa) return AbsoluteGreater(x_sign);
975 if (x_msd < compare_mantissa) return AbsoluteLess(x_sign);
976
977 // Then, compare additional digits against any remaining mantissa bits.
978 for (int digit_index = x_length - 2; digit_index >= 0; digit_index--) {
979 if (remaining_mantissa_bits > 0) {
980 remaining_mantissa_bits -= kDigitBits;
981 if (sizeof(mantissa) != sizeof(x_msd)) {
982 compare_mantissa = mantissa >> (64 - kDigitBits);
983 // "& 63" to appease compilers. kDigitBits is 32 here anyway.
984 mantissa = mantissa << (kDigitBits & 63);
985 } else {
986 compare_mantissa = mantissa;
987 mantissa = 0;
988 }
989 } else {
990 compare_mantissa = 0;
991 }
992 digit_t digit = x->digit(digit_index);
993 if (digit > compare_mantissa) return AbsoluteGreater(x_sign);
994 if (digit < compare_mantissa) return AbsoluteLess(x_sign);
995 }
996
997 // Integer parts are equal; check whether {y} has a fractional part.
998 if (mantissa != 0) {

1072double MutableBigInt::ToDouble(Handle<BigIntBase> x) {
1073 if (x->is_zero()) return 0.0;
1074 int x_length = x->length();
1075 digit_t x_msd = x->digit(x_length - 1);
1076 int msd_leading_zeros = base::bits::CountLeadingZeros(x_msd);
1077 int x_bitlength = x_length * kDigitBits - msd_leading_zeros;
1078 if (x_bitlength > 1024) return x->sign() ? -V8_INFINITYstd::numeric_limits<double>::infinity() : V8_INFINITYstd::numeric_limits<double>::infinity();
1079 uint64_t exponent = x_bitlength - 1;
1080 // We need the most significant bit shifted to the position of a double's
1081 // "hidden bit". We also need to hide that MSB, so we shift it out.
1082 uint64_t current_digit = x_msd;
1083 int digit_index = x_length - 1;
1084 int shift = msd_leading_zeros + 1 + (64 - kDigitBits);
1085 DCHECK_LE(1, shift)((void) 0);
1086 DCHECK_LE(shift, 64)((void) 0);
1087 uint64_t mantissa = (shift == 64) ? 0 : current_digit << shift;
1088 mantissa >>= 12;
1089 int mantissa_bits_unset = shift - 12;
1090 // If not all mantissa bits are defined yet, get more digits as needed.
1091 if (mantissa_bits_unset >= kDigitBits && digit_index > 0) {
1092 digit_index--;
1093 current_digit = static_cast<uint64_t>(x->digit(digit_index));
1094 mantissa |= (current_digit << (mantissa_bits_unset - kDigitBits));
1095 mantissa_bits_unset -= kDigitBits;
1096 }
1097 if (mantissa_bits_unset > 0 && digit_index > 0) {
1098 DCHECK_LT(mantissa_bits_unset, kDigitBits)((void) 0);
1099 digit_index--;
1100 current_digit = static_cast<uint64_t>(x->digit(digit_index));
1101 mantissa |= (current_digit >> (kDigitBits - mantissa_bits_unset));
1102 mantissa_bits_unset -= kDigitBits;
1103 }
1104 // If there are unconsumed digits left, we may have to round.
1105 Rounding rounding =
1106 DecideRounding(x, mantissa_bits_unset, digit_index, current_digit);
1107 if (rounding == kRoundUp || (rounding == kTie && (mantissa & 1) == 1)) {
1108 mantissa++;
1109 // Incrementing the mantissa can overflow the mantissa bits. In that case
1110 // the new mantissa will be all zero (plus hidden bit).
1111 if ((mantissa >> Double::kPhysicalSignificandSize) != 0) {
1112 mantissa = 0;
1113 exponent++;
1114 // Incrementing the exponent can overflow too.
1115 if (exponent > 1023) {
1116 return x->sign() ? -V8_INFINITYstd::numeric_limits<double>::infinity() : V8_INFINITYstd::numeric_limits<double>::infinity();
1117 }
1118 }
1119 }
1120 // Assemble the result.
1121 uint64_t sign_bit = x->sign() ? (static_cast<uint64_t>(1) << 63) : 0;
1122 exponent = (exponent + 0x3FF) << Double::kPhysicalSignificandSize;
1123 uint64_t double_bits = sign_bit | exponent | mantissa;
1124 return bit_cast<double>(double_bits);
1125}

1127// This is its own function to simplify control flow. The meaning of the
1128// parameters is defined by {ToDouble}'s local variable usage.
1129MutableBigInt::Rounding MutableBigInt::DecideRounding(Handle<BigIntBase> x,
1130 int mantissa_bits_unset,
1131 int digit_index,
1132 uint64_t current_digit) {
1133 if (mantissa_bits_unset > 0) return kRoundDown;
1134 int top_unconsumed_bit;
1135 if (mantissa_bits_unset < 0) {
1136 // There are unconsumed bits in {current_digit}.
1137 top_unconsumed_bit = -mantissa_bits_unset - 1;
1138 } else {
1139 DCHECK_EQ(mantissa_bits_unset, 0)((void) 0);
1140 // {current_digit} fit the mantissa exactly; look at the next digit.
1141 if (digit_index == 0) return kRoundDown;
1142 digit_index--;
1143 current_digit = static_cast<uint64_t>(x->digit(digit_index));
1144 top_unconsumed_bit = kDigitBits - 1;
1145 }
1146 // If the most significant remaining bit is 0, round down.
1147 uint64_t bitmask = static_cast<uint64_t>(1) << top_unconsumed_bit;
1148 if ((current_digit & bitmask) == 0) {
1149 return kRoundDown;
1150 }
1151 // If any other remaining bit is set, round up.
1152 bitmask -= 1;
1153 if ((current_digit & bitmask) != 0) return kRoundUp;
1154 while (digit_index > 0) {
1155 digit_index--;
1156 if (x->digit(digit_index) != 0) return kRoundUp;
1157 }
1158 return kTie;
1159}

1332// Helper for Absolute{And,AndNot,Or,Xor}.
1333// Performs the given binary {op} on digit pairs of {x} and {y}; when the
1334// end of the shorter of the two is reached, {extra_digits} configures how
1335// remaining digits in the longer input (if {symmetric} == kSymmetric, in
1336// {x} otherwise) are handled: copied to the result or ignored.
1337// If {result_storage} is non-nullptr, it will be used for the result and
1338// any extra digits in it will be zeroed out, otherwise a new BigInt (with
1339// the same length as the longer input) will be allocated.
1340// {result_storage} may alias {x} or {y} for in-place modification.
1341// Example:
1342// y: [ y2 ][ y1 ][ y0 ]
1343// x: [ x3 ][ x2 ][ x1 ][ x0 ]
1344// | | | |
1345// (kCopy) (op) (op) (op)
1346// | | | |
1347// v v v v
1348// result_storage: [ 0 ][ x3 ][ r2 ][ r1 ][ r0 ]
1349inline Handle<MutableBigInt> MutableBigInt::AbsoluteBitwiseOp(
1350 Isolate* isolate, Handle<BigIntBase> x, Handle<BigIntBase> y,
1351 MutableBigInt result_storage, ExtraDigitsHandling extra_digits,
1352 SymmetricOp symmetric, const std::function<digit_t(digit_t, digit_t)>& op) {
1353 int x_length = x->length();
1354 int y_length = y->length();
1355 int num_pairs = y_length;
1356 if (x_length < y_length) {
1357 num_pairs = x_length;
1358 if (symmetric == kSymmetric) {
1359 std::swap(x, y);
1360 std::swap(x_length, y_length);
1361 }
1362 }
1363 DCHECK(num_pairs == Min(x_length, y_length))((void) 0);
1364 Handle<MutableBigInt> result(result_storage, isolate);
1365 int result_length = extra_digits == kCopy ? x_length : num_pairs;
1366 if (result_storage.is_null()) {
1367 result = New(isolate, result_length).ToHandleChecked();
1368 } else {
1369 DCHECK(result_storage.length() >= result_length)((void) 0);
1370 result_length = result_storage.length();
1371 }
1372 int i = 0;
1373 for (; i < num_pairs; i++) {
1374 result->set_digit(i, op(x->digit(i), y->digit(i)));
1375 }
1376 if (extra_digits == kCopy) {
1377 for (; i < x_length; i++) {
1378 result->set_digit(i, x->digit(i));
1379 }
1380 }
1381 for (; i < result_length; i++) {
1382 result->set_digit(i, 0);
1383 }
1384 return result;
1385}

1446// Multiplies {multiplicand} with {multiplier} and adds the result to
1447// {accumulator}, starting at {accumulator_index} for the least-significant
1448// digit.
1449// Callers must ensure that {accumulator} is big enough to hold the result.
1450void MutableBigInt::MultiplyAccumulate(Handle<BigIntBase> multiplicand,
1451 digit_t multiplier,
1452 Handle<MutableBigInt> accumulator,
1453 int accumulator_index) {
1454 // This is a minimum requirement; the DCHECK in the second loop below
1455 // will enforce more as needed.
1456 DCHECK(accumulator->length() > multiplicand->length() + accumulator_index)((void) 0);
1457 if (multiplier == 0L) return;
1458 digit_t carry = 0;
1459 digit_t high = 0;
1460 for (int i = 0; i < multiplicand->length(); i++, accumulator_index++) {
1461 digit_t acc = accumulator->digit(accumulator_index);
1462 digit_t new_carry = 0;
1463 // Add last round's carryovers.
1464 acc = digit_add(acc, high, &new_carry);
1465 acc = digit_add(acc, carry, &new_carry);
1466 // Compute this round's multiplication.
1467 digit_t m_digit = multiplicand->digit(i);
1468 digit_t low = digit_mul(multiplier, m_digit, &high);
1469 acc = digit_add(acc, low, &new_carry);
1470 // Store result and prepare for next round.
1471 accumulator->set_digit(accumulator_index, acc);
1472 carry = new_carry;
1473 }
1474 for (; carry != 0 || high != 0; accumulator_index++) {
1475 DCHECK(accumulator_index < accumulator->length())((void) 0);
1476 digit_t acc = accumulator->digit(accumulator_index);
1477 digit_t new_carry = 0;
1478 acc = digit_add(acc, high, &new_carry);
1479 high = 0;
1480 acc = digit_add(acc, carry, &new_carry);
1481 accumulator->set_digit(accumulator_index, acc);
1482 carry = new_carry;
1483 }
1484}

1488void MutableBigInt::InternalMultiplyAdd(BigIntBase source, digit_t factor,
1489 digit_t summand, int n,

1491 DCHECK(source.length() >= n)((void) 0);
1492 DCHECK(result.length() >= n)((void) 0);

1494 digit_t high = 0;

1508 }

1514 }

1530// Divides {x} by {divisor}, returning the result in {quotient} and {remainder}.
1531// Mathematically, the contract is:
1532// quotient = (x - remainder) / divisor, with 0 <= remainder < divisor.
1533// If {quotient} is an empty handle, an appropriately sized BigInt will be
1534// allocated for it; otherwise the caller must ensure that it is big enough.
1535// {quotient} can be the same as {x} for an in-place division. {quotient} can
1536// also be nullptr if the caller is only interested in the remainder.

1538 digit_t divisor,
1539 Handle<MutableBigInt>* quotient,
1540 digit_t* remainder) {

1543 *remainder = 0;

1547 *quotient = New(isolate, length).ToHandleChecked();

1550 digit_t q = digit_div(*remainder, x->digit(i), divisor, remainder);
1551 (*quotient)->set_digit(i, q);
1552 }
1553 } else {
1554 for (int i = length - 1; i >= 0; i--) {
1555 digit_div(*remainder, x->digit(i), divisor, remainder);
1556 }
1557 }
1558}

1560// Divides {dividend} by {divisor}, returning the result in {quotient} and
1561// {remainder}. Mathematically, the contract is:
1562// quotient = (dividend - remainder) / divisor, with 0 <= remainder < divisor.
1563// Both {quotient} and {remainder} are optional, for callers that are only
1564// interested in one of them.
1565// See Knuth, Volume 2, section 4.3.1, Algorithm D.
1566bool MutableBigInt::AbsoluteDivLarge(Isolate* isolate,
1567 Handle<BigIntBase> dividend,
1568 Handle<BigIntBase> divisor,
1569 Handle<MutableBigInt>* quotient,
1570 Handle<MutableBigInt>* remainder) {
1571 DCHECK_GE(divisor->length(), 2)((void) 0);
1572 DCHECK(dividend->length() >= divisor->length())((void) 0);
1573 // The unusual variable names inside this function are consistent with
1574 // Knuth's book, as well as with Go's implementation of this algorithm.
1575 // Maintaining this consistency is probably more useful than trying to
1576 // come up with more descriptive names for them.
1577 int n = divisor->length();
1578 int m = dividend->length() - n;
1579
1580 // The quotient to be computed.
1581 Handle<MutableBigInt> q;
1582 if (quotient != nullptr) q = New(isolate, m + 1).ToHandleChecked();
1583 // In each iteration, {qhatv} holds {divisor} * {current quotient digit}.
1584 // "v" is the book's name for {divisor}, "qhat" the current quotient digit.
1585 Handle<MutableBigInt> qhatv;
1586 if (!New(isolate, n + 1).ToHandle(&qhatv)) return false;
1587
1588 // D1.
1589 // Left-shift inputs so that the divisor's MSB is set. This is necessary
1590 // to prevent the digit-wise divisions (see digit_div call below) from
1591 // overflowing (they take a two digits wide input, and return a one digit
1592 // result).
1593 int shift = base::bits::CountLeadingZeros(divisor->digit(n - 1));
1594 if (shift > 0) {
1595 divisor = SpecialLeftShift(isolate, divisor, shift, kSameSizeResult)
1596 .ToHandleChecked();
1597 }
1598 // Holds the (continuously updated) remaining part of the dividend, which
1599 // eventually becomes the remainder.
1600 Handle<MutableBigInt> u;
1601 if (!SpecialLeftShift(isolate, dividend, shift, kAlwaysAddOneDigit)
1602 .ToHandle(&u)) {
1603 return false;
1604 }
1605
1606 // D2.
1607 // Iterate over the dividend's digit (like the "grad school" algorithm).
1608 // {vn1} is the divisor's most significant digit.
1609 digit_t vn1 = divisor->digit(n - 1);
1610 uintptr_t work_estimate = 0;
1611 for (int j = m; j >= 0; j--) {
1612 // D3.
1613 // Estimate the current iteration's quotient digit (see Knuth for details).
1614 // {qhat} is the current quotient digit.
1615 digit_t qhat = std::numeric_limits<digit_t>::max();
1616 // {ujn} is the dividend's most significant remaining digit.
1617 digit_t ujn = u->digit(j + n);
1618 if (ujn != vn1) {
1619 // {rhat} is the current iteration's remainder.
1620 digit_t rhat = 0;
1621 // Estimate the current quotient digit by dividing the most significant
1622 // digits of dividend and divisor. The result will not be too small,
1623 // but could be a bit too large.
1624 qhat = digit_div(ujn, u->digit(j + n - 1), vn1, &rhat);
1625
1626 // Decrement the quotient estimate as needed by looking at the next
1627 // digit, i.e. by testing whether
1628 // qhat * v_{n-2} > (rhat << kDigitBits) + u_{j+n-2}.
1629 digit_t vn2 = divisor->digit(n - 2);
1630 digit_t ujn2 = u->digit(j + n - 2);
1631 while (ProductGreaterThan(qhat, vn2, rhat, ujn2)) {
1632 qhat--;
1633 digit_t prev_rhat = rhat;
1634 rhat += vn1;
1635 // v[n-1] >= 0, so this tests for overflow.
1636 if (rhat < prev_rhat) break;
1637 }
1638 }
1639
1640 // D4.
1641 // Multiply the divisor with the current quotient digit, and subtract
1642 // it from the dividend. If there was "borrow", then the quotient digit
1643 // was one too high, so we must correct it and undo one subtraction of
1644 // the (shifted) divisor.
1645 InternalMultiplyAdd(*divisor, qhat, 0, n, *qhatv);
1646 digit_t c = u->InplaceSub(qhatv, j);
1647 if (c != 0) {
1648 c = u->InplaceAdd(divisor, j);
1649 u->set_digit(j + n, u->digit(j + n) + c);
1650 qhat--;
1651 }
1652
1653 if (quotient != nullptr) q->set_digit(j, qhat);
1654
1655 // Division can take a long time. Check for interrupt requests every
1656 // now and then (roughly every 10-20 of milliseconds -- rarely enough
1657 // not to create noticeable overhead, frequently enough not to appear
1658 // frozen).
1659 work_estimate += n;
1660 if (work_estimate > 5000000) {
1661 work_estimate = 0;
1662 StackLimitCheck interrupt_check(isolate);
1663 if (interrupt_check.InterruptRequested() &&
1664 isolate->stack_guard()->HandleInterrupts().IsException(isolate)) {
1665 return false;
1666 }
1667 }
1668 }
1669 if (quotient != nullptr) {
1670 *quotient = q; // Caller will right-trim.
1671 }
1672 if (remainder != nullptr) {
1673 u->InplaceRightShift(shift);
1674 *remainder = u;
1675 }
1676 return true;
1677}

1723void MutableBigInt::InplaceRightShift(int shift) {
1724 DCHECK_GE(shift, 0)((void) 0);
1725 DCHECK_LT(shift, kDigitBits)((void) 0);
1726 DCHECK_GT(length(), 0)((void) 0);
1727 DCHECK_EQ(digit(0) & ((static_cast<digit_t>(1) << shift) - 1), 0)((void) 0);
1728 if (shift == 0) return;
1729 digit_t carry = digit(0) >> shift;
1730 int last = length() - 1;
1731 for (int i = 0; i < last; i++) {
1732 digit_t d = digit(i + 1);
1733 set_digit(i, (d << (kDigitBits - shift)) | carry);
1734 carry = d >> shift;
1735 }
1736 set_digit(last, carry);
1737}

1799 result->set_digit(i, x->digit(i - digit_shift));
1800 }
1801 } else {

1808 }

1810 result->set_digit(length + digit_shift, carry);
1811 } else {

1813 }

1819Handle<BigInt> MutableBigInt::RightShiftByAbsolute(Isolate* isolate,
1820 Handle<BigIntBase> x,
1821 Handle<BigIntBase> y) {
1822 int length = x->length();
1823 bool sign = x->sign();
1824 Maybe<digit_t> maybe_shift = ToShiftAmount(y);
1825 if (maybe_shift.IsNothing()) {
1826 return RightShiftByMaximum(isolate, sign);
1827 }
1828 digit_t shift = maybe_shift.FromJust();
1829 int digit_shift = static_cast<int>(shift / kDigitBits);
1830 int bits_shift = static_cast<int>(shift % kDigitBits);
1831 int result_length = length - digit_shift;
1832 if (result_length <= 0) {
1833 return RightShiftByMaximum(isolate, sign);
1834 }
1835 // For negative numbers, round down if any bit was shifted out (so that e.g.
1836 // -5n >> 1n == -3n and not -2n). Check now whether this will happen and
1837 // whether it can cause overflow into a new digit. If we allocate the result
1838 // large enough up front, it avoids having to do a second allocation later.
1839 bool must_round_down = false;
1840 if (sign) {
1841 const digit_t mask = (static_cast<digit_t>(1) << bits_shift) - 1;
1842 if ((x->digit(digit_shift) & mask) != 0) {
1843 must_round_down = true;
1844 } else {
1845 for (int i = 0; i < digit_shift; i++) {
1846 if (x->digit(i) != 0) {
1847 must_round_down = true;
1848 break;
1849 }
1850 }
1851 }
1852 }
1853 // If bits_shift is non-zero, it frees up bits, preventing overflow.
1854 if (must_round_down && bits_shift == 0) {
1855 // Overflow cannot happen if the most significant digit has unset bits.
1856 digit_t msd = x->digit(length - 1);
1857 bool rounding_can_overflow = digit_ismax(msd);
1858 if (rounding_can_overflow) result_length++;
1859 }
1860
1861 DCHECK_LE(result_length, length)((void) 0);
1862 Handle<MutableBigInt> result = New(isolate, result_length).ToHandleChecked();
1863 if (bits_shift == 0) {
1864 for (int i = digit_shift; i < length; i++) {
1865 result->set_digit(i - digit_shift, x->digit(i));
1866 }
1867 } else {
1868 digit_t carry = x->digit(digit_shift) >> bits_shift;
1869 int last = length - digit_shift - 1;
1870 for (int i = 0; i < last; i++) {
1871 digit_t d = x->digit(i + digit_shift + 1);
1872 result->set_digit(i, (d << (kDigitBits - bits_shift)) | carry);
1873 carry = d >> bits_shift;
1874 }
1875 result->set_digit(last, carry);
1876 }
1877
1878 if (sign) {
1879 result->set_sign(true);
1880 if (must_round_down) {
1881 // Since the result is negative, rounding down means adding one to
1882 // its absolute value. This cannot overflow.
1883 result = AbsoluteAddOne(isolate, result, true, *result).ToHandleChecked();
1884 }
1885 }
1886 return MakeImmutable(result);
1887}

2268Handle<BigInt> BigInt::AsIntN(Isolate* isolate, uint64_t n, Handle<BigInt> x) {
2269 if (x->is_zero()) return x;
2270 if (n == 0) return MutableBigInt::Zero(isolate);
2271 uint64_t needed_length = (n + kDigitBits - 1) / kDigitBits;
2272 uint64_t x_length = static_cast<uint64_t>(x->length());
2273 // If {x} has less than {n} bits, return it directly.
2274 if (x_length < needed_length) return x;
2275 DCHECK_LE(needed_length, kMaxInt)((void) 0);
2276 digit_t top_digit = x->digit(static_cast<int>(needed_length) - 1);
2277 digit_t compare_digit = static_cast<digit_t>(1) << ((n - 1) % kDigitBits);
2278 if (x_length == needed_length && top_digit < compare_digit) return x;
2279 // Otherwise we have to truncate (which is a no-op in the special case
2280 // of x == -2^(n-1)), and determine the right sign. We also might have
2281 // to subtract from 2^n to simulate having two's complement representation.
2282 // In most cases, the result's sign is x->sign() xor "(n-1)th bit present".
2283 // The only exception is when x is negative, has the (n-1)th bit, and all
2284 // its bits below (n-1) are zero. In that case, the result is the minimum
2285 // n-bit integer (example: asIntN(3, -12n) => -4n).
2286 bool has_bit = (top_digit & compare_digit) == compare_digit;
2287 DCHECK_LE(n, kMaxInt)((void) 0);
2288 int N = static_cast<int>(n);
2289 if (!has_bit) {
2290 return MutableBigInt::TruncateToNBits(isolate, N, x);
2291 }
2292 if (!x->sign()) {
2293 return MutableBigInt::TruncateAndSubFromPowerOfTwo(isolate, N, x, true);
2294 }
2295 // Negative numbers must subtract from 2^n, except for the special case
2296 // described above.
2297 if ((top_digit & (compare_digit - 1)) == 0) {
2298 for (int i = static_cast<int>(needed_length) - 2; i >= 0; i--) {
2299 if (x->digit(i) != 0) {
2300 return MutableBigInt::TruncateAndSubFromPowerOfTwo(isolate, N, x,
2301 false);
2302 }
2303 }
2304 // Truncation is no-op if x == -2^(n-1).
2305 if (x_length == needed_length && top_digit == compare_digit) return x;
2306 return MutableBigInt::TruncateToNBits(isolate, N, x);
2307 }
2308 return MutableBigInt::TruncateAndSubFromPowerOfTwo(isolate, N, x, false);
2309}

2366// Subtracts the least significant n bits of abs(x) from 2^n.
2367Handle<BigInt> MutableBigInt::TruncateAndSubFromPowerOfTwo(Isolate* isolate,
2368 int n,
2369 Handle<BigInt> x,
2370 bool result_sign) {
2371 DCHECK_NE(n, 0)((void) 0);
2372 DCHECK_LE(n, kMaxLengthBits)((void) 0);
2373
2374 int needed_digits = (n + (kDigitBits - 1)) / kDigitBits;
2375 DCHECK_LE(needed_digits, kMaxLength)((void) 0); // Follows from n <= kMaxLengthBits.
2376 Handle<MutableBigInt> result = New(isolate, needed_digits).ToHandleChecked();
2377
2378 // Process all digits except the MSD.
2379 int i = 0;
2380 int last = needed_digits - 1;
2381 int x_length = x->length();
2382 digit_t borrow = 0;
2383 // Take digits from {x} unless its length is exhausted.
2384 int limit = Min(last, x_length);
2385 for (; i < limit; i++) {
2386 digit_t new_borrow = 0;
2387 digit_t difference = digit_sub(0, x->digit(i), &new_borrow);
2388 difference = digit_sub(difference, borrow, &new_borrow);
2389 result->set_digit(i, difference);
2390 borrow = new_borrow;
2391 }
2392 // Then simulate leading zeroes in {x} as needed.
2393 for (; i < last; i++) {
2394 digit_t new_borrow = 0;
2395 digit_t difference = digit_sub(0, borrow, &new_borrow);
2396 result->set_digit(i, difference);
2397 borrow = new_borrow;
2398 }
2399
2400 // The MSD might contain extra bits that we don't want.
2401 digit_t msd = last < x_length ? x->digit(last) : 0;
2402 int msd_bits_consumed = n % kDigitBits;
2403 digit_t result_msd;
2404 if (msd_bits_consumed == 0) {
2405 digit_t new_borrow = 0;
2406 result_msd = digit_sub(0, msd, &new_borrow);
2407 result_msd = digit_sub(result_msd, borrow, &new_borrow);
2408 } else {
2409 int drop = kDigitBits - msd_bits_consumed;
2410 msd = (msd << drop) >> drop;
2411 digit_t minuend_msd = static_cast<digit_t>(1) << (kDigitBits - drop);
2412 digit_t new_borrow = 0;
2413 result_msd = digit_sub(minuend_msd, msd, &new_borrow);
2414 result_msd = digit_sub(result_msd, borrow, &new_borrow);
2415 DCHECK_EQ(new_borrow, 0)((void) 0); // result < 2^n.
2416 // If all subtracted bits were zero, we have to get rid of the
2417 // materialized minuend_msd again.
2418 result_msd &= (minuend_msd - 1);
2419 }
2420 result->set_digit(last, result_msd);
2421 result->set_sign(result_sign);
2422 return MakeImmutable(result);
2423}

2529 raw |= static_cast<uint64_t>(x.digit(1)) << 32;
2530 }
2531 // Simulate two's complement. MSVC dislikes "-raw".
2532 return x.sign() ? ((~raw) + 1u) : raw;
2533}

2559// {carry} must point to an initialized digit_t and will either be incremented
2560// by one or left alone.
2561inline BigInt::digit_t MutableBigInt::digit_add(digit_t a, digit_t b,
2562 digit_t* carry) {
2563#if HAVE_TWODIGIT_T
2564 twodigit_t result = static_cast<twodigit_t>(a) + static_cast<twodigit_t>(b);
2565 *carry += result >> kDigitBits;
2566 return static_cast<digit_t>(result);
2567#else
2568 digit_t result = a + b;
2569 if (result < a) *carry += 1;
2570 return result;
2571#endif
2572}

2574// {borrow} must point to an initialized digit_t and will either be incremented
2575// by one or left alone.
2576inline BigInt::digit_t MutableBigInt::digit_sub(digit_t a, digit_t b,
2577 digit_t* borrow) {
2578#if HAVE_TWODIGIT_T
2579 twodigit_t result = static_cast<twodigit_t>(a) - static_cast<twodigit_t>(b);
2580 *borrow += (result >> kDigitBits) & 1;
2581 return static_cast<digit_t>(result);
2582#else
2583 digit_t result = a - b;
2584 if (result > a) *borrow += 1;
2585 return static_cast<digit_t>(result);
2586#endif
2587}

2589// Returns the low half of the result. High half is in {high}.
2590inline BigInt::digit_t MutableBigInt::digit_mul(digit_t a, digit_t b,
2591 digit_t* high) {
2592#if HAVE_TWODIGIT_T
2593 twodigit_t result = static_cast<twodigit_t>(a) * static_cast<twodigit_t>(b);
2594 *high = result >> kDigitBits;
2595 return static_cast<digit_t>(result);
2596#else
2597 // Multiply in half-pointer-sized chunks.
2598 // For inputs [AH AL]*[BH BL], the result is:
2599 //
2600 // [AL*BL] // r_low
2601 // + [AL*BH] // r_mid1
2602 // + [AH*BL] // r_mid2
2603 // + [AH*BH] // r_high
2604 // = [R4 R3 R2 R1] // high = [R4 R3], low = [R2 R1]
2605 //
2606 // Where of course we must be careful with carries between the columns.
2607 digit_t a_low = a & kHalfDigitMask;
2608 digit_t a_high = a >> kHalfDigitBits;
2609 digit_t b_low = b & kHalfDigitMask;
2610 digit_t b_high = b >> kHalfDigitBits;
2611
2612 digit_t r_low = a_low * b_low;
2613 digit_t r_mid1 = a_low * b_high;
2614 digit_t r_mid2 = a_high * b_low;
2615 digit_t r_high = a_high * b_high;
2616
2617 digit_t carry = 0;
2618 digit_t low = digit_add(r_low, r_mid1 << kHalfDigitBits, &carry);
2619 low = digit_add(low, r_mid2 << kHalfDigitBits, &carry);
2620 *high =
2621 (r_mid1 >> kHalfDigitBits) + (r_mid2 >> kHalfDigitBits) + r_high + carry;
2622 return low;
2623#endif
2624}

2626// Returns the quotient.
2627// quotient = (high << kDigitBits + low - remainder) / divisor
2628BigInt::digit_t MutableBigInt::digit_div(digit_t high, digit_t low,
2629 digit_t divisor, digit_t* remainder) {
2630 DCHECK(high < divisor)((void) 0);
2631#if V8_TARGET_ARCH_X641 && (__GNUC__4 || __clang__1)
2632 digit_t quotient;
2633 digit_t rem;
2634 __asm__("divq %[divisor]"
2635 // Outputs: {quotient} will be in rax, {rem} in rdx.
2636 : "=a"(quotient), "=d"(rem)
2637 // Inputs: put {high} into rdx, {low} into rax, and {divisor} into
2638 // any register or stack slot.
2639 : "d"(high), "a"(low), [divisor] "rm"(divisor));
2640 *remainder = rem;
2641 return quotient;
2642#elif V8_TARGET_ARCH_IA32 && (__GNUC__4 || __clang__1)
2643 digit_t quotient;
2644 digit_t rem;
2645 __asm__("divl %[divisor]"
2646 // Outputs: {quotient} will be in eax, {rem} in edx.
2647 : "=a"(quotient), "=d"(rem)
2648 // Inputs: put {high} into edx, {low} into eax, and {divisor} into
2649 // any register or stack slot.
2650 : "d"(high), "a"(low), [divisor] "rm"(divisor));
2651 *remainder = rem;
2652 return quotient;
2653#else
2654 static const digit_t kHalfDigitBase = 1ull << kHalfDigitBits;
2655 // Adapted from Warren, Hacker's Delight, p. 152.
2656 int s = base::bits::CountLeadingZeros(divisor);
2657 DCHECK_NE(s, kDigitBits)((void) 0); // {divisor} is not 0.
2658 divisor <<= s;
2659
2660 digit_t vn1 = divisor >> kHalfDigitBits;
2661 digit_t vn0 = divisor & kHalfDigitMask;
2662 // {s} can be 0. {low >> kDigitBits} would be undefined behavior, so
2663 // we mask the shift amount with {kShiftMask}, and the result with
2664 // {s_zero_mask} which is 0 if s == 0 and all 1-bits otherwise.
2665 STATIC_ASSERT(sizeof(intptr_t) == sizeof(digit_t))static_assert(sizeof(intptr_t) == sizeof(digit_t), "sizeof(intptr_t) == sizeof(digit_t)"
);
2666 const int kShiftMask = kDigitBits - 1;
2667 digit_t s_zero_mask =
2668 static_cast<digit_t>(static_cast<intptr_t>(-s) >> (kDigitBits - 1));
2669 digit_t un32 =
2670 (high << s) | ((low >> ((kDigitBits - s) & kShiftMask)) & s_zero_mask);
2671 digit_t un10 = low << s;
2672 digit_t un1 = un10 >> kHalfDigitBits;
2673 digit_t un0 = un10 & kHalfDigitMask;
2674 digit_t q1 = un32 / vn1;
2675 digit_t rhat = un32 - q1 * vn1;
2676
2677 while (q1 >= kHalfDigitBase || q1 * vn0 > rhat * kHalfDigitBase + un1) {
2678 q1--;
2679 rhat += vn1;
2680 if (rhat >= kHalfDigitBase) break;
2681 }
2682
2683 digit_t un21 = un32 * kHalfDigitBase + un1 - q1 * divisor;
2684 digit_t q0 = un21 / vn1;
2685 rhat = un21 - q0 * vn1;
2686
2687 while (q0 >= kHalfDigitBase || q0 * vn0 > rhat * kHalfDigitBase + un0) {
2688 q0--;
2689 rhat += vn1;
2690 if (rhat >= kHalfDigitBase) break;
2691 }
2692
2693 *remainder = (un21 * kHalfDigitBase + un0 - q0 * divisor) >> s;
2694 return q1 * kHalfDigitBase + q0;
2695#endif
2696}

2698// Raises {base} to the power of {exponent}. Does not check for overflow.
2699BigInt::digit_t MutableBigInt::digit_pow(digit_t base, digit_t exponent) {
2700 digit_t result = 1ull;
2701 while (exponent > 0) {
2702 if (exponent & 1) {
2703 result *= base;
2704 }
2705 exponent >>= 1;
2706 base *= base;
2707 }
2708 return result;
2709}
