300 SmiCheck smi_check) {
301 // First, check if a write barrier is even needed. The tests below
302 // catch stores of Smis.
303 Label done;
304
305 // Skip barrier if writing a smi.
306 if (smi_check == INLINE_SMI_CHECK) {
307 JumpIfSmi(value, &done);
308 }
309
310 // Although the object register is tagged, the offset is relative to the start
311 // of the object, so the offset must be a multiple of kTaggedSize.
312 DCHECK(IsAligned(offset, kTaggedSize))((void) 0);
313
314 leaq(dst, FieldOperand(object, offset));
315 if (emit_debug_code()) {
316 Label ok;
317 testb(dst, Immediate(kTaggedSize - 1));
318 j(zero, &ok, Label::kNear);
319 int3();
320 bind(&ok);
321 }
322
323 RecordWrite(object, dst, value, save_fp, remembered_set_action,
324 OMIT_SMI_CHECK);
325
326 bind(&done);
327
328 // Clobber clobbered input registers when running with the debug-code flag
329 // turned on to provoke errors.
330 if (emit_debug_code()) {
331 Move(value, kZapValue, RelocInfo::NONE);
332 Move(dst, kZapValue, RelocInfo::NONE);
333 }
334}

402void TurboAssembler::CallRecordWriteStub(
403 Register object, Register address,
404 RememberedSetAction remembered_set_action, SaveFPRegsMode fp_mode,
405 Handle<Code> code_target, Address wasm_target) {
406 DCHECK_NE(code_target.is_null(), wasm_target == kNullAddress)((void) 0);
407
408 RecordWriteDescriptor descriptor;
409 RegList registers = descriptor.allocatable_registers();
410
411 SaveRegisters(registers);
412
413 Register object_parameter(
414 descriptor.GetRegisterParameter(RecordWriteDescriptor::kObject));
415 Register slot_parameter(
416 descriptor.GetRegisterParameter(RecordWriteDescriptor::kSlot));
417 Register remembered_set_parameter(
418 descriptor.GetRegisterParameter(RecordWriteDescriptor::kRememberedSet));
419 Register fp_mode_parameter(
420 descriptor.GetRegisterParameter(RecordWriteDescriptor::kFPMode));
421
422 // Prepare argument registers for calling RecordWrite
423 // slot_parameter <= address
424 // object_parameter <= object
425 MovePair(slot_parameter, address, object_parameter, object);
426
427 Smi smi_rsa = Smi::FromEnum(remembered_set_action);
428 Smi smi_fm = Smi::FromEnum(fp_mode);
429 Move(remembered_set_parameter, smi_rsa);
430 if (smi_rsa != smi_fm) {
431 Move(fp_mode_parameter, smi_fm);
432 } else {
433 movq(fp_mode_parameter, remembered_set_parameter);
434 }
435 if (code_target.is_null()) {
436 // Use {near_call} for direct Wasm call within a module.
437 near_call(wasm_target, RelocInfo::WASM_STUB_CALL);
438 } else {
439 Call(code_target, RelocInfo::CODE_TARGET);
440 }
441
442 RestoreRegisters(registers);
443}

445void MacroAssembler::RecordWrite(Register object, Register address,
446 Register value, SaveFPRegsMode fp_mode,
447 RememberedSetAction remembered_set_action,
448 SmiCheck smi_check) {
449 DCHECK(object != value)((void) 0);
450 DCHECK(object != address)((void) 0);
451 DCHECK(value != address)((void) 0);
452 AssertNotSmi(object);
453
454 if ((remembered_set_action == OMIT_REMEMBERED_SET &&
455 !FLAG_incremental_marking) ||
456 FLAG_disable_write_barriers) {
457 return;
458 }
459
460 if (emit_debug_code()) {
461 Label ok;
462 cmp_tagged(value, Operand(address, 0));
463 j(equal, &ok, Label::kNear);
464 int3();
465 bind(&ok);
466 }
467
468 // First, check if a write barrier is even needed. The tests below
469 // catch stores of smis and stores into the young generation.
470 Label done;
471
472 if (smi_check == INLINE_SMI_CHECK) {
473 // Skip barrier if writing a smi.
474 JumpIfSmi(value, &done);
475 }
476
477 CheckPageFlag(value,
478 value, // Used as scratch.
479 MemoryChunk::kPointersToHereAreInterestingMask, zero, &done,
480 Label::kNear);
481
482 CheckPageFlag(object,
483 value, // Used as scratch.
484 MemoryChunk::kPointersFromHereAreInterestingMask, zero, &done,
485 Label::kNear);
486
487 CallRecordWriteStub(object, address, remembered_set_action, fp_mode);
488
489 bind(&done);
490
491 // Clobber clobbered registers when running with the debug-code flag
492 // turned on to provoke errors.
493 if (emit_debug_code()) {
494 Move(address, kZapValue, RelocInfo::NONE);
495 Move(value, kZapValue, RelocInfo::NONE);
496 }

529void TurboAssembler::Abort(AbortReason reason) {
530#ifdef DEBUG
531 const char* msg = GetAbortReason(reason);
532 RecordComment("Abort message: ");
533 RecordComment(msg);
534#endif
535
536 // Avoid emitting call to builtin if requested.
537 if (trap_on_abort()) {
538 int3();
539 return;
540 }
541
542 if (should_abort_hard()) {
543 // We don't care if we constructed a frame. Just pretend we did.
544 FrameScope assume_frame(this, StackFrame::NONE);
545 movl(arg_reg_1, Immediate(static_cast<int>(reason)));
546 PrepareCallCFunction(1);
547 LoadAddress(rax, ExternalReference::abort_with_reason());
548 call(rax);
549 return;
550 }
551
552 Move(rdx, Smi::FromInt(static_cast<int>(reason)));
553
554 if (!has_frame()) {
555 // We don't actually want to generate a pile of code for this, so just
556 // claim there is a stack frame, without generating one.
557 FrameScope scope(this, StackFrame::NONE);
558 Call(BUILTIN_CODE(isolate(), Abort)(isolate())->builtins()->builtin_handle(Builtins::kAbort
), RelocInfo::CODE_TARGET);
559 } else {
560 Call(BUILTIN_CODE(isolate(), Abort)(isolate())->builtins()->builtin_handle(Builtins::kAbort
), RelocInfo::CODE_TARGET);
561 }
562 // Control will not return here.
563 int3();
564}

617int TurboAssembler::RequiredStackSizeForCallerSaved(SaveFPRegsMode fp_mode,
618 Register exclusion1,
619 Register exclusion2,
620 Register exclusion3) const {
621 int bytes = 0;
622 for (int i = 0; i < kNumberOfSavedRegs; i++) {
623 Register reg = saved_regs[i];
624 if (reg != exclusion1 && reg != exclusion2 && reg != exclusion3) {
625 bytes += kSystemPointerSize;
626 }
627 }
628
629 // R12 to r15 are callee save on all platforms.
630 if (fp_mode == kSaveFPRegs) {
631 bytes += kDoubleSize * XMMRegister::kNumRegisters;
632 }

637int TurboAssembler::PushCallerSaved(SaveFPRegsMode fp_mode, Register exclusion1,
638 Register exclusion2, Register exclusion3) {
639 // We don't allow a GC during a store buffer overflow so there is no need to
640 // store the registers in any particular way, but we do have to store and
641 // restore them.
642 int bytes = 0;
643 for (int i = 0; i < kNumberOfSavedRegs; i++) {
644 Register reg = saved_regs[i];
645 if (reg != exclusion1 && reg != exclusion2 && reg != exclusion3) {
646 pushq(reg);
647 bytes += kSystemPointerSize;
648 }
649 }
650
651 // R12 to r15 are callee save on all platforms.
652 if (fp_mode == kSaveFPRegs) {
653 int delta = kDoubleSize * XMMRegister::kNumRegisters;
654 AllocateStackSpace(delta);
655 for (int i = 0; i < XMMRegister::kNumRegisters; i++) {
656 XMMRegister reg = XMMRegister::from_code(i);
657 Movsd(Operand(rsp, i * kDoubleSize), reg);
658 }
659 bytes += delta;
660 }
661
662 return bytes;
663}

665int TurboAssembler::PopCallerSaved(SaveFPRegsMode fp_mode, Register exclusion1,
666 Register exclusion2, Register exclusion3) {
667 int bytes = 0;
668 if (fp_mode == kSaveFPRegs) {
669 for (int i = 0; i < XMMRegister::kNumRegisters; i++) {
670 XMMRegister reg = XMMRegister::from_code(i);
671 Movsd(reg, Operand(rsp, i * kDoubleSize));
672 }
673 int delta = kDoubleSize * XMMRegister::kNumRegisters;
674 addq(rsp, Immediate(kDoubleSize * XMMRegister::kNumRegisters));
675 bytes += delta;
676 }
677
678 for (int i = kNumberOfSavedRegs - 1; i >= 0; i--) {
679 Register reg = saved_regs[i];
680 if (reg != exclusion1 && reg != exclusion2 && reg != exclusion3) {
681 popq(reg);
682 bytes += kSystemPointerSize;
683 }
684 }
685
686 return bytes;
687}

950template <typename OperandOrXMMRegister, bool is_double>
951void ConvertFloatToUint64(TurboAssembler* tasm, Register dst,
952 OperandOrXMMRegister src, Label* fail) {
953 Label success;
954 // There does not exist a native float-to-uint instruction, so we have to use
955 // a float-to-int, and postprocess the result.
956 if (is_double) {
957 tasm->Cvttsd2siq(dst, src);
958 } else {
959 tasm->Cvttss2siq(dst, src);
960 }
961 // If the result of the conversion is positive, we are already done.
962 tasm->testq(dst, dst);
963 tasm->j(positive, &success);
964 // The result of the first conversion was negative, which means that the
965 // input value was not within the positive int64 range. We subtract 2^63
966 // and convert it again to see if it is within the uint64 range.
967 if (is_double) {
968 tasm->Move(kScratchDoubleReg, -9223372036854775808.0);
969 tasm->Addsd(kScratchDoubleReg, src);
970 tasm->Cvttsd2siq(dst, kScratchDoubleReg);
971 } else {
972 tasm->Move(kScratchDoubleReg, -9223372036854775808.0f);
973 tasm->Addss(kScratchDoubleReg, src);
974 tasm->Cvttss2siq(dst, kScratchDoubleReg);
975 }
976 tasm->testq(dst, dst);
977 // The only possible negative value here is 0x80000000000000000, which is
978 // used on x64 to indicate an integer overflow.
979 tasm->j(negative, fail ? fail : &success);
980 // The input value is within uint64 range and the second conversion worked
981 // successfully, but we still have to undo the subtraction we did
982 // earlier.
983 tasm->Set(kScratchRegister, 0x8000000000000000);
984 tasm->orq(dst, kScratchRegister);
985 tasm->bind(&success);
986}

1621void TurboAssembler::LoadCodeObjectEntry(Register destination,
1622 Register code_object) {
1623 // Code objects are called differently depending on whether we are generating
1624 // builtin code (which will later be embedded into the binary) or compiling
1625 // user JS code at runtime.
1626 // * Builtin code runs in --jitless mode and thus must not call into on-heap
1627 // Code targets. Instead, we dispatch through the builtins entry table.
1628 // * Codegen at runtime does not have this restriction and we can use the
1629 // shorter, branchless instruction sequence. The assumption here is that
1630 // targets are usually generated code and not builtin Code objects.
1631
1632 if (options().isolate_independent_code) {
1633 DCHECK(root_array_available())((void) 0);
1634 Label if_code_is_off_heap, out;
1635
1636 // Check whether the Code object is an off-heap trampoline. If so, call its
1637 // (off-heap) entry point directly without going through the (on-heap)
1638 // trampoline. Otherwise, just call the Code object as always.
1639 testl(FieldOperand(code_object, Code::kFlagsOffset),
1640 Immediate(Code::IsOffHeapTrampoline::kMask));
1641 j(not_equal, &if_code_is_off_heap);
1642
1643 // Not an off-heap trampoline, the entry point is at
1644 // Code::raw_instruction_start().
1645 Move(destination, code_object);
1646 addq(destination, Immediate(Code::kHeaderSize - kHeapObjectTag));
1647 jmp(&out);
1648
1649 // An off-heap trampoline, the entry point is loaded from the builtin entry
1650 // table.
1651 bind(&if_code_is_off_heap);
1652 movl(destination, FieldOperand(code_object, Code::kBuiltinIndexOffset));
1653 movq(destination,
1654 Operand(kRootRegister, destination, times_system_pointer_size,
1655 IsolateData::builtin_entry_table_offset()));
1656
1657 bind(&out);
1658 } else {
1659 Move(destination, code_object);
1660 addq(destination, Immediate(Code::kHeaderSize - kHeapObjectTag));
1661 }
1662}

2755 Condition cc, Label* condition_met,
2756 Label::Distance condition_met_distance) {
2757 DCHECK(cc == zero || cc == not_zero)((void) 0);
2758 if (scratch == object) {
2759 andq(scratch, Immediate(~kPageAlignmentMask));
2760 } else {
2761 movq(scratch, Immediate(~kPageAlignmentMask));
2762 andq(scratch, object);
2763 }
2764 if (mask < (1 << kBitsPerByte)) {
2765 testb(Operand(scratch, BasicMemoryChunk::kFlagsOffset),
2766 Immediate(static_cast<uint8_t>(mask)));
2767 } else {
2768 testl(Operand(scratch, BasicMemoryChunk::kFlagsOffset), Immediate(mask));
2769 }
2770 j(cc, condition_met, condition_met_distance);
2771}
