188// A set of work-stealing queues (vectors of units). Each background compile
189// task owns one of the queues and steals from all others once its own queue
190// runs empty.
191class CompilationUnitQueues {
192 public:
193 explicit CompilationUnitQueues(int max_tasks) : queues_(max_tasks) {
194 DCHECK_LT(0, max_tasks)((void) 0);
195 for (int task_id = 0; task_id < max_tasks; ++task_id) {
196 queues_[task_id].next_steal_task_id = next_task_id(task_id);
197 }
198 for (auto& atomic_counter : num_units_) {
199 std::atomic_init(&atomic_counter, size_t{0});
200 }
201 }
202
203 base::Optional<WasmCompilationUnit> GetNextUnit(
204 int task_id, CompileBaselineOnly baseline_only) {
205 DCHECK_LE(0, task_id)((void) 0);
206 DCHECK_GT(queues_.size(), task_id)((void) 0);
207
208 // As long as any lower-tier units are outstanding we need to steal them
209 // before executing own higher-tier units.
210 int max_tier = baseline_only ? kBaseline : kTopTier;
211 for (int tier = GetLowestTierWithUnits(); tier <= max_tier; ++tier) {
212 if (auto unit = GetNextUnitOfTier(task_id, tier)) {
213 size_t old_units_count =
214 num_units_[tier].fetch_sub(1, std::memory_order_relaxed);
215 DCHECK_LE(1, old_units_count)((void) 0);
216 USE(old_units_count)do { ::v8::base::Use unused_tmp_array_for_use_macro[]{old_units_count
}; (void)unused_tmp_array_for_use_macro; } while (false);
217 return unit;
218 }
219 }
220 return {};
221 }
222
223 void AddUnits(Vector<WasmCompilationUnit> baseline_units,
224 Vector<WasmCompilationUnit> top_tier_units,
225 const WasmModule* module) {
226 DCHECK_LT(0, baseline_units.size() + top_tier_units.size())((void) 0);
227 // Add to the individual queues in a round-robin fashion. No special care is
228 // taken to balance them; they will be balanced by work stealing.
229 int queue_to_add = next_queue_to_add.load(std::memory_order_relaxed);
230 while (!next_queue_to_add.compare_exchange_weak(
231 queue_to_add, next_task_id(queue_to_add), std::memory_order_relaxed)) {
232 // Retry with updated {queue_to_add}.
233 }
234
235 Queue* queue = &queues_[queue_to_add];
236 base::MutexGuard guard(&queue->mutex);
237 base::Optional<base::MutexGuard> big_units_guard;
238 for (auto pair : {std::make_pair(int{kBaseline}, baseline_units),
239 std::make_pair(int{kTopTier}, top_tier_units)}) {
240 int tier = pair.first;
241 Vector<WasmCompilationUnit> units = pair.second;
242 if (units.empty()) continue;
243 num_units_[tier].fetch_add(units.size(), std::memory_order_relaxed);
244 for (WasmCompilationUnit unit : units) {
245 size_t func_size = module->functions[unit.func_index()].code.length();
246 if (func_size <= kBigUnitsLimit) {
247 queue->units[tier].push_back(unit);
248 } else {
249 if (!big_units_guard) {
250 big_units_guard.emplace(&big_units_queue_.mutex);
251 }
252 big_units_queue_.has_units[tier].store(true,
253 std::memory_order_relaxed);
254 big_units_queue_.units[tier].emplace(func_size, unit);
255 }
256 }
257 }
258 }

260 // Get the current total number of units in all queues. This is only a
261 // momentary snapshot, it's not guaranteed that {GetNextUnit} returns a unit
262 // if this method returns non-zero.
263 size_t GetTotalSize() const {
264 size_t total = 0;
265 for (auto& atomic_counter : num_units_) {
266 total += atomic_counter.load(std::memory_order_relaxed);
267 }
268 return total;
269 }

271 private:
272 // Store tier in int so we can easily loop over it:
273 static constexpr int kBaseline = 0;
274 static constexpr int kTopTier = 1;
275 static constexpr int kNumTiers = kTopTier + 1;
276
277 // Functions bigger than {kBigUnitsLimit} will be compiled first, in ascending
278 // order of their function body size.
279 static constexpr size_t kBigUnitsLimit = 4096;

281 struct Queue {
282 base::Mutex mutex;
283
284 // Protected by {mutex}:
285 std::vector<WasmCompilationUnit> units[kNumTiers];
286 int next_steal_task_id;
287 // End of fields protected by {mutex}.
288 };

290 struct BigUnit {
291 BigUnit(size_t func_size, WasmCompilationUnit unit)
292 : func_size{func_size}, unit(unit) {}
293
294 size_t func_size;
295 WasmCompilationUnit unit;
296
297 bool operator<(const BigUnit& other) const {
298 return func_size < other.func_size;
299 }

302 struct BigUnitsQueue {
303 BigUnitsQueue() {
304 for (auto& atomic : has_units) std::atomic_init(&atomic, false);
305 }
306
307 base::Mutex mutex;
308
309 // Can be read concurrently to check whether any elements are in the queue.
310 std::atomic<bool> has_units[kNumTiers];
311
312 // Protected by {mutex}:
313 std::priority_queue<BigUnit> units[kNumTiers];
314 };
315
316 std::vector<Queue> queues_;
317 BigUnitsQueue big_units_queue_;
318
319 std::atomic<size_t> num_units_[kNumTiers];
320 std::atomic<int> next_queue_to_add{0};
321
322 int next_task_id(int task_id) const {
323 int next = task_id + 1;
324 return next == static_cast<int>(queues_.size()) ? 0 : next;
325 }
326
327 int GetLowestTierWithUnits() const {
328 for (int tier = 0; tier < kNumTiers; ++tier) {
329 if (num_units_[tier].load(std::memory_order_relaxed) > 0) return tier;
330 }
331 return kNumTiers;
332 }
333
334 base::Optional<WasmCompilationUnit> GetNextUnitOfTier(int task_id, int tier) {
335 Queue* queue = &queues_[task_id];
336 // First check whether there is a big unit of that tier. Execute that first.
337 if (auto unit = GetBigUnitOfTier(tier)) return unit;
338
339 // Then check whether our own queue has a unit of the wanted tier. If
340 // so, return it, otherwise get the task id to steal from.
341 int steal_task_id;
342 {
343 base::MutexGuard mutex_guard(&queue->mutex);
344 if (!queue->units[tier].empty()) {
345 auto unit = queue->units[tier].back();
346 queue->units[tier].pop_back();
347 return unit;
348 }
349 steal_task_id = queue->next_steal_task_id;
350 }
351
352 // Try to steal from all other queues. If this succeeds, return one of the
353 // stolen units.
354 size_t steal_trials = queues_.size();
355 for (; steal_trials > 0;
356 --steal_trials, steal_task_id = next_task_id(steal_task_id)) {
357 if (steal_task_id == task_id) continue;
358 if (auto unit = StealUnitsAndGetFirst(task_id, steal_task_id, tier)) {
359 return unit;
360 }
361 }
362
363 // If we reach here, we didn't find any unit of the requested tier.
364 return {};
365 }
366
367 base::Optional<WasmCompilationUnit> GetBigUnitOfTier(int tier) {
368 // Fast-path without locking.
369 if (!big_units_queue_.has_units[tier].load(std::memory_order_relaxed)) {
370 return {};
371 }
372 base::MutexGuard guard(&big_units_queue_.mutex);
373 if (big_units_queue_.units[tier].empty()) return {};
374 WasmCompilationUnit unit = big_units_queue_.units[tier].top().unit;
375 big_units_queue_.units[tier].pop();
376 if (big_units_queue_.units[tier].empty()) {
377 big_units_queue_.has_units[tier].store(false, std::memory_order_relaxed);
378 }
379 return unit;
380 }
381
382 // Steal units of {wanted_tier} from {steal_from_task_id} to {task_id}. Return
383 // first stolen unit (rest put in queue of {task_id}), or {nullopt} if
384 // {steal_from_task_id} had no units of {wanted_tier}.
385 base::Optional<WasmCompilationUnit> StealUnitsAndGetFirst(
386 int task_id, int steal_from_task_id, int wanted_tier) {
387 DCHECK_NE(task_id, steal_from_task_id)((void) 0);
388 std::vector<WasmCompilationUnit> stolen;
389 base::Optional<WasmCompilationUnit> returned_unit;
390 {
391 Queue* steal_queue = &queues_[steal_from_task_id];
392 base::MutexGuard guard(&steal_queue->mutex);
393 auto* steal_from_vector = &steal_queue->units[wanted_tier];
394 if (steal_from_vector->empty()) return {};
395 size_t remaining = steal_from_vector->size() / 2;
396 auto steal_begin = steal_from_vector->begin() + remaining;
397 returned_unit = *steal_begin;
398 stolen.assign(steal_begin + 1, steal_from_vector->end());
399 steal_from_vector->erase(steal_begin, steal_from_vector->end());
400 }
401 Queue* queue = &queues_[task_id];
402 base::MutexGuard guard(&queue->mutex);
403 auto* target_queue = &queue->units[wanted_tier];
404 target_queue->insert(target_queue->end(), stolen.begin(), stolen.end());
405 queue->next_steal_task_id = next_task_id(steal_from_task_id);
406 return returned_unit;
407 }
408};

1255// Returns the number of units added.
1256int AddImportWrapperUnits(NativeModule* native_module,
1257 CompilationUnitBuilder* builder) {
1258 std::unordered_set<WasmImportWrapperCache::CacheKey,
1259 WasmImportWrapperCache::CacheKeyHash>
1260 keys;
1261 int num_imported_functions = native_module->num_imported_functions();
1262 for (int func_index = 0; func_index < num_imported_functions; func_index++) {
1263 const FunctionSig* sig = native_module->module()->functions[func_index].sig;
1264 if (!IsJSCompatibleSignature(sig, native_module->enabled_features())) {
1265 continue;
1266 }
1267 WasmImportWrapperCache::CacheKey key(
1268 compiler::kDefaultImportCallKind, sig,
1269 static_cast<int>(sig->parameter_count()));
1270 auto it = keys.insert(key);
1271 if (it.second) {
1272 // Ensure that all keys exist in the cache, so that we can populate the
1273 // cache later without locking.
1274 (*native_module->import_wrapper_cache())[key] = nullptr;
1275 builder->AddUnits(func_index);
1276 }
1277 }
1278 return static_cast<int>(keys.size());

1281void InitializeCompilationUnits(Isolate* isolate, NativeModule* native_module) {
1282 CompilationStateImpl* compilation_state =
1283 Impl(native_module->compilation_state());
1284 const bool lazy_module = IsLazyModule(native_module->module());
1285 ModuleWireBytes wire_bytes(native_module->wire_bytes());
1286 CompilationUnitBuilder builder(native_module);
1287 auto* module = native_module->module();
1288 const bool prefer_liftoff = native_module->IsTieredDown();
1289
1290 uint32_t start = module->num_imported_functions;
1291 uint32_t end = start + module->num_declared_functions;
1292 for (uint32_t func_index = start; func_index < end; func_index++) {
1293 if (prefer_liftoff) {
1294 builder.AddRecompilationUnit(func_index, ExecutionTier::kLiftoff);
1295 continue;
1296 }
1297 CompileStrategy strategy = GetCompileStrategy(
1298 module, native_module->enabled_features(), func_index, lazy_module);
1299 if (strategy == CompileStrategy::kLazy) {
1300 native_module->UseLazyStub(func_index);
1301 } else if (strategy == CompileStrategy::kLazyBaselineEagerTopTier) {
1302 builder.AddTopTierUnit(func_index);
1303 native_module->UseLazyStub(func_index);
1304 } else {
1305 DCHECK_EQ(strategy, CompileStrategy::kEager)((void) 0);
1306 builder.AddUnits(func_index);
1307 }
1308 }
1309 int num_import_wrappers = AddImportWrapperUnits(native_module, &builder);
1310 int num_export_wrappers =
1311 AddExportWrapperUnits(isolate, isolate->wasm_engine(), native_module,
1312 &builder, WasmFeatures::FromIsolate(isolate));
1313 compilation_state->InitializeCompilationProgress(
1314 lazy_module, num_import_wrappers + num_export_wrappers);
1315 builder.Commit();
1316}

1852class AsyncCompileJob::CompilationStateCallback {
1853 public:
1854 explicit CompilationStateCallback(AsyncCompileJob* job) : job_(job) {}
1855
1856 void operator()(CompilationEvent event) {
1857 // This callback is only being called from a foreground task.
1858 switch (event) {
1859 case CompilationEvent::kFinishedBaselineCompilation:
1860 DCHECK(!last_event_.has_value())((void) 0);
1861 if (job_->DecrementAndCheckFinisherCount()) {
1862 // Install the native module in the cache, or reuse a conflicting one.
1863 // If we get a conflicting module, wait until we are back in the
1864 // main thread to update {job_->native_module_} to avoid a data race.
1865 std::shared_ptr<NativeModule> native_module = job_->native_module_;
1866 bool cache_hit =
1867 !job_->isolate_->wasm_engine()->UpdateNativeModuleCache(
1868 false, &native_module, job_->isolate_);
1869 DCHECK_EQ(cache_hit, native_module != job_->native_module_)((void) 0);
1870 job_->DoSync<CompileFinished>(cache_hit ? std::move(native_module)
1871 : nullptr);
1872 }
1873 break;
1874 case CompilationEvent::kFinishedTopTierCompilation:
1875 DCHECK_EQ(CompilationEvent::kFinishedBaselineCompilation, last_event_)((void) 0);
1876 // At this point, the job will already be gone, thus do not access it
1877 // here.
1878 break;
1879 case CompilationEvent::kFailedCompilation:
1880 DCHECK(!last_event_.has_value())((void) 0);
1881 if (job_->DecrementAndCheckFinisherCount()) {
1882 // Don't update {job_->native_module_} to avoid data races with other
1883 // compilation threads. Use a copy of the shared pointer instead.
1884 std::shared_ptr<NativeModule> native_module = job_->native_module_;
1885 job_->isolate_->wasm_engine()->UpdateNativeModuleCache(
1886 true, &native_module, job_->isolate_);
1887 job_->DoSync<CompileFailed>();
1888 }
1889 break;
1890 case CompilationEvent::kFinishedRecompilation:
1891 // This event can happen either before or after
1892 // {kFinishedTopTierCompilation}, hence don't remember this in
1893 // {last_event_}.
1894 return;
1895 }
1896#ifdef DEBUG
1897 last_event_ = event;
1898#endif
1899 }
1900
1901 private:
1902 AsyncCompileJob* job_;
1903#ifdef DEBUG
1904 // This will be modified by different threads, but they externally
1905 // synchronize, so no explicit synchronization (currently) needed here.
1906 base::Optional<CompilationEvent> last_event_;
1907#endif
1908};

2617 DCHECK(!failed())((void) 0);
2618 auto enabled_features = native_module_->enabled_features();
2619 auto* module = native_module_->module();
2620
2621 base::MutexGuard guard(&callbacks_mutex_);
2622 DCHECK_EQ(0, outstanding_baseline_units_)((void) 0);
2623 DCHECK_EQ(0, outstanding_top_tier_functions_)((void) 0);
2624 compilation_progress_.reserve(module->num_declared_functions);
2625 int start = module->num_imported_functions;
2626 int end = start + module->num_declared_functions;
2627
2628 const bool prefer_liftoff = native_module_->IsTieredDown();
2629 for (int func_index = start; func_index < end; func_index++) {

2632 RequiredTopTierField::encode(ExecutionTier::kLiftoff) |
2633 RequiredBaselineTierField::encode(ExecutionTier::kLiftoff) |
2634 ReachedTierField::encode(ExecutionTier::kNone);
2635 compilation_progress_.push_back(kLiftoffOnlyFunctionProgress);
2636 outstanding_baseline_units_++;
2637 outstanding_top_tier_functions_++;
2638 continue;
2639 }

2641 module, compile_mode(), enabled_features, func_index);
2642 CompileStrategy strategy =
2643 GetCompileStrategy(module, enabled_features, func_index, lazy_module);
2644
2645 bool required_for_baseline = strategy == CompileStrategy::kEager;
2646 bool required_for_top_tier = strategy != CompileStrategy::kLazy;
2647 DCHECK_EQ(required_for_top_tier,((void) 0)
2648 strategy == CompileStrategy::kEager ||((void) 0)
2649 strategy == CompileStrategy::kLazyBaselineEagerTopTier)((void) 0);
2650
2651 // Count functions to complete baseline and top tier compilation.
2652 if (required_for_baseline) outstanding_baseline_units_++;
2653 if (required_for_top_tier) outstanding_top_tier_functions_++;
2654
2655 // Initialize function's compilation progress.
2656 ExecutionTier required_baseline_tier = required_for_baseline
2657 ? requested_tiers.baseline_tier
2658 : ExecutionTier::kNone;
2659 ExecutionTier required_top_tier =
2660 required_for_top_tier ? requested_tiers.top_tier : ExecutionTier::kNone;
2661 uint8_t function_progress = ReachedTierField::encode(ExecutionTier::kNone);
2662 function_progress = RequiredBaselineTierField::update(
2663 function_progress, required_baseline_tier);
2664 function_progress =
2665 RequiredTopTierField::update(function_progress, required_top_tier);
2666 compilation_progress_.push_back(function_progress);

2669 DCHECK_IMPLIES(lazy_module, outstanding_top_tier_functions_ == 0)((void) 0);
2670 DCHECK_LE(0, outstanding_baseline_units_)((void) 0);
2671 DCHECK_LE(outstanding_baseline_units_, outstanding_top_tier_functions_)((void) 0);
2672 outstanding_baseline_units_ += num_wrappers;
2673
2674 // Trigger callbacks if module needs no baseline or top tier compilation. This
2675 // can be the case for an empty or fully lazy module.
2676 TriggerCallbacks();
2677}

2691void CompilationStateImpl::InitializeRecompilation(
2692 TieringState new_tiering_state,
2693 CompilationState::callback_t recompilation_finished_callback) {
2694 DCHECK(!failed())((void) 0);
2695
2696 // Generate necessary compilation units on the fly.
2697 CompilationUnitBuilder builder(native_module_);
2698
2699 // Information about compilation progress is shared between this class and the
2700 // NativeModule. Before updating information here, consult the NativeModule to
2701 // find all functions that need recompilation.
2702 // Since the current tiering state is updated on the NativeModule before
2703 // triggering recompilation, it's OK if the information is slightly outdated.
2704 // If we compile functions twice, the NativeModule will ignore all redundant
2705 // code (or code compiled for the wrong tier).
2706 std::vector<int> recompile_function_indexes =
2707 native_module_->FindFunctionsToRecompile(new_tiering_state);
2708
2709 {
2710 base::MutexGuard guard(&callbacks_mutex_);
2711
2712 callbacks_.emplace_back(std::move(recompilation_finished_callback));
2713 tiering_state_ = new_tiering_state;
2714
2715 // If compilation progress is not initialized yet, then compilation didn't
2716 // start yet, and new code will be kept tiered-down from the start. For
2717 // streaming compilation, there is a special path to tier down later, when
2718 // the module is complete. In any case, we don't need to recompile here.
2719 if (compilation_progress_.size() > 0) {
2720 const WasmModule* module = native_module_->module();
2721 DCHECK_EQ(module->num_declared_functions, compilation_progress_.size())((void) 0);
2722 DCHECK_GE(module->num_declared_functions,((void) 0)
2723 recompile_function_indexes.size())((void) 0);
2724 outstanding_recompilation_functions_ =
2725 static_cast<int>(recompile_function_indexes.size());
2726 // Restart recompilation if another recompilation is already happening.
2727 for (auto& progress : compilation_progress_) {
2728 progress = MissingRecompilationField::update(progress, false);
2729 }
2730 auto new_tier = new_tiering_state == kTieredDown
2731 ? ExecutionTier::kLiftoff
2732 : ExecutionTier::kTurbofan;
2733 int imported = module->num_imported_functions;
2734 for (int function_index : recompile_function_indexes) {
2735 DCHECK_LE(imported, function_index)((void) 0);
2736 int slot_index = function_index - imported;
2737 auto& progress = compilation_progress_[slot_index];
2738 progress = MissingRecompilationField::update(progress, true);
2739 builder.AddRecompilationUnit(function_index, new_tier);
2740 }
2741 }
2742
2743 // Trigger callback if module needs no recompilation.
2744 if (outstanding_recompilation_functions_ == 0) {
2745 TriggerCallbacks(base::EnumSet<CompilationEvent>(
2746 {CompilationEvent::kFinishedRecompilation}));
2747 }
2748 }
2749
2750 builder.Commit();
2751}
