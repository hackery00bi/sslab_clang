137template <typename T>
138Handle<T> HandleScope::CloseAndEscape(Handle<T> handle_value) {
139 HandleScopeData* current = isolate_->handle_scope_data();
140 T value = *handle_value;
141 // Throw away all handles in the current scope.
142 CloseScope(isolate_, prev_next_, prev_limit_);
143 // Allocate one handle in the parent scope.
144 DCHECK(current->level > current->sealed_level)((void) 0);
145 Handle<T> result(value, isolate_);
146 // Reinitialize the current scope (so that it's ready
147 // to be used or closed again).
148 prev_next_ = current->next;
149 prev_limit_ = current->limit;
150 current->level++;
151 return result;
152}

179inline SealHandleScope::SealHandleScope(Isolate* isolate) : isolate_(isolate) {
