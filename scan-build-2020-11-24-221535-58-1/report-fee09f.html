893// WebAssembly.instantiate(module, imports) -> WebAssembly.Instance
894// WebAssembly.instantiate(bytes, imports) ->
895// {module: WebAssembly.Module, instance: WebAssembly.Instance}
896void WebAssemblyInstantiate(const v8::FunctionCallbackInfo<v8::Value>& args) {
897 constexpr const char* kAPIMethodName = "WebAssembly.instantiate()";
898 v8::Isolate* isolate = args.GetIsolate();
899 i::Isolate* i_isolate = reinterpret_cast<i::Isolate*>(isolate);
900 i_isolate->CountUsage(
901 v8::Isolate::UseCounterFeature::kWebAssemblyInstantiation);
902
903 ScheduledErrorThrower thrower(i_isolate, kAPIMethodName);
904
905 HandleScope scope(isolate);
906
907 Local<Context> context = isolate->GetCurrentContext();
908
909 ASSIGN(Promise::Resolver, promise_resolver, Promise::Resolver::New(context));
910 Local<Promise> promise = promise_resolver->GetPromise();
911 args.GetReturnValue().Set(promise);
912
913 std::unique_ptr<i::wasm::InstantiationResultResolver> resolver(
914 new InstantiateModuleResultResolver(i_isolate,
915 Utils::OpenHandle(*promise)));
916
917 Local<Value> first_arg_value = args[0];
918 i::Handle<i::Object> first_arg = Utils::OpenHandle(*first_arg_value);
919 if (!first_arg->IsJSObject()) {
920 thrower.TypeError(
921 "Argument 0 must be a buffer source or a WebAssembly.Module object");
922 resolver->OnInstantiationFailed(thrower.Reify());
923 return;
924 }
925
926 // If args.Length < 2, this will be undefined - see FunctionCallbackInfo.
927 Local<Value> ffi = args[1];
928 i::MaybeHandle<i::JSReceiver> maybe_imports =
929 GetValueAsImports(ffi, &thrower);
930
931 if (thrower.error()) {
932 resolver->OnInstantiationFailed(thrower.Reify());
933 return;
934 }
935
936 if (first_arg->IsWasmModuleObject()) {
937 i::Handle<i::WasmModuleObject> module_obj =
938 i::Handle<i::WasmModuleObject>::cast(first_arg);
939
940 i_isolate->wasm_engine()->AsyncInstantiate(i_isolate, std::move(resolver),
941 module_obj, maybe_imports);
942 return;
943 }
944
945 bool is_shared = false;
946 auto bytes = GetFirstArgumentAsBytes(args, &thrower, &is_shared);
947 if (thrower.error()) {
948 resolver->OnInstantiationFailed(thrower.Reify());
949 return;
950 }
951
952 // We start compilation now, we have no use for the
953 // {InstantiationResultResolver}.
954 resolver.reset();
955
956 std::shared_ptr<i::wasm::CompilationResultResolver> compilation_resolver(
957 new AsyncInstantiateCompileResultResolver(
958 i_isolate, Utils::OpenHandle(*promise), maybe_imports));
959
960 // The first parameter is a buffer source, we have to check if we are allowed
961 // to compile it.
962 if (!i::wasm::IsWasmCodegenAllowed(i_isolate, i_isolate->native_context())) {
963 thrower.CompileError("Wasm code generation disallowed by embedder");
964 compilation_resolver->OnCompilationFailed(thrower.Reify());
965 return;
966 }
967
968 // Asynchronous compilation handles copying wire bytes if necessary.
969 auto enabled_features = i::wasm::WasmFeatures::FromIsolate(i_isolate);
970 i_isolate->wasm_engine()->AsyncCompile(i_isolate, enabled_features,
971 std::move(compilation_resolver), bytes,
972 is_shared, kAPIMethodName);
973}

1194// Determines the type encoded in a value type property (e.g. type reflection).
1195// Returns false if there was an exception, true upon success. On success the

1202 if (!maybe.ToLocal(&value)) return false;
1203 v8::Local<v8::String> string;

1205 if (string->StringEquals(v8_str(isolate, "i32"))) {

1210 *type = i::wasm::kWasmI64;
1211 } else if (string->StringEquals(v8_str(isolate, "f64"))) {

1213 } else if (enabled_features.has_reftypes() &&
1214 string->StringEquals(v8_str(isolate, "externref"))) {
1215 *type = i::wasm::kWasmExternRef;
1216 // The JS api spec uses 'anyfunc' instead of 'funcref'.
1217 } else if (enabled_features.has_reftypes() &&
1218 string->StringEquals(v8_str(isolate, "anyfunc"))) {
1219 *type = i::wasm::kWasmFuncRef;
1220 } else if (enabled_features.has_eh() &&
1221 string->StringEquals(v8_str(isolate, "exnref"))) {
1222 *type = i::wasm::kWasmExnRef;
1223 } else if (enabled_features.has_gc() &&
1224 string->StringEquals(v8_str(isolate, "eqref"))) {
1225 *type = i::wasm::kWasmEqRef;
1226 } else {
1227 // Unrecognized type.
1228 *type = i::wasm::kWasmStmt;
1229 }
1230 return true;

1233// WebAssembly.Global
1234void WebAssemblyGlobal(const v8::FunctionCallbackInfo<v8::Value>& args) {
1235 v8::Isolate* isolate = args.GetIsolate();
1236 i::Isolate* i_isolate = reinterpret_cast<i::Isolate*>(isolate);
1237 HandleScope scope(isolate);
1238 ScheduledErrorThrower thrower(i_isolate, "WebAssembly.Global()");
1239 if (!args.IsConstructCall()) {
1240 thrower.TypeError("WebAssembly.Global must be invoked with 'new'");
1241 return;
1242 }
1243 if (!args[0]->IsObject()) {
1244 thrower.TypeError("Argument 0 must be a global descriptor");
1245 return;
1246 }
1247 Local<Context> context = isolate->GetCurrentContext();
1248 Local<v8::Object> descriptor = Local<Object>::Cast(args[0]);
1249 auto enabled_features = i::wasm::WasmFeatures::FromIsolate(i_isolate);
1250
1251 // The descriptor's 'mutable'.
1252 bool is_mutable = false;
1253 {
1254 Local<String> mutable_key = v8_str(isolate, "mutable");
1255 v8::MaybeLocal<v8::Value> maybe = descriptor->Get(context, mutable_key);
1256 v8::Local<v8::Value> value;
1257 if (maybe.ToLocal(&value)) {
1258 is_mutable = value->BooleanValue(isolate);
1259 } else {
1260 DCHECK(i_isolate->has_scheduled_exception())((void) 0);
1261 return;
1262 }
1263 }
1264
1265 // The descriptor's type, called 'value'. It is called 'value' because this
1266 // descriptor is planned to be re-used as the global's type for reflection,
1267 // so calling it 'type' is redundant.
1268 i::wasm::ValueType type;
1269 {
1270 v8::MaybeLocal<v8::Value> maybe =
1271 descriptor->Get(context, v8_str(isolate, "value"));
1272 if (!GetValueType(isolate, maybe, context, &type, enabled_features)) return;
1273 if (type == i::wasm::kWasmStmt) {
1274 thrower.TypeError(
1275 "Descriptor property 'value' must be a WebAssembly type");
1276 return;
1277 }
1278 }
1279
1280 const uint32_t offset = 0;
1281 i::MaybeHandle<i::WasmGlobalObject> maybe_global_obj =
1282 i::WasmGlobalObject::New(i_isolate, i::MaybeHandle<i::JSArrayBuffer>(),
1283 i::MaybeHandle<i::FixedArray>(), type, offset,
1284 is_mutable);
1285
1286 i::Handle<i::WasmGlobalObject> global_obj;
1287 if (!maybe_global_obj.ToHandle(&global_obj)) {
1288 thrower.RangeError("could not allocate memory");
1289 return;
1290 }
1291
1292 // Convert value to a WebAssembly value, the default value is 0.
1293 Local<v8::Value> value = Local<Value>::Cast(args[1]);
1294 switch (type.kind()) {
1295 case i::wasm::ValueType::kI32: {
1296 int32_t i32_value = 0;
1297 if (!value->IsUndefined()) {
1298 v8::Local<v8::Int32> int32_value;
1299 if (!value->ToInt32(context).ToLocal(&int32_value)) return;
1300 if (!int32_value->Int32Value(context).To(&i32_value)) return;
1301 }
1302 global_obj->SetI32(i32_value);
1303 break;
1304 }
1305 case i::wasm::ValueType::kI64: {
1306 int64_t i64_value = 0;
1307 if (!value->IsUndefined()) {
1308 if (!enabled_features.has_bigint()) {
1309 thrower.TypeError("Can't set the value of i64 WebAssembly.Global");
1310 return;
1311 }
1312
1313 v8::Local<v8::BigInt> bigint_value;
1314 if (!value->ToBigInt(context).ToLocal(&bigint_value)) return;
1315 i64_value = bigint_value->Int64Value();
1316 }
1317 global_obj->SetI64(i64_value);
1318 break;
1319 }
1320 case i::wasm::ValueType::kF32: {
1321 float f32_value = 0;
1322 if (!value->IsUndefined()) {
1323 double f64_value = 0;
1324 v8::Local<v8::Number> number_value;
1325 if (!value->ToNumber(context).ToLocal(&number_value)) return;
1326 if (!number_value->NumberValue(context).To(&f64_value)) return;
1327 f32_value = i::DoubleToFloat32(f64_value);
1328 }
1329 global_obj->SetF32(f32_value);
1330 break;
1331 }
1332 case i::wasm::ValueType::kF64: {
1333 double f64_value = 0;
1334 if (!value->IsUndefined()) {
1335 v8::Local<v8::Number> number_value;
1336 if (!value->ToNumber(context).ToLocal(&number_value)) return;
1337 if (!number_value->NumberValue(context).To(&f64_value)) return;
1338 }
1339 global_obj->SetF64(f64_value);
1340 break;
1341 }
1342 case i::wasm::ValueType::kRef:
1343 case i::wasm::ValueType::kOptRef: {
1344 switch (type.heap_representation()) {
1345 case i::wasm::HeapType::kExtern:
1346 case i::wasm::HeapType::kExn: {
1347 if (args.Length() < 2) {
1348 // When no initial value is provided, we have to use the WebAssembly
1349 // default value 'null', and not the JS default value 'undefined'.
1350 global_obj->SetExternRef(i_isolate->factory()->null_value());
1351 break;
1352 }
1353 global_obj->SetExternRef(Utils::OpenHandle(*value));
1354 break;
1355 }
1356 case i::wasm::HeapType::kFunc: {
1357 if (args.Length() < 2) {
1358 // When no initial value is provided, we have to use the WebAssembly
1359 // default value 'null', and not the JS default value 'undefined'.
1360 global_obj->SetFuncRef(i_isolate,
1361 i_isolate->factory()->null_value());
1362 break;
1363 }
1364
1365 if (!global_obj->SetFuncRef(i_isolate, Utils::OpenHandle(*value))) {
1366 thrower.TypeError(
1367 "The value of funcref globals must be null or an "
1368 "exported function");
1369 }
1370 break;
1371 }
1372 case i::wasm::HeapType::kEq:
1373 default:
1374 // TODO(7748): Implement these.
1375 UNIMPLEMENTED()V8_Fatal("unimplemented code");
1376 }
1377 break;
1378 }
1379 case i::wasm::ValueType::kRtt:
1380 // TODO(7748): Implement.
1381 UNIMPLEMENTED()V8_Fatal("unimplemented code");
1382 case i::wasm::ValueType::kI8:
1383 case i::wasm::ValueType::kI16:
1384 case i::wasm::ValueType::kStmt:
1385 case i::wasm::ValueType::kS128:
1386 case i::wasm::ValueType::kBottom:
1387 UNREACHABLE()V8_Fatal("unreachable code");
1388 }
1389
1390 i::Handle<i::JSObject> global_js_object(global_obj);
1391 args.GetReturnValue().Set(Utils::ToLocal(global_js_object));
1392}

1419void WebAssemblyFunction(const v8::FunctionCallbackInfo<v8::Value>& args) {
1420 v8::Isolate* isolate = args.GetIsolate();
1421 i::Isolate* i_isolate = reinterpret_cast<i::Isolate*>(isolate);
1422 HandleScope scope(isolate);
1423 ScheduledErrorThrower thrower(i_isolate, "WebAssembly.Function()");
1424 if (!args.IsConstructCall()) {
1425 thrower.TypeError("WebAssembly.Function must be invoked with 'new'");
1426 return;
1427 }
1428 if (!args[0]->IsObject()) {
1429 thrower.TypeError("Argument 0 must be a function type");
1430 return;
1431 }
1432 Local<Object> function_type = Local<Object>::Cast(args[0]);
1433 Local<Context> context = isolate->GetCurrentContext();
1434 auto enabled_features = i::wasm::WasmFeatures::FromIsolate(i_isolate);
1435
1436 // Load the 'parameters' property of the function type.
1437 Local<String> parameters_key = v8_str(isolate, "parameters");

1439 function_type->Get(context, parameters_key);
1440 v8::Local<v8::Value> parameters_value;
1441 if (!parameters_maybe.ToLocal(&parameters_value)) return;
1442 if (!parameters_value->IsObject()) {
1443 thrower.TypeError("Argument 0 must be a function type with 'parameters'");
1444 return;

1448 if (parameters_len == i::kMaxUInt32) {
1449 thrower.TypeError("Argument 0 contains parameters without 'length'");
1450 return;
1451 }
1452 if (parameters_len > i::wasm::kV8MaxWasmFunctionParams) {
1453 thrower.TypeError("Argument 0 contains too many parameters");
1454 return;
1455 }
1456
1457 // Load the 'results' property of the function type.
1458 Local<String> results_key = v8_str(isolate, "results");
1459 v8::MaybeLocal<v8::Value> results_maybe =
1460 function_type->Get(context, results_key);
1461 v8::Local<v8::Value> results_value;
1462 if (!results_maybe.ToLocal(&results_value)) return;
1463 if (!results_value->IsObject()) {
1464 thrower.TypeError("Argument 0 must be a function type with 'results'");
1465 return;
1466 }
1467 Local<Object> results = results_value.As<Object>();
1468 uint32_t results_len = GetIterableLength(i_isolate, context, results);
1469 if (results_len == i::kMaxUInt32) {
1470 thrower.TypeError("Argument 0 contains results without 'length'");
1471 return;
1472 }
1473 if (results_len > (enabled_features.has_mv()
1474 ? i::wasm::kV8MaxWasmFunctionMultiReturns
1475 : i::wasm::kV8MaxWasmFunctionReturns)) {
1476 thrower.TypeError("Argument 0 contains too many results");
1477 return;
1478 }
1479
1480 // Decode the function type and construct a signature.
1481 i::Zone zone(i_isolate->allocator(), ZONE_NAME__func__);
1482 i::wasm::FunctionSig::Builder builder(&zone, results_len, parameters_len);
1483 for (uint32_t i = 0; i < parameters_len; ++i) {
1484 i::wasm::ValueType type;
1485 MaybeLocal<Value> maybe = parameters->Get(context, i);
1486 if (!GetValueType(isolate, maybe, context, &type, enabled_features)) return;
1487 if (type == i::wasm::kWasmStmt) {
1488 thrower.TypeError(
1489 "Argument 0 parameter type at index #%u must be a value type", i);
1490 return;
1491 }
1492 builder.AddParam(type);
1493 }
1494 for (uint32_t i = 0; i < results_len; ++i) {
1495 i::wasm::ValueType type;
1496 MaybeLocal<Value> maybe = results->Get(context, i);
1497 if (!GetValueType(isolate, maybe, context, &type, enabled_features)) return;
1498 if (type == i::wasm::kWasmStmt) {
1499 thrower.TypeError(
1500 "Argument 0 result type at index #%u must be a value type", i);
1501 return;
1502 }
1503 builder.AddReturn(type);
1504 }
1505
1506 if (!args[1]->IsFunction()) {
1507 thrower.TypeError("Argument 1 must be a function");
1508 return;
1509 }
1510 const i::wasm::FunctionSig* sig = builder.Build();
1511
1512 i::Handle<i::JSReceiver> callable =
1513 Utils::OpenHandle(*args[1].As<Function>());
1514 if (i::WasmExportedFunction::IsWasmExportedFunction(*callable)) {
1515 if (*i::Handle<i::WasmExportedFunction>::cast(callable)->sig() == *sig) {
1516 args.GetReturnValue().Set(Utils::ToLocal(callable));
1517 return;
1518 }
1519
1520 thrower.TypeError(
1521 "The signature of Argument 1 (a WebAssembly function) does "
1522 "not match the signature specified in Argument 0");
1523 return;
1524 }
1525
1526 if (i::WasmJSFunction::IsWasmJSFunction(*callable)) {
1527 if (i::Handle<i::WasmJSFunction>::cast(callable)->MatchesSignature(sig)) {
1528 args.GetReturnValue().Set(Utils::ToLocal(callable));
1529 return;
1530 }
1531
1532 thrower.TypeError(
1533 "The signature of Argument 1 (a WebAssembly function) does "
1534 "not match the signature specified in Argument 0");
1535 return;
1536 }
1537
1538 i::Handle<i::JSFunction> result =
1539 i::WasmJSFunction::New(i_isolate, sig, callable);
1540 args.GetReturnValue().Set(Utils::ToLocal(result));
1541}
