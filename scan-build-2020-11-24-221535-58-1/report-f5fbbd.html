466 // This is the legacy version of hrtime before BigInt was introduced in
467 // JavaScript.
468 // The value returned by uv_hrtime() is a 64-bit int representing nanoseconds,
469 // so this function instead fills in an Uint32Array with 3 entries,
470 // to avoid any integer overflow possibility.
471 // The first two entries contain the second part of the value
472 // broken into the upper/lower 32 bits to be converted back in JS,
473 // because there is no Uint64Array in JS.
474 // The third entry contains the remaining nanosecond part of the value.
475 static void NumberImpl(FastHrtime* receiver) {
476 uint64_t t = uv_hrtime();
477 uint32_t* fields = static_cast<uint32_t*>(receiver->backing_store_->Data());
478 fields[0] = (t / NANOS_PER_SEC1000000000) >> 32;
479 fields[1] = (t / NANOS_PER_SEC1000000000) & 0xffffffff;
480 fields[2] = t % NANOS_PER_SEC1000000000;
481 }
482
483 static void FastNumber(v8::ApiObject receiver) {
484 NumberImpl(FromV8ApiObject(receiver));
485 }
486
487 static void SlowNumber(const FunctionCallbackInfo<Value>& args) {
488 NumberImpl(FromJSObject<FastHrtime>(args.Holder()));
489 }
490
491 static void BigIntImpl(FastHrtime* receiver) {
492 uint64_t t = uv_hrtime();
493 uint64_t* fields = static_cast<uint64_t*>(receiver->backing_store_->Data());
494 fields[0] = t;
495 }
496
497 static void FastBigInt(v8::ApiObject receiver) {
498 BigIntImpl(FromV8ApiObject(receiver));
499 }
500
501 static void SlowBigInt(const FunctionCallbackInfo<Value>& args) {
502 BigIntImpl(FromJSObject<FastHrtime>(args.Holder()));
503 }
