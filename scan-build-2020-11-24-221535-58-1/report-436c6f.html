579 Address lazy_deopt_start =
580 deopt_start + non_lazy_deopt_count * kNonLazyDeoptExitSize;
581 // The deoptimization exits are sorted so that lazy deopt exits appear last.
582 static_assert(DeoptimizeKind::kLazy == DeoptimizeKind::kLastDeoptimizeKind,
583 "lazy deopts are expected to be emitted last");
584 // from_ is the value of the link register after the call to the
585 // deoptimizer, so for the last lazy deopt, from_ points to the first
586 // non-lazy deopt, so we use <=.
587 if (from_ <= lazy_deopt_start) {
588 int offset =
589 static_cast<int>(from_ - kNonLazyDeoptExitSize - deopt_start);
590 DCHECK_EQ(0, offset % kNonLazyDeoptExitSize)((void) 0);
591 bailout_id_ = offset / kNonLazyDeoptExitSize;
592 } else {
593 int offset =
594 static_cast<int>(from_ - kLazyDeoptExitSize - lazy_deopt_start);
595 DCHECK_EQ(0, offset % kLazyDeoptExitSize)((void) 0);
596 bailout_id_ = non_lazy_deopt_count + (offset / kLazyDeoptExitSize);
597 }
598 }

906 int frame_index,
907 bool goto_catch_handler) {
908 SharedFunctionInfo shared = translated_frame->raw_shared_info();
909
910 TranslatedFrame::iterator value_iterator = translated_frame->begin();
911 const bool is_bottommost = (0 == frame_index);
912 const bool is_topmost = (output_count_ - 1 == frame_index);
913
914 const int real_bytecode_offset = translated_frame->node_id().ToInt();
915 const int bytecode_offset =
916 goto_catch_handler ? catch_handler_pc_offset_ : real_bytecode_offset;
917
918 const int parameters_count = InternalFormalParameterCountWithReceiver(shared);
919 const int locals_count = translated_frame->height();
920 InterpretedFrameInfo frame_info =
921 InterpretedFrameInfo::Precise(parameters_count, locals_count, is_topmost);
922 const uint32_t output_frame_size = frame_info.frame_size_in_bytes();

924 TranslatedFrame::iterator function_iterator = value_iterator++;
925 if (trace_scope_ != nullptr) {
926 PrintF(trace_scope_->file(), " translating interpreted frame ");

929 PrintF(trace_scope_->file(),
930 " => bytecode_offset=%d, variable_frame_size=%d, frame_size=%d%s\n",
931 real_bytecode_offset, frame_info.frame_size_in_bytes_without_fixed(),
932 output_frame_size, goto_catch_handler ? " (throw)" : "");
933 }
934
935 // Allocate and store the output frame description.
936 FrameDescription* output_frame = new (output_frame_size)
937 FrameDescription(output_frame_size, parameters_count);
938 FrameWriter frame_writer(this, output_frame, trace_scope_);
939
940 CHECK(frame_index >= 0 && frame_index < output_count_)do { if ((__builtin_expect(!!(!(frame_index >= 0 &&
frame_index < output_count_)), 0))) { V8_Fatal("Check failed: %s."
, "frame_index >= 0 && frame_index < output_count_"
); } } while (false);
941 CHECK_NULL(output_[frame_index])do { if ((__builtin_expect(!!(!((output_[frame_index]) == nullptr
)), 0))) { V8_Fatal("Check failed: %s.", "(output_[frame_index]) == nullptr"
); } } while (false);
942 output_[frame_index] = output_frame;
943
944 // The top address of the frame is computed from the previous frame's top and
945 // this frame's size.
946 const intptr_t top_address =
947 is_bottommost ? caller_frame_top_ - output_frame_size
948 : output_[frame_index - 1]->GetTop() - output_frame_size;
949 output_frame->SetTop(top_address);
950
951 // Compute the incoming parameter translation.
952 ReadOnlyRoots roots(isolate());
953 if (ShouldPadArguments(parameters_count)) {
954 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
955 }
956 frame_writer.PushStackJSArguments(value_iterator, parameters_count);

965 // context, the function and the bytecode offset. Synthesize
966 // their values and set them up
967 // explicitly.

969 // The caller's pc for the bottommost output frame is the same as in the
970 // input frame. For all subsequent output frames, it can be read from the
971 // previous one. This frame's pc can be computed from the non-optimized
972 // function code and AST id of the bailout.
973 const intptr_t caller_pc =
974 is_bottommost ? caller_pc_ : output_[frame_index - 1]->GetPc();

976

978 // as in the input frame. For all subsequent output frames, it can be
979 // read from the previous one. Also compute and set this frame's frame
980 // pointer.
981 const intptr_t caller_fp =
982 is_bottommost ? caller_fp_ : output_[frame_index - 1]->GetFp();
983 frame_writer.PushCallerFp(caller_fp);

990 }

992 if (FLAG_enable_embedded_constant_pool) {
993 // For the bottommost output frame the constant pool pointer can be gotten
994 // from the input frame. For subsequent output frames, it can be read from
995 // the previous frame.
996 const intptr_t caller_cp =
997 is_bottommost ? caller_constant_pool_
998 : output_[frame_index - 1]->GetConstantPool();
999 frame_writer.PushCallerConstantPool(caller_cp);
1000 }

1002 // For the bottommost output frame the context can be gotten from the input
1003 // frame. For all subsequent output frames it can be gotten from the function
1004 // so long as we don't inline functions that need local contexts.
1005
1006 // When deoptimizing into a catch block, we need to take the context
1007 // from a register that was specified in the handler table.
1008 TranslatedFrame::iterator context_pos = value_iterator++;
1009 if (goto_catch_handler) {
1010 // Skip to the translated value of the register specified
1011 // in the handler table.
1012 for (int i = 0; i < catch_handler_data_ + 1; ++i) {
1013 context_pos++;
1014 }
1015 }
1016 // Read the context from the translations.
1017 Object context = context_pos->GetRawValue();
1018 output_frame->SetContext(static_cast<intptr_t>(context.ptr()));
1019 frame_writer.PushTranslatedValue(context_pos, "context");
1020
1021 // The function was mentioned explicitly in the BEGIN_FRAME.
1022 frame_writer.PushTranslatedValue(function_iterator, "function");
1023
1024 // Set the bytecode array pointer.
1025 Object bytecode_array = shared.HasBreakInfo()
1026 ? shared.GetDebugInfo().DebugBytecodeArray()
1027 : shared.GetBytecodeArray();
1028 frame_writer.PushRawObject(bytecode_array, "bytecode array\n");
1029
1030 // The bytecode offset was mentioned explicitly in the BEGIN_FRAME.
1031 const int raw_bytecode_offset =
1032 BytecodeArray::kHeaderSize - kHeapObjectTag + bytecode_offset;
1033 Smi smi_bytecode_offset = Smi::FromInt(raw_bytecode_offset);
1034 frame_writer.PushRawObject(smi_bytecode_offset, "bytecode offset\n");
1035
1036 if (trace_scope_ != nullptr) {
1037 PrintF(trace_scope_->file(), " -------------------------\n");
1038 }
1039
1040 // Translate the rest of the interpreter registers in the frame.
1041 // The return_value_offset is counted from the top. Here, we compute the
1042 // register index (counted from the start).
1043 const int return_value_first_reg =
1044 locals_count - translated_frame->return_value_offset();
1045 const int return_value_count = translated_frame->return_value_count();
1046 for (int i = 0; i < locals_count; ++i, ++value_iterator) {
1047 // Ensure we write the return value if we have one and we are returning
1048 // normally to a lazy deopt point.
1049 if (is_topmost && !goto_catch_handler &&
1050 deopt_kind_ == DeoptimizeKind::kLazy && i >= return_value_first_reg &&
1051 i < return_value_first_reg + return_value_count) {
1052 const int return_index = i - return_value_first_reg;
1053 if (return_index == 0) {
1054 frame_writer.PushRawValue(input_->GetRegister(kReturnRegister0.code()),
1055 "return value 0\n");
1056 // We do not handle the situation when one return value should go into
1057 // the accumulator and another one into an ordinary register. Since
1058 // the interpreter should never create such situation, just assert
1059 // this does not happen.
1060 CHECK_LE(return_value_first_reg + return_value_count, locals_count)do { bool _cmp = ::v8::base::CmpLEImpl< typename ::v8::base
::pass_value_or_ref<decltype(return_value_first_reg + return_value_count
)>::type, typename ::v8::base::pass_value_or_ref<decltype
(locals_count)>::type>((return_value_first_reg + return_value_count
), (locals_count)); do { if ((__builtin_expect(!!(!(_cmp)), 0
))) { V8_Fatal("Check failed: %s.", "return_value_first_reg + return_value_count"
" " "<=" " " "locals_count"); } } while (false); } while (
false);
1061 } else {
1062 CHECK_EQ(return_index, 1)do { bool _cmp = ::v8::base::CmpEQImpl< typename ::v8::base
::pass_value_or_ref<decltype(return_index)>::type, typename
::v8::base::pass_value_or_ref<decltype(1)>::type>((
return_index), (1)); do { if ((__builtin_expect(!!(!(_cmp)), 0
))) { V8_Fatal("Check failed: %s.", "return_index" " " "==" " "
"1"); } } while (false); } while (false);
1063 frame_writer.PushRawValue(input_->GetRegister(kReturnRegister1.code()),
1064 "return value 1\n");
1065 }
1066 } else {
1067 // This is not return value, just write the value from the translations.
1068 frame_writer.PushTranslatedValue(value_iterator, "stack parameter");
1069 }
1070 }
1071
1072 uint32_t register_slots_written = static_cast<uint32_t>(locals_count);
1073 DCHECK_LE(register_slots_written, frame_info.register_stack_slot_count())((void) 0);
1074 // Some architectures must pad the stack frame with extra stack slots
1075 // to ensure the stack frame is aligned. Do this now.
1076 while (register_slots_written < frame_info.register_stack_slot_count()) {
1077 register_slots_written++;
1078 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
1079 }
1080
1081 // Translate the accumulator register (depending on frame position).
1082 if (is_topmost) {
1083 if (kPadArguments) {
1084 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
1085 }
1086 // For topmost frame, put the accumulator on the stack. The
1087 // {NotifyDeoptimized} builtin pops it off the topmost frame (possibly
1088 // after materialization).
1089 if (goto_catch_handler) {
1090 // If we are lazy deopting to a catch handler, we set the accumulator to
1091 // the exception (which lives in the result register).
1092 intptr_t accumulator_value =
1093 input_->GetRegister(kInterpreterAccumulatorRegister.code());
1094 frame_writer.PushRawObject(Object(accumulator_value), "accumulator\n");
1095 } else {
1096 // If we are lazily deoptimizing make sure we store the deopt
1097 // return value into the appropriate slot.
1098 if (deopt_kind_ == DeoptimizeKind::kLazy &&
1099 translated_frame->return_value_offset() == 0 &&
1100 translated_frame->return_value_count() > 0) {
1101 CHECK_EQ(translated_frame->return_value_count(), 1)do { bool _cmp = ::v8::base::CmpEQImpl< typename ::v8::base
::pass_value_or_ref<decltype(translated_frame->return_value_count
())>::type, typename ::v8::base::pass_value_or_ref<decltype
(1)>::type>((translated_frame->return_value_count())
, (1)); do { if ((__builtin_expect(!!(!(_cmp)), 0))) { V8_Fatal
("Check failed: %s.", "translated_frame->return_value_count()"
" " "==" " " "1"); } } while (false); } while (false);
1102 frame_writer.PushRawValue(input_->GetRegister(kReturnRegister0.code()),
1103 "return value 0\n");
1104 } else {
1105 frame_writer.PushTranslatedValue(value_iterator, "accumulator");
1106 }
1107 }
1108 ++value_iterator; // Move over the accumulator.
1109 } else {
1110 // For non-topmost frames, skip the accumulator translation. For those
1111 // frames, the return value from the callee will become the accumulator.
1112 ++value_iterator;
1113 }
1114 CHECK_EQ(translated_frame->end(), value_iterator)do { bool _cmp = ::v8::base::CmpEQImpl< typename ::v8::base
::pass_value_or_ref<decltype(translated_frame->end())>
::type, typename ::v8::base::pass_value_or_ref<decltype(value_iterator
)>::type>((translated_frame->end()), (value_iterator
)); do { if ((__builtin_expect(!!(!(_cmp)), 0))) { V8_Fatal("Check failed: %s."
, "translated_frame->end()" " " "==" " " "value_iterator")
; } } while (false); } while (false);
1115 CHECK_EQ(0u, frame_writer.top_offset())do { bool _cmp = ::v8::base::CmpEQImpl< typename ::v8::base
::pass_value_or_ref<decltype(0u)>::type, typename ::v8::
base::pass_value_or_ref<decltype(frame_writer.top_offset()
)>::type>((0u), (frame_writer.top_offset())); do { if (
(__builtin_expect(!!(!(_cmp)), 0))) { V8_Fatal("Check failed: %s."
, "0u" " " "==" " " "frame_writer.top_offset()"); } } while (
false); } while (false);
1116
1117 // Compute this frame's PC and state. The PC will be a special builtin that
1118 // continues the bytecode dispatch. Note that non-topmost and lazy-style
1119 // bailout handlers also advance the bytecode offset before dispatch, hence
1120 // simulating what normal handlers do upon completion of the operation.
1121 Builtins* builtins = isolate_->builtins();
1122 Code dispatch_builtin =
1123 (!is_topmost || (deopt_kind_ == DeoptimizeKind::kLazy)) &&
1124 !goto_catch_handler
1125 ? builtins->builtin(Builtins::kInterpreterEnterBytecodeAdvance)
1126 : builtins->builtin(Builtins::kInterpreterEnterBytecodeDispatch);
1127 output_frame->SetPc(
1128 static_cast<intptr_t>(dispatch_builtin.InstructionStart()));
1129
1130 // Update constant pool.
1131 if (FLAG_enable_embedded_constant_pool) {
1132 intptr_t constant_pool_value =
1133 static_cast<intptr_t>(dispatch_builtin.constant_pool());
1134 output_frame->SetConstantPool(constant_pool_value);
1135 if (is_topmost) {
1136 Register constant_pool_reg =
1137 InterpretedFrame::constant_pool_pointer_register();
1138 output_frame->SetRegister(constant_pool_reg.code(), constant_pool_value);
1139 }
1140 }
1141
1142 // Clear the context register. The context might be a de-materialized object
1143 // and will be materialized by {Runtime_NotifyDeoptimized}. For additional
1144 // safety we use Smi(0) instead of the potential {arguments_marker} here.
1145 if (is_topmost) {
1146 intptr_t context_value = static_cast<intptr_t>(Smi::zero().ptr());
1147 Register context_reg = JavaScriptFrame::context_register();
1148 output_frame->SetRegister(context_reg.code(), context_value);
1149 // Set the continuation for the topmost frame.
1150 Code continuation = builtins->builtin(Builtins::kNotifyDeoptimized);
1151 output_frame->SetContinuation(
1152 static_cast<intptr_t>(continuation.InstructionStart()));
1153 }
1154}

1156void Deoptimizer::DoComputeArgumentsAdaptorFrame(
1157 TranslatedFrame* translated_frame, int frame_index) {
1158 TranslatedFrame::iterator value_iterator = translated_frame->begin();
1159 const bool is_bottommost = (0 == frame_index);
1160
1161 const int parameters_count = translated_frame->height();
1162 ArgumentsAdaptorFrameInfo frame_info =
1163 ArgumentsAdaptorFrameInfo::Precise(parameters_count);
1164 const uint32_t output_frame_size = frame_info.frame_size_in_bytes();
1165
1166 TranslatedFrame::iterator function_iterator = value_iterator++;
1167 if (trace_scope_ != nullptr) {
1168 PrintF(trace_scope_->file(),
1169 " translating arguments adaptor => variable_frame_size=%d, "
1170 "frame_size=%d\n",
1171 frame_info.frame_size_in_bytes_without_fixed(), output_frame_size);
1172 }
1173
1174 // Allocate and store the output frame description.

1176 FrameDescription(output_frame_size, parameters_count);

1178
1179 // Arguments adaptor can not be topmost.
1180 CHECK(frame_index < output_count_ - 1)do { if ((__builtin_expect(!!(!(frame_index < output_count_
- 1)), 0))) { V8_Fatal("Check failed: %s.", "frame_index < output_count_ - 1"
); } } while (false);
1181 CHECK_NULL(output_[frame_index])do { if ((__builtin_expect(!!(!((output_[frame_index]) == nullptr
)), 0))) { V8_Fatal("Check failed: %s.", "(output_[frame_index]) == nullptr"
); } } while (false);
1182 output_[frame_index] = output_frame;
1183
1184 // The top address of the frame is computed from the previous frame's top and
1185 // this frame's size.
1186 const intptr_t top_address =
1187 is_bottommost ? caller_frame_top_ - output_frame_size
1188 : output_[frame_index - 1]->GetTop() - output_frame_size;
1189 output_frame->SetTop(top_address);
1190
1191 ReadOnlyRoots roots(isolate());
1192 if (ShouldPadArguments(parameters_count)) {
1193 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
1194 }
1195
1196 // Compute the incoming parameter translation.
1197 for (int i = 0; i < parameters_count; ++i, ++value_iterator) {
1198 frame_writer.PushTranslatedValue(value_iterator, "stack parameter");
1199 }
1200
1201 DCHECK_EQ(output_frame->GetLastArgumentSlotOffset(),((void) 0)
1202 frame_writer.top_offset())((void) 0);
1203
1204 // Read caller's PC from the previous frame.
1205 const intptr_t caller_pc =
1206 is_bottommost ? caller_pc_ : output_[frame_index - 1]->GetPc();
1207 frame_writer.PushCallerPc(caller_pc);
1208
1209 // Read caller's FP from the previous frame, and set this frame's FP.
1210 const intptr_t caller_fp =
1211 is_bottommost ? caller_fp_ : output_[frame_index - 1]->GetFp();
1212 frame_writer.PushCallerFp(caller_fp);
1213
1214 intptr_t fp_value = top_address + frame_writer.top_offset();
1215 output_frame->SetFp(fp_value);
1216
1217 if (FLAG_enable_embedded_constant_pool) {
1218 // Read the caller's constant pool from the previous frame.
1219 const intptr_t caller_cp =
1220 is_bottommost ? caller_constant_pool_
1221 : output_[frame_index - 1]->GetConstantPool();
1222 frame_writer.PushCallerConstantPool(caller_cp);
1223 }
1224
1225 // A marker value is used in place of the context.
1226 intptr_t marker = StackFrame::TypeToMarker(StackFrame::ARGUMENTS_ADAPTOR);
1227 frame_writer.PushRawValue(marker, "context (adaptor sentinel)\n");
1228
1229 // The function was mentioned explicitly in the ARGUMENTS_ADAPTOR_FRAME.
1230 frame_writer.PushTranslatedValue(function_iterator, "function\n");
1231
1232 // Number of incoming arguments.
1233 const uint32_t parameters_count_without_receiver = parameters_count - 1;
1234 frame_writer.PushRawObject(Smi::FromInt(parameters_count_without_receiver),
1235 "argc\n");
1236
1237 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
1238
1239 CHECK_EQ(translated_frame->end(), value_iterator)do { bool _cmp = ::v8::base::CmpEQImpl< typename ::v8::base
::pass_value_or_ref<decltype(translated_frame->end())>
::type, typename ::v8::base::pass_value_or_ref<decltype(value_iterator
)>::type>((translated_frame->end()), (value_iterator
)); do { if ((__builtin_expect(!!(!(_cmp)), 0))) { V8_Fatal("Check failed: %s."
, "translated_frame->end()" " " "==" " " "value_iterator")
; } } while (false); } while (false);
1240 DCHECK_EQ(0, frame_writer.top_offset())((void) 0);
1241
1242 Builtins* builtins = isolate_->builtins();
1243 Code adaptor_trampoline =
1244 builtins->builtin(Builtins::kArgumentsAdaptorTrampoline);
1245 intptr_t pc_value = static_cast<intptr_t>(
1246 adaptor_trampoline.InstructionStart() +
1247 isolate_->heap()->arguments_adaptor_deopt_pc_offset().value());
1248 output_frame->SetPc(pc_value);
1249 if (FLAG_enable_embedded_constant_pool) {
1250 intptr_t constant_pool_value =
1251 static_cast<intptr_t>(adaptor_trampoline.constant_pool());
1252 output_frame->SetConstantPool(constant_pool_value);
1253 }

1261 // call which does a tail call (otherwise the tail callee's frame would be

1271 ConstructStubFrameInfo::Precise(parameters_count, is_topmost);
1272 const uint32_t output_frame_size = frame_info.frame_size_in_bytes();
1273
1274 TranslatedFrame::iterator function_iterator = value_iterator++;
1275 if (trace_scope_ != nullptr) {

1278 "variable_frame_size=%d, frame_size=%d\n",
1279 bailout_id.ToInt(),

1281 frame_info.frame_size_in_bytes_without_fixed(), output_frame_size);
1282 }

1284 // Allocate and store the output frame description.
1285 FrameDescription* output_frame = new (output_frame_size)
1286 FrameDescription(output_frame_size, parameters_count);

1288

1291 DCHECK_NULL(output_[frame_index])((void) 0);
1292 output_[frame_index] = output_frame;
1293
1294 // The top address of the frame is computed from the previous frame's top and
1295 // this frame's size.
1296 const intptr_t top_address =
1297 output_[frame_index - 1]->GetTop() - output_frame_size;
1298 output_frame->SetTop(top_address);

1301 if (ShouldPadArguments(parameters_count)) {
1302 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
1303 }
1304
1305 // The allocated receiver of a construct stub frame is passed as the
1306 // receiver parameter through the translation. It might be encoding
1307 // a captured object, so we need save it for later.
1308 TranslatedFrame::iterator receiver_iterator = value_iterator;
1309

1311 frame_writer.PushStackJSArguments(value_iterator, parameters_count);

1314 frame_writer.top_offset())((void) 0);

1317 const intptr_t caller_pc = output_[frame_index - 1]->GetPc();
1318 frame_writer.PushCallerPc(caller_pc);
1319

1321 const intptr_t caller_fp = output_[frame_index - 1]->GetFp();
1322 frame_writer.PushCallerFp(caller_fp);
1323
1324 const intptr_t fp_value = top_address + frame_writer.top_offset();
1325 output_frame->SetFp(fp_value);
1326 if (is_topmost) {
1327 Register fp_reg = JavaScriptFrame::fp_register();
1328 output_frame->SetRegister(fp_reg.code(), fp_value);
1329 }
1330
1331 if (FLAG_enable_embedded_constant_pool) {
1332 // Read the caller's constant pool from the previous frame.
1333 const intptr_t caller_cp = output_[frame_index - 1]->GetConstantPool();
1334 frame_writer.PushCallerConstantPool(caller_cp);
1335 }
1336

1345 frame_writer.PushRawObject(Smi::FromInt(parameters_count_without_receiver),
1346 "argc\n");

1348 // The constructor function was mentioned explicitly in the
1349 // CONSTRUCT_STUB_FRAME.
1350 frame_writer.PushTranslatedValue(function_iterator, "constructor function\n");

1353 // position of the receiver. Copy it to the top of stack, with the hole value
1354 // as padding to maintain alignment.
1355
1356 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");

() || bailout_id == BailoutId::ConstructStubInvoke())), 0))) {

1361 ? "new target\n"
1362 : "allocated receiver\n";

1364
1365 if (is_topmost) {
1366 if (kPadArguments) {
1367 frame_writer.PushRawObject(roots.the_hole_value(), "padding\n");
1368 }
1369 // Ensure the result is restored back when we return to the stub.
1370 Register result_reg = kReturnRegister0;
1371 intptr_t result = input_->GetRegister(result_reg.code());
1372 frame_writer.PushRawValue(result, "subcall result\n");
1373 }

)>::type>((translated_frame->end()), (value_iterator
)); do { if ((__builtin_expect(!!(!(_cmp)), 0))) { V8_Fatal("Check failed: %s."

; } } while (false); } while (false);
1376 CHECK_EQ(0u, frame_writer.top_offset())do { bool _cmp = ::v8::base::CmpEQImpl< typename ::v8::base
::pass_value_or_ref<decltype(0u)>::type, typename ::v8::
base::pass_value_or_ref<decltype(frame_writer.top_offset()
)>::type>((0u), (frame_writer.top_offset())); do { if (
(__builtin_expect(!!(!(_cmp)), 0))) { V8_Fatal("Check failed: %s."
, "0u" " " "==" " " "frame_writer.top_offset()"); } } while (
false); } while (false);
1377

1379 DCHECK(bailout_id.IsValidForConstructStub())((void) 0);

1381 const int pc_offset =
1382 bailout_id == BailoutId::ConstructStubCreate()
1383 ? isolate_->heap()->construct_stub_create_deopt_pc_offset().value()
1384 : isolate_->heap()->construct_stub_invoke_deopt_pc_offset().value();
1385 intptr_t pc_value = static_cast<intptr_t>(start + pc_offset);
1386 output_frame->SetPc(pc_value);
1387
1388 // Update constant pool.
1389 if (FLAG_enable_embedded_constant_pool) {
1390 intptr_t constant_pool_value =
1391 static_cast<intptr_t>(construct_stub.constant_pool());
1392 output_frame->SetConstantPool(constant_pool_value);
1393 if (is_topmost) {
1394 Register constant_pool_reg =
1395 JavaScriptFrame::constant_pool_pointer_register();
1396 output_frame->SetRegister(constant_pool_reg.code(), constant_pool_value);
1397 }
1398 }

1401 // and will be materialized by {Runtime_NotifyDeoptimized}. For additional
1402 // safety we use Smi(0) instead of the potential {arguments_marker} here.
1403 if (is_topmost) {
1404 intptr_t context_value = static_cast<intptr_t>(Smi::zero().ptr());
1405 Register context_reg = JavaScriptFrame::context_register();
1406 output_frame->SetRegister(context_reg.code(), context_value);
1407 }
1408
1409 // Set the continuation for the topmost frame.
1410 if (is_topmost) {
1411 Builtins* builtins = isolate_->builtins();
1412 DCHECK_EQ(DeoptimizeKind::kLazy, deopt_kind_)((void) 0);
1413 Code continuation = builtins->builtin(Builtins::kNotifyDeoptimized);
1414 output_frame->SetContinuation(
1415 static_cast<intptr_t>(continuation.InstructionStart()));
1416 }

1466// BuiltinContinuationFrames capture the machine state that is expected as input
1467// to a builtin, including both input register values and stack parameters. When
1468// the frame is reactivated (i.e. the frame below it returns), a
1469// ContinueToBuiltin stub restores the register state from the frame and tail
1470// calls to the actual target builtin, making it appear that the stub had been
1471// directly called by the frame above it. The input values to populate the frame
1472// are taken from the deopt's FrameState.
1473//
1474// Frame translation happens in two modes, EAGER and LAZY. In EAGER mode, all of
1475// the parameters to the Builtin are explicitly specified in the TurboFan
1476// FrameState node. In LAZY mode, there is always one fewer parameters specified
1477// in the FrameState than expected by the Builtin. In that case, construction of
1478// BuiltinContinuationFrame adds the final missing parameter during
1479// deoptimization, and that parameter is always on the stack and contains the
1480// value returned from the callee of the call site triggering the LAZY deopt
1481// (e.g. rax on x64). This requires that continuation Builtins for LAZY deopts
1482// must have at least one stack parameter.
1483//
1484// TO
1485// | .... |
1486// +-------------------------+
1487// | arg padding (arch dept) |<- at most 1*kSystemPointerSize
1488// +-------------------------+
1489// | builtin param 0 |<- FrameState input value n becomes
1490// +-------------------------+
1491// | ... |
1492// +-------------------------+
1493// | builtin param m |<- FrameState input value n+m-1, or in
1494// +-----needs-alignment-----+ the LAZY case, return LAZY result value
1495// | ContinueToBuiltin entry |
1496// +-------------------------+
1497// | | saved frame (FP) |
1498// | +=====needs=alignment=====+<- fpreg
1499// | |constant pool (if ool_cp)|
1500// v +-------------------------+
1501// |BUILTIN_CONTINUATION mark|
1502// +-------------------------+
1503// | JSFunction (or zero) |<- only if JavaScript builtin
1504// +-------------------------+
1505// | frame height above FP |
1506// +-------------------------+
1507// | context |<- this non-standard context slot contains
1508// +-------------------------+ the context, even for non-JS builtins.
1509// | builtin index |
1510// +-------------------------+
1511// | builtin input GPR reg0 |<- populated from deopt FrameState using
1512// +-------------------------+ the builtin's CallInterfaceDescriptor
1513// | ... | to map a FrameState's 0..n-1 inputs to
1514// +-------------------------+ the builtin's n input register params.
1515// | builtin input GPR regn |
1516// +-------------------------+
1517// | reg padding (arch dept) |
1518// +-----needs--alignment----+
1519// | res padding (arch dept) |<- only if {is_topmost}; result is pop'd by
1520// +-------------------------+<- kNotifyDeopt ASM stub and moved to acc
1521// | result value |<- reg, as ContinueToBuiltin stub expects.
1522// +-----needs-alignment-----+<- spreg
1523//

1956void TranslationBuffer::Add(int32_t value) {
1957 // This wouldn't handle kMinInt correctly if it ever encountered it.
1958 DCHECK_NE(value, kMinInt)((void) 0);
1959 // Encode the sign bit in the least significant bit.
1960 bool is_negative = (value < 0);
1961 uint32_t bits = (static_cast<uint32_t>(is_negative ? -value : value) << 1) |
1962 static_cast<uint32_t>(is_negative);
1963 // Encode the individual bytes using the least significant bit of
1964 // each byte to indicate whether or not more bytes follow.
1965 do {
1966 uint32_t next = bits >> 7;
1967 contents_.push_back(((bits << 1) & 0xFF) | (next != 0));
1968 bits = next;
1969 } while (bits != 0);
1970}

1977int32_t TranslationIterator::Next() {
1978 // Run through the bytes until we reach one with a least significant
1979 // bit of zero (marks the end).
1980 uint32_t bits = 0;
1981 for (int i = 0; true; i += 7) {
1982 DCHECK(HasNext())((void) 0);
1983 uint8_t next = buffer_.get(index_++);
1984 bits |= (next >> 1) << i;
1985 if ((next & 1) == 0) break;
1986 }
1987 // The bits encode the sign in the least significant bit.
1988 bool is_negative = (bits & 1) == 1;
1989 int32_t result = bits >> 1;
1990 return is_negative ? -result : result;

2503Object TranslatedValue::GetRawValue() const {
2504 // If we have a value, return it.
2505 if (materialization_state() == kFinished) {
2506 int smi;
2507 if (storage_->IsHeapNumber() &&
2508 DoubleToSmiInteger(storage_->Number(), &smi)) {
2509 return Smi::FromInt(smi);
2510 }
2511 return *storage_;
2512 }
2513
2514 // Otherwise, do a best effort to get the value without allocation.
2515 switch (kind()) {
2516 case kTagged:
2517 return raw_literal();
2518
2519 case kInt32: {
2520 bool is_smi = Smi::IsValid(int32_value());
2521 if (is_smi) {
2522 return Smi::FromInt(int32_value());
2523 }
2524 break;
2525 }
2526
2527 case kInt64: {
2528 bool is_smi = (int64_value() >= static_cast<int64_t>(Smi::kMinValue) &&
2529 int64_value() <= static_cast<int64_t>(Smi::kMaxValue));
2530 if (is_smi) {
2531 return Smi::FromIntptr(static_cast<intptr_t>(int64_value()));
2532 }
2533 break;
2534 }
2535
2536 case kUInt32: {
2537 bool is_smi = (uint32_value() <= static_cast<uintptr_t>(Smi::kMaxValue));
2538 if (is_smi) {
2539 return Smi::FromInt(static_cast<int32_t>(uint32_value()));
2540 }
2541 break;
2542 }
2543
2544 case kBoolBit: {
2545 if (uint32_value() == 0) {
2546 return ReadOnlyRoots(isolate()).false_value();
2547 } else {
2548 CHECK_EQ(1U, uint32_value())do { bool _cmp = ::v8::base::CmpEQImpl< typename ::v8::base
::pass_value_or_ref<decltype(1U)>::type, typename ::v8::
base::pass_value_or_ref<decltype(uint32_value())>::type
>((1U), (uint32_value())); do { if ((__builtin_expect(!!(!
(_cmp)), 0))) { V8_Fatal("Check failed: %s.", "1U" " " "==" " "
"uint32_value()"); } } while (false); } while (false);
2549 return ReadOnlyRoots(isolate()).true_value();
2550 }
2551 }
2552
2553 case kFloat: {
2554 int smi;
2555 if (DoubleToSmiInteger(float_value().get_scalar(), &smi)) {
2556 return Smi::FromInt(smi);
2557 }
2558 break;
2559 }
2560
2561 case kDouble: {
2562 int smi;
2563 if (DoubleToSmiInteger(double_value().get_scalar(), &smi)) {
2564 return Smi::FromInt(smi);
2565 }
2566 break;
2567 }
2568
2569 default:
2570 break;
2571 }
2572
2573 // If we could not get the value without allocation, return the arguments
2574 // marker.
2575 return ReadOnlyRoots(isolate()).arguments_marker();

2584Handle<Object> TranslatedValue::GetValue() {
2585 Handle<Object> value(GetRawValue(), isolate());
2586 if (materialization_state() == kFinished) return value;
2587
2588 if (value->IsSmi()) {
2589 // Even though stored as a Smi, this number might instead be needed as a
2590 // HeapNumber when materializing a JSObject with a field of HeapObject
2591 // representation. Since we don't have this information available here, we
2592 // just always allocate a HeapNumber and later extract the Smi again if we
2593 // don't need a HeapObject.
2594 set_initialized_storage(
2595 isolate()->factory()->NewHeapNumber(value->Number()));
2596 return value;
2597 }
2598
2599 if (*value != ReadOnlyRoots(isolate()).arguments_marker()) {
2600 set_initialized_storage(Handle<HeapObject>::cast(value));
2601 return storage_;
2602 }
2603
2604 // Otherwise we have to materialize.
2605
2606 if (kind() == TranslatedValue::kCapturedObject ||
2607 kind() == TranslatedValue::kDuplicatedObject) {
2608 // We need to materialize the object (or possibly even object graphs).
2609 // To make the object verifier happy, we materialize in two steps.
2610
2611 // 1. Allocate storage for reachable objects. This makes sure that for
2612 // each object we have allocated space on heap. The space will be
2613 // a byte array that will be later initialized, or a fully
2614 // initialized object if it is safe to allocate one that will
2615 // pass the verifier.
2616 container_->EnsureObjectAllocatedAt(this);
2617
2618 // 2. Initialize the objects. If we have allocated only byte arrays
2619 // for some objects, we now overwrite the byte arrays with the
2620 // correct object fields. Note that this phase does not allocate
2621 // any new objects, so it does not trigger the object verifier.
2622 return container_->InitializeObjectAt(this);
2623 }
2624
2625 double number;
2626 switch (kind()) {
2627 case TranslatedValue::kInt32:
2628 number = int32_value();
2629 break;
2630 case TranslatedValue::kInt64:
2631 number = int64_value();
2632 break;
2633 case TranslatedValue::kUInt32:
2634 number = uint32_value();
2635 break;
2636 case TranslatedValue::kFloat:
2637 number = float_value().get_scalar();
2638 break;
2639 case TranslatedValue::kDouble:
2640 number = double_value().get_scalar();
2641 break;
2642 default:
2643 UNREACHABLE()V8_Fatal("unreachable code");
2644 }
2645 DCHECK(!IsSmiDouble(number))((void) 0);
2646 set_initialized_storage(isolate()->factory()->NewHeapNumber(number));
2647 return storage_;
2648}
