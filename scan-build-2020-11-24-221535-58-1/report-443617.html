192 // Adds a new entry, and returns a pointer to a builder for modifying that
193 // entry ({stack_height} includes {num_locals}).
194 EntryBuilder* NewEntry(int pc_offset, int num_locals, int stack_height,
195 LiftoffAssembler::VarState* stack_state,
196 AssumeSpilling assume_spilling) {
197 DCHECK_LE(num_locals, stack_height)((void) 0);
198 // Record stack types.
199 std::vector<DebugSideTable::Entry::Value> values(stack_height);
200 for (int i = 0; i < stack_height; ++i) {
201 const auto& slot = stack_state[i];
202 values[i].type = slot.type();
203 values[i].stack_offset = slot.offset();
204 switch (slot.loc()) {
205 case kIntConst:
206 values[i].kind = DebugSideTable::Entry::kConstant;
207 values[i].i32_const = slot.i32_const();
208 break;
209 case kRegister:
210 DCHECK_NE(kDidSpill, assume_spilling)((void) 0);
211 if (assume_spilling == kAllowRegisters) {
212 values[i].kind = DebugSideTable::Entry::kRegister;
213 values[i].reg_code = slot.reg().liftoff_code();
214 break;
215 }
216 DCHECK_EQ(kAssumeSpilling, assume_spilling)((void) 0);
217 V8_FALLTHROUGH[[clang::fallthrough]];
218 case kStack:
219 values[i].kind = DebugSideTable::Entry::kStack;
220 values[i].stack_offset = slot.offset();
221 break;
222 }
223 }
224 entries_.emplace_back(pc_offset, std::move(values));
225 return &entries_.back();

517 bool SpillLocalsInitially(FullDecoder* decoder, uint32_t num_params) {
518 int actual_locals = __ num_locals() - num_params;
519 DCHECK_LE(0, actual_locals)((void) 0);
520 constexpr int kNumCacheRegisters = NumRegs(kLiftoffAssemblerGpCacheRegs);
521 // If we have many locals, we put them on the stack initially. This avoids
522 // having to spill them on merge points. Use of these initial values should
523 // be rare anyway.
524 if (actual_locals > kNumCacheRegisters / 2) return true;
525 // If there are locals which are not i32 or i64, we also spill all locals,
526 // because other types cannot be initialized to constants.

1477 });
1478 case kExprI32RemS:
1479 return EmitBinOp<kI32, kI32>(
1480 [this, decoder](LiftoffRegister dst, LiftoffRegister lhs,
1481 LiftoffRegister rhs) {
1482 Label* rem_by_zero = AddOutOfLineTrap(
1483 decoder->position(), WasmCode::kThrowWasmTrapRemByZero);
1484 __ emit_i32_rems(dst.gp(), lhs.gp(), rhs.gp(), rem_by_zero);
1485 });
1486 case kExprI32RemU:
1487 return EmitBinOp<kI32, kI32>(
1488 [this, decoder](LiftoffRegister dst, LiftoffRegister lhs,
1489 LiftoffRegister rhs) {
1490 Label* rem_by_zero = AddOutOfLineTrap(
1491 decoder->position(), WasmCode::kThrowWasmTrapRemByZero);
1492 __ emit_i32_remu(dst.gp(), lhs.gp(), rhs.gp(), rem_by_zero);
1493 });
1494 case kExprI64DivS:
1495 return EmitBinOp<kI64, kI64>([this, decoder](LiftoffRegister dst,
1496 LiftoffRegister lhs,
1497 LiftoffRegister rhs) {
1498 WasmCodePosition position = decoder->position();
1499 AddOutOfLineTrap(position, WasmCode::kThrowWasmTrapDivByZero);
1500 // Adding the second trap might invalidate the pointer returned for
1501 // the first one, thus get both pointers afterwards.
1502 AddOutOfLineTrap(position,
1503 WasmCode::kThrowWasmTrapDivUnrepresentable);
1504 Label* div_by_zero = out_of_line_code_.end()[-2].label.get();
1505 Label* div_unrepresentable = out_of_line_code_.end()[-1].label.get();
1506 if (!__ emit_i64_divs(dst, lhs, rhs, div_by_zero,
1507 div_unrepresentable)) {
1508 ExternalReference ext_ref = ExternalReference::wasm_int64_div();
1509 EmitDivOrRem64CCall(dst, lhs, rhs, ext_ref, div_by_zero,
1510 div_unrepresentable);
1511 }
1512 });
1513 case kExprI64DivU:
1514 return EmitBinOp<kI64, kI64>([this, decoder](LiftoffRegister dst,
1515 LiftoffRegister lhs,
1516 LiftoffRegister rhs) {
1517 Label* div_by_zero = AddOutOfLineTrap(
1518 decoder->position(), WasmCode::kThrowWasmTrapDivByZero);
1519 if (!__ emit_i64_divu(dst, lhs, rhs, div_by_zero)) {
1520 ExternalReference ext_ref = ExternalReference::wasm_uint64_div();
1521 EmitDivOrRem64CCall(dst, lhs, rhs, ext_ref, div_by_zero);
1522 }
1523 });
1524 case kExprI64RemS:
1525 return EmitBinOp<kI64, kI64>(
1526 [this, decoder](LiftoffRegister dst, LiftoffRegister lhs,
1527 LiftoffRegister rhs) {
1528 Label* rem_by_zero = AddOutOfLineTrap(
1529 decoder->position(), WasmCode::kThrowWasmTrapRemByZero);
1530 if (!__ emit_i64_rems(dst, lhs, rhs, rem_by_zero)) {
1531 ExternalReference ext_ref = ExternalReference::wasm_int64_mod();
1532 EmitDivOrRem64CCall(dst, lhs, rhs, ext_ref, rem_by_zero);
1533 }
1534 });
1535 case kExprI64RemU:
1536 return EmitBinOp<kI64, kI64>([this, decoder](LiftoffRegister dst,
1537 LiftoffRegister lhs,
1538 LiftoffRegister rhs) {
1539 Label* rem_by_zero = AddOutOfLineTrap(
1540 decoder->position(), WasmCode::kThrowWasmTrapRemByZero);
1541 if (!__ emit_i64_remu(dst, lhs, rhs, rem_by_zero)) {
1542 ExternalReference ext_ref = ExternalReference::wasm_uint64_mod();
1543 EmitDivOrRem64CCall(dst, lhs, rhs, ext_ref, rem_by_zero);
1544 }
1545 });
1546 default:
1547 UNREACHABLE()V8_Fatal("unreachable code");
1548 }

1550#undef CASE_CCALL_BINOP
1551 }

1603 void TraceFunctionExit(FullDecoder* decoder) {
1604 DEBUG_CODE_COMMENT("trace function exit");
1605 // Before making the runtime call, spill all cache registers.
1606 __ SpillAllRegisters();
1607 LiftoffRegList pinned;
1608 // Get a register to hold the stack slot for the return value.
1609 LiftoffRegister info = pinned.set(__ GetUnusedRegister(kGpReg, pinned));
1610 __ AllocateStackSlot(info.gp(), sizeof(int64_t));
1611
1612 // Store the return value if there is exactly one. Multiple return values
1613 // are not handled yet.
1614 size_t num_returns = decoder->sig_->return_count();
1615 if (num_returns == 1) {
1616 ValueType return_type = decoder->sig_->GetReturn(0);
1617 LiftoffRegister return_reg =
1618 __ LoadToRegister(__ cache_state()->stack_state.back(), pinned);
1619 __ Store(info.gp(), no_reg, 0, return_reg,
1620 StoreType::ForValueType(return_type), pinned);
1621 }
1622 // Put the parameter in its place.
1623 WasmTraceExitDescriptor descriptor;
1624 DCHECK_EQ(0, descriptor.GetStackParameterCount())((void) 0);
1625 DCHECK_EQ(1, descriptor.GetRegisterParameterCount())((void) 0);
1626 Register param_reg = descriptor.GetRegisterParameter(0);
1627 if (info.gp() != param_reg) {
1628 __ Move(param_reg, info.gp(), LiftoffAssembler::kWasmIntPtr);
1629 }
1630
1631 source_position_table_builder_.AddPosition(
1632 __ pc_offset(), SourcePosition(decoder->position()), false);
1633 __ CallRuntimeStub(WasmCode::kWasmTraceExit);
1634 safepoint_table_builder_.DefineSafepoint(&asm_, Safepoint::kNoLazyDeopt);
1635
1636 __ DeallocateStackSlot(sizeof(int64_t));
1637 }

1697 auto& state = *__ cache_state();

1842 void BrIf(FullDecoder* decoder, const Value& /* cond */, uint32_t depth) {
1843 Label cont_false;
1844 Register value = __ PopToRegister().gp();
1845
1846 if (!has_outstanding_op()) {
1847 // Unary "equal" means "equals zero".
1848 __ emit_cond_jump(kEqual, &cont_false, kWasmI32, value);
1849 } else if (outstanding_op_ == kExprI32Eqz) {
1850 // Unary "unequal" means "not equals zero".
1851 __ emit_cond_jump(kUnequal, &cont_false, kWasmI32, value);
1852 outstanding_op_ = kNoOutstandingOp;
1853 } else {
1854 // Otherwise, it's an i32 compare opcode.
1855 Condition cond = NegateCondition(GetCompareCondition(outstanding_op_));
1856 Register rhs = value;
1857 Register lhs = __ PopToRegister(LiftoffRegList::ForRegs(rhs)).gp();
1858 __ emit_cond_jump(cond, &cont_false, kWasmI32, lhs, rhs);
1859 outstanding_op_ = kNoOutstandingOp;
1860 }
1861
1862 BrOrRet(decoder, depth);
1863 __ bind(&cont_false);
1864 }

1879 // Generate a branch table for input in [min, max).
1880 // TODO(wasm): Generate a real branch table (like TF TableSwitch).
1881 void GenerateBrTable(FullDecoder* decoder, LiftoffRegister tmp,
1882 LiftoffRegister value, uint32_t min, uint32_t max,
1883 BranchTableIterator<validate>* table_iterator,
1884 std::map<uint32_t, MovableLabel>* br_targets) {
1885 DCHECK_LT(min, max)((void) 0);
1886 // Check base case.
1887 if (max == min + 1) {
1888 DCHECK_EQ(min, table_iterator->cur_index())((void) 0);
1889 GenerateBrCase(decoder, table_iterator->next(), br_targets);
1890 return;
1891 }
1892
1893 uint32_t split = min + (max - min) / 2;
1894 Label upper_half;
1895 __ LoadConstant(tmp, WasmValue(split));
1896 __ emit_cond_jump(kUnsignedGreaterEqual, &upper_half, kWasmI32, value.gp(),
1897 tmp.gp());
1898 // Emit br table for lower half:
1899 GenerateBrTable(decoder, tmp, value, min, split, table_iterator,
1900 br_targets);
1901 __ bind(&upper_half);
1902 // table_iterator will trigger a DCHECK if we don't stop decoding now.
1903 if (did_bailout()) return;
1904 // Emit br table for upper half:
1905 GenerateBrTable(decoder, tmp, value, split, max, table_iterator,
1906 br_targets);
1907 }

1909 void BrTable(FullDecoder* decoder, const BranchTableImmediate<validate>& imm,
1910 const Value& key) {
1911 LiftoffRegList pinned;
1912 LiftoffRegister value = pinned.set(__ PopToRegister());
1913 BranchTableIterator<validate> table_iterator(decoder, imm);
1914 std::map<uint32_t, MovableLabel> br_targets;
1915
1916 if (imm.table_count > 0) {
1917 LiftoffRegister tmp = __ GetUnusedRegister(kGpReg, pinned);
1918 __ LoadConstant(tmp, WasmValue(uint32_t{imm.table_count}));
1919 Label case_default;
1920 __ emit_cond_jump(kUnsignedGreaterEqual, &case_default, kWasmI32,
1921 value.gp(), tmp.gp());
1922
1923 GenerateBrTable(decoder, tmp, value, 0, imm.table_count, &table_iterator,
1924 &br_targets);
1925
1926 __ bind(&case_default);
1927 // table_iterator will trigger a DCHECK if we don't stop decoding now.
1928 if (did_bailout()) return;
1929 }
1930
1931 // Generate the default case.
1932 GenerateBrCase(decoder, table_iterator.next(), &br_targets);
1933 DCHECK(!table_iterator.has_next())((void) 0);
1934 }

1978 // Returns true if the memory access is statically known to be out of bounds
1979 // (a jump to the trap was generated then); return false otherwise.
1980 bool BoundsCheckMem(FullDecoder* decoder, uint32_t access_size,
1981 uint32_t offset, Register index, LiftoffRegList pinned,
1982 ForceCheck force_check) {
1983 const bool statically_oob =
1984 !base::IsInBounds<uint64_t>(offset, access_size, env_->max_memory_size);
1985
1986 if (!force_check && !statically_oob &&
1987 (!FLAG_wasm_bounds_checks || env_->use_trap_handler)) {
1988 return false;
1989 }
1990
1991 // TODO(wasm): This adds protected instruction information for the jump
1992 // instruction we are about to generate. It would be better to just not add
1993 // protected instruction info when the pc is 0.
1994 Label* trap_label = AddOutOfLineTrap(
1995 decoder->position(), WasmCode::kThrowWasmTrapMemOutOfBounds,
1996 env_->use_trap_handler ? __ pc_offset() : 0);
1997
1998 if (statically_oob) {
1999 __ emit_jump(trap_label);
2000 decoder->SetSucceedingCodeDynamicallyUnreachable();
2001 return true;
2002 }
2003
2004 uint64_t end_offset = uint64_t{offset} + access_size - 1u;
2005
2006 // If the end offset is larger than the smallest memory, dynamically check
2007 // the end offset against the actual memory size, which is not known at
2008 // compile time. Otherwise, only one check is required (see below).
2009 LiftoffRegister end_offset_reg =
2010 pinned.set(__ GetUnusedRegister(kGpReg, pinned));
2011 Register mem_size = __ GetUnusedRegister(kGpReg, pinned).gp();
2012 LOAD_INSTANCE_FIELD(mem_size, MemorySize, kSystemPointerSize);
2013
2014 if (kSystemPointerSize == 8) {
2015 __ LoadConstant(end_offset_reg, WasmValue(end_offset));
2016 } else {
2017 __ LoadConstant(end_offset_reg,
2018 WasmValue(static_cast<uint32_t>(end_offset)));
2019 }
2020
2021 if (end_offset >= env_->min_memory_size) {
2022 __ emit_cond_jump(kUnsignedGreaterEqual, trap_label,
2023 LiftoffAssembler::kWasmIntPtr, end_offset_reg.gp(),
2024 mem_size);
2025 }
2026
2027 // Just reuse the end_offset register for computing the effective size.
2028 LiftoffRegister effective_size_reg = end_offset_reg;
2029 __ emit_ptrsize_sub(effective_size_reg.gp(), mem_size, end_offset_reg.gp());
2030

2749 typename EmitFn>
2750 void EmitSimdExtractLaneOp(EmitFn fn,
2751 const SimdLaneImmediate<validate>& imm) {
2752 static constexpr RegClass src_rc = reg_class_for(src_type);
2753 static constexpr RegClass result_rc = reg_class_for(result_type);
2754 LiftoffRegister lhs = __ PopToRegister();
2755 LiftoffRegister dst = src_rc == result_rc
2756 ? __ GetUnusedRegister(result_rc, {lhs}, {})
2757 : __ GetUnusedRegister(result_rc, {});
2758 fn(dst, lhs, imm.lane);
2759 __ PushRegister(ValueType::Primitive(result_type), dst);
2760 }

2762 template <ValueType::Kind src2_type, typename EmitFn>
2763 void EmitSimdReplaceLaneOp(EmitFn fn,
2764 const SimdLaneImmediate<validate>& imm) {
2765 static constexpr RegClass src1_rc = reg_class_for(kS128);
2766 static constexpr RegClass src2_rc = reg_class_for(src2_type);
2767 static constexpr RegClass result_rc = reg_class_for(kS128);
2768 // On backends which need fp pair, src1_rc and result_rc end up being
2769 // kFpRegPair, which is != kFpReg, but we still want to pin src2 when it is
2770 // kFpReg, since it can overlap with those pairs.
2771 static constexpr bool pin_src2 = kNeedS128RegPair && src2_rc == kFpReg;
2772
2773 // Does not work for arm
2774 LiftoffRegister src2 = __ PopToRegister();
2775 LiftoffRegister src1 = (src1_rc == src2_rc || pin_src2)
2776 ? __ PopToRegister(LiftoffRegList::ForRegs(src2))
2777 : __
2778 PopToRegister();
2779 LiftoffRegister dst =
2780 (src2_rc == result_rc || pin_src2)
2781 ? __ GetUnusedRegister(result_rc, {src1},
2782 LiftoffRegList::ForRegs(src2))
2783 : __ GetUnusedRegister(result_rc, {src1}, {});
2784 fn(dst, src1, src2, imm.lane);
2785 __ PushRegister(kWasmS128, dst);
2786 }

2833 void S128Const(FullDecoder* decoder, const Simd128Immediate<validate>& imm,
2834 Value* result) {
2835 constexpr RegClass result_rc = reg_class_for(ValueType::kS128);
2836 LiftoffRegister dst = __ GetUnusedRegister(result_rc, {});
2837 bool all_zeroes = std::all_of(std::begin(imm.value), std::end(imm.value),
2838 [](uint8_t v) { return v == 0; });
2839 bool all_ones = std::all_of(std::begin(imm.value), std::end(imm.value),
2840 [](uint8_t v) { return v == 0xff; });
2841 if (all_zeroes) {
2842 __ LiftoffAssembler::emit_s128_xor(dst, dst, dst);
2843 } else if (all_ones) {
2844 // Any SIMD eq will work, i32x4 is efficient on all archs.
2845 __ LiftoffAssembler::emit_i32x4_eq(dst, dst, dst);
2846 } else {
2847 __ LiftoffAssembler::emit_s128_const(dst, imm.value);
2848 }
2849 __ PushRegister(kWasmS128, dst);
2850 }

2852 void Simd8x16ShuffleOp(FullDecoder* decoder,
2853 const Simd128Immediate<validate>& imm,
2854 const Value& input0, const Value& input1,
2855 Value* result) {
2856 static constexpr RegClass result_rc = reg_class_for(ValueType::kS128);
2857 LiftoffRegister rhs = __ PopToRegister();
2858 LiftoffRegister lhs = __ PopToRegister(LiftoffRegList::ForRegs(rhs));
2859 LiftoffRegister dst = __ GetUnusedRegister(result_rc, {lhs, rhs}, {});
2860
2861 uint8_t shuffle[kSimd128Size];
2862 memcpy(shuffle, imm.value, sizeof(shuffle));
2863 bool is_swizzle;
2864 bool needs_swap;
2865 wasm::SimdShuffle::CanonicalizeShuffle(lhs == rhs, shuffle, &needs_swap,
2866 &is_swizzle);
2867 if (needs_swap) {

3684 void CallIndirect(FullDecoder* decoder, const Value& index_val,
3685 const CallIndirectImmediate<validate>& imm,
3686 CallKind call_kind) {
3687 if (imm.table_index != 0) {
3688 return unsupported(decoder, kRefTypes, "table index != 0");
3689 }
3690 for (ValueType ret : imm.sig->returns()) {
3691 if (!CheckSupportedType(decoder, kSupportedTypes, ret, "return")) {
3692 return;
3693 }
3694 }
3695
3696 // Pop the index.
3697 Register index = __ PopToRegister().gp();
3698 // If that register is still being used after popping, we move it to another
3699 // register, because we want to modify that register.
3700 if (__ cache_state()->is_used(LiftoffRegister(index))) {
3701 Register new_index =
3702 __ GetUnusedRegister(kGpReg, LiftoffRegList::ForRegs(index)).gp();
3703 __ Move(new_index, index, kWasmI32);
3704 index = new_index;
3705 }
3706
3707 LiftoffRegList pinned = LiftoffRegList::ForRegs(index);
3708 // Get three temporary registers.
3709 Register table = pinned.set(__ GetUnusedRegister(kGpReg, pinned)).gp();
3710 Register tmp_const = pinned.set(__ GetUnusedRegister(kGpReg, pinned)).gp();
3711 Register scratch = pinned.set(__ GetUnusedRegister(kGpReg, pinned)).gp();
3712
3713 // Bounds check against the table size.
3714 Label* invalid_func_label = AddOutOfLineTrap(
3715 decoder->position(), WasmCode::kThrowWasmTrapFuncInvalid);
3716
3717 uint32_t canonical_sig_num = env_->module->signature_ids[imm.sig_index];
3718 DCHECK_GE(canonical_sig_num, 0)((void) 0);
3719 DCHECK_GE(kMaxInt, canonical_sig_num)((void) 0);
3720
3721 // Compare against table size stored in
3722 // {instance->indirect_function_table_size}.
3723 LOAD_INSTANCE_FIELD(tmp_const, IndirectFunctionTableSize, kUInt32Size);
3724 __ emit_cond_jump(kUnsignedGreaterEqual, invalid_func_label, kWasmI32,
3725 index, tmp_const);
3726
3727 // Mask the index to prevent SSCA.
3728 if (FLAG_untrusted_code_mitigations) {
3729 DEBUG_CODE_COMMENT("Mask indirect call index");
3730 // mask = ((index - size) & ~index) >> 31
3731 // Reuse allocated registers; note: size is still stored in {tmp_const}.
3732 Register diff = table;
3733 Register neg_index = tmp_const;
3734 Register mask = scratch;
3735 // 1) diff = index - size
3736 __ emit_i32_sub(diff, index, tmp_const);
3737 // 2) neg_index = ~index
3738 __ LoadConstant(LiftoffRegister(neg_index), WasmValue(int32_t{-1}));
3739 __ emit_i32_xor(neg_index, neg_index, index);
3740 // 3) mask = diff & neg_index
3741 __ emit_i32_and(mask, diff, neg_index);
3742 // 4) mask = mask >> 31
3743 __ emit_i32_sari(mask, mask, 31);
3744
3745 // Apply mask.
3746 __ emit_i32_and(index, index, mask);
3747 }
3748
3749 DEBUG_CODE_COMMENT("Check indirect call signature");
3750 // Load the signature from {instance->ift_sig_ids[key]}
3751 LOAD_INSTANCE_FIELD(table, IndirectFunctionTableSigIds, kSystemPointerSize);
3752 // Shift {index} by 2 (multiply by 4) to represent kInt32Size items.
3753 STATIC_ASSERT((1 << 2) == kInt32Size)static_assert((1 << 2) == kInt32Size, "(1 << 2) == kInt32Size"
);
3754 __ emit_i32_shli(index, index, 2);
3755 __ Load(LiftoffRegister(scratch), table, index, 0, LoadType::kI32Load,
3756 pinned);
3757
3758 // Compare against expected signature.
3759 __ LoadConstant(LiftoffRegister(tmp_const), WasmValue(canonical_sig_num));
3760
3761 Label* sig_mismatch_label = AddOutOfLineTrap(
3762 decoder->position(), WasmCode::kThrowWasmTrapFuncSigMismatch);
3763 __ emit_cond_jump(kUnequal, sig_mismatch_label,
3764 LiftoffAssembler::kWasmIntPtr, scratch, tmp_const);
3765
3766 // At this point {index} has already been multiplied by 4.
3767 DEBUG_CODE_COMMENT("Execute indirect call");
3768 if (kTaggedSize != kInt32Size) {
3769 DCHECK_EQ(kTaggedSize, kInt32Size * 2)((void) 0);
3770 // Multiply {index} by another 2 to represent kTaggedSize items.
3771 __ emit_i32_add(index, index, index);
3772 }
3773 // At this point {index} has already been multiplied by kTaggedSize.
3774
3775 // Load the instance from {instance->ift_instances[key]}
3776 LOAD_TAGGED_PTR_INSTANCE_FIELD(table, IndirectFunctionTableRefs);
3777 __ LoadTaggedPointer(tmp_const, table, index,
3778 ObjectAccess::ElementOffsetInTaggedFixedArray(0),
3779 pinned);
3780
3781 if (kTaggedSize != kSystemPointerSize) {
3782 DCHECK_EQ(kSystemPointerSize, kTaggedSize * 2)((void) 0);
3783 // Multiply {index} by another 2 to represent kSystemPointerSize items.
3784 __ emit_i32_add(index, index, index);
3785 }
3786 // At this point {index} has already been multiplied by kSystemPointerSize.
3787
3788 Register* explicit_instance = &tmp_const;
3789
3790 // Load the target from {instance->ift_targets[key]}
3791 LOAD_INSTANCE_FIELD(table, IndirectFunctionTableTargets,
3792 kSystemPointerSize);
3793 __ Load(LiftoffRegister(scratch), table, index, 0, kPointerLoadType,
3794 pinned);
3795
3796 auto call_descriptor =
3797 compiler::GetWasmCallDescriptor(compilation_zone_, imm.sig);
3798 call_descriptor =
3799 GetLoweredCallDescriptor(compilation_zone_, call_descriptor);
3800
3801 Register target = scratch;
3802 __ PrepareCall(imm.sig, call_descriptor, &target, explicit_instance);
3803 if (call_kind == kReturnCall) {
3804 __ PrepareTailCall(
3805 static_cast<int>(call_descriptor->StackParameterCount()),
3806 static_cast<int>(
3807 call_descriptor->GetStackParameterDelta(descriptor_)));
3808 __ TailCallIndirect(target);
3809 } else {
3810 source_position_table_builder_.AddPosition(
3811 __ pc_offset(), SourcePosition(decoder->position()), true);
3812 __ CallIndirect(imm.sig, call_descriptor, target);
3813 }
3814
3815 RegisterDebugSideTableEntry(DebugSideTableBuilder::kDidSpill);
3816 safepoint_table_builder_.DefineSafepoint(&asm_, Safepoint::kNoLazyDeopt);
3817
3818 __ FinishCall(imm.sig, call_descriptor);
3819 }
