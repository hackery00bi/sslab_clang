102void StackFrameIterator::Advance() {
103 DCHECK(!done())((void) 0);
104 // Compute the state of the calling frame before restoring
105 // callee-saved registers and unwinding handlers. This allows the
106 // frame code that computes the caller state to access the top
107 // handler and the value of any callee-saved register if needed.
108 StackFrame::State state;
109 StackFrame::Type type = frame_->GetCallerState(&state);
110
111 // Unwind handlers corresponding to the current frame.
112 StackHandlerIterator it(frame_, handler_);
113 while (!it.done()) it.Advance();
114 handler_ = it.handler();
115
116 // Advance to the calling frame.
117 frame_ = SingletonFor(type, &state);
118
119 // When we're done iterating over the stack frames, the handler
120 // chain must have been completely unwound.
121 DCHECK(!done() || handler_ == nullptr)((void) 0);
122}

445void SafeStackFrameIterator::Advance() {
446 while (true) {
447 AdvanceOneFrame();
448 if (done()) break;
449 ExternalCallbackScope* last_callback_scope = nullptr;
450 while (external_callback_scope_ != nullptr &&
451 external_callback_scope_->scope_address() < frame_->fp()) {
452 // As long as the setup of a frame is not atomic, we may happen to be
453 // in an interval where an ExternalCallbackScope is already created,
454 // but the frame is not yet entered. So we are actually observing
455 // the previous frame.
456 // Skip all the ExternalCallbackScope's that are below the current fp.
457 last_callback_scope = external_callback_scope_;
458 external_callback_scope_ = external_callback_scope_->previous();
459 }
460 if (frame_->is_java_script() || frame_->is_wasm() ||
461 frame_->is_wasm_to_js()) {

465 // Some of the EXIT frames may have ExternalCallbackScope allocated on
466 // top of them. In that case the scope corresponds to the first EXIT
467 // frame beneath it. There may be other EXIT frames on top of the
468 // ExternalCallbackScope, just skip them as we cannot collect any useful
469 // information about them.

472 last_callback_scope->callback_entrypoint_address();

912void StandardFrame::IterateCompiledFrame(RootVisitor* v) const {
913 // Make sure that we're not doing "safe" stack frame iteration. We cannot
914 // possibly find pointers in optimized frames in that state.
915 DCHECK(can_access_heap_objects())((void) 0);
916
917 // Find the code and compute the safepoint information.
918 Address inner_pointer = pc();
919 const wasm::WasmCode* wasm_code =
920 isolate()->wasm_engine()->code_manager()->LookupCode(inner_pointer);
921 SafepointEntry safepoint_entry;
922 uint32_t stack_slots;
923 Code code;
924 bool has_tagged_params = false;
925 uint32_t tagged_parameter_slots = 0;
926 if (wasm_code != nullptr) {
927 SafepointTable table(wasm_code);
928 safepoint_entry = table.FindEntry(inner_pointer);
929 stack_slots = wasm_code->stack_slots();
930 has_tagged_params = wasm_code->kind() != wasm::WasmCode::kFunction &&
931 wasm_code->kind() != wasm::WasmCode::kWasmToCapiWrapper;
932 tagged_parameter_slots = wasm_code->tagged_parameter_slots();
933 } else {
934 InnerPointerToCodeCache::InnerPointerToCodeCacheEntry* entry =
935 isolate()->inner_pointer_to_code_cache()->GetCacheEntry(inner_pointer);
936 if (!entry->safepoint_entry.is_valid()) {
937 entry->safepoint_entry = entry->code.GetSafepointEntry(inner_pointer);
938 DCHECK(entry->safepoint_entry.is_valid())((void) 0);
939 } else {
940 DCHECK(entry->safepoint_entry.Equals(((void) 0)
941 entry->code.GetSafepointEntry(inner_pointer)))((void) 0);
942 }
943
944 code = entry->code;
945 safepoint_entry = entry->safepoint_entry;
946 stack_slots = code.stack_slots();
947 has_tagged_params = code.has_tagged_params();
948 }
949 uint32_t slot_space = stack_slots * kSystemPointerSize;
950
951 // Determine the fixed header and spill slot area size.
952 int frame_header_size = StandardFrameConstants::kFixedFrameSizeFromFp;
953 intptr_t marker =
954 Memory<intptr_t>(fp() + CommonFrameConstants::kContextOrFrameTypeOffset);
955 if (StackFrame::IsTypeMarker(marker)) {
956 StackFrame::Type candidate = StackFrame::MarkerToType(marker);
957 switch (candidate) {
958 case ENTRY:
959 case CONSTRUCT_ENTRY:
960 case EXIT:
961 case BUILTIN_CONTINUATION:
962 case JAVA_SCRIPT_BUILTIN_CONTINUATION:
963 case JAVA_SCRIPT_BUILTIN_CONTINUATION_WITH_CATCH:
964 case BUILTIN_EXIT:
965 case ARGUMENTS_ADAPTOR:
966 case STUB:
967 case INTERNAL:
968 case CONSTRUCT:
969 case JS_TO_WASM:
970 case C_WASM_ENTRY:
971 case WASM_DEBUG_BREAK:
972 frame_header_size = TypedFrameConstants::kFixedFrameSizeFromFp;
973 break;
974 case WASM_TO_JS:
975 case WASM:
976 case WASM_COMPILE_LAZY:
977 frame_header_size = WasmFrameConstants::kFixedFrameSizeFromFp;
978 break;
979 case WASM_EXIT:
980 // The last value in the frame header is the calling PC, which should
981 // not be visited.
982 static_assert(WasmExitFrameConstants::kFixedSlotCountFromFp ==
983 WasmFrameConstants::kFixedSlotCountFromFp + 1,
984 "WasmExitFrame has one slot more than WasmFrame");
985 frame_header_size = WasmFrameConstants::kFixedFrameSizeFromFp;
986 break;
987 case OPTIMIZED:
988 case INTERPRETED:
989 case BUILTIN:
990 // These frame types have a context, but they are actually stored
991 // in the place on the stack that one finds the frame type.
992 UNREACHABLE()V8_Fatal("unreachable code");
993 break;
994 case NATIVE:
995 case NONE:
996 case NUMBER_OF_TYPES:
997 case MANUAL:
998 UNREACHABLE()V8_Fatal("unreachable code");
999 break;
1000 }
1001 }
1002 slot_space -=
1003 (frame_header_size + StandardFrameConstants::kFixedFrameSizeAboveFp);
1004
1005 FullObjectSlot frame_header_base(&Memory<Address>(fp() - frame_header_size));
1006 FullObjectSlot frame_header_limit(
1007 &Memory<Address>(fp() - StandardFrameConstants::kCPSlotSize));
1008 FullObjectSlot parameters_base(&Memory<Address>(sp()));
1009 FullObjectSlot parameters_limit(frame_header_base.address() - slot_space);
1010
1011 // Visit the rest of the parameters if they are tagged.
1012 if (has_tagged_params) {
1013 v->VisitRootPointers(Root::kTop, nullptr, parameters_base,
1014 parameters_limit);
1015 }
1016
1017 // Visit pointer spill slots and locals.
1018 uint8_t* safepoint_bits = safepoint_entry.bits();
1019 for (unsigned index = 0; index < stack_slots; index++) {
1020 int byte_index = index >> kBitsPerByteLog2;
1021 int bit_index = index & (kBitsPerByte - 1);
1022 if ((safepoint_bits[byte_index] & (1U << bit_index)) != 0) {
1023 FullObjectSlot spill_slot = parameters_limit + index;
1024#ifdef V8_COMPRESS_POINTERS
1025 // Spill slots may contain compressed values in which case the upper
1026 // 32-bits will contain zeros. In order to simplify handling of such
1027 // slots in GC we ensure that the slot always contains full value.
1028
1029 // The spill slot may actually contain weak references so we load/store
1030 // values using spill_slot.location() in order to avoid dealing with
1031 // FullMaybeObjectSlots here.
1032 Tagged_t compressed_value = static_cast<Tagged_t>(*spill_slot.location());
1033 if (!HAS_SMI_TAG(compressed_value)((static_cast<i::Tagged_t>(compressed_value) & ::i::
kSmiTagMask) == ::i::kSmiTag)) {
1034 // We don't need to update smi values.
1035 *spill_slot.location() =
1036 DecompressTaggedPointer(isolate(), compressed_value);
1037 }
1038#endif
1039 v->VisitRootPointer(Root::kTop, nullptr, spill_slot);
1040 }
1041 }
1042
1043 // Visit tagged parameters that have been passed to the function of this
1044 // frame. Conceptionally these parameters belong to the parent frame. However,
1045 // the exact count is only known by this frame (in the presence of tail calls,
1046 // this information cannot be derived from the call site).
1047 if (tagged_parameter_slots > 0) {
1048 FullObjectSlot tagged_parameter_base(&Memory<Address>(caller_sp()));
1049 FullObjectSlot tagged_parameter_limit =
1050 tagged_parameter_base + tagged_parameter_slots;
1051
1052 v->VisitRootPointers(Root::kTop, nullptr, tagged_parameter_base,
1053 tagged_parameter_limit);
1054 }
1055
1056 // For the off-heap code cases, we can skip this.
1057 if (!code.is_null()) {
1058 // Visit the return address in the callee and incoming arguments.
1059 IteratePc(v, pc_address(), constant_pool_address(), code);
1060 }
1061
1062 // If this frame has JavaScript ABI, visit the context (in stub and JS
1063 // frames) and the function (in JS frames). If it has WebAssembly ABI, visit
1064 // the instance object.
1065 v->VisitRootPointers(Root::kTop, nullptr, frame_header_base,
1066 frame_header_limit);
1067}

1508void OptimizedFrame::Summarize(std::vector<FrameSummary>* frames) const {
1509 DCHECK(frames->empty())((void) 0);
1510 DCHECK(is_optimized())((void) 0);
1511
1512 // Delegate to JS frame in absence of turbofan deoptimization.
1513 // TODO(turbofan): Revisit once we support deoptimization across the board.
1514 Code code = LookupCode();
1515 if (code.kind() == CodeKind::BUILTIN) {
1516 return JavaScriptFrame::Summarize(frames);
1517 }
1518
1519 int deopt_index = Safepoint::kNoDeoptimizationIndex;
1520 DeoptimizationData const data = GetDeoptimizationData(&deopt_index);
1521 if (deopt_index == Safepoint::kNoDeoptimizationIndex) {
1522 CHECK(data.is_null())do { if ((__builtin_expect(!!(!(data.is_null())), 0))) { V8_Fatal
("Check failed: %s.", "data.is_null()"); } } while (false);
1523 FATAL("Missing deoptimization information for OptimizedFrame::Summarize.")V8_Fatal("Missing deoptimization information for OptimizedFrame::Summarize."
);
1524 }
1525
1526 // Prepare iteration over translation. Note that the below iteration might
1527 // materialize objects without storing them back to the Isolate, this will
1528 // lead to objects being re-materialized again for each summary.
1529 TranslatedState translated(this);
1530 translated.Prepare(fp());
1531
1532 // We create the summary in reverse order because the frames
1533 // in the deoptimization translation are ordered bottom-to-top.
1534 bool is_constructor = IsConstructor();
1535 for (auto it = translated.begin(); it != translated.end(); it++) {
1536 if (it->kind() == TranslatedFrame::kInterpretedFunction ||
1537 it->kind() == TranslatedFrame::kJavaScriptBuiltinContinuation ||
1538 it->kind() ==
1539 TranslatedFrame::kJavaScriptBuiltinContinuationWithCatch) {
1540 Handle<SharedFunctionInfo> shared_info = it->shared_info();
1541
1542 // The translation commands are ordered and the function is always
1543 // at the first position, and the receiver is next.
1544 TranslatedFrame::iterator translated_values = it->begin();
1545
1546 // Get or materialize the correct function in the optimized frame.
1547 Handle<JSFunction> function =
1548 Handle<JSFunction>::cast(translated_values->GetValue());
1549 translated_values++;
1550
1551 // Get or materialize the correct receiver in the optimized frame.
1552 Handle<Object> receiver = translated_values->GetValue();
1553 translated_values++;
1554
1555 // Determine the underlying code object and the position within it from
1556 // the translation corresponding to the frame type in question.
1557 Handle<AbstractCode> abstract_code;
1558 unsigned code_offset;
1559 if (it->kind() == TranslatedFrame::kJavaScriptBuiltinContinuation ||
1560 it->kind() ==
1561 TranslatedFrame::kJavaScriptBuiltinContinuationWithCatch) {
1562 code_offset = 0;
1563 abstract_code =
1564 handle(AbstractCode::cast(isolate()->builtins()->builtin(
1565 Builtins::GetBuiltinFromBailoutId(it->node_id()))),
1566 isolate());
1567 } else {
1568 DCHECK_EQ(it->kind(), TranslatedFrame::kInterpretedFunction)((void) 0);
1569 code_offset = it->node_id().ToInt(); // Points to current bytecode.
1570 abstract_code = handle(shared_info->abstract_code(), isolate());
1571 }
1572
1573 // Append full summary of the encountered JS frame.
1574 Handle<FixedArray> params = GetParameters();
1575 FrameSummary::JavaScriptFrameSummary summary(
1576 isolate(), *receiver, *function, *abstract_code, code_offset,
1577 is_constructor, *params);
1578 frames->push_back(summary);
1579 is_constructor = false;
1580 } else if (it->kind() == TranslatedFrame::kConstructStub) {
1581 // The next encountered JS frame will be marked as a constructor call.
1582 DCHECK(!is_constructor)((void) 0);
1583 is_constructor = true;
1584 }
1585 }

1588int OptimizedFrame::LookupExceptionHandlerInTable(
1589 int* data, HandlerTable::CatchPrediction* prediction) {
1590 // We cannot perform exception prediction on optimized code. Instead, we need
1591 // to use FrameSummary to find the corresponding code offset in unoptimized
1592 // code to perform prediction there.
1593 DCHECK_NULL(prediction)((void) 0);
1594 Code code = LookupCode();
1595 HandlerTable table(code);
1596 int pc_offset = static_cast<int>(pc() - code.InstructionStart());
1597 DCHECK_NULL(data)((void) 0); // Data is not used and will not return a value.
1598
1599 // When the return pc has been replaced by a trampoline there won't be
1600 // a handler for this trampoline. Thus we need to use the return pc that
1601 // _used to be_ on the stack to get the right ExceptionHandler.
1602 if (CodeKindCanDeoptimize(code.kind()) && code.marked_for_deoptimization()) {
1603 SafepointTable safepoints(code);
1604 pc_offset = safepoints.find_return_pc(pc_offset);
1605 }
1606 return table.LookupReturn(pc_offset);

1651void OptimizedFrame::GetFunctions(
1652 std::vector<SharedFunctionInfo>* functions) const {
1653 DCHECK(functions->empty())((void) 0);
1654 DCHECK(is_optimized())((void) 0);
1655
1656 // Delegate to JS frame in absence of turbofan deoptimization.
1657 // TODO(turbofan): Revisit once we support deoptimization across the board.
1658 Code code = LookupCode();
1659 if (code.kind() == CodeKind::BUILTIN) {
1660 return JavaScriptFrame::GetFunctions(functions);
1661 }
1662
1663 DisallowHeapAllocation no_gc;
1664 int deopt_index = Safepoint::kNoDeoptimizationIndex;
1665 DeoptimizationData const data = GetDeoptimizationData(&deopt_index);
1666 DCHECK(!data.is_null())((void) 0);
1667 DCHECK_NE(Safepoint::kNoDeoptimizationIndex, deopt_index)((void) 0);
1668 FixedArray const literal_array = data.LiteralArray();
1669
1670 TranslationIterator it(data.TranslationByteArray(),
1671 data.TranslationIndex(deopt_index).value());
1672 Translation::Opcode opcode = static_cast<Translation::Opcode>(it.Next());
1673 DCHECK_EQ(Translation::BEGIN, opcode)((void) 0);
1674 it.Next(); // Skip frame count.
1675 int jsframe_count = it.Next();
1676 it.Next(); // Skip update feedback count.
1677
1678 // We insert the frames in reverse order because the frames
1679 // in the deoptimization translation are ordered bottom-to-top.
1680 while (jsframe_count != 0) {
1681 opcode = static_cast<Translation::Opcode>(it.Next());
1682 if (opcode == Translation::INTERPRETED_FRAME ||
1683 opcode == Translation::JAVA_SCRIPT_BUILTIN_CONTINUATION_FRAME ||
1684 opcode ==
1685 Translation::JAVA_SCRIPT_BUILTIN_CONTINUATION_WITH_CATCH_FRAME) {
1686 it.Next(); // Skip bailout id.
1687 jsframe_count--;
1688
1689 // The second operand of the frame points to the function.
1690 Object shared = literal_array.get(it.Next());
1691 functions->push_back(SharedFunctionInfo::cast(shared));
1692
1693 // Skip over remaining operands to advance to the next opcode.
1694 it.Skip(Translation::NumberOfOperandsFor(opcode) - 2);
1695 } else {
1696 // Skip over operands to advance to the next opcode.
1697 it.Skip(Translation::NumberOfOperandsFor(opcode));
1698 }
1699 }
