1007Object Object::GetSimpleHash(Object object) {
1008 DisallowHeapAllocation no_gc;
1009 if (object.IsSmi()) {
1010 uint32_t hash = ComputeUnseededHash(Smi::ToInt(object));
1011 return Smi::FromInt(hash & Smi::kMaxValue);
1012 }
1013 if (object.IsHeapNumber()) {
1014 double num = HeapNumber::cast(object).value();
1015 if (std::isnan(num)) return Smi::FromInt(Smi::kMaxValue);
1016 // Use ComputeUnseededHash for all values in Signed32 range, including -0,
1017 // which is considered equal to 0 because collections use SameValueZero.
1018 uint32_t hash;
1019 // Check range before conversion to avoid undefined behavior.
1020 if (num >= kMinInt && num <= kMaxInt && FastI2D(FastD2I(num)) == num) {
1021 hash = ComputeUnseededHash(FastD2I(num));
1022 } else {
1023 hash = ComputeLongHash(double_to_uint64(num));
1024 }
1025 return Smi::FromInt(hash & Smi::kMaxValue);
1026 }
1027 if (object.IsName()) {
1028 uint32_t hash = Name::cast(object).Hash();
1029 return Smi::FromInt(hash);
1030 }
1031 if (object.IsOddball()) {
1032 uint32_t hash = Oddball::cast(object).to_string().Hash();
1033 return Smi::FromInt(hash);
1034 }
1035 if (object.IsBigInt()) {
1036 uint32_t hash = BigInt::cast(object).Hash();
1037 return Smi::FromInt(hash & Smi::kMaxValue);
1038 }
1039 if (object.IsSharedFunctionInfo()) {
1040 uint32_t hash = SharedFunctionInfo::cast(object).Hash();
1041 return Smi::FromInt(hash & Smi::kMaxValue);
1042 }
1043 DCHECK(object.IsJSReceiver())((void) 0);
1044 return object;
1045}
