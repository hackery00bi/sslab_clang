50base::AddressRegion DisjointAllocationPool::Merge(
51 base::AddressRegion new_region) {
52 // Find the possible insertion position by identifying the first region whose
53 // start address is not less than that of {new_region}. Since there cannot be
54 // any overlap between regions, this also means that the start of {above} is
55 // bigger or equal than the *end* of {new_region}.
56 auto above = regions_.lower_bound(new_region);
57 DCHECK(above == regions_.end() || above->begin() >= new_region.end())((void) 0);
58
59 // Check whether to merge with {above}.
60 if (above != regions_.end() && new_region.end() == above->begin()) {
61 base::AddressRegion merged_region{new_region.begin(),
62 new_region.size() + above->size()};
63 DCHECK_EQ(merged_region.end(), above->end())((void) 0);
64 // Check whether to also merge with the region below.
65 if (above != regions_.begin()) {
66 auto below = above;
67 --below;
68 if (below->end() == new_region.begin()) {
69 merged_region = {below->begin(), below->size() + merged_region.size()};
70 regions_.erase(below);
71 }
72 }
73 auto insert_pos = regions_.erase(above);
74 regions_.insert(insert_pos, merged_region);
75 return merged_region;
76 }
77
78 // No element below, and not adjavent to {above}: insert and done.
79 if (above == regions_.begin()) {
80 regions_.insert(above, new_region);
81 return new_region;
82 }
83
84 auto below = above;
85 --below;
86 // Consistency check:
87 DCHECK(above == regions_.end() || below->end() < above->begin())((void) 0);
88
89 // Adjacent to {below}: merge and done.
90 if (below->end() == new_region.begin()) {
91 base::AddressRegion merged_region{below->begin(),
92 below->size() + new_region.size()};
93 DCHECK_EQ(merged_region.end(), new_region.end())((void) 0);
94 regions_.erase(below);
95 regions_.insert(above, merged_region);
96 return merged_region;
97 }
98
99 // Not adjacent to any existing region: insert between {below} and {above}.
100 DCHECK_LT(below->end(), new_region.begin())((void) 0);
101 regions_.insert(above, new_region);
102 return new_region;
103}

110base::AddressRegion DisjointAllocationPool::AllocateInRegion(
111 size_t size, base::AddressRegion region) {
112 // Get an iterator to the first contained region whose start address is not
113 // smaller than the start address of {region}. Start the search from the
114 // region one before that (the last one whose start address is smaller).
115 auto it = regions_.lower_bound(region);
116 if (it != regions_.begin()) --it;
117
118 for (auto end = regions_.end(); it != end; ++it) {
119 base::AddressRegion overlap = it->GetOverlap(region);
120 if (size > overlap.size()) continue;
121 base::AddressRegion ret{overlap.begin(), size};
122 base::AddressRegion old = *it;
123 auto insert_pos = regions_.erase(it);
124 if (size == old.size()) {
125 // We use the full region --> nothing to add back.
126 } else if (ret.begin() == old.begin()) {
127 // We return a region at the start --> shrink old region from front.
128 regions_.insert(insert_pos, {old.begin() + size, old.size() - size});
129 } else if (ret.end() == old.end()) {
130 // We return a region at the end --> shrink remaining region.
131 regions_.insert(insert_pos, {old.begin(), old.size() - size});
132 } else {
133 // We return something in the middle --> split the remaining region
134 // (insert the region with smaller address first).
135 regions_.insert(insert_pos, {old.begin(), ret.begin() - old.begin()});
136 regions_.insert(insert_pos, {ret.end(), old.end() - ret.end()});
137 }
138 return ret;
139 }
140 return {};

520// On Windows, we cannot commit a region that straddles different reservations
521// of virtual memory. Because we bump-allocate, and because, if we need more
522// memory, we append that memory at the end of the owned_code_space_ list, we
523// traverse that list in reverse order to find the reservation(s) that guide how
524// to chunk the region to commit.
525#if V8_OS_WIN
526constexpr bool kNeedsToSplitRangeByReservations = true;
527#else
528constexpr bool kNeedsToSplitRangeByReservations = false;
529#endif
530
531base::SmallVector<base::AddressRegion, 1> SplitRangeByReservationsIfNeeded(
532 base::AddressRegion range,
533 const std::vector<VirtualMemory>& owned_code_space) {
534 if (!kNeedsToSplitRangeByReservations) return {range};
535
536 base::SmallVector<base::AddressRegion, 1> split_ranges;
537 size_t missing_begin = range.begin();
538 size_t missing_end = range.end();
539 for (auto& vmem : base::Reversed(owned_code_space)) {
540 Address overlap_begin = std::max(missing_begin, vmem.address());
541 Address overlap_end = std::min(missing_end, vmem.end());
542 if (overlap_begin >= overlap_end) continue;
543 split_ranges.emplace_back(overlap_begin, overlap_end - overlap_begin);
544 // Opportunistically reduce the missing range. This might terminate the loop
545 // early.
546 if (missing_begin == overlap_begin) missing_begin = overlap_end;
547 if (missing_end == overlap_end) missing_end = overlap_begin;
548 if (missing_begin >= missing_end) break;
549 }
550#ifdef ENABLE_SLOW_DCHECKS
551 // The returned vector should cover the full range.
552 size_t total_split_size = 0;
553 for (auto split : split_ranges) total_split_size += split.size();
554 DCHECK_EQ(range.size(), total_split_size)((void) 0);
555#endif
556 return split_ranges;
557}

565// Returns an overapproximation of the code size overhead per new code space
566// created by the jump tables.
567size_t OverheadPerCodeSpace(uint32_t num_declared_functions) {
568 // Overhead for the jump table.
569 size_t overhead = RoundUp<kCodeAlignment>(
570 JumpTableAssembler::SizeForNumberOfSlots(num_declared_functions));
571
572#if defined(V8_OS_WIN64)
573 // On Win64, we need to reserve some pages at the beginning of an executable
574 // space. See {AddCodeSpace}.
575 overhead += Heap::GetCodeRangeReservedAreaSize();
576#endif // V8_OS_WIN64
577
578 // Overhead for the far jump table.
579 overhead +=
580 RoundUp<kCodeAlignment>(JumpTableAssembler::SizeForNumberOfFarJumpSlots(
581 WasmCode::kRuntimeStubCount,
582 NumWasmFunctionsInFarJumpTable(num_declared_functions)));
583
584 return overhead;
585}
586
587size_t ReservationSize(size_t code_size_estimate, int num_declared_functions,
588 size_t total_reserved) {
589 size_t overhead = OverheadPerCodeSpace(num_declared_functions);
590
591 // Reserve a power of two at least as big as any of
592 // a) needed size + overhead (this is the minimum needed)
593 // b) 2 * overhead (to not waste too much space by overhead)
594 // c) 1/4 of current total reservation size (to grow exponentially)
595 size_t reserve_size = base::bits::RoundUpToPowerOfTwo(
596 std::max(std::max(RoundUp<kCodeAlignment>(code_size_estimate) + overhead,
597 2 * overhead),
598 total_reserved / 4));
599
600 // Limit by the maximum supported code space size.
601 return std::min(kMaxWasmCodeSpaceSize, reserve_size);
602}

612Vector<byte> WasmCodeAllocator::AllocateForCodeInRegion(
613 NativeModule* native_module, size_t size, base::AddressRegion region,
614 const WasmCodeAllocator::OptionalLock& optional_lock) {
615 OptionalLock new_lock;
616 if (!optional_lock.is_locked()) new_lock.Lock(this);
617 const auto& locked_lock =
618 optional_lock.is_locked() ? optional_lock : new_lock;
619 DCHECK(locked_lock.is_locked())((void) 0);
620 DCHECK_EQ(code_manager_, native_module->engine()->code_manager())((void) 0);
621 DCHECK_LT(0, size)((void) 0);
622 v8::PageAllocator* page_allocator = GetPlatformPageAllocator();
623 size = RoundUp<kCodeAlignment>(size);
624 base::AddressRegion code_space =
625 free_code_space_.AllocateInRegion(size, region);
626 if (V8_UNLIKELY(code_space.is_empty())(__builtin_expect(!!(code_space.is_empty()), 0))) {
627 // Only allocations without a specific region are allowed to fail. Otherwise
628 // the region must have been allocated big enough to hold all initial
629 // allocations (jump tables etc).
630 CHECK_EQ(kUnrestrictedRegion, region)do { bool _cmp = ::v8::base::CmpEQImpl< typename ::v8::base
::pass_value_or_ref<decltype(kUnrestrictedRegion)>::type
, typename ::v8::base::pass_value_or_ref<decltype(region)>
::type>((kUnrestrictedRegion), (region)); do { if ((__builtin_expect
(!!(!(_cmp)), 0))) { V8_Fatal("Check failed: %s.", "kUnrestrictedRegion"
" " "==" " " "region"); } } while (false); } while (false);
631
632 Address hint = owned_code_space_.empty() ? kNullAddress
633 : owned_code_space_.back().end();
634
635 size_t total_reserved = 0;
636 for (auto& vmem : owned_code_space_) total_reserved += vmem.size();
637 size_t reserve_size = ReservationSize(
638 size, native_module->module()->num_declared_functions, total_reserved);
639 VirtualMemory new_mem =
640 code_manager_->TryAllocate(reserve_size, reinterpret_cast<void*>(hint));
641 if (!new_mem.IsReserved()) {
642 V8::FatalProcessOutOfMemory(nullptr, "wasm code reservation");
643 UNREACHABLE()V8_Fatal("unreachable code");
644 }
645
646 base::AddressRegion new_region = new_mem.region();
647 code_manager_->AssignRange(new_region, native_module);
648 free_code_space_.Merge(new_region);
649 owned_code_space_.emplace_back(std::move(new_mem));
650 native_module->AddCodeSpace(new_region, locked_lock);
651
652 code_space = free_code_space_.Allocate(size);
653 DCHECK(!code_space.is_empty())((void) 0);
654 async_counters_->wasm_module_num_code_spaces()->AddSample(
655 static_cast<int>(owned_code_space_.size()));
656 }
657 const Address commit_page_size = page_allocator->CommitPageSize();
658 Address commit_start = RoundUp(code_space.begin(), commit_page_size);
659 Address commit_end = RoundUp(code_space.end(), commit_page_size);
660 // {commit_start} will be either code_space.start or the start of the next
661 // page. {commit_end} will be the start of the page after the one in which
662 // the allocation ends.
663 // We start from an aligned start, and we know we allocated vmem in
664 // page multiples.
665 // We just need to commit what's not committed. The page in which we
666 // start is already committed (or we start at the beginning of a page).
667 // The end needs to be committed all through the end of the page.
668 if (commit_start < commit_end) {
669 committed_code_space_.fetch_add(commit_end - commit_start);
670 // Committed code cannot grow bigger than maximum code space size.
671 DCHECK_LE(committed_code_space_.load(), kMaxWasmCodeMemory)((void) 0);
672 for (base::AddressRegion split_range : SplitRangeByReservationsIfNeeded(
673 {commit_start, commit_end - commit_start}, owned_code_space_)) {
674 code_manager_->Commit(split_range);
675 }
676 }
677 DCHECK(IsAligned(code_space.begin(), kCodeAlignment))((void) 0);
678 allocated_code_space_.Merge(code_space);
679 generated_code_size_.fetch_add(code_space.size(), std::memory_order_relaxed);
680
681 TRACE_HEAP("Code alloc for %p: 0x%" PRIxPTR"l" "x" ",+%zu\n", this,
682 code_space.begin(), size);
683 return {reinterpret_cast<byte*>(code_space.begin()), code_space.size()};
684}

686bool WasmCodeAllocator::SetExecutable(bool executable) {
687 base::MutexGuard lock(&mutex_);
688 if (is_executable_ == executable) return true;
689 TRACE_HEAP("Setting module %p as executable: %d.\n", this, executable);
690
691 v8::PageAllocator* page_allocator = GetPlatformPageAllocator();
692
693 if (FLAG_wasm_write_protect_code_memory) {
694 PageAllocator::Permission permission =
695 executable ? PageAllocator::kReadExecute : PageAllocator::kReadWrite;
696#if V8_OS_WIN
697 // On windows, we need to switch permissions per separate virtual memory
698 // reservation.
699 // For now, in that case, we commit at reserved memory granularity.
700 // Technically, that may be a waste, because we may reserve more than we
701 // use. On 32-bit though, the scarce resource is the address space -
702 // committed or not.
703 for (auto& vmem : owned_code_space_) {
704 if (!SetPermissions(page_allocator, vmem.address(), vmem.size(),
705 permission)) {
706 return false;
707 }
708 TRACE_HEAP("Set %p:%p to executable:%d\n", vmem.address(), vmem.end(),
709 executable);
710 }
711#else // V8_OS_WIN
712 size_t commit_page_size = page_allocator->CommitPageSize();
713 for (auto& region : allocated_code_space_.regions()) {
714 // allocated_code_space_ is fine-grained, so we need to
715 // page-align it.
716 size_t region_size = RoundUp(region.size(), commit_page_size);
717 if (!SetPermissions(page_allocator, region.begin(), region_size,
718 permission)) {
719 return false;
720 }
721 TRACE_HEAP("Set 0x%" PRIxPTR"l" "x" ":0x%" PRIxPTR"l" "x" " to executable:%d\n",
722 region.begin(), region.end(), executable);
723 }
724#endif // V8_OS_WIN
725 }
726 is_executable_ = executable;
727 return true;
728}

730void WasmCodeAllocator::FreeCode(Vector<WasmCode* const> codes) {
731 // Zap code area and collect freed code regions.
732 DisjointAllocationPool freed_regions;
733 size_t code_size = 0;
734 for (WasmCode* code : codes) {
735 ZapCode(code->instruction_start(), code->instructions().size());
736 FlushInstructionCache(code->instruction_start(),
737 code->instructions().size());
738 code_size += code->instructions().size();
739 freed_regions.Merge(base::AddressRegion{code->instruction_start(),
740 code->instructions().size()});
741 }
742 freed_code_size_.fetch_add(code_size);
743
744 // Merge {freed_regions} into {freed_code_space_} and discard full pages.
745 base::MutexGuard guard(&mutex_);
746 PageAllocator* allocator = GetPlatformPageAllocator();
747 size_t commit_page_size = allocator->CommitPageSize();
748 for (auto region : freed_regions.regions()) {
749 auto merged_region = freed_code_space_.Merge(region);
750 Address discard_start =
751 std::max(RoundUp(merged_region.begin(), commit_page_size),
752 RoundDown(region.begin(), commit_page_size));
753 Address discard_end =
754 std::min(RoundDown(merged_region.end(), commit_page_size),
755 RoundUp(region.end(), commit_page_size));
756 if (discard_start >= discard_end) continue;
757 size_t discard_size = discard_end - discard_start;
758 size_t old_committed = committed_code_space_.fetch_sub(discard_size);
759 DCHECK_GE(old_committed, discard_size)((void) 0);
760 USE(old_committed)do { ::v8::base::Use unused_tmp_array_for_use_macro[]{old_committed
}; (void)unused_tmp_array_for_use_macro; } while (false);
761 for (base::AddressRegion split_range : SplitRangeByReservationsIfNeeded(
762 {discard_start, discard_size}, owned_code_space_)) {
763 code_manager_->Decommit(split_range);
764 }
765 }

865 source_pos_table->copy_out(0, source_pos.start(),

906 } else {
907 it.rinfo()->apply(delta);
908 }
909 }
910
911 // Flush the i-cache after relocation.
912 FlushInstructionCache(dst_code_bytes.begin(), dst_code_bytes.size());
913
914 std::unique_ptr<WasmCode> new_code{
915 new WasmCode{this, // native_module
916 kAnonymousFuncIndex, // index
917 dst_code_bytes, // instructions
918 stack_slots, // stack_slots
919 0, // tagged_parameter_slots
920 safepoint_table_offset, // safepoint_table_offset
921 handler_table_offset, // handler_table_offset
922 constant_pool_offset, // constant_pool_offset
923 code_comments_offset, // code_comments_offset
924 instructions.length(), // unpadded_binary_size
925 {}, // protected_instructions
926 reloc_info.as_vector(), // reloc_info
927 source_pos.as_vector(), // source positions
928 WasmCode::kFunction, // kind
929 ExecutionTier::kNone, // tier
930 kNoDebugging}}; // for_debugging
931 new_code->MaybePrint();
932 new_code->Validate();
933
934 return PublishCode(std::move(new_code));
935}

1412NativeModule::JumpTablesRef NativeModule::FindJumpTablesForRegion(
1413 base::AddressRegion code_region) const {
1414 auto jump_table_usable = [code_region](const WasmCode* jump_table) {
1415 Address table_start = jump_table->instruction_start();
1416 Address table_end = table_start + jump_table->instructions().size();
1417 // Compute the maximum distance from anywhere in the code region to anywhere
1418 // in the jump table, avoiding any underflow.
1419 size_t max_distance = std::max(
1420 code_region.end() > table_start ? code_region.end() - table_start : 0,
1421 table_end > code_region.begin() ? table_end - code_region.begin() : 0);
1422 return max_distance < kMaxWasmCodeSpaceSize;
1423 };
1424
1425 // Fast path: Try to use {main_jump_table_} and {main_far_jump_table_}.
1426 // Access to these fields is possible without locking, since these fields are
1427 // initialized on construction of the {NativeModule}.
1428 if (main_far_jump_table_ && jump_table_usable(main_far_jump_table_) &&
1429 (main_jump_table_ == nullptr || jump_table_usable(main_jump_table_))) {
1430 return {
1431 main_jump_table_ ? main_jump_table_->instruction_start() : kNullAddress,
1432 main_far_jump_table_->instruction_start()};
1433 }
1434
1435 // Otherwise, take the mutex and look for another suitable jump table.
1436 base::MutexGuard guard(&allocation_mutex_);
1437 for (auto& code_space_data : code_space_data_) {
1438 DCHECK_IMPLIES(code_space_data.jump_table, code_space_data.far_jump_table)((void) 0);
1439 if (!code_space_data.far_jump_table) continue;
1440 // Only return these jump tables if they are reachable from the whole
1441 // {code_region}.
1442 if (kNeedsFarJumpsBetweenCodeSpaces &&
1443 (!jump_table_usable(code_space_data.far_jump_table) ||
1444 (code_space_data.jump_table &&
1445 !jump_table_usable(code_space_data.jump_table)))) {
1446 continue;
1447 }
1448 return {code_space_data.jump_table
1449 ? code_space_data.jump_table->instruction_start()
1450 : kNullAddress,
1451 code_space_data.far_jump_table->instruction_start()};
1452 }
1453 return {};
1454}

1533 if (V8_UNLIKELY(FLAG_perf_prof)(__builtin_expect(!!(FLAG_perf_prof), 0))) return;

1537 // {total_committed_code_space_}.

1730std::shared_ptr<NativeModule> WasmCodeManager::NewNativeModule(
1731 WasmEngine* engine, Isolate* isolate, const WasmFeatures& enabled,
1732 size_t code_size_estimate, std::shared_ptr<const WasmModule> module) {
1733 DCHECK_EQ(this, isolate->wasm_engine()->code_manager())((void) 0);
1734 if (total_committed_code_space_.load() >
1735 critical_committed_code_space_.load()) {
1736 (reinterpret_cast<v8::Isolate*>(isolate))
1737 ->MemoryPressureNotification(MemoryPressureLevel::kCritical);
1738 size_t committed = total_committed_code_space_.load();
1739 DCHECK_GE(max_committed_code_space_, committed)((void) 0);
1740 critical_committed_code_space_.store(
1741 committed + (max_committed_code_space_ - committed) / 2);
1742 }
1743
1744 // If we cannot add code space later, reserve enough address space up front.
1745 size_t code_vmem_size =
1746 ReservationSize(code_size_estimate, module->num_declared_functions, 0);
1747
1748 // The '--wasm-max-code-space-reservation' testing flag can be used to reduce
1749 // the maximum size of the initial code space reservation (in MB).
1750 if (FLAG_wasm_max_initial_code_space_reservation > 0) {
1751 size_t flag_max_bytes =
1752 static_cast<size_t>(FLAG_wasm_max_initial_code_space_reservation) * MB;
1753 if (flag_max_bytes < code_vmem_size) code_vmem_size = flag_max_bytes;
1754 }
1755
1756 // Try up to two times; getting rid of dead JSArrayBuffer allocations might
1757 // require two GCs because the first GC maybe incremental and may have
1758 // floating garbage.
1759 static constexpr int kAllocationRetries = 2;
1760 VirtualMemory code_space;
1761 for (int retries = 0;; ++retries) {
1762 code_space = TryAllocate(code_vmem_size);
1763 if (code_space.IsReserved()) break;
1764 if (retries == kAllocationRetries) {
1765 V8::FatalProcessOutOfMemory(isolate, "NewNativeModule");
1766 UNREACHABLE()V8_Fatal("unreachable code");
1767 }
1768 // Run one GC, then try the allocation again.
1769 isolate->heap()->MemoryPressureNotification(MemoryPressureLevel::kCritical,
1770 true);
1771 }
1772
1773 Address start = code_space.address();
1774 size_t size = code_space.size();
1775 Address end = code_space.end();
1776 std::shared_ptr<NativeModule> ret;
1777 new NativeModule(engine, enabled, std::move(code_space), std::move(module),
1778 isolate->async_counters(), &ret);
1779 // The constructor initialized the shared_ptr.
1780 DCHECK_NOT_NULL(ret)((void) 0);
1781 TRACE_HEAP("New NativeModule %p: Mem: %" PRIuPTR"l" "u" ",+%zu\n", ret.get(), start,
1782 size);
1783
1784 base::MutexGuard lock(&native_modules_mutex_);
1785 lookup_map_.insert(std::make_pair(start, std::make_pair(end, ret.get())));
1786 return ret;
1787}

1789void NativeModule::SampleCodeSize(
1790 Counters* counters, NativeModule::CodeSamplingTime sampling_time) const {
1791 size_t code_size = sampling_time == kSampling
1792 ? code_allocator_.committed_code_space()
1793 : code_allocator_.generated_code_size();
1794 int code_size_mb = static_cast<int>(code_size / MB);
1795 Histogram* histogram = nullptr;
1796 switch (sampling_time) {
1797 case kAfterBaseline:
1798 histogram = counters->wasm_module_code_size_mb_after_baseline();
1799 break;
1800 case kAfterTopTier:
1801 histogram = counters->wasm_module_code_size_mb_after_top_tier();
1802 break;
1803 case kSampling: {
1804 histogram = counters->wasm_module_code_size_mb();
1805 // If this is a wasm module of >= 2MB, also sample the freed code size,
1806 // absolute and relative. Code GC does not happen on asm.js modules, and
1807 // small modules will never trigger GC anyway.
1808 size_t generated_size = code_allocator_.generated_code_size();
1809 if (generated_size >= 2 * MB && module()->origin == kWasmOrigin) {
1810 size_t freed_size = code_allocator_.freed_code_size();
1811 DCHECK_LE(freed_size, generated_size)((void) 0);
1812 int freed_percent = static_cast<int>(100 * freed_size / generated_size);
1813 counters->wasm_module_freed_code_size_percent()->AddSample(
1814 freed_percent);
1815 }
1816 break;
1817 }
1818 }
1819 histogram->AddSample(code_size_mb);
1820}

1828std::vector<std::unique_ptr<WasmCode>> NativeModule::AddCompiledCode(
1829 Vector<WasmCompilationResult> results) {
1830 DCHECK(!results.empty())((void) 0);
1831 // First, allocate code space for all the results.
1832 size_t total_code_space = 0;
1833 for (auto& result : results) {
1834 DCHECK(result.succeeded())((void) 0);
1835 total_code_space += RoundUp<kCodeAlignment>(result.code_desc.instr_size);
1836 }
1837 Vector<byte> code_space =
1838 code_allocator_.AllocateForCode(this, total_code_space);
1839 // Lookup the jump tables to use once, then use for all code objects.
1840 auto jump_tables = FindJumpTablesForRegion(base::AddressRegionOf(code_space));
1841
1842 std::vector<std::unique_ptr<WasmCode>> generated_code;
1843 generated_code.reserve(results.size());
1844
1845 // Now copy the generated code into the code space and relocate it.
1846 for (auto& result : results) {
1847 DCHECK_EQ(result.code_desc.buffer, result.instr_buffer.get())((void) 0);
1848 size_t code_size = RoundUp<kCodeAlignment>(result.code_desc.instr_size);
1849 Vector<byte> this_code_space = code_space.SubVector(0, code_size);
1850 code_space += code_size;
1851 generated_code.emplace_back(AddCodeWithCodeSpace(
1852 result.func_index, result.code_desc, result.frame_slot_count,
1853 result.tagged_parameter_slots,
1854 result.protected_instructions_data.as_vector(),
1855 result.source_positions.as_vector(), GetCodeKind(result),
1856 result.result_tier, result.for_debugging, this_code_space,
1857 jump_tables));
1858 }
1859 DCHECK_EQ(0, code_space.size())((void) 0);
1860
1861 return generated_code;
1862}

611 /** @brief Replace the stored pointer.
612 *
613 * @param __p The new pointer to store.
614 *
615 * The deleter will be invoked if a pointer is already owned.
616 */
617 template <typename _Up,
618 typename = _Require<
619 __or_<is_same<_Up, pointer>,
620 __and_<is_same<pointer, element_type*>,
621 is_pointer<_Up>,
622 is_convertible<
623 typename remove_pointer<_Up>::type(*)[],
624 element_type(*)[]
625 >
626 >
627 >
628 >>
629 void
630 reset(_Up __p) noexcept
631 {
632 pointer __ptr = __p;
633 using std::swap;
634 swap(_M_t._M_ptr(), __ptr);
635 if (__ptr != nullptr)
636 get_deleter()(__ptr);
637 }

642 }
