67static int i2r_pci(X509V3_EXT_METHOD *method, PROXY_CERT_INFO_EXTENSION *pci,
68 BIO *out, int indent)
69{
70 BIO_printf(out, "%*sPath Length Constraint: ", indent, "");

74 BIO_printf(out, "infinite");
75 BIO_puts(out, "\n");
