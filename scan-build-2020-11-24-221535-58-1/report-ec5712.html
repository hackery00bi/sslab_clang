177static void LookupForRead(LookupIterator* it, bool is_has_property) {
178 for (; it->IsFound(); it->Next()) {
179 switch (it->state()) {
180 case LookupIterator::NOT_FOUND:
181 case LookupIterator::TRANSITION:
182 UNREACHABLE()V8_Fatal("unreachable code");
183 case LookupIterator::JSPROXY:
184 return;
185 case LookupIterator::INTERCEPTOR: {
186 // If there is a getter, return; otherwise loop to perform the lookup.
187 Handle<JSObject> holder = it->GetHolder<JSObject>();
188 if (!holder->GetNamedInterceptor().getter().IsUndefined(
189 it->isolate())) {
190 return;
191 }
192 if (is_has_property &&
193 !holder->GetNamedInterceptor().query().IsUndefined(it->isolate())) {
194 return;
195 }
196 break;
197 }
198 case LookupIterator::ACCESS_CHECK:
199 // ICs know how to perform access checks on global proxies.
200 if (it->GetHolder<JSObject>()->IsJSGlobalProxy() && it->HasAccess()) {
201 break;
202 }
203 return;
204 case LookupIterator::ACCESSOR:
205 case LookupIterator::INTEGER_INDEXED_EXOTIC:
206 case LookupIterator::DATA:
207 return;
208 }
209 }

213 if (!RecomputeHandlerForName(name)) return false;

216 // monomorphic.
217 if (IsGlobalIC()) return true;

222 // *unless* we're moving from a deprecated map to its replacement, or

232 return IsMoreGeneralElementsKindTransition(old_map->elements_kind(),
233 receiver_map()->elements_kind());
234 }
235
236 return true;
237}

560bool IC::UpdatePolymorphicIC(Handle<Name> name,
561 const MaybeObjectHandle& handler) {
562 DCHECK(IsHandler(*handler))((void) 0);
563 if (is_keyed() && state() != RECOMPUTE_HANDLER) {
564 if (nexus()->GetName() != *name) return false;
565 }
566 Handle<Map> map = receiver_map();
567
568 std::vector<MapAndHandler> maps_and_handlers;
569 nexus()->ExtractMapsAndHandlers(&maps_and_handlers);
570 int number_of_maps = static_cast<int>(maps_and_handlers.size());
571 int deprecated_maps = 0;
572 int handler_to_overwrite = -1;
573
574 for (int i = 0; i < number_of_maps; i++) {
575 Handle<Map> current_map = maps_and_handlers.at(i).first;
576 MaybeObjectHandle current_handler = maps_and_handlers.at(i).second;
577 if (current_map->is_deprecated()) {
578 // Filter out deprecated maps to ensure their instances get migrated.
579 ++deprecated_maps;
580 } else if (map.is_identical_to(current_map)) {
581 // If both map and handler stayed the same (and the name is also the
582 // same as checked above, for keyed accesses), we're not progressing
583 // in the lattice and need to go MEGAMORPHIC instead. There's one
584 // exception to this rule, which is when we're in RECOMPUTE_HANDLER
585 // state, there we allow to migrate to a new handler.
586 if (handler.is_identical_to(current_handler) &&
587 state() != RECOMPUTE_HANDLER) {
588 return false;
589 }
590 // If the receiver type is already in the polymorphic IC, this indicates
591 // there was a prototoype chain failure. In that case, just overwrite the
592 // handler.
593 handler_to_overwrite = i;
594 } else if (handler_to_overwrite == -1 &&
595 IsTransitionOfMonomorphicTarget(*current_map, *map)) {
596 handler_to_overwrite = i;
597 }
598 }
599
600 int number_of_valid_maps =
601 number_of_maps - deprecated_maps - (handler_to_overwrite != -1);
602
603 if (number_of_valid_maps >= FLAG_max_valid_polymorphic_map_count)
604 return false;
605 if (number_of_maps == 0 && state() != MONOMORPHIC && state() != POLYMORPHIC) {
606 return false;
607 }
608
609 number_of_valid_maps++;
610 if (number_of_valid_maps == 1) {
611 ConfigureVectorState(name, receiver_map(), handler);
612 } else {
613 if (is_keyed() && nexus()->GetName() != *name) return false;
614 if (handler_to_overwrite >= 0) {
615 maps_and_handlers[handler_to_overwrite].second = handler;
616 if (!map.is_identical_to(
617 maps_and_handlers.at(handler_to_overwrite).first)) {
618 maps_and_handlers[handler_to_overwrite].first = map;
619 }
620 } else {
621 maps_and_handlers.push_back(MapAndHandler(map, handler));
622 }
623
624 ConfigureVectorState(name, maps_and_handlers);
625 }
626
627 return true;
628}

1024void KeyedLoadIC::UpdateLoadElement(Handle<HeapObject> receiver,
1025 KeyedAccessLoadMode load_mode) {
1026 Handle<Map> receiver_map(receiver->map(), isolate());
1027 DCHECK(receiver_map->instance_type() !=((void) 0)
1028 JS_PRIMITIVE_WRAPPER_TYPE)((void) 0); // Checked by caller.
1029 MapHandles target_receiver_maps;
1030 TargetMaps(&target_receiver_maps);
1031
1032 if (target_receiver_maps.empty()) {
1033 Handle<Object> handler = LoadElementHandler(receiver_map, load_mode);
1034 return ConfigureVectorState(Handle<Name>(), receiver_map, handler);
1035 }
1036
1037 for (Handle<Map> map : target_receiver_maps) {
1038 if (map.is_null()) continue;
1039 if (map->instance_type() == JS_PRIMITIVE_WRAPPER_TYPE) {
1040 set_slow_stub_reason("JSPrimitiveWrapper");
1041 return;
1042 }
1043 if (map->instance_type() == JS_PROXY_TYPE) {
1044 set_slow_stub_reason("JSProxy");
1045 return;
1046 }
1047 }
1048
1049 // The first time a receiver is seen that is a transitioned version of the
1050 // previous monomorphic receiver type, assume the new ElementsKind is the
1051 // monomorphic type. This benefits global arrays that only transition
1052 // once, and all call sites accessing them are faster if they remain
1053 // monomorphic. If this optimistic assumption is not true, the IC will
1054 // miss again and it will become polymorphic and support both the
1055 // untransitioned and transitioned maps.
1056 if (state() == MONOMORPHIC && !receiver->IsString() &&
1057 !receiver->IsJSProxy() &&
1058 IsMoreGeneralElementsKindTransition(
1059 target_receiver_maps.at(0)->elements_kind(),
1060 Handle<JSObject>::cast(receiver)->GetElementsKind())) {
1061 Handle<Object> handler = LoadElementHandler(receiver_map, load_mode);
1062 return ConfigureVectorState(Handle<Name>(), receiver_map, handler);
1063 }
1064
1065 DCHECK(state() != GENERIC)((void) 0);
1066
1067 // Determine the list of receiver maps that this call site has seen,
1068 // adding the map that was just encountered.
1069 if (!AddOneReceiverMapIfMissing(&target_receiver_maps, receiver_map)) {
1070 // If the {receiver_map} previously had a handler that didn't handle
1071 // out-of-bounds access, but can generally handle it, we can just go
1072 // on and update the handler appropriately below.
1073 if (load_mode != LOAD_IGNORE_OUT_OF_BOUNDS ||
1074 !CanChangeToAllowOutOfBounds(receiver_map)) {
1075 // If the miss wasn't due to an unseen map, a polymorphic stub
1076 // won't help, use the generic stub.
1077 set_slow_stub_reason("same map added twice");
1078 return;
1079 }
1080 }
1081
1082 // If the maximum number of receiver maps has been exceeded, use the generic
1083 // version of the IC.
1084 if (static_cast<int>(target_receiver_maps.size()) >
1085 FLAG_max_valid_polymorphic_map_count) {
1086 set_slow_stub_reason("max polymorph exceeded");
1087 return;
1088 }
1089
1090 MaybeObjectHandles handlers;
1091 handlers.reserve(target_receiver_maps.size());
1092 LoadElementPolymorphicHandlers(&target_receiver_maps, &handlers, load_mode);
1093 DCHECK_LE(1, target_receiver_maps.size())((void) 0);
1094 if (target_receiver_maps.size() == 1) {
1095 ConfigureVectorState(Handle<Name>(), target_receiver_maps[0], handlers[0]);
1096 } else {
1097 ConfigureVectorState(Handle<Name>(), target_receiver_maps, &handlers);
1098 }
1099}

1370bool StoreIC::LookupForWrite(LookupIterator* it, Handle<Object> value,
1371 StoreOrigin store_origin) {
1372 // Disable ICs for non-JSObjects for now.
1373 Handle<Object> object = it->GetReceiver();
1374 if (object->IsJSProxy()) return true;
1375 if (!object->IsJSObject()) return false;

1377 DCHECK(!receiver->map().is_deprecated())((void) 0);
1378
1379 if (it->state() != LookupIterator::TRANSITION) {
1380 for (; it->IsFound(); it->Next()) {

1382 case LookupIterator::NOT_FOUND:
1383 case LookupIterator::TRANSITION:
1384 UNREACHABLE()V8_Fatal("unreachable code");

1402 case LookupIterator::INTEGER_INDEXED_EXOTIC:

1410 // so reload it.

1824void KeyedStoreIC::UpdateStoreElement(Handle<Map> receiver_map,
1825 KeyedAccessStoreMode store_mode,
1826 Handle<Map> new_receiver_map) {
1827 std::vector<MapAndHandler> target_maps_and_handlers;
1828 nexus()->ExtractMapsAndHandlers(&target_maps_and_handlers, true);
1829 if (target_maps_and_handlers.empty()) {
1830 Handle<Map> monomorphic_map = receiver_map;
1831 // If we transitioned to a map that is a more general map than incoming
1832 // then use the new map.
1833 if (IsTransitionOfMonomorphicTarget(*receiver_map, *new_receiver_map)) {
1834 monomorphic_map = new_receiver_map;
1835 }
1836 Handle<Object> handler = StoreElementHandler(monomorphic_map, store_mode);
1837 return ConfigureVectorState(Handle<Name>(), monomorphic_map, handler);
1838 }
1839
1840 for (const MapAndHandler& map_and_handler : target_maps_and_handlers) {
1841 Handle<Map> map = map_and_handler.first;
1842 if (!map.is_null() && map->instance_type() == JS_PRIMITIVE_WRAPPER_TYPE) {
1843 DCHECK(!IsStoreInArrayLiteralICKind(kind()))((void) 0);
1844 set_slow_stub_reason("JSPrimitiveWrapper");
1845 return;
1846 }
1847 }
1848
1849 // There are several special cases where an IC that is MONOMORPHIC can still
1850 // transition to a different IC that handles a superset of the original IC.
1851 // Handle those here if the receiver map hasn't changed or it has transitioned
1852 // to a more general kind.
1853 KeyedAccessStoreMode old_store_mode = GetKeyedAccessStoreMode();
1854 Handle<Map> previous_receiver_map = target_maps_and_handlers.at(0).first;
1855 if (state() == MONOMORPHIC) {
1856 Handle<Map> transitioned_receiver_map = new_receiver_map;
1857 if (IsTransitionOfMonomorphicTarget(*previous_receiver_map,
1858 *transitioned_receiver_map)) {
1859 // If the "old" and "new" maps are in the same elements map family, or
1860 // if they at least come from the same origin for a transitioning store,
1861 // stay MONOMORPHIC and use the map for the most generic ElementsKind.
1862 Handle<Object> handler =
1863 StoreElementHandler(transitioned_receiver_map, store_mode);
1864 ConfigureVectorState(Handle<Name>(), transitioned_receiver_map, handler);
1865 return;
1866 }
1867 // If there is no transition and if we have seen the same map earlier and
1868 // there is only a change in the store_mode we can still stay monomorphic.
1869 if (receiver_map.is_identical_to(previous_receiver_map) &&
1870 new_receiver_map.is_identical_to(receiver_map) &&
1871 old_store_mode == STANDARD_STORE && store_mode != STANDARD_STORE) {
1872 if (receiver_map->IsJSArrayMap() &&
1873 JSArray::MayHaveReadOnlyLength(*receiver_map)) {
1874 set_slow_stub_reason(
1875 "can't generalize store mode (potentially read-only length)");
1876 return;
1877 }
1878 // A "normal" IC that handles stores can switch to a version that can
1879 // grow at the end of the array, handle OOB accesses or copy COW arrays
1880 // and still stay MONOMORPHIC.
1881 Handle<Object> handler = StoreElementHandler(receiver_map, store_mode);
1882 return ConfigureVectorState(Handle<Name>(), receiver_map, handler);
1883 }
1884 }
1885
1886 DCHECK(state() != GENERIC)((void) 0);
1887
1888 bool map_added =
1889 AddOneReceiverMapIfMissing(&target_maps_and_handlers, receiver_map);
1890
1891 if (IsTransitionOfMonomorphicTarget(*receiver_map, *new_receiver_map)) {
1892 map_added |=
1893 AddOneReceiverMapIfMissing(&target_maps_and_handlers, new_receiver_map);
1894 }
1895
1896 if (!map_added) {
1897 // If the miss wasn't due to an unseen map, a polymorphic stub
1898 // won't help, use the megamorphic stub which can handle everything.
1899 set_slow_stub_reason("same map added twice");
1900 return;
1901 }
1902
1903 // If the maximum number of receiver maps has been exceeded, use the
1904 // megamorphic version of the IC.
1905 if (static_cast<int>(target_maps_and_handlers.size()) >
1906 FLAG_max_valid_polymorphic_map_count) {
1907 return;
1908 }
1909
1910 // Make sure all polymorphic handlers have the same store mode, otherwise the
1911 // megamorphic stub must be used.
1912 if (old_store_mode != STANDARD_STORE) {
1913 if (store_mode == STANDARD_STORE) {
1914 store_mode = old_store_mode;
1915 } else if (store_mode != old_store_mode) {
1916 set_slow_stub_reason("store mode mismatch");
1917 return;
1918 }
1919 }
1920
1921 // If the store mode isn't the standard mode, make sure that all polymorphic
1922 // receivers are either external arrays, or all "normal" arrays with writable
1923 // length. Otherwise, use the megamorphic stub.
1924 if (store_mode != STANDARD_STORE) {

1926 for (MapAndHandler map_and_handler : target_maps_and_handlers) {
1927 Handle<Map> map = map_and_handler.first;
1928 if (map->IsJSArrayMap() && JSArray::MayHaveReadOnlyLength(*map)) {
1929 set_slow_stub_reason(
1930 "unsupported combination of arrays (potentially read-only length)");
1931 return;
1932
1933 } else if (map->has_typed_array_elements()) {
1934 DCHECK(!IsStoreInArrayLiteralICKind(kind()))((void) 0);
1935 external_arrays++;
1936 }
1937 }
1938 if (external_arrays != 0 &&
1939 external_arrays != target_maps_and_handlers.size()) {
1940 DCHECK(!IsStoreInArrayLiteralICKind(kind()))((void) 0);
1941 set_slow_stub_reason(
1942 "unsupported combination of external and normal arrays");
1943 return;
1944 }
1945 }
1946
1947 StoreElementPolymorphicHandlers(&target_maps_and_handlers, store_mode);
1948 if (target_maps_and_handlers.size() == 0) {
1949 Handle<Object> handler = StoreElementHandler(receiver_map, store_mode);
1950 ConfigureVectorState(Handle<Name>(), receiver_map, handler);
1951 } else if (target_maps_and_handlers.size() == 1) {
1952 ConfigureVectorState(Handle<Name>(), target_maps_and_handlers[0].first,

1954 } else {

1956 }
