336// Find the arguments of the JavaScript function invocation that called
337// into C++ code. Collect these in a newly allocated array of handles.
338std::unique_ptr<Handle<Object>[]> GetCallerArguments(Isolate* isolate,
339 int* total_argc) {
340 // Find frame containing arguments passed to the caller.
341 JavaScriptFrameIterator it(isolate);
342 JavaScriptFrame* frame = it.frame();
343 std::vector<SharedFunctionInfo> functions;
344 frame->GetFunctions(&functions);
345 if (functions.size() > 1) {
346 int inlined_jsframe_index = static_cast<int>(functions.size()) - 1;
347 TranslatedState translated_values(frame);
348 translated_values.Prepare(frame->fp());
349
350 int argument_count = 0;
351 TranslatedFrame* translated_frame =
352 translated_values.GetArgumentsInfoFromJSFrameIndex(
353 inlined_jsframe_index, &argument_count);
354 TranslatedFrame::iterator iter = translated_frame->begin();
355
356 // Skip the function.
357 iter++;
358
359 // Skip the receiver.
360 iter++;
361 argument_count--;
362
363 *total_argc = argument_count;
364 std::unique_ptr<Handle<Object>[]> param_data(
365 NewArray<Handle<Object>>(*total_argc));
366 bool should_deoptimize = false;
367 for (int i = 0; i < argument_count; i++) {
368 // If we materialize any object, we should deoptimize the frame because we
369 // might alias an object that was eliminated by escape analysis.
370 should_deoptimize = should_deoptimize || iter->IsMaterializedObject();
371 Handle<Object> value = iter->GetValue();
372 param_data[i] = value;
373 iter++;
374 }
375
376 if (should_deoptimize) {
377 translated_values.StoreMaterializedValuesAndDeopt(frame);
378 }
379
380 return param_data;
381 } else {
382 if (it.frame()->has_adapted_arguments()) {
383 it.AdvanceOneFrame();
384 DCHECK(it.frame()->is_arguments_adaptor())((void) 0);
385 }
386 frame = it.frame();
387 int args_count = frame->ComputeParametersCount();
388
389 *total_argc = args_count;
390 std::unique_ptr<Handle<Object>[]> param_data(
391 NewArray<Handle<Object>>(*total_argc));
392 for (int i = 0; i < args_count; i++) {
393 Handle<Object> val = Handle<Object>(frame->GetParameter(i), isolate);
394 param_data[i] = val;
395 }
396 return param_data;
397 }

444 }

447 // found parameter as mapped.
448 for (int i = 0; i < scope_info->ContextLocalCount(); i++) {
