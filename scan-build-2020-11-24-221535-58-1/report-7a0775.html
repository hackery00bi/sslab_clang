110// Reads a length-prefixed string, checking that it is within bounds. Returns
111// the offset of the string, and the length as an out parameter.
112WireBytesRef consume_string(Decoder* decoder, bool validate_utf8,
113 const char* name) {
114 uint32_t length = decoder->consume_u32v("string length");
115 uint32_t offset = decoder->pc_offset();
116 const byte* string_start = decoder->pc();
117 // Consume bytes before validation to guarantee that the string is not oob.
118 if (length > 0) {
119 decoder->consume_bytes(length, name);
120 if (decoder->ok() && validate_utf8 &&
121 !unibrow::Utf8::ValidateEncoding(string_start, length)) {
122 decoder->errorf(string_start, "%s: no valid UTF-8 string", name);
123 }
124 }
125 return {offset, decoder->failed() ? 0 : length};
126}

194 // Advances to the next section, checking that decoding the current section
195 // stopped at {section_end_}.
196 void advance(bool move_to_section_end = false) {
197 if (move_to_section_end && decoder_->pc() < section_end_) {
198 decoder_->consume_bytes(
199 static_cast<uint32_t>(section_end_ - decoder_->pc()));
200 }
201 if (decoder_->pc() != section_end_) {
202 const char* msg = decoder_->pc() < section_end_ ? "shorter" : "longer";
203 decoder_->errorf(decoder_->pc(),
204 "section was %s than expected size "
205 "(%u bytes expected, %zu decoded)",
206 msg, section_length(),
207 static_cast<size_t>(decoder_->pc() - section_start_));
208 }
209 next();
210 }

212 private:
213 Decoder* decoder_;
214 SectionCode section_code_;
215 const byte* section_start_;
216 const byte* payload_start_;
217 const byte* section_end_;
218
219 // Reads the section code/name at the current position and sets up
220 // the embedder fields.
221 void next() {
222 if (!decoder_->more()) {
223 section_code_ = kUnknownSectionCode;
224 return;
225 }
226 section_start_ = decoder_->pc();
227 uint8_t section_code = decoder_->consume_u8("section code");
228 // Read and check the section size.
229 uint32_t section_length = decoder_->consume_u32v("section length");
230
231 payload_start_ = decoder_->pc();
232 if (decoder_->checkAvailable(section_length)) {
233 // Get the limit of the section within the module.
234 section_end_ = payload_start_ + section_length;
235 } else {
236 // The section would extend beyond the end of the module.
237 section_end_ = payload_start_;
238 }
239
240 if (section_code == kUnknownSectionCode) {
241 // Check for the known "name", "sourceMappingURL", or "compilationHints"
242 // section.
243 // To identify the unknown section we set the end of the decoder bytes to
244 // the end of the custom section, so that we do not read the section name
245 // beyond the end of the section.
246 const byte* module_end = decoder_->end();
247 decoder_->set_end(section_end_);
248 section_code = IdentifyUnknownSectionInternal(decoder_);
249 if (decoder_->ok()) decoder_->set_end(module_end);
250 // As a side effect, the above function will forward the decoder to after
251 // the identifier string.
252 payload_start_ = decoder_->pc();
253 } else if (!IsValidSectionCode(section_code)) {
254 decoder_->errorf(decoder_->pc(), "unknown section code #0x%02x",
255 section_code);
256 section_code = kUnknownSectionCode;
257 }
258 section_code_ = decoder_->failed() ? kUnknownSectionCode
259 : static_cast<SectionCode>(section_code);
260
261 if (section_code_ == kUnknownSectionCode && section_end_ > decoder_->pc()) {
262 // skip to the end of the unknown section.
263 uint32_t remaining = static_cast<uint32_t>(section_end_ - decoder_->pc());
264 decoder_->consume_bytes(remaining, "section payload");
265 }

404 void DecodeSection(SectionCode section_code, Vector<const uint8_t> bytes,
405 uint32_t offset, bool verify_functions = true) {
406 if (failed()) return;
407 Reset(bytes, offset);
408 TRACE("Section: %s\n", SectionName(section_code));
409 TRACE("Decode Section %p - %p\n", bytes.begin(), bytes.end());
410
411 // Check if the section is out-of-order.
412 if (section_code < next_ordered_section_ &&
413 section_code < kFirstUnorderedSection) {
414 errorf(pc(), "unexpected section <%s>", SectionName(section_code));
415 return;
416 }
417
418 switch (section_code) {
419 case kUnknownSectionCode:
420 break;
421 case kDataCountSectionCode:
422 if (!CheckUnorderedSection(section_code)) return;
423 if (!CheckSectionOrder(section_code, kElementSectionCode,
424 kCodeSectionCode))
425 return;
426 break;
427 case kExceptionSectionCode:
428 if (!CheckUnorderedSection(section_code)) return;
429 if (!CheckSectionOrder(section_code, kMemorySectionCode,
430 kGlobalSectionCode))
431 return;
432 break;
433 case kNameSectionCode:
434 // TODO(titzer): report out of place name section as a warning.
435 // Be lenient with placement of name section. All except first
436 // occurrence are ignored.
437 case kSourceMappingURLSectionCode:
438 // sourceMappingURL is a custom section and currently can occur anywhere
439 // in the module. In case of multiple sourceMappingURL sections, all
440 // except the first occurrence are ignored.
441 case kDebugInfoSectionCode:
442 // .debug_info is a custom section containing core DWARF information
443 // if produced by compiler. Its presence likely means that Wasm was
444 // built in a debug mode.
445 case kExternalDebugInfoSectionCode:
446 // external_debug_info is a custom section containing a reference to an
447 // external symbol file.
448 case kCompilationHintsSectionCode:
449 // TODO(frgossen): report out of place compilation hints section as a
450 // warning.
451 // Be lenient with placement of compilation hints section. All except
452 // first occurrence after function section and before code section are
453 // ignored.
454 break;
455 default:
456 next_ordered_section_ = section_code + 1;
457 break;
458 }

461 case kUnknownSectionCode:
462 break;
463 case kTypeSectionCode:
464 DecodeTypeSection();
465 break;

468 break;

477 break;

551 module_->types.reserve(signatures_count);

593 module_->import_table.reserve(import_table_count);

595 TRACE("DecodeImportTable[%d] module+%d\n", i,
596 static_cast<int>(pc_ - start_));
597
598 module_->import_table.push_back({
599 {0, 0}, // module_name
600 {0, 0}, // field_name
601 kExternalFunction, // kind
602 0 // index
603 });
604 WasmImport* import = &module_->import_table.back();
605 const byte* pos = pc_;
606 import->module_name = consume_string(this, true, "module name");
607 import->field_name = consume_string(this, true, "field name");
608 import->kind =
609 static_cast<ImportExportKindCode>(consume_u8("import kind"));
610 switch (import->kind) {
611 case kExternalFunction: {
612 // ===== Imported function ===========================================
613 import->index = static_cast<uint32_t>(module_->functions.size());
614 module_->num_imported_functions++;
615 module_->functions.push_back({nullptr, // sig
616 import->index, // func_index
617 0, // sig_index
618 {0, 0}, // code
619 true, // imported
620 false, // exported
621 false}); // declared
622 WasmFunction* function = &module_->functions.back();
623 function->sig_index =
624 consume_sig_index(module_.get(), &function->sig);
625 break;
626 }
627 case kExternalTable: {
628 // ===== Imported table ==============================================
629 if (!AddTable(module_.get())) break;
630 import->index = static_cast<uint32_t>(module_->tables.size());
631 module_->num_imported_tables++;
632 module_->tables.emplace_back();
633 WasmTable* table = &module_->tables.back();
634 table->imported = true;
635 ValueType type = consume_reference_type();
636 table->type = type;
637 uint8_t flags = validate_table_flags("element count");
638 consume_resizable_limits(
639 "element count", "elements", std::numeric_limits<uint32_t>::max(),
640 &table->initial_size, &table->has_maximum_size,
641 std::numeric_limits<uint32_t>::max(), &table->maximum_size,
642 flags);
643 break;
644 }
645 case kExternalMemory: {
646 // ===== Imported memory =============================================
647 if (!AddMemory(module_.get())) break;
648 uint8_t flags = validate_memory_flags(&module_->has_shared_memory);
649 consume_resizable_limits(
650 "memory", "pages", max_initial_mem_pages(),
651 &module_->initial_pages, &module_->has_maximum_pages,
652 max_maximum_mem_pages(), &module_->maximum_pages, flags);
653 break;
654 }
655 case kExternalGlobal: {
656 // ===== Imported global =============================================
657 import->index = static_cast<uint32_t>(module_->globals.size());
658 module_->globals.push_back(
659 {kWasmStmt, false, WasmInitExpr(), {0}, true, false});
660 WasmGlobal* global = &module_->globals.back();
661 global->type = consume_value_type();
662 global->mutability = consume_mutability();
663 if (global->mutability) {
664 module_->num_imported_mutable_globals++;
665 }
666 break;
667 }
668 case kExternalException: {
669 // ===== Imported exception ==========================================
670 if (!enabled_features_.has_eh()) {
671 errorf(pos, "unknown import kind 0x%02x", import->kind);
672 break;
673 }
674 import->index = static_cast<uint32_t>(module_->exceptions.size());
675 const WasmExceptionSig* exception_sig = nullptr;
676 consume_exception_attribute(); // Attribute ignored for now.
677 consume_exception_sig_index(module_.get(), &exception_sig);
678 module_->exceptions.emplace_back(exception_sig);
679 break;
680 }
681 default:
682 errorf(pos, "unknown import kind 0x%02x", import->kind);
683 break;
684 }
685 }

769 void DecodeExportSection() {
770 uint32_t export_table_count =
771 consume_count("exports count", kV8MaxWasmExports);
772 module_->export_table.reserve(export_table_count);
773 for (uint32_t i = 0; ok() && i < export_table_count; ++i) {
774 TRACE("DecodeExportTable[%d] module+%d\n", i,
775 static_cast<int>(pc_ - start_));
776
777 module_->export_table.push_back({
778 {0, 0}, // name
779 kExternalFunction, // kind
780 0 // index
781 });
782 WasmExport* exp = &module_->export_table.back();
783
784 exp->name = consume_string(this, true, "field name");
785
786 const byte* pos = pc();
787 exp->kind = static_cast<ImportExportKindCode>(consume_u8("export kind"));
788 switch (exp->kind) {
789 case kExternalFunction: {
790 WasmFunction* func = nullptr;
791 exp->index =
792 consume_func_index(module_.get(), &func, "export function index");
793
794 if (failed()) break;
795 DCHECK_NOT_NULL(func)((void) 0);
796
797 module_->num_exported_functions++;
798 func->exported = true;
799 // Exported functions are considered "declared".
800 func->declared = true;
801 break;
802 }
803 case kExternalTable: {
804 WasmTable* table = nullptr;
805 exp->index = consume_table_index(module_.get(), &table);
806 if (table) table->exported = true;
807 break;
808 }
809 case kExternalMemory: {
810 uint32_t index = consume_u32v("memory index");
811 // TODO(titzer): This should become more regular
812 // once we support multiple memories.
813 if (!module_->has_memory || index != 0) {
814 error("invalid memory index != 0");
815 }
816 module_->mem_export = true;
817 break;
818 }
819 case kExternalGlobal: {
820 WasmGlobal* global = nullptr;
821 exp->index = consume_global_index(module_.get(), &global);
822 if (global) {
823 global->exported = true;
824 }
825 break;
826 }
827 case kExternalException: {
828 if (!enabled_features_.has_eh()) {
829 errorf(pos, "invalid export kind 0x%02x", exp->kind);
830 break;
831 }
832 WasmException* exception = nullptr;
833 exp->index = consume_exception_index(module_.get(), &exception);
834 break;
835 }
836 default:
837 errorf(pos, "invalid export kind 0x%02x", exp->kind);
838 break;
839 }
840 }
841 // Check for duplicate exports (except for asm.js).
842 if (ok() && origin_ == kWasmOrigin && module_->export_table.size() > 1) {
843 std::vector<WasmExport> sorted_exports(module_->export_table);
844
845 auto cmp_less = [this](const WasmExport& a, const WasmExport& b) {
846 // Return true if a < b.
847 if (a.name.length() != b.name.length()) {
848 return a.name.length() < b.name.length();
849 }
850 const byte* left = start() + GetBufferRelativeOffset(a.name.offset());
851 const byte* right = start() + GetBufferRelativeOffset(b.name.offset());
852 return memcmp(left, right, a.name.length()) < 0;
853 };
854 std::stable_sort(sorted_exports.begin(), sorted_exports.end(), cmp_less);
855
856 auto it = sorted_exports.begin();
857 WasmExport* last = &*it++;
858 for (auto end = sorted_exports.end(); it != end; last = &*it++) {
859 DCHECK(!cmp_less(*it, *last))((void) 0); // Vector must be sorted.
860 if (!cmp_less(*last, *it)) {
861 const byte* pc = start() + GetBufferRelativeOffset(it->name.offset());
862 TruncatedUserString<> name(pc, it->name.length());
863 errorf(pc, "Duplicate export name '%.*s' for %s %d and %s %d",
864 name.length(), name.start(), ExternalKindName(last->kind),
865 last->index, ExternalKindName(it->kind), it->index);
866 break;
867 }
868 }
869 }

883 void DecodeElementSection() {
884 uint32_t element_count =
885 consume_count("element count", FLAG_wasm_max_table_size);
886
887 for (uint32_t i = 0; ok() && i < element_count; ++i) {
888 const byte* pos = pc();
889
890 WasmElemSegment::Status status;
891 bool functions_as_elements;
892 uint32_t table_index;
893 WasmInitExpr offset;
894 ValueType type = kWasmBottom;
895 consume_element_segment_header(&status, &functions_as_elements, &type,
896 &table_index, &offset);
897 if (failed()) return;
898 DCHECK_NE(type, kWasmBottom)((void) 0);
899
900 if (status == WasmElemSegment::kStatusActive) {
901 if (table_index >= module_->tables.size()) {

915 consume_count("number of elements", max_table_init_entries());

917 module_->elem_segments.emplace_back(table_index, std::move(offset));
918 } else {
919 module_->elem_segments.emplace_back(
920 status == WasmElemSegment::kStatusDeclarative);
921 }
922
923 WasmElemSegment* init = &module_->elem_segments.back();
924 init->type = type;

926 uint32_t index = functions_as_elements ? consume_element_expr()
927 : consume_element_func_index();

1090 void DecodeCompilationHintsSection() {
1091 TRACE("DecodeCompilationHints module+%d\n", static_cast<int>(pc_ - start_));
1092
1093 // TODO(frgossen): Find a way to report compilation hint errors as warnings.
1094 // All except first occurrence after function section and before code
1095 // section are ignored.
1096 const bool before_function_section =
1097 next_ordered_section_ <= kFunctionSectionCode;
1098 const bool after_code_section = next_ordered_section_ > kCodeSectionCode;
1099 if (before_function_section || after_code_section ||
1100 has_seen_unordered_section(kCompilationHintsSectionCode)) {
1101 return;
1102 }
1103 set_seen_unordered_section(kCompilationHintsSectionCode);
1104
1105 // TODO(frgossen) Propagate errors to outer decoder in experimental phase.
1106 // We should use an inner decoder later and propagate its errors as
1107 // warnings.
1108 Decoder& decoder = *this;
1109 // Decoder decoder(start_, pc_, end_, buffer_offset_);
1110
1111 // Ensure exactly one compilation hint per function.
1112 uint32_t hint_count = decoder.consume_u32v("compilation hint count");
1113 if (hint_count != module_->num_declared_functions) {
1114 decoder.errorf(decoder.pc(), "Expected %u compilation hints (%u found)",
1115 module_->num_declared_functions, hint_count);
1116 }
1117
1118 // Decode sequence of compilation hints.
1119 if (decoder.ok()) {
1120 module_->compilation_hints.reserve(hint_count);
1121 }
1122 for (uint32_t i = 0; decoder.ok() && i < hint_count; i++) {
1123 TRACE("DecodeCompilationHints[%d] module+%d\n", i,
1124 static_cast<int>(pc_ - start_));
1125
1126 // Compilation hints are encoded in one byte each.
1127 // +-------+----------+---------------+----------+
1128 // | 2 bit | 2 bit | 2 bit | 2 bit |
1129 // | ... | Top tier | Baseline tier | Strategy |
1130 // +-------+----------+---------------+----------+
1131 uint8_t hint_byte = decoder.consume_u8("compilation hint");
1132 if (!decoder.ok()) break;
1133
1134 // Decode compilation hint.
1135 WasmCompilationHint hint;
1136 hint.strategy =
1137 static_cast<WasmCompilationHintStrategy>(hint_byte & 0x03);
1138 hint.baseline_tier =
1139 static_cast<WasmCompilationHintTier>(hint_byte >> 2 & 0x3);
1140 hint.top_tier =
1141 static_cast<WasmCompilationHintTier>(hint_byte >> 4 & 0x3);
1142
1143 // Ensure that the top tier never downgrades a compilation result.
1144 // If baseline and top tier are the same compilation will be invoked only
1145 // once.
1146 if (hint.top_tier < hint.baseline_tier &&
1147 hint.top_tier != WasmCompilationHintTier::kDefault) {
1148 decoder.errorf(decoder.pc(),
1149 "Invalid compilation hint %#x (forbidden downgrade)",
1150 hint_byte);
1151 }
1152
1153 // Happily accept compilation hint.
1154 if (decoder.ok()) {
1155 module_->compilation_hints.push_back(std::move(hint));
1156 }
1157 }
1158
1159 // If section was invalid reset compilation hints.
1160 if (decoder.failed()) {
1161 module_->compilation_hints.clear();
1162 }
1163
1164 // @TODO(frgossen) Skip the whole compilation hints section in the outer
1165 // decoder if inner decoder was used.
1166 // consume_bytes(static_cast<uint32_t>(end_ - start_), nullptr);
1167 }

383 constexpr int shift = byte_index * 7;
384 constexpr bool is_last_byte = byte_index == kMaxLength - 1;

398 // following call is unreachable if is_last_byte is false.

400 return read_leb_tail<IntType, validate, advance_pc, trace, size_in_bits,

412 // most-significant bit; e.g. for a 32-bit LEB128:
413 // kExtraBits = 4 (== 32 - (5-1) * 7)
414 // For unsigned values, the extra bits must be all zero.
415 // For signed values, the extra bits *plus* the most significant bit must
416 // either be 0, or all ones.
417 constexpr int kExtraBits = size_in_bits - ((kMaxLength - 1) * 7);
418 constexpr int kSignExtBits = kExtraBits - (is_signed ? 1 : 0);
419 const byte checked_bits = b & (0xFF << kSignExtBits);
420 constexpr byte kSignExtendedExtraBits = 0x7f & (0xFF << kSignExtBits);
421 const bool valid_extra_bits =
422 checked_bits == 0 ||

427 error(pc, "extra bits in varint");

429 }
430 }

432 is_signed ? Max(0, int{8 * sizeof(IntType)} - shift - 7) : 0;
433 // Perform sign extension.
434 result = (result << sign_ext_shift) >> sign_ext_shift;
435 if (trace && is_signed) {

437 } else if (trace) {
438 TRACE("= %" PRIu64"l" "u" "\n", static_cast<uint64_t>(result));
439 }
440 return result;
441 }
