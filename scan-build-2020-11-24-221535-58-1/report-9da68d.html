444int ParseDecimalInteger(const uint8_t* s, int from, int to) {
445 DCHECK_LT(to - from, 10)((void) 0); // Overflow is not possible.

459Handle<Object> String::ToNumber(Isolate* isolate, Handle<String> subject) {
460 // Flatten {subject} string first.
461 subject = String::Flatten(isolate, subject);
462
463 // Fast array index case.
464 uint32_t index;
465 if (subject->AsArrayIndex(&index)) {
466 return isolate->factory()->NewNumberFromUint(index);
467 }
468
469 // Fast case: short integer or some sorts of junk values.
470 if (subject->IsSeqOneByteString()) {
471 int len = subject->length();
472 if (len == 0) return handle(Smi::zero(), isolate);
473
474 DisallowHeapAllocation no_gc;
475 uint8_t const* data =
476 Handle<SeqOneByteString>::cast(subject)->GetChars(no_gc);
477 bool minus = (data[0] == '-');
478 int start_pos = (minus ? 1 : 0);
479
480 if (start_pos == len) {
481 return isolate->factory()->nan_value();
482 } else if (data[start_pos] > '9') {
483 // Fast check for a junk value. A valid string may start from a
484 // whitespace, a sign ('+' or '-'), the decimal point, a decimal digit
485 // or the 'I' character ('Infinity'). All of that have codes not greater
486 // than '9' except 'I' and &nbsp;.
487 if (data[start_pos] != 'I' && data[start_pos] != 0xA0) {
488 return isolate->factory()->nan_value();
489 }
490 } else if (len - start_pos < 10 && AreDigits(data, start_pos, len)) {
491 // The maximal/minimal smi has 10 digits. If the string has less digits
492 // we know it will fit into the smi-data type.
493 int d = ParseDecimalInteger(data, start_pos, len);
494 if (minus) {
495 if (d == 0) return isolate->factory()->minus_zero_value();
496 d = -d;
497 } else if (!subject->HasHashCode() && len <= String::kMaxArrayIndexSize &&
498 (len == 1 || data[0] != '0')) {
499 // String hash is not calculated yet but all the data are present.
500 // Update the hash field to speed up sequential convertions.
501 uint32_t hash = StringHasher::MakeArrayIndexHash(d, len);
502#ifdef DEBUG
503 subject->Hash(); // Force hash calculation.
504 DCHECK_EQ(static_cast<int>(subject->hash_field()),((void) 0)
505 static_cast<int>(hash))((void) 0);
506#endif
507 subject->set_hash_field(hash);
508 }
509 return handle(Smi::FromInt(d), isolate);
510 }
511 }
512
513 // Slower case.
514 int flags = ALLOW_HEX | ALLOW_OCTAL | ALLOW_BINARY;
515 return isolate->factory()->NewNumber(StringToDouble(isolate, subject, flags));
516}

572 return std::unique_ptr<char[]>();

574 // Negative length means the to the end of the string.
575 if (length < 0) length = kMaxInt - offset;

577 // Compute the size of the UTF-8 string. Start at the specified offset.
578 StringCharacterStream stream(*this, offset);
579 int character_position = offset;
580 int utf8_bytes = 0;
581 int last = unibrow::Utf16::kNoPreviousCharacter;
582 while (stream.HasMore() && character_position++ < offset + length) {
583 uint16_t character = stream.GetNext();
584 utf8_bytes += unibrow::Utf8::Length(character, last);
585 last = character;
586 }

588 if (length_return) {
589 *length_return = utf8_bytes;
590 }

592 char* result = NewArray<char>(utf8_bytes + 1);

594 // Convert the UTF-16 string to a UTF-8 buffer. Start at the specified offset.
595 stream.Reset(*this, offset);
596 character_position = offset;
597 int utf8_byte_position = 0;
598 last = unibrow::Utf16::kNoPreviousCharacter;
599 while (stream.HasMore() && character_position++ < offset + length) {
600 uint16_t character = stream.GetNext();

602 character = ' ';

606 last = character;
607 }
608 result[utf8_byte_position] = 0;
609 return std::unique_ptr<char[]>(result);

618template <typename sinkchar>
619void String::WriteToFlat(String src, sinkchar* sink, int f, int t) {
620 DisallowHeapAllocation no_gc;
621 String source = src;
622 int from = f;
623 int to = t;
624 while (from < to) {
625 DCHECK_LE(0, from)((void) 0);
626 DCHECK_LE(to, source.length())((void) 0);
627 switch (StringShape(source).full_representation_tag()) {
628 case kOneByteStringTag | kExternalStringTag: {
629 CopyChars(sink, ExternalOneByteString::cast(source).GetChars() + from,
630 to - from);
631 return;
632 }
633 case kTwoByteStringTag | kExternalStringTag: {
634 const uc16* data = ExternalTwoByteString::cast(source).GetChars();
635 CopyChars(sink, data + from, to - from);
636 return;
637 }
638 case kOneByteStringTag | kSeqStringTag: {
639 CopyChars(sink, SeqOneByteString::cast(source).GetChars(no_gc) + from,
640 to - from);
641 return;
642 }
643 case kTwoByteStringTag | kSeqStringTag: {
644 CopyChars(sink, SeqTwoByteString::cast(source).GetChars(no_gc) + from,
645 to - from);
646 return;
647 }
648 case kOneByteStringTag | kConsStringTag:
649 case kTwoByteStringTag | kConsStringTag: {
650 ConsString cons_string = ConsString::cast(source);
651 String first = cons_string.first();
652 int boundary = first.length();
653 if (to - boundary >= boundary - from) {
654 // Right hand side is longer. Recurse over left.
655 if (from < boundary) {
656 WriteToFlat(first, sink, from, boundary);
657 if (from == 0 && cons_string.second() == first) {
658 CopyChars(sink + boundary, sink, boundary);
659 return;
660 }
661 sink += boundary - from;
662 from = 0;
663 } else {
664 from -= boundary;
665 }
666 to -= boundary;
667 source = cons_string.second();
668 } else {
669 // Left hand side is longer. Recurse over right.
670 if (to > boundary) {
671 String second = cons_string.second();
672 // When repeatedly appending to a string, we get a cons string that
673 // is unbalanced to the left, a list, essentially. We inline the
674 // common case of sequential one-byte right child.
675 if (to - boundary == 1) {
676 sink[boundary - from] = static_cast<sinkchar>(second.Get(0));
677 } else if (second.IsSeqOneByteString()) {
678 CopyChars(sink + boundary - from,
679 SeqOneByteString::cast(second).GetChars(no_gc),
680 to - boundary);
681 } else {
682 WriteToFlat(second, sink + boundary - from, 0, to - boundary);
683 }
684 to = boundary;
685 }
686 source = first;
687 }
688 break;
689 }
690 case kOneByteStringTag | kSlicedStringTag:
691 case kTwoByteStringTag | kSlicedStringTag: {
692 SlicedString slice = SlicedString::cast(source);
693 unsigned offset = slice.offset();
694 WriteToFlat(slice.parent(), sink, from + offset, to + offset);
695 return;
696 }
697 case kOneByteStringTag | kThinStringTag:
698 case kTwoByteStringTag | kThinStringTag:
699 source = ThinString::cast(source).actual();
700 break;
701 }
702 }
703 DCHECK_EQ(from, to)((void) 0);
704}

722 // position for the implicit return statement.
723 line_ends->push_back(src_len);
