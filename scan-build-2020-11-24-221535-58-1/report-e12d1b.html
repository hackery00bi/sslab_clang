518 // displacements and scale factors that are used as inputs, so instead of
519 // enumerating all possible patterns by brute force, checking for node
520 // clusters using the following templates in the following order suffices to
521 // find all of the interesting cases (S = index * scale, B = base input, D =
522 // displacement input):
523 // (S + (B + D))
524 // (S + (B + B))
525 // (S + D)
526 // (S + B)
527 // ((S + D) + B)

87Reduction CommonOperatorReducer::ReduceBranch(Node* node) {
88 DCHECK_EQ(IrOpcode::kBranch, node->opcode())((void) 0);
89 Node* const cond = node->InputAt(0);
90 // Swap IfTrue/IfFalse on {branch} if {cond} is a BooleanNot and use the input
91 // to BooleanNot as new condition for {branch}. Note we assume that {cond} was
92 // already properly optimized before we get here (as guaranteed by the graph
93 // reduction logic). The same applies if {cond} is a Select acting as boolean
94 // not (i.e. true being returned in the false case and vice versa).
95 if (cond->opcode() == IrOpcode::kBooleanNot ||
96 (cond->opcode() == IrOpcode::kSelect &&
97 DecideCondition(broker(), cond->InputAt(1)) == Decision::kFalse &&
98 DecideCondition(broker(), cond->InputAt(2)) == Decision::kTrue)) {
99 for (Node* const use : node->uses()) {
100 switch (use->opcode()) {
101 case IrOpcode::kIfTrue:
102 NodeProperties::ChangeOp(use, common()->IfFalse());
103 break;
104 case IrOpcode::kIfFalse:
105 NodeProperties::ChangeOp(use, common()->IfTrue());
106 break;
107 default:
108 UNREACHABLE()V8_Fatal("unreachable code");
109 }
110 }
111 // Update the condition of {branch}. No need to mark the uses for revisit,
112 // since we tell the graph reducer that the {branch} was changed and the
113 // graph reduction logic will ensure that the uses are revisited properly.
114 node->ReplaceInput(0, cond->InputAt(0));
115 // Negate the hint for {branch}.
116 NodeProperties::ChangeOp(
117 node, common()->Branch(NegateBranchHint(BranchHintOf(node->op()))));
118 return Changed(node);
119 }
120 Decision const decision = DecideCondition(broker(), cond);
121 if (decision == Decision::kUnknown) return NoChange();
122 Node* const control = node->InputAt(1);
123 for (Node* const use : node->uses()) {
124 switch (use->opcode()) {
125 case IrOpcode::kIfTrue:
126 Replace(use, (decision == Decision::kTrue) ? control : dead());
127 break;
128 case IrOpcode::kIfFalse:
129 Replace(use, (decision == Decision::kFalse) ? control : dead());
130 break;
131 default:
132 UNREACHABLE()V8_Fatal("unreachable code");
133 }
134 }
135 return Replace(dead());
136}

138Reduction CommonOperatorReducer::ReduceDeoptimizeConditional(Node* node) {
139 DCHECK(node->opcode() == IrOpcode::kDeoptimizeIf ||((void) 0)
140 node->opcode() == IrOpcode::kDeoptimizeUnless)((void) 0);
141 bool condition_is_true = node->opcode() == IrOpcode::kDeoptimizeUnless;
142 DeoptimizeParameters p = DeoptimizeParametersOf(node->op());
143 Node* condition = NodeProperties::GetValueInput(node, 0);
144 Node* frame_state = NodeProperties::GetValueInput(node, 1);
145 Node* effect = NodeProperties::GetEffectInput(node);
146 Node* control = NodeProperties::GetControlInput(node);
147 // Swap DeoptimizeIf/DeoptimizeUnless on {node} if {cond} is a BooleaNot
148 // and use the input to BooleanNot as new condition for {node}. Note we
149 // assume that {cond} was already properly optimized before we get here
150 // (as guaranteed by the graph reduction logic).
151 if (condition->opcode() == IrOpcode::kBooleanNot) {
152 NodeProperties::ReplaceValueInput(node, condition->InputAt(0), 0);
153 NodeProperties::ChangeOp(
154 node,
155 condition_is_true
156 ? common()->DeoptimizeIf(p.kind(), p.reason(), p.feedback())
157 : common()->DeoptimizeUnless(p.kind(), p.reason(), p.feedback()));
158 return Changed(node);
159 }
160 Decision const decision = DecideCondition(broker(), condition);
161 if (decision == Decision::kUnknown) return NoChange();
162 if (condition_is_true == (decision == Decision::kTrue)) {
163 ReplaceWithValue(node, dead(), effect, control);
164 } else {
165 control = graph()->NewNode(
166 common()->Deoptimize(p.kind(), p.reason(), p.feedback()), frame_state,
167 effect, control);
168 // TODO(bmeurer): This should be on the AdvancedReducer somehow.
169 NodeProperties::MergeControlToEnd(graph(), common(), control);
170 Revisit(graph()->end());
171 }
172 return Replace(dead());
173}

175Reduction CommonOperatorReducer::ReduceMerge(Node* node) {
176 DCHECK_EQ(IrOpcode::kMerge, node->opcode())((void) 0);
177 //
178 // Check if this is a merge that belongs to an unused diamond, which means
179 // that:
180 //
181 // a) the {Merge} has no {Phi} or {EffectPhi} uses, and
182 // b) the {Merge} has two inputs, one {IfTrue} and one {IfFalse}, which are
183 // both owned by the Merge, and
184 // c) and the {IfTrue} and {IfFalse} nodes point to the same {Branch}.
185 //
186 if (node->InputCount() == 2) {
187 for (Node* const use : node->uses()) {
188 if (IrOpcode::IsPhiOpcode(use->opcode())) return NoChange();
189 }
190 Node* if_true = node->InputAt(0);
191 Node* if_false = node->InputAt(1);
192 if (if_true->opcode() != IrOpcode::kIfTrue) std::swap(if_true, if_false);
193 if (if_true->opcode() == IrOpcode::kIfTrue &&
194 if_false->opcode() == IrOpcode::kIfFalse &&
195 if_true->InputAt(0) == if_false->InputAt(0) && if_true->OwnedBy(node) &&
196 if_false->OwnedBy(node)) {
197 Node* const branch = if_true->InputAt(0);
198 DCHECK_EQ(IrOpcode::kBranch, branch->opcode())((void) 0);
199 DCHECK(branch->OwnedBy(if_true, if_false))((void) 0);
200 Node* const control = branch->InputAt(1);
201 // Mark the {branch} as {Dead}.
202 branch->TrimInputCount(0);
203 NodeProperties::ChangeOp(branch, common()->Dead());
204 return Replace(control);
205 }
206 }
207 return NoChange();
208}

237 DCHECK_EQ(IrOpcode::kPhi, node->opcode())((void) 0);
238 Node::Inputs inputs = node->inputs();
239 int const value_input_count = inputs.count() - 1;
240 DCHECK_LE(1, value_input_count)((void) 0);
241 Node* const merge = inputs[value_input_count];
242 DCHECK(IrOpcode::IsMergeOpcode(merge->opcode()))((void) 0);
243 DCHECK_EQ(value_input_count, merge->InputCount())((void) 0);
244 if (value_input_count == 2) {
245 Node* vtrue = inputs[0];
246 Node* vfalse = inputs[1];

248 Node* if_true = merge_inputs[0];
249 Node* if_false = merge_inputs[1];
250 if (if_true->opcode() != IrOpcode::kIfTrue) {
251 std::swap(if_true, if_false);
252 std::swap(vtrue, vfalse);
253 }
254 if (if_true->opcode() == IrOpcode::kIfTrue &&
255 if_false->opcode() == IrOpcode::kIfFalse &&
256 if_true->InputAt(0) == if_false->InputAt(0)) {
257 Node* const branch = if_true->InputAt(0);
258 // Check that the branch is not dead already.
259 if (branch->opcode() != IrOpcode::kBranch) return NoChange();
260 Node* const cond = branch->InputAt(0);
261 if (cond->opcode() == IrOpcode::kFloat32LessThan) {
262 Float32BinopMatcher mcond(cond);
263 if (mcond.left().Is(0.0) && mcond.right().Equals(vtrue) &&
264 vfalse->opcode() == IrOpcode::kFloat32Sub) {
265 Float32BinopMatcher mvfalse(vfalse);
266 if (mvfalse.left().IsZero() && mvfalse.right().Equals(vtrue)) {
267 // We might now be able to further reduce the {merge} node.
268 Revisit(merge);
269 return Change(node, machine()->Float32Abs(), vtrue);
270 }
271 }
272 } else if (cond->opcode() == IrOpcode::kFloat64LessThan) {
273 Float64BinopMatcher mcond(cond);
274 if (mcond.left().Is(0.0) && mcond.right().Equals(vtrue) &&
275 vfalse->opcode() == IrOpcode::kFloat64Sub) {
276 Float64BinopMatcher mvfalse(vfalse);
277 if (mvfalse.left().IsZero() && mvfalse.right().Equals(vtrue)) {
278 // We might now be able to further reduce the {merge} node.
279 Revisit(merge);
280 return Change(node, machine()->Float64Abs(), vtrue);
281 }
282 }
283 }
284 }
285 }
286 Node* const value = inputs[0];
287 DCHECK_NE(node, value)((void) 0);
288 for (int i = 1; i < value_input_count; ++i) {
289 Node* const input = inputs[i];
290 if (input == node) {
291 // Ignore redundant inputs.
292 DCHECK_EQ(IrOpcode::kLoop, merge->opcode())((void) 0);
293 continue;
294 }
295 if (input != value) return NoChange();
296 }
297 // We might now be able to further reduce the {merge} node.
298 Revisit(merge);
299 return Replace(value);

302Reduction CommonOperatorReducer::ReduceReturn(Node* node) {
303 DCHECK_EQ(IrOpcode::kReturn, node->opcode())((void) 0);
304 Node* effect = NodeProperties::GetEffectInput(node);
305 if (effect->opcode() == IrOpcode::kCheckpoint) {
306 // Any {Return} node can never be used to insert a deoptimization point,
307 // hence checkpoints can be cut out of the effect chain flowing into it.
308 effect = NodeProperties::GetEffectInput(effect);
309 NodeProperties::ReplaceEffectInput(node, effect);
310 return Changed(node).FollowedBy(ReduceReturn(node));
311 }
312 // TODO(ahaas): Extend the reduction below to multiple return values.
313 if (ValueInputCountOfReturn(node->op()) != 1) {
314 return NoChange();
315 }
316 Node* pop_count = NodeProperties::GetValueInput(node, 0);
317 Node* value = NodeProperties::GetValueInput(node, 1);
318 Node* control = NodeProperties::GetControlInput(node);
319 if (value->opcode() == IrOpcode::kPhi &&
320 NodeProperties::GetControlInput(value) == control &&
321 control->opcode() == IrOpcode::kMerge) {
322 // This optimization pushes {Return} nodes through merges. It checks that
323 // the return value is actually a {Phi} and the return control dependency
324 // is the {Merge} to which the {Phi} belongs.
325
326 // Value1 ... ValueN Control1 ... ControlN
327 // ^ ^ ^ ^
328 // | | | |
329 // +----+-----+ +------+-----+
330 // | |
331 // Phi --------------> Merge
332 // ^ ^
333 // | |
334 // | +-----------------+
335 // | |
336 // Return -----> Effect
337 // ^
338 // |
339 // End
340
341 // Now the effect input to the {Return} node can be either an {EffectPhi}
342 // hanging off the same {Merge}, or the effect chain doesn't depend on the
343 // {Phi} or the {Merge}, in which case we know that the effect input must
344 // somehow dominate all merged branches.
345
346 Node::Inputs control_inputs = control->inputs();
347 Node::Inputs value_inputs = value->inputs();
348 DCHECK_NE(0, control_inputs.count())((void) 0);
349 DCHECK_EQ(control_inputs.count(), value_inputs.count() - 1)((void) 0);
350 DCHECK_EQ(IrOpcode::kEnd, graph()->end()->opcode())((void) 0);
351 DCHECK_NE(0, graph()->end()->InputCount())((void) 0);
352 if (control->OwnedBy(node, value) && value->OwnedBy(node)) {
353 for (int i = 0; i < control_inputs.count(); ++i) {
354 // Create a new {Return} and connect it to {end}. We don't need to mark
355 // {end} as revisit, because we mark {node} as {Dead} below, which was
356 // previously connected to {end}, so we know for sure that at some point
357 // the reducer logic will visit {end} again.
358 Node* ret = graph()->NewNode(node->op(), pop_count, value_inputs[i],
359 effect, control_inputs[i]);
360 NodeProperties::MergeControlToEnd(graph(), common(), ret);
361 }
362 // Mark the Merge {control} and Return {node} as {dead}.
363 Replace(control, dead());
364 return Replace(dead());
365 } else if (effect->opcode() == IrOpcode::kEffectPhi &&
366 NodeProperties::GetControlInput(effect) == control) {
367 Node::Inputs effect_inputs = effect->inputs();
368 DCHECK_EQ(control_inputs.count(), effect_inputs.count() - 1)((void) 0);
369 for (int i = 0; i < control_inputs.count(); ++i) {
370 // Create a new {Return} and connect it to {end}. We don't need to mark
371 // {end} as revisit, because we mark {node} as {Dead} below, which was
372 // previously connected to {end}, so we know for sure that at some point
373 // the reducer logic will visit {end} again.
374 Node* ret = graph()->NewNode(node->op(), pop_count, value_inputs[i],
375 effect_inputs[i], control_inputs[i]);
376 NodeProperties::MergeControlToEnd(graph(), common(), ret);
377 }
378 // Mark the Merge {control} and Return {node} as {dead}.
379 Replace(control, dead());
380 return Replace(dead());
381 }
382 }
383 return NoChange();
384}

429Reduction CommonOperatorReducer::ReduceSwitch(Node* node) {
430 DCHECK_EQ(IrOpcode::kSwitch, node->opcode())((void) 0);
431 Node* const switched_value = node->InputAt(0);
432 Node* const control = node->InputAt(1);
433
434 // Attempt to constant match the switched value against the IfValue cases. If
435 // no case matches, then use the IfDefault. We don't bother marking
436 // non-matching cases as dead code (same for an unused IfDefault), because the
437 // Switch itself will be marked as dead code.
438 Int32Matcher mswitched(switched_value);
439 if (mswitched.HasValue()) {
440 bool matched = false;
441
442 size_t const projection_count = node->op()->ControlOutputCount();
443 Node** projections = zone_->NewArray<Node*>(projection_count);
444 NodeProperties::CollectControlProjections(node, projections,
445 projection_count);
446 for (size_t i = 0; i < projection_count - 1; i++) {
447 Node* if_value = projections[i];
448 DCHECK_EQ(IrOpcode::kIfValue, if_value->opcode())((void) 0);
449 const IfValueParameters& p = IfValueParametersOf(if_value->op());
450 if (p.value() == mswitched.Value()) {
451 matched = true;
452 Replace(if_value, control);
453 break;
454 }
455 }
456 if (!matched) {
457 Node* if_default = projections[projection_count - 1];
458 DCHECK_EQ(IrOpcode::kIfDefault, if_default->opcode())((void) 0);
459 Replace(if_default, control);
460 }
461 return Replace(dead());
462 }
463 return NoChange();
464}
