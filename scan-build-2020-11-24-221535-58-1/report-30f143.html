164std::shared_ptr<NativeModule> NativeModuleCache::MaybeGetNativeModule(
165 ModuleOrigin origin, Vector<const uint8_t> wire_bytes) {
166 if (origin != kWasmOrigin) return nullptr;
167 base::MutexGuard lock(&mutex_);
168 size_t prefix_hash = PrefixHash(wire_bytes);
169 NativeModuleCache::Key key{prefix_hash, wire_bytes};
170 while (true) {
171 auto it = map_.find(key);
172 if (it == map_.end()) {
173 // Even though this exact key is not in the cache, there might be a
174 // matching prefix hash indicating that a streaming compilation is
175 // currently compiling a module with the same prefix. {OnFinishedStream}
176 // happens on the main thread too, so waiting for streaming compilation to
177 // finish would create a deadlock. Instead, compile the module twice and
178 // handle the conflict in {UpdateNativeModuleCache}.
179
180 // Insert a {nullopt} entry to let other threads know that this
181 // {NativeModule} is already being created on another thread.
182 auto p = map_.emplace(key, base::nullopt);
183 USE(p)do { ::v8::base::Use unused_tmp_array_for_use_macro[]{p}; (void
)unused_tmp_array_for_use_macro; } while (false);
184 DCHECK(p.second)((void) 0);
185 return nullptr;
186 }
187 if (it->second.has_value()) {
188 if (auto shared_native_module = it->second.value().lock()) {
189 DCHECK_EQ(shared_native_module->wire_bytes(), wire_bytes)((void) 0);
190 return shared_native_module;
191 }
192 }
193 cache_cv_.Wait(&mutex_);
194 }
195}

219std::shared_ptr<NativeModule> NativeModuleCache::Update(
220 std::shared_ptr<NativeModule> native_module, bool error) {
221 DCHECK_NOT_NULL(native_module)((void) 0);
222 if (native_module->module()->origin != kWasmOrigin) return native_module;
223 Vector<const uint8_t> wire_bytes = native_module->wire_bytes();
224 DCHECK(!wire_bytes.empty())((void) 0);
225 size_t prefix_hash = PrefixHash(native_module->wire_bytes());
226 base::MutexGuard lock(&mutex_);
227 map_.erase(Key{prefix_hash, {}});
228 const Key key{prefix_hash, wire_bytes};
229 auto it = map_.find(key);
230 if (it != map_.end()) {
231 if (it->second.has_value()) {
232 auto conflicting_module = it->second.value().lock();
233 if (conflicting_module != nullptr) {
234 DCHECK_EQ(conflicting_module->wire_bytes(), wire_bytes)((void) 0);
235 return conflicting_module;
236 }
237 }
238 map_.erase(it);
239 }
240 if (!error) {
241 // The key now points to the new native module's owned copy of the bytes,
242 // so that it stays valid until the native module is freed and erased from
243 // the map.
244 auto p = map_.emplace(
245 key, base::Optional<std::weak_ptr<NativeModule>>(native_module));
246 USE(p)do { ::v8::base::Use unused_tmp_array_for_use_macro[]{p}; (void
)unused_tmp_array_for_use_macro; } while (false);
247 DCHECK(p.second)((void) 0);
248 }
249 cache_cv_.NotifyAll();
250 return native_module;
251}

270// static
271size_t NativeModuleCache::PrefixHash(Vector<const uint8_t> wire_bytes) {
272 // Compute the hash as a combined hash of the sections up to the code section
273 // header, to mirror the way streaming compilation does it.
274 Decoder decoder(wire_bytes.begin(), wire_bytes.end());
275 decoder.consume_bytes(8, "module header");
276 size_t hash = NativeModuleCache::WireBytesHash(wire_bytes.SubVector(0, 8));
277 SectionCode section_id = SectionCode::kUnknownSectionCode;
278 while (decoder.ok() && decoder.more()) {
279 section_id = static_cast<SectionCode>(decoder.consume_u8());
280 uint32_t section_size = decoder.consume_u32v("section size");
281 if (section_id == SectionCode::kCodeSectionCode) {
282 uint32_t num_functions = decoder.consume_u32v("num functions");
283 // If {num_functions} is 0, the streaming decoder skips the section. Do
284 // the same here to ensure hashes are consistent.
285 if (num_functions != 0) {
286 hash = base::hash_combine(hash, section_size);
287 }
288 break;
289 }
290 const uint8_t* payload_start = decoder.pc();
291 decoder.consume_bytes(section_size, "section payload");
292 size_t section_hash = NativeModuleCache::WireBytesHash(
293 Vector<const uint8_t>(payload_start, section_size));
294 hash = base::hash_combine(hash, section_hash);
295 }
296 return hash;
297}

565 // start function and report thrown exception explicitly via out argument.
566 v8::TryCatch catcher(reinterpret_cast<v8::Isolate*>(isolate));
567 catcher.SetVerbose(false);
568 catcher.SetCaptureMessage(false);
569
570 MaybeHandle<WasmInstanceObject> instance_object = SyncInstantiate(
571 isolate, &thrower, module_object, imports, Handle<JSArrayBuffer>::null());
572
573 if (!instance_object.is_null()) {
574 resolver->OnInstantiationSucceeded(instance_object.ToHandleChecked());
575 return;
576 }
577
578 if (isolate->has_pending_exception()) {
579 // The JS code executed during instantiation has thrown an exception.
580 // We have to move the exception to the promise chain.
581 Handle<Object> exception(isolate->pending_exception(), isolate);
582 isolate->clear_pending_exception();
583 *isolate->external_caught_exception_address() = false;
584 resolver->OnInstantiationFailed(exception);
585 thrower.Reset();
586 } else {
587 DCHECK(thrower.error())((void) 0);
588 resolver->OnInstantiationFailed(thrower.Reify());
589 }
590}

685void WasmEngine::TierUpAllModulesPerIsolate(Isolate* isolate) {
686 // Only trigger recompilation after releasing the mutex, otherwise we risk
687 // deadlocks because of lock inversion. The bool tells whether the module
688 // needs recompilation for tier up.
689 std::vector<std::pair<std::shared_ptr<NativeModule>, bool>> native_modules;
690 {
691 base::MutexGuard lock(&mutex_);
692 isolates_[isolate]->keep_tiered_down = false;
693 auto test_can_tier_up = [this](NativeModule* native_module) {
694 DCHECK_EQ(1, native_modules_.count(native_module))((void) 0);
695 for (auto* isolate : native_modules_[native_module]->isolates) {
696 DCHECK_EQ(1, isolates_.count(isolate))((void) 0);
697 if (isolates_[isolate]->keep_tiered_down) return false;
698 }
699 return true;
700 };
701 for (auto* native_module : isolates_[isolate]->native_modules) {
702 DCHECK_EQ(1, native_modules_.count(native_module))((void) 0);
703 auto shared_ptr = native_modules_[native_module]->weak_ptr.lock();
704 if (!shared_ptr) continue; // The module is not used any more.
705 if (!native_module->IsTieredDown()) continue;
706 // Only start tier-up if no other isolate needs this module in tiered
707 // down state.
708 bool tier_up = test_can_tier_up(native_module);
709 if (tier_up) native_module->SetTieringState(kTieredUp);
710 native_modules.emplace_back(std::move(shared_ptr), tier_up);
711 }
712 }
713 for (auto& entry : native_modules) {
714 auto& native_module = entry.first;
715 bool tier_up = entry.second;
716 // Remove all breakpoints set by this isolate.
717 if (native_module->HasDebugInfo()) {
718 native_module->GetDebugInfo()->RemoveIsolate(isolate);
719 }
720 if (tier_up) native_module->RecompileForTiering();
721 }

730Handle<Script> CreateWasmScript(Isolate* isolate,
731 std::shared_ptr<NativeModule> native_module,
732 Vector<const char> source_url = {}) {
733 Handle<Script> script =
734 isolate->factory()->NewScript(isolate->factory()->empty_string());
735 script->set_compilation_state(Script::COMPILATION_STATE_COMPILED);
736 script->set_context_data(isolate->native_context()->debug_context_id());
737 script->set_type(Script::TYPE_WASM);
738
739 Vector<const uint8_t> wire_bytes = native_module->wire_bytes();
740 int hash = StringHasher::HashSequentialString(
741 reinterpret_cast<const char*>(wire_bytes.begin()), wire_bytes.length(),
742 kZeroHashSeed);
743
744 const int kBufferSize = 32;
745 char buffer[kBufferSize];
746
747 // Script name is "<module_name>-hash" if name is available and "hash"
748 // otherwise.
749 const WasmModule* module = native_module->module();
750 Handle<String> name_str;
751 if (module->name.is_set()) {
752 int name_chars = SNPrintF(ArrayVector(buffer), "-%08x", hash);
753 DCHECK(name_chars >= 0 && name_chars < kBufferSize)((void) 0);
754 Handle<String> name_hash =
755 isolate->factory()
756 ->NewStringFromOneByte(
757 VectorOf(reinterpret_cast<uint8_t*>(buffer), name_chars),
758 AllocationType::kOld)
759 .ToHandleChecked();
760 Handle<String> module_name =
761 WasmModuleObject::ExtractUtf8StringFromModuleBytes(
762 isolate, wire_bytes, module->name, kNoInternalize);
763 name_str = isolate->factory()
764 ->NewConsString(module_name, name_hash)
765 .ToHandleChecked();
766 } else {
767 int name_chars = SNPrintF(ArrayVector(buffer), "%08x", hash);
768 DCHECK(name_chars >= 0 && name_chars < kBufferSize)((void) 0);
769 name_str = isolate->factory()
770 ->NewStringFromOneByte(
771 VectorOf(reinterpret_cast<uint8_t*>(buffer), name_chars),
772 AllocationType::kOld)
773 .ToHandleChecked();
774 }
775 script->set_name(*name_str);
776 MaybeHandle<String> url_str;
777 if (!source_url.empty()) {
778 url_str =
779 isolate->factory()->NewStringFromUtf8(source_url, AllocationType::kOld);
780 } else {
781 Handle<String> url_prefix =
782 isolate->factory()->InternalizeString(StaticCharVector("wasm://wasm/"));
783 url_str = isolate->factory()->NewConsString(url_prefix, name_str);
784 }
785 script->set_source_url(*url_str.ToHandleChecked());
786
787 const WasmDebugSymbols& debug_symbols =
788 native_module->module()->debug_symbols;
789 if (debug_symbols.type == WasmDebugSymbols::Type::SourceMap &&
790 !debug_symbols.external_url.is_empty()) {
791 Vector<const char> external_url =
792 ModuleWireBytes(wire_bytes).GetNameOrNull(debug_symbols.external_url);
793 MaybeHandle<String> src_map_str = isolate->factory()->NewStringFromUtf8(
794 external_url, AllocationType::kOld);
795 script->set_source_mapping_url(*src_map_str.ToHandleChecked());
796 }
797
798 // Use the given shared {NativeModule}, but increase its reference count by
799 // allocating a new {Managed<T>} that the {Script} references.
800 size_t code_size_estimate = native_module->committed_code_space();
801 size_t memory_estimate =
802 code_size_estimate +
803 wasm::WasmCodeManager::EstimateNativeModuleMetaDataSize(module);
804 Handle<Managed<wasm::NativeModule>> managed_native_module =
805 Managed<wasm::NativeModule>::FromSharedPtr(isolate, memory_estimate,
806 std::move(native_module));
807 script->set_wasm_managed_native_module(*managed_native_module);
808 script->set_wasm_breakpoint_infos(ReadOnlyRoots(isolate).empty_fixed_array());
809 script->set_wasm_weak_instance_list(
810 ReadOnlyRoots(isolate).empty_weak_array_list());
811 return script;
812}

940 // bias samples towards apps with high memory pressure. We should switch to
941 // using sampling based on regular intervals independent of the GC.
942 auto callback = [](v8::Isolate* v8_isolate, v8::GCType type,
943 v8::GCCallbackFlags flags, void* data) {

950 native_module->SampleCodeSize(counters, NativeModule::kSampling);

1140void WasmEngine::FreeNativeModule(NativeModule* native_module) {
1141 base::MutexGuard guard(&mutex_);
1142 auto it = native_modules_.find(native_module);
1143 DCHECK_NE(native_modules_.end(), it)((void) 0);
1144 for (Isolate* isolate : it->second->isolates) {
1145 DCHECK_EQ(1, isolates_.count(isolate))((void) 0);
1146 IsolateInfo* info = isolates_[isolate].get();
1147 DCHECK_EQ(1, info->native_modules.count(native_module))((void) 0);
1148 info->native_modules.erase(native_module);
1149 info->scripts.erase(native_module);
1150 // If there are {WasmCode} objects of the deleted {NativeModule}
1151 // outstanding to be logged in this isolate, remove them. Decrementing the
1152 // ref count is not needed, since the {NativeModule} dies anyway.
1153 size_t remaining = info->code_to_log.size();
1154 if (remaining > 0) {
1155 for (size_t i = 0; i < remaining; ++i) {
1156 while (i < remaining &&
1157 info->code_to_log[i]->native_module() == native_module) {
1158 // Move the last remaining item to this slot (this can be the same
1159 // as {i}, which is OK).
1160 info->code_to_log[i] = info->code_to_log[--remaining];
1161 }
1162 }
1163 info->code_to_log.resize(remaining);
1164 }
1165 }
1166 // If there is a GC running which has references to code contained in the
1167 // deleted {NativeModule}, remove those references.
1168 if (current_gc_info_) {
1169 for (auto it = current_gc_info_->dead_code.begin(),
1170 end = current_gc_info_->dead_code.end();
1171 it != end;) {
1172 if ((*it)->native_module() == native_module) {
1173 it = current_gc_info_->dead_code.erase(it);
1174 } else {
1175 ++it;
1176 }
1177 }
1178 TRACE_CODE_GC("Native module %p died, reducing dead code objects to %zu.\n",
1179 native_module, current_gc_info_->dead_code.size());
1180 }
1181 native_module_cache_.Erase(native_module);
1182 native_modules_.erase(it);
1183}

1250bool WasmEngine::AddPotentiallyDeadCode(WasmCode* code) {
1251 base::MutexGuard guard(&mutex_);
1252 auto it = native_modules_.find(code->native_module());

1257 if (!added.second) return false; // An entry already existed.
1258 new_potentially_dead_code_size_ += code->instructions().size();

1260 // Trigger a GC if 64kB plus 10% of committed code are potentially dead.
1261 size_t dead_code_limit =
1262 FLAG_stress_wasm_code_gc
1263 ? 0
1264 : 64 * KB + code_manager_.committed_code_space() / 10;
1265 if (new_potentially_dead_code_size_ > dead_code_limit) {
1266 bool inc_gc_count =
1267 info->num_code_gcs_triggered < std::numeric_limits<int8_t>::max();

1269 if (inc_gc_count) ++info->num_code_gcs_triggered;
1270 TRACE_CODE_GC(
1271 "Triggering GC (potentially dead: %zu bytes; limit: %zu bytes).\n",
1272 new_potentially_dead_code_size_, dead_code_limit);
1273 TriggerGC(info->num_code_gcs_triggered);

1283 }

1348void WasmEngine::TriggerGC(int8_t gc_sequence_index) {
1349 DCHECK(!mutex_.TryLock())((void) 0);
1350 DCHECK_NULL(current_gc_info_)((void) 0);
1351 DCHECK(FLAG_wasm_code_gc)((void) 0);
1352 new_potentially_dead_code_size_ = 0;
1353 current_gc_info_.reset(new CurrentGCInfo(gc_sequence_index));
1354 // Add all potentially dead code to this GC, and trigger a GC task in each
1355 // isolate.
1356 for (auto& entry : native_modules_) {
1357 NativeModuleInfo* info = entry.second.get();
1358 if (info->potentially_dead_code.empty()) continue;
1359 for (auto* isolate : native_modules_[entry.first]->isolates) {
1360 auto& gc_task = current_gc_info_->outstanding_isolates[isolate];
1361 if (!gc_task) {
1362 auto new_task = std::make_unique<WasmGCForegroundTask>(isolate);
1363 gc_task = new_task.get();
1364 DCHECK_EQ(1, isolates_.count(isolate))((void) 0);
1365 isolates_[isolate]->foreground_task_runner->PostTask(
1366 std::move(new_task));
1367 }
1368 isolate->stack_guard()->RequestWasmCodeGC();
1369 }
1370 for (WasmCode* code : info->potentially_dead_code) {
1371 current_gc_info_->dead_code.insert(code);
1372 }
1373 }
1374 TRACE_CODE_GC(
1375 "Starting GC. Total number of potentially dead code objects: %zu\n",
1376 current_gc_info_->dead_code.size());
1377 // Ensure that there are outstanding isolates that will eventually finish this
1378 // GC. If there are no outstanding isolates, we finish the GC immediately.
1379 PotentiallyFinishCurrentGC();
1380 DCHECK(current_gc_info_ == nullptr ||((void) 0)
1381 !current_gc_info_->outstanding_isolates.empty())((void) 0);
1382}

1390void WasmEngine::PotentiallyFinishCurrentGC() {
1391 DCHECK(!mutex_.TryLock())((void) 0);
1392 TRACE_CODE_GC(
1393 "Remaining dead code objects: %zu; outstanding isolates: %zu.\n",
1394 current_gc_info_->dead_code.size(),
1395 current_gc_info_->outstanding_isolates.size());
1396
1397 // If there are more outstanding isolates, return immediately.
1398 if (!current_gc_info_->outstanding_isolates.empty()) return;
1399
1400 // All remaining code in {current_gc_info->dead_code} is really dead.
1401 // Move it from the set of potentially dead code to the set of dead code,
1402 // and decrement its ref count.
1403 size_t num_freed = 0;
1404 DeadCodeMap dead_code;
1405 for (WasmCode* code : current_gc_info_->dead_code) {
1406 DCHECK_EQ(1, native_modules_.count(code->native_module()))((void) 0);
1407 auto* native_module_info = native_modules_[code->native_module()].get();
1408 DCHECK_EQ(1, native_module_info->potentially_dead_code.count(code))((void) 0);
1409 native_module_info->potentially_dead_code.erase(code);
1410 DCHECK_EQ(0, native_module_info->dead_code.count(code))((void) 0);
1411 native_module_info->dead_code.insert(code);
1412 if (code->DecRefOnDeadCode()) {
1413 dead_code[code->native_module()].push_back(code);
1414 ++num_freed;
1415 }
1416 }
1417
1418 FreeDeadCodeLocked(dead_code);
1419
1420 TRACE_CODE_GC("Found %zu dead code objects, freed %zu.\n",
1421 current_gc_info_->dead_code.size(), num_freed);
1422 USE(num_freed)do { ::v8::base::Use unused_tmp_array_for_use_macro[]{num_freed
}; (void)unused_tmp_array_for_use_macro; } while (false);
1423
1424 int8_t next_gc_sequence_index = current_gc_info_->next_gc_sequence_index;
1425 current_gc_info_.reset();
1426 if (next_gc_sequence_index != 0) TriggerGC(next_gc_sequence_index);
1427}
