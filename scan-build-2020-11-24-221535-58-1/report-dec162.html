322int WasmTableObject::Grow(Isolate* isolate, Handle<WasmTableObject> table,
323 uint32_t count, Handle<Object> init_value) {
324 uint32_t old_size = table->current_length();
325 if (count == 0) return old_size; // Degenerate case: nothing to do.
326
327 // Check if growing by {count} is valid.
328 uint32_t max_size;
329 if (!table->maximum_length().ToUint32(&max_size)) {
330 max_size = FLAG_wasm_max_table_size;
331 }
332 max_size = std::min(max_size, FLAG_wasm_max_table_size);
333 DCHECK_LE(old_size, max_size)((void) 0);
334 if (max_size - old_size < count) return -1;
335
336 uint32_t new_size = old_size + count;
337 // Even with 2x over-allocation, there should not be an integer overflow.
338 STATIC_ASSERT(wasm::kV8MaxWasmTableSize <= kMaxInt / 2)static_assert(wasm::kV8MaxWasmTableSize <= kMaxInt / 2, "wasm::kV8MaxWasmTableSize <= kMaxInt / 2"
);
339 DCHECK_GE(kMaxInt, new_size)((void) 0);
340 int old_capacity = table->entries().length();
341 if (new_size > static_cast<uint32_t>(old_capacity)) {
342 int grow = static_cast<int>(new_size) - old_capacity;
343 // Grow at least by the old capacity, to implement exponential growing.
344 grow = std::max(grow, old_capacity);
345 // Never grow larger than the max size.
346 grow = std::min(grow, static_cast<int>(max_size - old_capacity));
347 auto new_store = isolate->factory()->CopyFixedArrayAndGrow(
348 handle(table->entries(), isolate), grow);
349 table->set_entries(*new_store, WriteBarrierMode::UPDATE_WRITE_BARRIER);
350 }
351 table->set_current_length(new_size);
352
353 Handle<FixedArray> dispatch_tables(table->dispatch_tables(), isolate);
354 DCHECK_EQ(0, dispatch_tables->length() % kDispatchTableNumElements)((void) 0);
355 // Tables are stored in the instance object, no code patching is
356 // necessary. We simply have to grow the raw tables in each instance
357 // that has imported this table.
358
359 // TODO(titzer): replace the dispatch table with a weak list of all
360 // the instances that import a given table.
361 for (int i = 0; i < dispatch_tables->length();
362 i += kDispatchTableNumElements) {
363 int table_index =
364 Smi::cast(dispatch_tables->get(i + kDispatchTableIndexOffset)).value();
365
366 Handle<WasmInstanceObject> instance(
367 WasmInstanceObject::cast(dispatch_tables->get(i)), isolate);
368
369 DCHECK_EQ(old_size, WasmInstanceObject::IndirectFunctionTableSize(((void) 0)
370 isolate, instance, table_index))((void) 0);
371 WasmInstanceObject::EnsureIndirectFunctionTableWithMinimumSize(
372 instance, table_index, new_size);
373 }
374
375 for (uint32_t entry = old_size; entry < new_size; ++entry) {
376 WasmTableObject::Set(isolate, table, entry, init_value);
377 }
378 return old_size;
379}

403void WasmTableObject::Set(Isolate* isolate, Handle<WasmTableObject> table,
404 uint32_t index, Handle<Object> entry) {
405 // Callers need to perform bounds checks, type check, and error handling.
406 DCHECK(IsInBounds(isolate, table, index))((void) 0);
407 DCHECK(IsValidElement(isolate, table, entry))((void) 0);
408
409 Handle<FixedArray> entries(table->entries(), isolate);
410 // The FixedArray is addressed with int's.
411 int entry_index = static_cast<int>(index);
412 if (table->type().is_reference_to(wasm::HeapType::kExtern) ||
413 table->type().is_reference_to(wasm::HeapType::kExn)) {
414 entries->set(entry_index, *entry);
415 return;
416 }
417
418 if (entry->IsNull(isolate)) {
419 ClearDispatchTables(isolate, table, entry_index); // Degenerate case.
420 entries->set(entry_index, ReadOnlyRoots(isolate).null_value());
421 return;
422 }
423
424 if (WasmExportedFunction::IsWasmExportedFunction(*entry)) {
425 auto exported_function = Handle<WasmExportedFunction>::cast(entry);
426 Handle<WasmInstanceObject> target_instance(exported_function->instance(),
427 isolate);
428 int func_index = exported_function->function_index();
429 auto* wasm_function = &target_instance->module()->functions[func_index];
430 DCHECK_NOT_NULL(wasm_function)((void) 0);
431 DCHECK_NOT_NULL(wasm_function->sig)((void) 0);
432 UpdateDispatchTables(isolate, table, entry_index, wasm_function->sig,
433 target_instance, func_index);
434 } else if (WasmJSFunction::IsWasmJSFunction(*entry)) {
435 UpdateDispatchTables(isolate, table, entry_index,
436 Handle<WasmJSFunction>::cast(entry));
437 } else {
438 DCHECK(WasmCapiFunction::IsWasmCapiFunction(*entry))((void) 0);
439 UpdateDispatchTables(isolate, table, entry_index,
440 Handle<WasmCapiFunction>::cast(entry));
441 }
442 entries->set(entry_index, *entry);
443}

445Handle<Object> WasmTableObject::Get(Isolate* isolate,
446 Handle<WasmTableObject> table,
447 uint32_t index) {
448 Handle<FixedArray> entries(table->entries(), isolate);
449 // Callers need to perform bounds checks and error handling.
450 DCHECK(IsInBounds(isolate, table, index))((void) 0);
451
452 // The FixedArray is addressed with int's.
453 int entry_index = static_cast<int>(index);
454
455 Handle<Object> entry(entries->get(entry_index), isolate);
456
457 // First we handle the easy externref and exnref table case.
458 if (table->type().is_reference_to(wasm::HeapType::kExtern) ||
459 table->type().is_reference_to(wasm::HeapType::kExn)) {
460 return entry;
461 }
462
463 // Now we handle the funcref case.
464 if (WasmExportedFunction::IsWasmExportedFunction(*entry) ||
465 WasmJSFunction::IsWasmJSFunction(*entry) ||
466 WasmCapiFunction::IsWasmCapiFunction(*entry)) {
467 return entry;
468 }
469
470 if (entry->IsNull(isolate)) {
471 return entry;
472 }
473
474 // {entry} is not a valid entry in the table. It has to be a placeholder
475 // for lazy initialization.
476 Handle<Tuple2> tuple = Handle<Tuple2>::cast(entry);
477 auto instance = handle(WasmInstanceObject::cast(tuple->value1()), isolate);
478 int function_index = Smi::cast(tuple->value2()).value();
479
480 // Check if we already compiled a wrapper for the function but did not store
481 // it in the table slot yet.
482 entry = WasmInstanceObject::GetOrCreateWasmExternalFunction(isolate, instance,
483 function_index);
484 entries->set(entry_index, *entry);
485 return entry;
486}

548void WasmTableObject::UpdateDispatchTables(
549 Isolate* isolate, Handle<WasmTableObject> table, int entry_index,
550 Handle<WasmCapiFunction> capi_function) {
551 // We simply need to update the IFTs for each instance that imports
552 // this table.
553 Handle<FixedArray> dispatch_tables(table->dispatch_tables(), isolate);
554 DCHECK_EQ(0, dispatch_tables->length() % kDispatchTableNumElements)((void) 0);
555
556 // Reconstruct signature.
557 // TODO(jkummerow): Unify with "SignatureHelper" in c-api.cc.
558 PodArray<wasm::ValueType> serialized_sig =
559 capi_function->GetSerializedSignature();
560 int total_count = serialized_sig.length() - 1;
561 std::unique_ptr<wasm::ValueType[]> reps(new wasm::ValueType[total_count]);
562 int result_count;
563 static const wasm::ValueType kMarker = wasm::kWasmStmt;
564 for (int i = 0, j = 0; i <= total_count; i++) {
565 if (serialized_sig.get(i) == kMarker) {
566 result_count = i;
567 continue;
568 }
569 reps[j++] = serialized_sig.get(i);
570 }
571 int param_count = total_count - result_count;
572 wasm::FunctionSig sig(result_count, param_count, reps.get());
573
574 for (int i = 0; i < dispatch_tables->length();
575 i += kDispatchTableNumElements) {
576 int table_index =
577 Smi::cast(dispatch_tables->get(i + kDispatchTableIndexOffset)).value();
578 Handle<WasmInstanceObject> instance(
579 WasmInstanceObject::cast(
580 dispatch_tables->get(i + kDispatchTableInstanceOffset)),
581 isolate);
582 // TODO(jkummerow): Find a way to avoid recompiling wrappers.
583 wasm::NativeModule* native_module =
584 instance->module_object().native_module();
585 Address host_address = capi_function->GetHostCallTarget();
586 wasm::WasmCodeRefScope code_ref_scope;
587 wasm::WasmCode* wasm_code = compiler::CompileWasmCapiCallWrapper(
588 isolate->wasm_engine(), native_module, &sig, host_address);
589 isolate->counters()->wasm_generated_code_size()->Increment(
590 wasm_code->instructions().length());
591 isolate->counters()->wasm_reloc_size()->Increment(
592 wasm_code->reloc_info().length());
593 Handle<Tuple2> tuple = isolate->factory()->NewTuple2(
594 instance, capi_function, AllocationType::kOld);
595 // Note that {SignatureMap::Find} may return {-1} if the signature is
596 // not found; it will simply never match any check.
597 auto sig_id = instance->module()->signature_map.Find(sig);
598 IndirectFunctionTableEntry(instance, table_index, entry_index)
599 .Set(sig_id, wasm_code->instruction_start(), *tuple);
600 }

1431 // Note that {SignatureMap::Find} may return {-1} if the signature is
1432 // not found; it will simply never match any check.
1433 Zone zone(isolate->allocator(), ZONE_NAME__func__);
1434 const wasm::FunctionSig* sig = js_function->GetSignature(&zone);
1435 auto sig_id = instance->module()->signature_map.Find(*sig);
1436

1438 Handle<JSReceiver> callable(js_function->GetCallable(), isolate);
1439 wasm::WasmCodeRefScope code_ref_scope;
1440 Address call_target = kNullAddress;
1441 if (sig_id >= 0) {
1442 wasm::NativeModule* native_module =
1443 instance->module_object().native_module();
1444 // TODO(wasm): Cache and reuse wrapper code.
1445 const wasm::WasmFeatures enabled = native_module->enabled_features();

1447 compiler::WasmImportCallKind kind = resolved.first;
1448 callable = resolved.second; // Update to ultimate target.
1449 DCHECK_NE(compiler::WasmImportCallKind::kLinkError, kind)((void) 0);
1450 wasm::CompilationEnv env = native_module->CreateCompilationEnv();
1451 SharedFunctionInfo shared = js_function->shared();

1453 isolate->wasm_engine(), &env, kind, sig, false,

1456 result.func_index, result.code_desc, result.frame_slot_count,

1459 result.source_positions.as_vector(), GetCodeKind(result),

1462 native_module->PublishCode(std::move(wasm_code));

1464 published_code->instructions().length());
1465 isolate->counters()->wasm_reloc_size()->Increment(

1468 }

1471 Handle<Tuple2> tuple =
1472 isolate->factory()->NewTuple2(instance, callable, AllocationType::kOld);
1473 IndirectFunctionTableEntry(instance, table_index, entry_index)
1474 .Set(sig_id, call_target, *tuple);
1475}

1715 const wasm::WasmException* exception) {
1716 const wasm::WasmExceptionSig* sig = exception->sig;
1717 uint32_t encoded_size = 0;
1718 for (size_t i = 0; i < sig->parameter_count(); ++i) {
1719 switch (sig->GetParam(i).kind()) {
1720 case wasm::ValueType::kI32:
1721 case wasm::ValueType::kF32:
1722 DCHECK_EQ(2, ComputeEncodedElementSize(sig->GetParam(i)))((void) 0);
1723 encoded_size += 2;
1724 break;
1725 case wasm::ValueType::kI64:
1726 case wasm::ValueType::kF64:
1727 DCHECK_EQ(4, ComputeEncodedElementSize(sig->GetParam(i)))((void) 0);
1728 encoded_size += 4;
1729 break;
1730 case wasm::ValueType::kS128:
1731 DCHECK_EQ(8, ComputeEncodedElementSize(sig->GetParam(i)))((void) 0);
1732 encoded_size += 8;
1733 break;
1734 case wasm::ValueType::kRef:
1735 case wasm::ValueType::kOptRef:
1736 encoded_size += 1;
1737 break;
1738 case wasm::ValueType::kRtt:
1739 case wasm::ValueType::kStmt:
1740 case wasm::ValueType::kBottom:
1741 case wasm::ValueType::kI8:
1742 case wasm::ValueType::kI16:
1743 UNREACHABLE()V8_Fatal("unreachable code");

1801Handle<WasmExportedFunction> WasmExportedFunction::New(
1802 Isolate* isolate, Handle<WasmInstanceObject> instance, int func_index,
1803 int arity, Handle<Code> export_wrapper) {
1804 DCHECK(((void) 0)
1805 CodeKind::JS_TO_WASM_FUNCTION == export_wrapper->kind() ||((void) 0)
1806 (export_wrapper->is_builtin() &&((void) 0)
1807 export_wrapper->builtin_index() == Builtins::kGenericJSToWasmWrapper))((void) 0);
1808 int num_imported_functions = instance->module()->num_imported_functions;
1809 int jump_table_offset = -1;
1810 if (func_index >= num_imported_functions) {
1811 uint32_t jump_table_diff =
1812 instance->module_object().native_module()->GetJumpTableOffset(
1813 func_index);
1814 DCHECK_GE(kMaxInt, jump_table_diff)((void) 0);
1815 jump_table_offset = static_cast<int>(jump_table_diff);
1816 }
1817 Handle<WasmExportedFunctionData> function_data =
1818 Handle<WasmExportedFunctionData>::cast(isolate->factory()->NewStruct(
1819 WASM_EXPORTED_FUNCTION_DATA_TYPE, AllocationType::kOld));
1820 function_data->set_wrapper_code(*export_wrapper);
1821 function_data->set_instance(*instance);
1822 function_data->set_jump_table_offset(jump_table_offset);
1823 function_data->set_function_index(func_index);
1824 function_data->set_c_wrapper_code(Smi::zero(), SKIP_WRITE_BARRIER);
1825 function_data->set_wasm_call_target(Smi::zero(), SKIP_WRITE_BARRIER);
1826 function_data->set_packed_args_size(0);
1827 const wasm::FunctionSig* sig = instance->module()->functions[func_index].sig;
1828 sig->parameters().empty() ? function_data->set_signature_type(0)
1829 : function_data->set_signature_type(1);
1830
1831 MaybeHandle<String> maybe_name;
1832 bool is_asm_js_module = instance->module_object().is_asm_js();
1833 if (is_asm_js_module) {
1834 // We can use the function name only for asm.js. For WebAssembly, the
1835 // function name is specified as the function_index.toString().
1836 maybe_name = WasmModuleObject::GetFunctionNameOrNull(
1837 isolate, handle(instance->module_object(), isolate), func_index);
1838 }
1839 Handle<String> name;
1840 if (!maybe_name.ToHandle(&name)) {
1841 EmbeddedVector<char, 16> buffer;
1842 int length = SNPrintF(buffer, "%d", func_index);
1843 name = isolate->factory()
1844 ->NewStringFromOneByte(
1845 Vector<uint8_t>::cast(buffer.SubVector(0, length)))
1846 .ToHandleChecked();
1847 }
1848 Handle<Map> function_map;
1849 switch (instance->module()->origin) {
1850 case wasm::kWasmOrigin:
1851 if (instance->module_object()
1852 .native_module()
1853 ->enabled_features()
1854 .has_gc()) {
1855 uint32_t sig_index =
1856 instance->module()->functions[func_index].sig_index;
1857 function_map = handle(
1858 Map::cast(instance->managed_object_maps().get(sig_index)), isolate);
1859 } else {
1860 function_map = isolate->wasm_exported_function_map();
1861 }
1862 break;
1863 case wasm::kAsmJsSloppyOrigin:
1864 function_map = isolate->sloppy_function_map();
1865 break;
1866 case wasm::kAsmJsStrictOrigin:
1867 function_map = isolate->strict_function_map();
1868 break;
1869 }
1870 NewFunctionArgs args =
1871 NewFunctionArgs::ForWasm(name, function_data, function_map);
1872 Handle<JSFunction> js_function = isolate->factory()->NewFunction(args);
1873 // According to the spec, exported functions should not have a [[Construct]]
1874 // method. This does not apply to functions exported from asm.js however.
1875 DCHECK_EQ(is_asm_js_module, js_function->IsConstructor())((void) 0);
1876 js_function->shared().set_length(arity);
1877 js_function->shared().set_internal_formal_parameter_count(arity);
1878 js_function->shared().set_script(instance->module_object().script());
1879 return Handle<WasmExportedFunction>::cast(js_function);
1880}

1954bool WasmJSFunction::MatchesSignature(const wasm::FunctionSig* sig) {
1955 DCHECK_LE(sig->all().size(), kMaxInt)((void) 0);
1956 int sig_size = static_cast<int>(sig->all().size());

1958 int parameter_count = static_cast<int>(sig->parameter_count());

1960 if (return_count != function_data.serialized_return_count() ||
1961 parameter_count != function_data.serialized_parameter_count()) {

1964 if (sig_size == 0) return true; // Prevent undefined behavior.
1965 const wasm::ValueType* expected = sig->all().begin();
1966 return function_data.serialized_signature().matches(expected, sig_size);
