165Address FindNewPC(WasmFrame* frame, WasmCode* wasm_code, int byte_offset,
166 ReturnLocation return_location) {
167 Vector<const uint8_t> new_pos_table = wasm_code->source_positions();
168
169 DCHECK_LE(0, byte_offset)((void) 0);
170
171 // Find the size of the call instruction by computing the distance from the
172 // source position entry to the return address.
173 WasmCode* old_code = frame->wasm_code();
174 int pc_offset = static_cast<int>(frame->pc() - old_code->instruction_start());
175 Vector<const uint8_t> old_pos_table = old_code->source_positions();
176 SourcePositionTableIterator old_it(old_pos_table);
177 int call_offset = -1;
178 while (!old_it.done() && old_it.code_offset() < pc_offset) {
179 call_offset = old_it.code_offset();
180 old_it.Advance();
181 }
182 DCHECK_LE(0, call_offset)((void) 0);
183 int call_instruction_size = pc_offset - call_offset;
184
185 // If {return_location == kAfterBreakpoint} we search for the first code
186 // offset which is marked as instruction (i.e. not the breakpoint).
187 // If {return_location == kAfterWasmCall} we return the last code offset
188 // associated with the byte offset.
189 SourcePositionTableIterator it(new_pos_table);
190 while (!it.done() && it.source_position().ScriptOffset() != byte_offset) {
191 it.Advance();
192 }
193 if (return_location == kAfterBreakpoint) {
194 while (!it.is_statement()) it.Advance();
195 DCHECK_EQ(byte_offset, it.source_position().ScriptOffset())((void) 0);
196 return wasm_code->instruction_start() + it.code_offset() +
197 call_instruction_size;
198 }
199
200 DCHECK_EQ(kAfterWasmCall, return_location)((void) 0);
201 int code_offset;
202 do {
203 code_offset = it.code_offset();
204 it.Advance();
205 } while (!it.done() && it.source_position().ScriptOffset() == byte_offset);
206 return wasm_code->instruction_start() + code_offset + call_instruction_size;
207}

441 void SetBreakpoint(int func_index, int offset, Isolate* isolate) {
442 // Put the code ref scope outside of the mutex, so we don't unnecessarily
443 // hold the mutex while freeing code.
444 WasmCodeRefScope wasm_code_ref_scope;
445
446 // Hold the mutex while modifying breakpoints, to ensure consistency when
447 // multiple isolates set/remove breakpoints at the same time.
448 base::MutexGuard guard(&mutex_);
449
450 // offset == 0 indicates flooding and should not happen here.
451 DCHECK_NE(0, offset)((void) 0);
452
453 // Get the set of previously set breakpoints, to check later whether a new
454 // breakpoint was actually added.
455 std::vector<int> all_breakpoints = FindAllBreakpoints(func_index);
456
457 auto& isolate_data = per_isolate_data_[isolate];
458 std::vector<int>& breakpoints =
459 isolate_data.breakpoints_per_function[func_index];
460 auto insertion_point =
461 std::lower_bound(breakpoints.begin(), breakpoints.end(), offset);
462 if (insertion_point != breakpoints.end() && *insertion_point == offset) {
463 // The breakpoint is already set for this isolate.
464 return;
465 }
466 breakpoints.insert(insertion_point, offset);
467
468 DCHECK(std::is_sorted(all_breakpoints.begin(), all_breakpoints.end()))((void) 0);
469 // Find the insertion position within {all_breakpoints}.
470 insertion_point = std::lower_bound(all_breakpoints.begin(),
471 all_breakpoints.end(), offset);
472 bool breakpoint_exists =
473 insertion_point != all_breakpoints.end() && *insertion_point == offset;
474 // If the breakpoint was already set before, then we can just reuse the old
475 // code. Otherwise, recompile it. In any case, rewrite this isolate's stack
476 // to make sure that it uses up-to-date code containing the breakpoint.
477 WasmCode* new_code;
478 if (breakpoint_exists) {
479 new_code = native_module_->GetCode(func_index);
480 } else {
481 all_breakpoints.insert(insertion_point, offset);
482 new_code = RecompileLiftoffWithBreakpoints(func_index,

519 void PrepareStep(Isolate* isolate, StackFrameId break_frame_id) {
520 StackTraceFrameIterator it(isolate, break_frame_id);
521 DCHECK(!it.done())((void) 0);
522 DCHECK(it.frame()->is_wasm())((void) 0);
523 WasmFrame* frame = WasmFrame::cast(it.frame());
524 StepAction step_action = isolate->debug()->last_step_action();
525
526 // If we are flooding the top frame, the return location is after a
527 // breakpoints. Otherwise, it's after a call.
528 ReturnLocation return_location = kAfterBreakpoint;
529
530 // If we are at a return instruction, then any stepping action is equivalent
531 // to StepOut, and we need to flood the parent function.
532 if (IsAtReturn(frame) || step_action == StepOut) {
533 it.Advance();
534 if (it.done() || !it.frame()->is_wasm()) return;
535 frame = WasmFrame::cast(it.frame());
536 return_location = kAfterWasmCall;
537 }
538
539 FloodWithBreakpoints(frame, return_location);

541 base::MutexGuard guard(&mutex_);
542 per_isolate_data_[isolate].stepping_frame = frame->id();
543 }

560 void RemoveBreakpoint(int func_index, int position, Isolate* isolate) {
561 // TODO(thibaudm): Cannot remove the breakpoint we are currently paused at,
562 // because the new code would be missing the call instruction and the
563 // corresponding source position that we rely on for OSR.
564 StackTraceFrameIterator it(isolate);
565 if (!it.done() && it.is_wasm()) {
566 WasmFrame* frame = WasmFrame::cast(it.frame());
567 if (static_cast<int32_t>(frame->function_index()) == func_index &&
568 frame->position() == position) {
569 return;
570 }
571 }
572
573 // Put the code ref scope outside of the mutex, so we don't unnecessarily
574 // hold the mutex while freeing code.
575 WasmCodeRefScope wasm_code_ref_scope;
576
577 // Hold the mutex while modifying breakpoints, to ensure consistency when
578 // multiple isolates set/remove breakpoints at the same time.
579 base::MutexGuard guard(&mutex_);
580
581 const auto& function = native_module_->module()->functions[func_index];
582 int offset = position - function.code.offset();
583
584 auto& isolate_data = per_isolate_data_[isolate];
585 std::vector<int>& breakpoints =
586 isolate_data.breakpoints_per_function[func_index];
587 DCHECK_LT(0, offset)((void) 0);
588 auto insertion_point =
589 std::lower_bound(breakpoints.begin(), breakpoints.end(), offset);
590 if (insertion_point == breakpoints.end()) return;
591 if (*insertion_point != offset) return;
592 breakpoints.erase(insertion_point);
593
594 std::vector<int> remaining = FindAllBreakpoints(func_index);
595 // If the breakpoint is still set in another isolate, don't remove it.
596 DCHECK(std::is_sorted(remaining.begin(), remaining.end()))((void) 0);
597 if (std::binary_search(remaining.begin(), remaining.end(), offset)) return;
598 UpdateBreakpoints(func_index, VectorOf(remaining), isolate,
599 isolate_data.stepping_frame);
600 }
