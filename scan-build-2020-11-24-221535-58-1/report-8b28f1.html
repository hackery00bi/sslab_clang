132 for (TSlot slot = start; slot < end; ++slot) {

143 HeapObject target) {
144 static_assert(
145 std::is_same<THeapObjectSlot, FullHeapObjectSlot>::value ||
146 std::is_same<THeapObjectSlot, HeapObjectSlot>::value,
147 "Only FullHeapObjectSlot and HeapObjectSlot are expected here");
148 scavenger_->PageMemoryFence(MaybeObject::FromObject(target));
149
150 if (Heap::InFromPage(target)) {
151 SlotCallbackResult result = scavenger_->ScavengeObject(slot, target);
152 bool success = (*slot)->GetHeapObject(&target);
153 USE(success)do { ::v8::base::Use unused_tmp_array_for_use_macro[]{success
}; (void)unused_tmp_array_for_use_macro; } while (false);
154 DCHECK(success)((void) 0);

157 SLOW_DCHECK(target.IsHeapObject())((void)0);
158 MemoryChunk* chunk = MemoryChunk::FromHeapObject(host);
159
160 // Sweeper is stopped during scavenge, so we can directly
161 // insert into its remembered set here.
162 if (chunk->sweeping_slot_set()) {
163 RememberedSetSweeping::Insert<AccessMode::ATOMIC>(chunk,
164 slot.address());
165 } else {
166 RememberedSet<OLD_TO_NEW>::Insert<AccessMode::ATOMIC>(chunk,
167 slot.address());
168 }
169 }
170 SLOW_DCHECK(!MarkCompactCollector::IsOnEvacuationCandidate(((void)0)
171 HeapObject::cast(target)))((void)0);
172 } else if (record_slots_ && MarkCompactCollector::IsOnEvacuationCandidate(

174 // We should never try to record off-heap slots.
175 DCHECK((std::is_same<THeapObjectSlot, HeapObjectSlot>::value))((void) 0);
176 // We cannot call MarkCompactCollector::RecordSlot because that checks
177 // that the host page is not in young generation, which does not hold

515void Scavenger::IterateAndScavengePromotedObject(HeapObject target, Map map,
516 int size) {
517 // We are not collecting slots on new space objects during mutation thus we
518 // have to scan for pointers to evacuation candidates when we promote
519 // objects. But we should not record any slots in non-black objects. Grey
520 // object's slots would be rescanned. White object might not survive until
521 // the end of collection it would be a violation of the invariant to record
522 // its slots.
523 const bool record_slots =
524 is_compacting_ &&
525 heap()->incremental_marking()->atomic_marking_state()->IsBlack(target);
526
527 IterateAndScavengePromotedObjectsVisitor visitor(this, record_slots);
528 target.IterateBodyFast(map, size, &visitor);
529
530 if (map.IsJSArrayBufferMap()) {
531 DCHECK(!BasicMemoryChunk::FromHeapObject(target)->IsLargePage())((void) 0);
532 JSArrayBuffer::cast(target).YoungMarkExtensionPromoted();
533 }
534}

551void Scavenger::ScavengePage(MemoryChunk* page) {
552 CodePageMemoryModificationScope memory_modification_scope(page);
553
554 if (page->slot_set<OLD_TO_NEW, AccessMode::NON_ATOMIC>() != nullptr) {
555 InvalidatedSlotsFilter filter = InvalidatedSlotsFilter::OldToNew(page);
556 RememberedSet<OLD_TO_NEW>::IterateAndTrackEmptyBuckets(
557 page,
558 [this, &filter](MaybeObjectSlot slot) {
559 if (!filter.IsValid(slot.address())) return REMOVE_SLOT;
560 return CheckAndScavengeObject(heap_, slot);
561 },
562 empty_chunks_);
563 }
564
565 if (page->sweeping_slot_set<AccessMode::NON_ATOMIC>() != nullptr) {
566 InvalidatedSlotsFilter filter = InvalidatedSlotsFilter::OldToNew(page);
567 RememberedSetSweeping::Iterate(
568 page,
569 [this, &filter](MaybeObjectSlot slot) {
570 if (!filter.IsValid(slot.address())) return REMOVE_SLOT;
571 return CheckAndScavengeObject(heap_, slot);
572 },
573 SlotSet::KEEP_EMPTY_BUCKETS);
574 }
575
576 if (page->invalidated_slots<OLD_TO_NEW>() != nullptr) {
577 // The invalidated slots are not needed after old-to-new slots were
578 // processed.
579 page->ReleaseInvalidatedSlots<OLD_TO_NEW>();
580 }
581
582 RememberedSet<OLD_TO_NEW>::IterateTyped(
583 page, [=](SlotType type, Address addr) {
584 return UpdateTypedSlotHelper::UpdateTypedSlot(
585 heap_, type, addr, [this](FullMaybeObjectSlot slot) {
586 return CheckAndScavengeObject(heap(), slot);
587 });
588 });
589
590 AddPageToSweeperIfNecessary(page);
591}

593void Scavenger::Process(OneshotBarrier* barrier) {
594 ScavengeVisitor scavenge_visitor(this);
595
596 const bool have_barrier = barrier != nullptr;
597 bool done;
598 size_t objects = 0;
599 do {
600 done = true;
601 ObjectAndSize object_and_size;
602 while (promotion_list_.ShouldEagerlyProcessPromotionList() &&
603 copied_list_.Pop(&object_and_size)) {
604 scavenge_visitor.Visit(object_and_size.first);
605 done = false;
606 if (have_barrier && ((++objects % kInterruptThreshold) == 0)) {
607 if (!copied_list_.IsGlobalPoolEmpty()) {
608 barrier->NotifyAll();
609 }
610 }
611 }
612
613 struct PromotionListEntry entry;
614 while (promotion_list_.Pop(&entry)) {
615 HeapObject target = entry.heap_object;
616 IterateAndScavengePromotedObject(target, entry.map, entry.size);
617 done = false;
618 if (have_barrier && ((++objects % kInterruptThreshold) == 0)) {
619 if (!promotion_list_.IsGlobalPoolEmpty()) {
620 barrier->NotifyAll();
621 }
622 }
623 }
624 } while (!done);
625}

635// Clear ephemeron entries from EphemeronHashTables in new-space whenever the
636// entry has a dead new-space key.
637void ScavengerCollector::ClearYoungEphemerons(
638 EphemeronTableList* ephemeron_table_list) {
639 ephemeron_table_list->Iterate([this](EphemeronHashTable table) {
640 for (InternalIndex i : table.IterateEntries()) {
641 // Keys in EphemeronHashTables must be heap objects.
642 HeapObjectSlot key_slot(
643 table.RawFieldOfElementAt(EphemeronHashTable::EntryToIndex(i)));
644 HeapObject key = key_slot.ToHeapObject();
645 if (IsUnscavengedHeapObject(heap_, key)) {
646 table.RemoveEntry(i);
647 } else {
648 HeapObject forwarded = ForwardingAddress(key);
649 key_slot.StoreHeapObject(forwarded);
650 }
651 }
652 });
653 ephemeron_table_list->Clear();
654}
655
656// Clear ephemeron entries from EphemeronHashTables in old-space whenever the
657// entry has a dead new-space key.
658void ScavengerCollector::ClearOldEphemerons() {
659 for (auto it = heap_->ephemeron_remembered_set_.begin();
660 it != heap_->ephemeron_remembered_set_.end();) {
661 EphemeronHashTable table = it->first;
662 auto& indices = it->second;
663 for (auto iti = indices.begin(); iti != indices.end();) {
664 // Keys in EphemeronHashTables must be heap objects.
665 HeapObjectSlot key_slot(table.RawFieldOfElementAt(
666 EphemeronHashTable::EntryToIndex(InternalIndex(*iti))));
667 HeapObject key = key_slot.ToHeapObject();
668 if (IsUnscavengedHeapObject(heap_, key)) {
669 table.RemoveEntry(InternalIndex(*iti));
670 iti = indices.erase(iti);
671 } else {
672 HeapObject forwarded = ForwardingAddress(key);
673 key_slot.StoreHeapObject(forwarded);
674 if (!Heap::InYoungGeneration(forwarded)) {
675 iti = indices.erase(iti);
676 } else {
677 ++iti;
678 }
679 }
680 }
681
682 if (indices.size() == 0) {
683 it = heap_->ephemeron_remembered_set_.erase(it);
684 } else {
685 ++it;
686 }
687 }
688}

690void Scavenger::Finalize() {
691 heap()->MergeAllocationSitePretenuringFeedback(local_pretenuring_feedback_);
692 heap()->IncrementSemiSpaceCopiedObjectSize(copied_size_);
693 heap()->IncrementPromotedObjectsSize(promoted_size_);
694 collector_->MergeSurvivingNewLargeObjects(surviving_new_large_objects_);
695 allocator_.Finalize();
696 empty_chunks_.FlushToGlobal();
697 ephemeron_table_list_.FlushToGlobal();
698 for (auto it = ephemeron_remembered_set_.begin();
699 it != ephemeron_remembered_set_.end(); ++it) {
700 auto insert_result = heap()->ephemeron_remembered_set_.insert(
701 {it->first, std::unordered_set<int>()});
702 for (int entry : it->second) {
703 insert_result.first->second.insert(entry);
704 }
705 }
