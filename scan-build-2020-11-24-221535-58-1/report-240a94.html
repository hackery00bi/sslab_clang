597void CodeGenerator::GetPushCompatibleMoves(Instruction* instr,
598 PushTypeFlags push_type,
599 ZoneVector<MoveOperands*>* pushes) {
600 static constexpr int first_push_compatible_index =
601 kReturnAddressStackSlotCount;
602 pushes->clear();
603 for (int i = Instruction::FIRST_GAP_POSITION;
604 i <= Instruction::LAST_GAP_POSITION; ++i) {
605 Instruction::GapPosition inner_pos =
606 static_cast<Instruction::GapPosition>(i);
607 ParallelMove* parallel_move = instr->GetParallelMove(inner_pos);
608 if (parallel_move != nullptr) {
609 for (auto move : *parallel_move) {
610 InstructionOperand source = move->source();
611 InstructionOperand destination = move->destination();
612 // If there are any moves from slots that will be overridden by pushes,
613 // then the full gap resolver must be used since optimization with
614 // pushes don't participate in the parallel move and might clobber
615 // values needed for the gap resolve.
616 if (source.IsStackSlot() && LocationOperand::cast(source).index() >=
617 first_push_compatible_index) {
618 pushes->clear();
619 return;
620 }
621 // TODO(danno): Right now, only consider moves from the FIRST gap for
622 // pushes. Theoretically, we could extract pushes for both gaps (there
623 // are cases where this happens), but the logic for that would also have
624 // to check to make sure that non-memory inputs to the pushes from the
625 // LAST gap don't get clobbered in the FIRST gap.
626 if (i == Instruction::FIRST_GAP_POSITION) {
627 if (destination.IsStackSlot() &&
628 LocationOperand::cast(destination).index() >=
629 first_push_compatible_index) {
630 int index = LocationOperand::cast(destination).index();
631 if (IsValidPush(source, push_type)) {
632 if (index >= static_cast<int>(pushes->size())) {
633 pushes->resize(index + 1);
634 }
635 (*pushes)[index] = move;
636 }
637 }
638 }
639 }
640 }
641 }
642
643 // For now, only support a set of continuous pushes at the end of the list.
644 size_t push_count_upper_bound = pushes->size();
645 size_t push_begin = push_count_upper_bound;
646 for (auto move : base::Reversed(*pushes)) {
647 if (move == nullptr) break;
648 push_begin--;
649 }
650 size_t push_count = pushes->size() - push_begin;
651 std::copy(pushes->begin() + push_begin,
652 pushes->begin() + push_begin + push_count, pushes->begin());
653 pushes->resize(push_count);
654}

727CodeGenerator::CodeGenResult CodeGenerator::AssembleInstruction(
728 int instruction_index, const InstructionBlock* block) {
729 Instruction* instr = instructions()->InstructionAt(instruction_index);
730 if (info()->trace_turbo_json()) {
731 instr_starts_[instruction_index].gap_pc_offset = tasm()->pc_offset();
732 }
733 int first_unused_stack_slot;
734 FlagsMode mode = FlagsModeField::decode(instr->opcode());
735 if (mode != kFlags_trap) {
736 AssembleSourcePosition(instr);
737 }
738 bool adjust_stack =
739 GetSlotAboveSPBeforeTailCall(instr, &first_unused_stack_slot);
740 if (adjust_stack) AssembleTailCallBeforeGap(instr, first_unused_stack_slot);
741 AssembleGaps(instr);
742 if (adjust_stack) AssembleTailCallAfterGap(instr, first_unused_stack_slot);
743 DCHECK_IMPLIES(((void) 0)
744 block->must_deconstruct_frame(),((void) 0)
745 instr != instructions()->InstructionAt(block->last_instruction_index()) ||((void) 0)
746 instr->IsRet() || instr->IsJump())((void) 0);
747 if (instr->IsJump() && block->must_deconstruct_frame()) {
748 AssembleDeconstructFrame();
749 }
750 if (info()->trace_turbo_json()) {
751 instr_starts_[instruction_index].arch_instr_pc_offset = tasm()->pc_offset();
752 }
753 // Assemble architecture-specific code for the instruction.
754 CodeGenResult result = AssembleArchInstruction(instr);
755 if (result != kSuccess) return result;
756
757 if (info()->trace_turbo_json()) {
758 instr_starts_[instruction_index].condition_pc_offset = tasm()->pc_offset();
759 }
760
761 FlagsCondition condition = FlagsConditionField::decode(instr->opcode());
762 switch (mode) {
763 case kFlags_branch:
764 case kFlags_branch_and_poison: {
765 BranchInfo branch;
766 RpoNumber target = ComputeBranchInfo(&branch, instr);
767 if (target.IsValid()) {
768 // redundant branch.
769 if (!IsNextInAssemblyOrder(target)) {
770 AssembleArchJump(target);
771 }
772 return kSuccess;
773 }
774 // Assemble architecture-specific branch.
775 AssembleArchBranch(instr, &branch);
776 break;
777 }
778 case kFlags_deoptimize:
779 case kFlags_deoptimize_and_poison: {
780 // Assemble a conditional eager deoptimization after this instruction.
781 InstructionOperandConverter i(this, instr);
782 size_t frame_state_offset = MiscField::decode(instr->opcode());
783 DeoptimizationExit* const exit =
784 AddDeoptimizationExit(instr, frame_state_offset);
785 Label continue_label;
786 BranchInfo branch;
787 branch.condition = condition;
788 branch.true_label = exit->label();
789 branch.false_label = &continue_label;
790 branch.fallthru = true;
791 // Assemble architecture-specific branch.
792 AssembleArchDeoptBranch(instr, &branch);
793 tasm()->bind(&continue_label);
794 if (mode == kFlags_deoptimize_and_poison) {
795 AssembleBranchPoisoning(NegateFlagsCondition(branch.condition), instr);
796 }
797 break;
798 }
799 case kFlags_set: {
800 // Assemble a boolean materialization after this instruction.
801 AssembleArchBoolean(instr, condition);
802 break;
803 }
804 case kFlags_trap: {
805 AssembleArchTrap(instr, condition);
806 break;
807 }
808 case kFlags_none: {
809 break;
810 }
811 }
812
813 // TODO(jarin) We should thread the flag through rather than set it.
814 if (instr->IsCall()) {
815 ResetSpeculationPoison();
816 }
817
818 return kSuccess;
819}

1090 return; // Stub with no SharedFunctionInfo.
1091 }
1092 shared_info = info()->shared_info();
1093 }
1094
1095 const BailoutId bailout_id = descriptor->bailout_id();
1096 const int shared_info_id =
1097 DefineDeoptimizationLiteral(DeoptimizationLiteral(shared_info));
1098 const unsigned int height =
1099 static_cast<unsigned int>(descriptor->GetHeight());
1100
1101 switch (descriptor->type()) {
1102 case FrameStateType::kInterpretedFunction: {
1103 int return_offset = 0;
1104 int return_count = 0;
1105 if (!state_combine.IsOutputIgnored()) {
1106 return_offset = static_cast<int>(state_combine.GetOffsetToPokeAt());
1107 return_count = static_cast<int>(iter->instruction()->OutputCount());
1108 }
1109 translation->BeginInterpretedFrame(bailout_id, shared_info_id, height,
1110 return_offset, return_count);
1111 break;
1112 }
1113 case FrameStateType::kArgumentsAdaptor:
1114 translation->BeginArgumentsAdaptorFrame(shared_info_id, height);
1115 break;
1116 case FrameStateType::kConstructStub:
1117 DCHECK(bailout_id.IsValidForConstructStub())((void) 0);
1118 translation->BeginConstructStubFrame(bailout_id, shared_info_id, height);
1119 break;
1120 case FrameStateType::kBuiltinContinuation: {
1121 translation->BeginBuiltinContinuationFrame(bailout_id, shared_info_id,
1122 height);
1123 break;
1124 }
1125 case FrameStateType::kJavaScriptBuiltinContinuation: {
1126 translation->BeginJavaScriptBuiltinContinuationFrame(
1127 bailout_id, shared_info_id, height);
1128 break;
1129 }
1130 case FrameStateType::kJavaScriptBuiltinContinuationWithCatch: {
1131 translation->BeginJavaScriptBuiltinContinuationWithCatchFrame(
1132 bailout_id, shared_info_id, height);
1133 break;
1134 }
1135 }
1136
1137 TranslateFrameStateDescriptorOperands(descriptor, iter, translation);
1138}

1140DeoptimizationExit* CodeGenerator::BuildTranslation(
1141 Instruction* instr, int pc_offset, size_t frame_state_offset,
1142 OutputFrameStateCombine state_combine) {
1143 DeoptimizationEntry const& entry =
1144 GetDeoptimizationEntry(instr, frame_state_offset);
1145 FrameStateDescriptor* const descriptor = entry.descriptor();
1146 frame_state_offset++;
1147
1148 int update_feedback_count = entry.feedback().IsValid() ? 1 : 0;
1149 Translation translation(&translations_,
1150 static_cast<int>(descriptor->GetFrameCount()),
1151 static_cast<int>(descriptor->GetJSFrameCount()),
1152 update_feedback_count, zone());
1153 if (entry.feedback().IsValid()) {
1154 DeoptimizationLiteral literal =
1155 DeoptimizationLiteral(entry.feedback().vector);
1156 int literal_id = DefineDeoptimizationLiteral(literal);
1157 translation.AddUpdateFeedback(literal_id, entry.feedback().slot.ToInt());
1158 }
1159 InstructionOperandIterator iter(instr, frame_state_offset);
1160 BuildTranslationForFrameStateDescriptor(descriptor, &iter, &translation,
1161 state_combine);
1162
1163 DeoptimizationExit* const exit = zone()->New<DeoptimizationExit>(
1164 current_source_position_, descriptor->bailout_id(), translation.index(),
1165 pc_offset, entry.kind(), entry.reason());
1166
1167 if (!Deoptimizer::kSupportsFixedDeoptExitSizes) {
1168 exit->set_deoptimization_id(next_deoptimization_id_++);
1169 }
1170
1171 deoptimization_exits_.push_back(exit);
1172 return exit;
