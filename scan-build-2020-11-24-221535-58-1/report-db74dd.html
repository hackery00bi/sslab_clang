637 // Initialize the indirect tables.
638 //--------------------------------------------------------------------------
639 if (table_count > 0) {
640 InitializeIndirectFunctionTables(instance);
641 }
642
643 //--------------------------------------------------------------------------
644 // Initialize the exceptions table.
645 //--------------------------------------------------------------------------
646 if (exceptions_count > 0) {
647 InitializeExceptions(instance);
648 }
649
650 // The bulk memory proposal changes the MVP behavior here; the segments are
651 // written as if `memory.init` and `table.init` are executed directly, and
652 // not bounds checked ahead of time.
653 if (!enabled_.has_bulk_memory()) {
654 //--------------------------------------------------------------------------
655 // Check that indirect function table segments are within bounds.
656 //--------------------------------------------------------------------------
657 for (const WasmElemSegment& elem_segment : module_->elem_segments) {
658 if (elem_segment.status != WasmElemSegment::kStatusActive) continue;
659 DCHECK_LT(elem_segment.table_index, table_count)((void) 0);
660 uint32_t base = EvalUint32InitExpr(instance, elem_segment.offset);
661 // Because of imported tables, {table_size} has to come from the table
662 // object itself.
663 auto table_object = handle(WasmTableObject::cast(instance->tables().get(
664 elem_segment.table_index)),
665 isolate_);
666 uint32_t table_size = table_object->current_length();
667 if (!base::IsInBounds<uint32_t>(
668 base, static_cast<uint32_t>(elem_segment.entries.size()),
669 table_size)) {
670 thrower_->LinkError("table initializer is out of bounds");
671 return {};
672 }
673 }
674
675 //--------------------------------------------------------------------------
676 // Check that memory segments are within bounds.
677 //--------------------------------------------------------------------------
678 for (const WasmDataSegment& seg : module_->data_segments) {
679 if (!seg.active) continue;
680 uint32_t base = EvalUint32InitExpr(instance, seg.dest_addr);
681 if (!base::IsInBounds<uint64_t>(base, seg.source.length(),
682 instance->memory_size())) {
683 thrower_->LinkError("data segment is out of bounds");
684 return {};
685 }
686 }
687 }
688
689 //--------------------------------------------------------------------------
690 // Set up the exports object for the new instance.
691 //--------------------------------------------------------------------------
692 ProcessExports(instance);
693 if (thrower_->error()) return {};
694
695 //--------------------------------------------------------------------------
696 // Initialize the indirect function tables.
697 //--------------------------------------------------------------------------
698 if (table_count > 0) {
699 LoadTableSegments(instance);
700 if (thrower_->error()) return {};
701 }
702
703 //--------------------------------------------------------------------------
704 // Initialize the memory by loading data segments.
705 //--------------------------------------------------------------------------
706 if (module_->data_segments.size() > 0) {
707 LoadDataSegments(instance);
708 if (thrower_->error()) return {};
709 }
710
711 //--------------------------------------------------------------------------
712 // Create a wrapper for the start function.
713 //--------------------------------------------------------------------------
714 if (module_->start_function_index >= 0) {
715 int start_index = module_->start_function_index;
716 auto& function = module_->functions[start_index];
717 Handle<Code> wrapper_code =
718 JSToWasmWrapperCompilationUnit::CompileJSToWasmWrapper(
719 isolate_, function.sig, function.imported);
720 // TODO(clemensb): Don't generate an exported function for the start
721 // function. Use CWasmEntry instead.
722 start_function_ = WasmExportedFunction::New(
723 isolate_, instance, start_index,
724 static_cast<int>(function.sig->parameter_count()), wrapper_code);
725 }
726
727 DCHECK(!isolate_->has_pending_exception())((void) 0);
728 TRACE("Successfully built instance for module %p\n",
729 module_object_->native_module());
730 wasm_module_instantiated.success = true;
731 wasm_module_instantiated_timed_scope.Stop();
732 isolate_->metrics_recorder()->DelayMainThreadEvent(wasm_module_instantiated,
733 context_id_);
734 return instance;
735}

786// Look up an import value in the {ffi_} object specifically for linking an
787// asm.js module. This only performs non-observable lookups, which allows
788// falling back to JavaScript proper (and hence re-executing all lookups) if
789// module instantiation fails.
790MaybeHandle<Object> InstanceBuilder::LookupImportAsm(
791 uint32_t index, Handle<String> import_name) {
792 // Check that a foreign function interface object was provided.
793 if (ffi_.is_null()) {
794 return ReportLinkError("missing imports object", index, import_name);
795 }
796
797 // Perform lookup of the given {import_name} without causing any observable
798 // side-effect. We only accept accesses that resolve to data properties,
799 // which is indicated by the asm.js spec in section 7 ("Linking") as well.
800 Handle<Object> result;
801 LookupIterator::Key key(isolate_, Handle<Name>::cast(import_name));
802 LookupIterator it(isolate_, ffi_.ToHandleChecked(), key);
803 switch (it.state()) {
804 case LookupIterator::ACCESS_CHECK:
805 case LookupIterator::INTEGER_INDEXED_EXOTIC:
806 case LookupIterator::INTERCEPTOR:
807 case LookupIterator::JSPROXY:
808 case LookupIterator::ACCESSOR:
809 case LookupIterator::TRANSITION:
810 return ReportLinkError("not a data property", index, import_name);
811 case LookupIterator::NOT_FOUND:
812 // Accepting missing properties as undefined does not cause any
813 // observable difference from JavaScript semantics, we are lenient.
814 result = isolate_->factory()->undefined_value();
815 break;
816 case LookupIterator::DATA:
817 result = it.GetDataValue();
818 break;
819 }
820
821 return result;

824// Load data segments into the memory.
825void InstanceBuilder::LoadDataSegments(Handle<WasmInstanceObject> instance) {
826 Vector<const uint8_t> wire_bytes =
827 module_object_->native_module()->wire_bytes();
828 for (const WasmDataSegment& segment : module_->data_segments) {
829 uint32_t size = segment.source.length();
830
831 if (enabled_.has_bulk_memory()) {
832 // Passive segments are not copied during instantiation.
833 if (!segment.active) continue;
834
835 uint32_t dest_offset = EvalUint32InitExpr(instance, segment.dest_addr);
836 bool ok = base::ClampToBounds(
837 dest_offset, &size, static_cast<uint32_t>(instance->memory_size()));
838 if (!ok) {
839 thrower_->RuntimeError("data segment is out of bounds");
840 return;
841 }
842 // No need to copy empty segments.
843 if (size == 0) continue;
844 std::memcpy(instance->memory_start() + dest_offset,
845 wire_bytes.begin() + segment.source.offset(), size);
846 } else {
847 DCHECK(segment.active)((void) 0);
848 // Segments of size == 0 are just nops.
849 if (size == 0) continue;
850
851 uint32_t dest_offset = EvalUint32InitExpr(instance, segment.dest_addr);
852 DCHECK(base::IsInBounds<uint64_t>(dest_offset, size,((void) 0)
853 instance->memory_size()))((void) 0);
854 byte* dest = instance->memory_start() + dest_offset;
855 const byte* src = wire_bytes.begin() + segment.source.offset();
856 memcpy(dest, src, size);
857 }
858 }
859}

990bool InstanceBuilder::ProcessImportedFunction(
991 Handle<WasmInstanceObject> instance, int import_index, int func_index,
992 Handle<String> module_name, Handle<String> import_name,
993 Handle<Object> value) {
994 // Function imports must be callable.
995 if (!value->IsCallable()) {
996 ReportLinkError("function import requires a callable", import_index,
997 module_name, import_name);
998 return false;
999 }
1000 // Store any {WasmExternalFunction} callable in the instance before the call
1001 // is resolved to preserve its identity. This handles exported functions as
1002 // well as functions constructed via other means (e.g. WebAssembly.Function).
1003 if (WasmExternalFunction::IsWasmExternalFunction(*value)) {
1004 WasmInstanceObject::SetWasmExternalFunction(
1005 isolate_, instance, func_index,
1006 Handle<WasmExternalFunction>::cast(value));
1007 }
1008 auto js_receiver = Handle<JSReceiver>::cast(value);
1009 const FunctionSig* expected_sig = module_->functions[func_index].sig;
1010 auto resolved =
1011 compiler::ResolveWasmImportCall(js_receiver, expected_sig, enabled_);
1012 compiler::WasmImportCallKind kind = resolved.first;
1013 js_receiver = resolved.second;
1014 switch (kind) {
1015 case compiler::WasmImportCallKind::kLinkError:
1016 ReportLinkError("imported function does not match the expected type",
1017 import_index, module_name, import_name);
1018 return false;
1019 case compiler::WasmImportCallKind::kWasmToWasm: {
1020 // The imported function is a Wasm function from another instance.
1021 auto imported_function = Handle<WasmExportedFunction>::cast(js_receiver);
1022 Handle<WasmInstanceObject> imported_instance(
1023 imported_function->instance(), isolate_);
1024 // The import reference is the instance object itself.
1025 Address imported_target = imported_function->GetWasmCallTarget();
1026 ImportedFunctionEntry entry(instance, func_index);
1027 entry.SetWasmToWasm(*imported_instance, imported_target);
1028 break;
1029 }
1030 case compiler::WasmImportCallKind::kWasmToCapi: {
1031 NativeModule* native_module = instance->module_object().native_module();
1032 Address host_address =
1033 WasmCapiFunction::cast(*js_receiver).GetHostCallTarget();
1034 WasmCodeRefScope code_ref_scope;
1035 WasmCode* wasm_code = compiler::CompileWasmCapiCallWrapper(
1036 isolate_->wasm_engine(), native_module, expected_sig, host_address);
1037 isolate_->counters()->wasm_generated_code_size()->Increment(
1038 wasm_code->instructions().length());
1039 isolate_->counters()->wasm_reloc_size()->Increment(
1040 wasm_code->reloc_info().length());
1041
1042 ImportedFunctionEntry entry(instance, func_index);
1043 // We re-use the SetWasmToJs infrastructure because it passes the
1044 // callable to the wrapper, which we need to get the function data.
1045 entry.SetWasmToJs(isolate_, js_receiver, wasm_code);
1046 break;
1047 }
1048 default: {
1049 // The imported function is a callable.
1050
1051 int expected_arity = static_cast<int>(expected_sig->parameter_count());
1052 if (kind ==
1053 compiler::WasmImportCallKind::kJSFunctionArityMismatchSkipAdaptor) {
1054 Handle<JSFunction> function = Handle<JSFunction>::cast(js_receiver);
1055 SharedFunctionInfo shared = function->shared();
1056 expected_arity = shared.internal_formal_parameter_count();
1057 }
1058
1059 NativeModule* native_module = instance->module_object().native_module();
1060 WasmCode* wasm_code = native_module->import_wrapper_cache()->Get(
1061 kind, expected_sig, expected_arity);
1062 DCHECK_NOT_NULL(wasm_code)((void) 0);
1063 ImportedFunctionEntry entry(instance, func_index);
1064 if (wasm_code->kind() == WasmCode::kWasmToJsWrapper) {
1065 // Wasm to JS wrappers are treated specially in the import table.
1066 entry.SetWasmToJs(isolate_, js_receiver, wasm_code);
1067 } else {
1068 // Wasm math intrinsics are compiled as regular Wasm functions.
1069 DCHECK(kind >= compiler::WasmImportCallKind::kFirstMathIntrinsic &&((void) 0)
1070 kind <= compiler::WasmImportCallKind::kLastMathIntrinsic)((void) 0);
1071 entry.SetWasmToWasm(*instance, wasm_code->instruction_start());
1072 }
1073 break;
1074 }
1075 }
1076 return true;
1077}

1079bool InstanceBuilder::InitializeImportedIndirectFunctionTable(
1080 Handle<WasmInstanceObject> instance, int table_index, int import_index,
1081 Handle<WasmTableObject> table_object) {
1082 int imported_table_size = table_object->current_length();
1083 // Allocate a new dispatch table.
1084 WasmInstanceObject::EnsureIndirectFunctionTableWithMinimumSize(
1085 instance, table_index, imported_table_size);
1086 // Initialize the dispatch table with the (foreign) JS functions
1087 // that are already in the table.
1088 for (int i = 0; i < imported_table_size; ++i) {
1089 bool is_valid;
1090 bool is_null;
1091 MaybeHandle<WasmInstanceObject> maybe_target_instance;
1092 int function_index;
1093 MaybeHandle<WasmJSFunction> maybe_js_function;
1094 WasmTableObject::GetFunctionTableEntry(isolate_, table_object, i, &is_valid,
1095 &is_null, &maybe_target_instance,
1096 &function_index, &maybe_js_function);
1097 if (!is_valid) {
1098 thrower_->LinkError("table import %d[%d] is not a wasm function",
1099 import_index, i);
1100 return false;
1101 }
1102 if (is_null) continue;
1103 Handle<WasmJSFunction> js_function;
1104 if (maybe_js_function.ToHandle(&js_function)) {
1105 WasmInstanceObject::ImportWasmJSFunctionIntoTable(
1106 isolate_, instance, table_index, i, js_function);
1107 continue;
1108 }
1109
1110 Handle<WasmInstanceObject> target_instance =
1111 maybe_target_instance.ToHandleChecked();
1112 const FunctionSig* sig = target_instance->module_object()
1113 .module()
1114 ->functions[function_index]
1115 .sig;
1116
1117 // Look up the signature's canonical id. If there is no canonical
1118 // id, then the signature does not appear at all in this module,
1119 // so putting {-1} in the table will cause checks to always fail.
1120 IndirectFunctionTableEntry(instance, table_index, i)
1121 .Set(module_->signature_map.Find(*sig), target_instance,
1122 function_index);
1123 }
1124 return true;
1125}

1127bool InstanceBuilder::ProcessImportedTable(Handle<WasmInstanceObject> instance,
1128 int import_index, int table_index,
1129 Handle<String> module_name,
1130 Handle<String> import_name,
1131 Handle<Object> value) {
1132 if (!value->IsWasmTableObject()) {
1133 ReportLinkError("table import requires a WebAssembly.Table", import_index,
1134 module_name, import_name);
1135 return false;
1136 }
1137 const WasmTable& table = module_->tables[table_index];
1138
1139 auto table_object = Handle<WasmTableObject>::cast(value);
1140
1141 uint32_t imported_table_size =
1142 static_cast<uint32_t>(table_object->current_length());
1143 if (imported_table_size < table.initial_size) {
1144 thrower_->LinkError("table import %d is smaller than initial %u, got %u",
1145 import_index, table.initial_size, imported_table_size);
1146 return false;
1147 }
1148
1149 if (table.has_maximum_size) {
1150 if (table_object->maximum_length().IsUndefined(isolate_)) {
1151 thrower_->LinkError("table import %d has no maximum length, expected %u",
1152 import_index, table.maximum_size);
1153 return false;
1154 }
1155 int64_t imported_maximum_size = table_object->maximum_length().Number();
1156 if (imported_maximum_size < 0) {
1157 thrower_->LinkError("table import %d has no maximum length, expected %u",
1158 import_index, table.maximum_size);
1159 return false;
1160 }
1161 if (imported_maximum_size > table.maximum_size) {
1162 thrower_->LinkError("table import %d has a larger maximum size %" PRIx64"l" "x"
1163 " than the module's declared maximum %u",
1164 import_index, imported_maximum_size,
1165 table.maximum_size);
1166 return false;
1167 }
1168 }
1169
1170 if (table.type != table_object->type()) {
1171 ReportLinkError("imported table does not match the expected type",
1172 import_index, module_name, import_name);
1173 return false;
1174 }
1175
1176 if (table.type.is_reference_to(HeapType::kFunc) &&
1177 !InitializeImportedIndirectFunctionTable(instance, table_index,
1178 import_index, table_object)) {
1179 return false;
1180 }
1181
1182 instance->tables().set(table_index, *value);
1183 return true;
1184}

1209 imported_cur_pages);

1224 "module's declared maximum %u",

1232 module_->has_shared_memory, buffer->is_shared());

1239bool InstanceBuilder::ProcessImportedWasmGlobalObject(
1240 Handle<WasmInstanceObject> instance, int import_index,
1241 Handle<String> module_name, Handle<String> import_name,
1242 const WasmGlobal& global, Handle<WasmGlobalObject> global_object) {
1243 if (static_cast<bool>(global_object->is_mutable()) != global.mutability) {
1244 ReportLinkError("imported global does not match the expected mutability",
1245 import_index, module_name, import_name);
1246 return false;
1247 }
1248
1249 bool is_sub_type =
1250 IsSubtypeOf(global_object->type(), global.type, instance->module());
1251 bool is_same_type = global_object->type() == global.type;
1252 bool valid_type = global.mutability ? is_same_type : is_sub_type;
1253
1254 if (!valid_type) {
1255 ReportLinkError("imported global does not match the expected type",
1256 import_index, module_name, import_name);
1257 return false;
1258 }
1259 if (global.mutability) {
1260 DCHECK_LT(global.index, module_->num_imported_mutable_globals)((void) 0);
1261 Handle<Object> buffer;
1262 Address address_or_offset;
1263 if (global.type.is_reference_type()) {
1264 static_assert(sizeof(global_object->offset()) <= sizeof(Address),

1266 "the imported_mutable_globals array");

1269 // address.
1270 address_or_offset = static_cast<Address>(global_object->offset());
1271 } else {
1272 buffer = handle(global_object->untagged_buffer(), isolate_);

1274 // since the backing store of the JSArrayBuffer will not be
1275 // relocated.
1276 address_or_offset = reinterpret_cast<Address>(raw_buffer_ptr(
1277 Handle<JSArrayBuffer>::cast(buffer), global_object->offset()));
1278 }
1279 instance->imported_mutable_globals_buffers().set(global.index, *buffer);
1280 instance->imported_mutable_globals()[global.index] = address_or_offset;
1281 return true;
1282 }
1283
1284 WriteGlobalValue(global, global_object);
1285 return true;
1286}

1288bool InstanceBuilder::ProcessImportedGlobal(Handle<WasmInstanceObject> instance,
1289 int import_index, int global_index,
1290 Handle<String> module_name,
1291 Handle<String> import_name,
1292 Handle<Object> value) {
1293 // Immutable global imports are converted to numbers and written into
1294 // the {untagged_globals_} array buffer.
1295 //
1296 // Mutable global imports instead have their backing array buffers
1297 // referenced by this instance, and store the address of the imported
1298 // global in the {imported_mutable_globals_} array.
1299 const WasmGlobal& global = module_->globals[global_index];
1300
1301 // The mutable-global proposal allows importing i64 values, but only if
1302 // they are passed as a WebAssembly.Global object.
1303 //
1304 // However, the bigint proposal allows importing constant i64 values,
1305 // as non WebAssembly.Global object.
1306 if (global.type == kWasmI64 && !enabled_.has_bigint() &&
1307 !value->IsWasmGlobalObject()) {
1308 ReportLinkError("global import cannot have type i64", import_index,
1309 module_name, import_name);
1310 return false;
1311 }
1312 if (is_asmjs_module(module_)) {
1313 // Accepting {JSFunction} on top of just primitive values here is a
1314 // workaround to support legacy asm.js code with broken binding. Note
1315 // that using {NaN} (or Smi::zero()) here is what using the observable
1316 // conversion via {ToPrimitive} would produce as well.
1317 // TODO(wasm): Still observable if Function.prototype.valueOf or friends
1318 // are patched, we might need to check for that as well.
1319 if (value->IsJSFunction()) value = isolate_->factory()->nan_value();
1320 if (value->IsPrimitive() && !value->IsSymbol()) {
1321 if (global.type == kWasmI32) {
1322 value = Object::ToInt32(isolate_, value).ToHandleChecked();
1323 } else {
1324 value = Object::ToNumber(isolate_, value).ToHandleChecked();
1325 }
1326 }
1327 }
1328
1329 if (value->IsWasmGlobalObject()) {
1330 auto global_object = Handle<WasmGlobalObject>::cast(value);
1331 return ProcessImportedWasmGlobalObject(instance, import_index, module_name,
1332 import_name, global, global_object);
1333 }
1334
1335 if (global.mutability) {
1336 ReportLinkError(
1337 "imported mutable global must be a WebAssembly.Global object",
1338 import_index, module_name, import_name);
1339 return false;
1340 }
1341
1342 if (global.type.is_reference_type()) {
1343 if (global.type.is_reference_to(HeapType::kFunc)) {
1344 if (!value->IsNull(isolate_) &&
1345 !WasmExportedFunction::IsWasmExportedFunction(*value)) {
1346 ReportLinkError(
1347 "imported funcref global must be null or an exported function",
1348 import_index, module_name, import_name);
1349 return false;
1350 }
1351 }
1352 WriteGlobalExternRef(global, value);
1353 return true;
1354 }
1355
1356 if (value->IsNumber() && global.type != kWasmI64) {
1357 WriteGlobalValue(global, value->Number());
1358 return true;
1359 }
1360
1361 if (enabled_.has_bigint() && global.type == kWasmI64 && value->IsBigInt()) {
1362 WriteGlobalValue(global, BigInt::cast(*value).AsInt64());
1363 return true;
1364 }
1365
1366 ReportLinkError("global import must be a number or WebAssembly.Global object",
1367 import_index, module_name, import_name);
1368 return false;
1369}

1371void InstanceBuilder::CompileImportWrappers(
1372 Handle<WasmInstanceObject> instance) {
1373 int num_imports = static_cast<int>(module_->import_table.size());
1374 NativeModule* native_module = instance->module_object().native_module();
1375 WasmImportWrapperCache::ModificationScope cache_scope(
1376 native_module->import_wrapper_cache());
1377
1378 // Compilation is done in two steps:
1379 // 1) Insert nullptr entries in the cache for wrappers that need to be
1380 // compiled. 2) Compile wrappers in background tasks using the
1381 // ImportWrapperQueue. This way the cache won't invalidate other iterators
1382 // when inserting a new WasmCode, since the key will already be there.
1383 ImportWrapperQueue import_wrapper_queue;
1384 for (int index = 0; index < num_imports; ++index) {
1385 Handle<Object> value = sanitized_imports_[index].value;
1386 if (module_->import_table[index].kind != kExternalFunction ||
1387 !value->IsCallable()) {
1388 continue;
1389 }
1390 auto js_receiver = Handle<JSReceiver>::cast(value);
1391 uint32_t func_index = module_->import_table[index].index;
1392 const FunctionSig* sig = module_->functions[func_index].sig;
1393 auto resolved = compiler::ResolveWasmImportCall(js_receiver, sig, enabled_);
1394 compiler::WasmImportCallKind kind = resolved.first;
1395 if (kind == compiler::WasmImportCallKind::kWasmToWasm ||
1396 kind == compiler::WasmImportCallKind::kLinkError ||
1397 kind == compiler::WasmImportCallKind::kWasmToCapi) {
1398 continue;
1399 }
1400
1401 int expected_arity = static_cast<int>(sig->parameter_count());
1402 if (resolved.first ==
1403 compiler::WasmImportCallKind::kJSFunctionArityMismatchSkipAdaptor) {
1404 Handle<JSFunction> function = Handle<JSFunction>::cast(resolved.second);
1405 SharedFunctionInfo shared = function->shared();
1406 expected_arity = shared.internal_formal_parameter_count();
1407 }
1408
1409 WasmImportWrapperCache::CacheKey key(kind, sig, expected_arity);
1410 if (cache_scope[key] != nullptr) {
1411 // Cache entry already exists, no need to compile it again.
1412 continue;
1413 }
1414 import_wrapper_queue.insert(key);
1415 }
1416
1417 CancelableTaskManager task_manager;
1418 // TODO(wasm): Switch this to the Jobs API.
1419 const int max_background_tasks = GetMaxCompileConcurrency();
1420 for (int i = 0; i < max_background_tasks; ++i) {
1421 auto task = std::make_unique<CompileImportWrapperTask>(
1422 &task_manager, isolate_->wasm_engine(), isolate_->counters(),
1423 native_module, &import_wrapper_queue, &cache_scope);
1424 V8::GetCurrentPlatform()->CallOnWorkerThread(std::move(task));
1425 }
1426
1427 // Also compile in the current thread, in case there are no worker threads.
1428 while (base::Optional<WasmImportWrapperCache::CacheKey> key =
1429 import_wrapper_queue.pop()) {
1430 CompileImportWrapper(isolate_->wasm_engine(), native_module,
1431 isolate_->counters(), key->kind, key->signature,
1432 key->expected_arity, &cache_scope);
1433 }
1434 task_manager.CancelAndWait();
1435}

1437// Process the imports, including functions, tables, globals, and memory, in
1438// order, loading them from the {ffi_} object. Returns the number of imported
1439// functions.
1440int InstanceBuilder::ProcessImports(Handle<WasmInstanceObject> instance) {
1441 int num_imported_functions = 0;
1442 int num_imported_tables = 0;
1443
1444 DCHECK_EQ(module_->import_table.size(), sanitized_imports_.size())((void) 0);
1445
1446 CompileImportWrappers(instance);
1447 int num_imports = static_cast<int>(module_->import_table.size());
1448 for (int index = 0; index < num_imports; ++index) {
1449 const WasmImport& import = module_->import_table[index];
1450
1451 Handle<String> module_name = sanitized_imports_[index].module_name;
1452 Handle<String> import_name = sanitized_imports_[index].import_name;
1453 Handle<Object> value = sanitized_imports_[index].value;
1454
1455 switch (import.kind) {
1456 case kExternalFunction: {
1457 uint32_t func_index = import.index;
1458 DCHECK_EQ(num_imported_functions, func_index)((void) 0);
1459 if (!ProcessImportedFunction(instance, index, func_index, module_name,
1460 import_name, value)) {
1461 return -1;
1462 }
1463 num_imported_functions++;
1464 break;
1465 }
1466 case kExternalTable: {
1467 uint32_t table_index = import.index;
1468 DCHECK_EQ(table_index, num_imported_tables)((void) 0);
1469 if (!ProcessImportedTable(instance, index, table_index, module_name,
1470 import_name, value)) {
1471 return -1;
1472 }
1473 num_imported_tables++;
1474 break;
1475 }
1476 case kExternalMemory: {
1477 if (!ProcessImportedMemory(instance, index, module_name, import_name,
1478 value)) {
1479 return -1;
1480 }
1481 break;
1482 }
1483 case kExternalGlobal: {
1484 if (!ProcessImportedGlobal(instance, index, import.index, module_name,
1485 import_name, value)) {
1486 return -1;
1487 }
1488 break;
1489 }
1490 case kExternalException: {
1491 if (!value->IsWasmExceptionObject()) {
1492 ReportLinkError("exception import requires a WebAssembly.Exception",
1493 index, module_name, import_name);
1494 return -1;
1495 }
1496 Handle<WasmExceptionObject> imported_exception =
1497 Handle<WasmExceptionObject>::cast(value);
1498 if (!imported_exception->IsSignatureEqual(
1499 module_->exceptions[import.index].sig)) {
1500 ReportLinkError("imported exception does not match the expected type",
1501 index, module_name, import_name);
1502 return -1;
1503 }
1504 Object exception_tag = imported_exception->exception_tag();
1505 DCHECK(instance->exceptions_table().get(import.index).IsUndefined())((void) 0);
1506 instance->exceptions_table().set(import.index, exception_tag);
1507 exception_wrappers_[import.index] = imported_exception;
1508 break;
1509 }
1510 default:
1511 UNREACHABLE()V8_Fatal("unreachable code");
1512 break;
1513 }
1514 }
1515 return num_imported_functions;
1516}

1523Handle<Object> InstanceBuilder::RecursivelyEvaluateGlobalInitializer(
1524 const WasmInitExpr& init, Handle<WasmInstanceObject> instance) {
1525 switch (init.kind()) {
1526 case WasmInitExpr::kI32Const:
1527 case WasmInitExpr::kI64Const:
1528 case WasmInitExpr::kF32Const:
1529 case WasmInitExpr::kF64Const:
1530 case WasmInitExpr::kS128Const:
1531 case WasmInitExpr::kRefNullConst:
1532 case WasmInitExpr::kRefFuncConst:
1533 case WasmInitExpr::kNone:
1534 // Handled directly by {InitGlobals()}, can't occur as recursive case.
1535 UNREACHABLE()V8_Fatal("unreachable code");
1536 break;
1537 case WasmInitExpr::kGlobalGet: {
1538 // We can only get here for reference-type globals, but we don't have
1539 // enough information to DCHECK that directly.
1540 DCHECK(enabled_.has_reftypes() || enabled_.has_eh())((void) 0);
1541 uint32_t old_offset = module_->globals[init.immediate().index].offset;
1542 DCHECK(static_cast<int>(old_offset) < tagged_globals_->length())((void) 0);
1543 return handle(tagged_globals_->get(old_offset), isolate_);
1544 }
1545 case WasmInitExpr::kRttCanon: {
1546 switch (init.immediate().heap_type) {
1547 case wasm::HeapType::kEq:
1548 return isolate_->root_handle(RootIndex::kWasmRttEqrefMap);
1549 case wasm::HeapType::kExtern:
1550 return isolate_->root_handle(RootIndex::kWasmRttExternrefMap);
1551 case wasm::HeapType::kFunc:
1552 return isolate_->root_handle(RootIndex::kWasmRttFuncrefMap);
1553 case wasm::HeapType::kI31:
1554 return isolate_->root_handle(RootIndex::kWasmRttI31refMap);
1555 case wasm::HeapType::kExn:
1556 UNIMPLEMENTED()V8_Fatal("unimplemented code"); // TODO(jkummerow): This is going away?
1557 case wasm::HeapType::kBottom:
1558 UNREACHABLE()V8_Fatal("unreachable code");
1559 }
1560 // Non-generic types fall through.
1561 int map_index = init.immediate().heap_type;
1562 return handle(instance->managed_object_maps().get(map_index), isolate_);
1563 }
1564 case WasmInitExpr::kRttSub: {
1565 uint32_t type = static_cast<uint32_t>(init.immediate().heap_type);
1566 Handle<Object> parent =
1567 RecursivelyEvaluateGlobalInitializer(*init.operand(), instance);
1568 return AllocateSubRtt(isolate_, instance, type,
1569 Handle<Map>::cast(parent));
1570 }
1571 }

1574// Process initialization of globals.
1575void InstanceBuilder::InitGlobals(Handle<WasmInstanceObject> instance) {
1576 for (const WasmGlobal& global : module_->globals) {
1577 if (global.mutability && global.imported) {
1578 continue;
1579 }
1580
1581 switch (global.init.kind()) {
1582 case WasmInitExpr::kI32Const:
1583 WriteLittleEndianValue<int32_t>(GetRawGlobalPtr<int32_t>(global),
1584 global.init.immediate().i32_const);
1585 break;
1586 case WasmInitExpr::kI64Const:
1587 WriteLittleEndianValue<int64_t>(GetRawGlobalPtr<int64_t>(global),
1588 global.init.immediate().i64_const);
1589 break;
1590 case WasmInitExpr::kF32Const:
1591 WriteLittleEndianValue<float>(GetRawGlobalPtr<float>(global),
1592 global.init.immediate().f32_const);
1593 break;
1594 case WasmInitExpr::kF64Const:
1595 WriteLittleEndianValue<double>(GetRawGlobalPtr<double>(global),
1596 global.init.immediate().f64_const);
1597 break;
1598 case WasmInitExpr::kS128Const:
1599 DCHECK(enabled_.has_simd())((void) 0);
1600 WriteLittleEndianValue<std::array<uint8_t, kSimd128Size>>(
1601 GetRawGlobalPtr<std::array<uint8_t, kSimd128Size>>(global),
1602 global.init.immediate().s128_const);
1603 break;
1604 case WasmInitExpr::kRefNullConst:
1605 DCHECK(enabled_.has_reftypes() || enabled_.has_eh())((void) 0);
1606 if (global.imported) break; // We already initialized imported globals.
1607
1608 tagged_globals_->set(global.offset,
1609 ReadOnlyRoots(isolate_).null_value(),
1610 SKIP_WRITE_BARRIER);
1611 break;
1612 case WasmInitExpr::kRefFuncConst: {
1613 DCHECK(enabled_.has_reftypes())((void) 0);
1614 auto function = WasmInstanceObject::GetOrCreateWasmExternalFunction(
1615 isolate_, instance, global.init.immediate().index);
1616 tagged_globals_->set(global.offset, *function);
1617 break;
1618 }
1619 case WasmInitExpr::kGlobalGet: {
1620 // Initialize with another global.
1621 uint32_t new_offset = global.offset;
1622 uint32_t old_offset =
1623 module_->globals[global.init.immediate().index].offset;
1624 TRACE("init [globals+%u] = [globals+%d]\n", global.offset, old_offset);
1625 if (global.type.is_reference_type()) {
1626 DCHECK(enabled_.has_reftypes() || enabled_.has_eh())((void) 0);
1627 tagged_globals_->set(new_offset, tagged_globals_->get(old_offset));
1628 } else {
1629 size_t size = (global.type == kWasmI64 || global.type == kWasmF64)
1630 ? sizeof(double)
1631 : sizeof(int32_t);
1632 memcpy(raw_buffer_ptr(untagged_globals_, new_offset),
1633 raw_buffer_ptr(untagged_globals_, old_offset), size);
1634 }
1635 break;
1636 }
1637 case WasmInitExpr::kRttCanon:
1638 case WasmInitExpr::kRttSub:
1639 tagged_globals_->set(
1640 global.offset,
1641 *RecursivelyEvaluateGlobalInitializer(global.init, instance));
1642 break;
1643 case WasmInitExpr::kNone:
1644 // Happens with imported globals.
1645 break;
1646 }
1647 }
1648}

1676// Process the exports, creating wrappers for functions, tables, memories,
1677// globals, and exceptions.
1678void InstanceBuilder::ProcessExports(Handle<WasmInstanceObject> instance) {
1679 std::unordered_map<int, Handle<Object>> imported_globals;
1680
1681 // If an imported WebAssembly function or global gets exported, the export
1682 // has to be identical to to import. Therefore we cache all imported
1683 // WebAssembly functions in the instance, and all imported globals in a map
1684 // here.
1685 for (int index = 0, end = static_cast<int>(module_->import_table.size());
1686 index < end; ++index) {
1687 const WasmImport& import = module_->import_table[index];
1688 if (import.kind == kExternalFunction) {
1689 Handle<Object> value = sanitized_imports_[index].value;
1690 if (WasmExternalFunction::IsWasmExternalFunction(*value)) {
1691 WasmInstanceObject::SetWasmExternalFunction(
1692 isolate_, instance, import.index,
1693 Handle<WasmExternalFunction>::cast(value));
1694 }
1695 } else if (import.kind == kExternalGlobal) {
1696 Handle<Object> value = sanitized_imports_[index].value;
1697 if (value->IsWasmGlobalObject()) {
1698 imported_globals[import.index] = value;
1699 }
1700 }
1701 }
1702
1703 Handle<JSObject> exports_object;
1704 MaybeHandle<String> single_function_name;
1705 bool is_asm_js = is_asmjs_module(module_);
1706 if (is_asm_js) {
1707 Handle<JSFunction> object_function = Handle<JSFunction>(
1708 isolate_->native_context()->object_function(), isolate_);
1709 exports_object = isolate_->factory()->NewJSObject(object_function);
1710 single_function_name =
1711 isolate_->factory()->InternalizeUtf8String(AsmJs::kSingleFunctionName);
1712 } else {
1713 exports_object = isolate_->factory()->NewJSObjectWithNullProto();
1714 }
1715 instance->set_exports_object(*exports_object);
1716
1717 PropertyDescriptor desc;
1718 desc.set_writable(is_asm_js);
1719 desc.set_enumerable(true);
1720 desc.set_configurable(is_asm_js);
1721
1722 // Process each export in the export table.
1723 for (const WasmExport& exp : module_->export_table) {
1724 Handle<String> name = WasmModuleObject::ExtractUtf8StringFromModuleBytes(
1725 isolate_, module_object_, exp.name, kInternalize);
1726 Handle<JSObject> export_to = exports_object;
1727 switch (exp.kind) {
1728 case kExternalFunction: {
1729 // Wrap and export the code as a JSFunction.
1730 // TODO(wasm): reduce duplication with LoadElemSegment() further below
1731 Handle<WasmExternalFunction> wasm_external_function =
1732 WasmInstanceObject::GetOrCreateWasmExternalFunction(
1733 isolate_, instance, exp.index);
1734 desc.set_value(wasm_external_function);
1735
1736 if (is_asm_js &&
1737 String::Equals(isolate_, name,
1738 single_function_name.ToHandleChecked())) {
1739 export_to = instance;
1740 }
1741 break;
1742 }
1743 case kExternalTable: {
1744 desc.set_value(handle(instance->tables().get(exp.index), isolate_));
1745 break;
1746 }
1747 case kExternalMemory: {
1748 // Export the memory as a WebAssembly.Memory object. A WasmMemoryObject
1749 // should already be available if the module has memory, since we always
1750 // create or import it when building an WasmInstanceObject.
1751 DCHECK(instance->has_memory_object())((void) 0);
1752 desc.set_value(
1753 Handle<WasmMemoryObject>(instance->memory_object(), isolate_));
1754 break;
1755 }
1756 case kExternalGlobal: {
1757 const WasmGlobal& global = module_->globals[exp.index];
1758 if (global.imported) {
1759 auto cached_global = imported_globals.find(exp.index);
1760 if (cached_global != imported_globals.end()) {
1761 desc.set_value(cached_global->second);
1762 break;
1763 }
1764 }
1765 Handle<JSArrayBuffer> untagged_buffer;
1766 Handle<FixedArray> tagged_buffer;

1780 offset = static_cast<uint32_t>(addr);
1781 } else {
1782 untagged_buffer =
1783 handle(JSArrayBuffer::cast(buffers_array->get(global.index)),
1784 isolate_);
1785 Address global_addr =

1822 WasmExceptionObject::New(isolate_, exception.sig, exception_tag);
1823 exception_wrappers_[exp.index] = wrapper;
1824 }
1825 desc.set_value(wrapper);
1826 break;
1827 }

1830 break;
1831 }

1864bool LoadElemSegmentImpl(Isolate* isolate, Handle<WasmInstanceObject> instance,
1865 Handle<WasmTableObject> table_object,
1866 uint32_t table_index, uint32_t segment_index,
1867 uint32_t dst, uint32_t src, size_t count) {
1868 DCHECK_LT(segment_index, instance->module()->elem_segments.size())((void) 0);
1869 auto& elem_segment = instance->module()->elem_segments[segment_index];
1870 // TODO(wasm): Move this functionality into wasm-objects, since it is used
1871 // for both instantiation and in the implementation of the table.init
1872 // instruction.
1873 if (!base::IsInBounds<uint64_t>(dst, count, table_object->current_length()) ||
1874 !base::IsInBounds<uint64_t>(
1875 src, count,
1876 instance->dropped_elem_segments()[segment_index] == 0
1877 ? elem_segment.entries.size()
1878 : 0)) {
1879 return false;
1880 }
1881
1882 const WasmModule* module = instance->module();
1883 for (size_t i = 0; i < count; ++i) {
1884 uint32_t func_index = elem_segment.entries[src + i];
1885 int entry_index = static_cast<int>(dst + i);
1886
1887 if (func_index == WasmElemSegment::kNullIndex) {
1888 if (table_object->type().is_reference_to(HeapType::kFunc)) {
1889 IndirectFunctionTableEntry(instance, table_index, entry_index).clear();
1890 }
1891 WasmTableObject::Set(isolate, table_object, entry_index,
1892 isolate->factory()->null_value());
1893 continue;
1894 }
1895
1896 const WasmFunction* function = &module->functions[func_index];
1897
1898 // Update the local dispatch table first if necessary.
1899 // TODO(9495): Make sure tables work with all function types.
1900 if (table_object->type().is_reference_to(HeapType::kFunc)) {
1901 uint32_t sig_id = module->signature_ids[function->sig_index];
1902 IndirectFunctionTableEntry(instance, table_index, entry_index)
1903 .Set(sig_id, instance, func_index);
1904 }
1905
1906 // For ExternRef tables, we have to generate the WasmExternalFunction
1907 // eagerly. Later we cannot know if an entry is a placeholder or not.
1908 if (table_object->type().is_reference_to(HeapType::kExtern)) {
1909 Handle<WasmExternalFunction> wasm_external_function =
1910 WasmInstanceObject::GetOrCreateWasmExternalFunction(isolate, instance,
1911 func_index);
1912 WasmTableObject::Set(isolate, table_object, entry_index,
1913 wasm_external_function);
1914 } else {
1915 // Update the table object's other dispatch tables.
1916 MaybeHandle<WasmExternalFunction> wasm_external_function =
1917 WasmInstanceObject::GetWasmExternalFunction(isolate, instance,
1918 func_index);
1919 if (wasm_external_function.is_null()) {
1920 // No JSFunction entry yet exists for this function. Create a {Tuple2}
1921 // holding the information to lazily allocate one.
1922 WasmTableObject::SetFunctionTablePlaceholder(
1923 isolate, table_object, entry_index, instance, func_index);
1924 } else {
1925 table_object->entries().set(entry_index,
1926 *wasm_external_function.ToHandleChecked());
1927 }
1928 // UpdateDispatchTables() updates all other dispatch tables, since
1929 // we have not yet added the dispatch table we are currently building.
1930 WasmTableObject::UpdateDispatchTables(isolate, table_object, entry_index,
1931 function->sig, instance,
1932 func_index);
1933 }
1934 }
1935 return true;
1936}

1938void InstanceBuilder::LoadTableSegments(Handle<WasmInstanceObject> instance) {
1939 for (uint32_t segment_index = 0;
1940 segment_index < module_->elem_segments.size(); ++segment_index) {
1941 auto& elem_segment = instance->module()->elem_segments[segment_index];
1942 // Passive segments are not copied during instantiation.
1943 if (elem_segment.status != WasmElemSegment::kStatusActive) continue;
1944
1945 uint32_t table_index = elem_segment.table_index;
1946 uint32_t dst = EvalUint32InitExpr(instance, elem_segment.offset);
1947 uint32_t src = 0;
1948 size_t count = elem_segment.entries.size();
1949
1950 bool success = LoadElemSegmentImpl(
1951 isolate_, instance,
1952 handle(WasmTableObject::cast(
1953 instance->tables().get(elem_segment.table_index)),
1954 isolate_),
1955 table_index, segment_index, dst, src, count);
1956 // Set the active segments to being already dropped, since memory.init on
1957 // a dropped passive segment and an active segment have the same
1958 // behavior.
1959 instance->dropped_elem_segments()[segment_index] = 1;
1960 if (enabled_.has_bulk_memory()) {
1961 if (!success) {
1962 thrower_->RuntimeError("table initializer is out of bounds");
1963 // Break out instead of returning; we don't want to continue to
1964 // initialize any further element segments, but still need to add
1965 // dispatch tables below.
1966 break;
1967 }
1968 } else {
1969 CHECK(success)do { if ((__builtin_expect(!!(!(success)), 0))) { V8_Fatal("Check failed: %s."
, "success"); } } while (false);
1970 }
1971 }
1972
1973 int table_count = static_cast<int>(module_->tables.size());
1974 for (int index = 0; index < table_count; ++index) {
1975 if (module_->tables[index].type.is_reference_to(HeapType::kFunc)) {
1976 auto table_object = handle(
1977 WasmTableObject::cast(instance->tables().get(index)), isolate_);
1978
1979 // Add the new dispatch table at the end to avoid redundant lookups.
1980 WasmTableObject::AddDispatchTable(isolate_, table_object, instance,
1981 index);
1982 }
1983 }
1984}
