167/***************************************************************************
168 * The receive thread doesn't transmit packets. Instead, it queues them
169 * up on the transmit thread. Every so often, the transmit thread needs
170 * to flush this transmit queue and send everything.
171 *
172 * This is an inherent design issue trying to send things as batches rather
173 * than individually. It increases latency, but increases performance. We
174 * don't really care about latency.

180 uint64_t *packets_sent,

183 /*
184 * Send a batch of queued packets
185 */
186 for ( ; (*batchsize); (*batchsize)--) {
187 int err;
188 struct PacketBuffer *p;
189

191 * Get the next packet from the transmit queue. This packet was
192 * put there by a receive thread, and will contain things like
193 * an ACK or an HTTP request
194 */
195 err = rte_ring_sc_dequeue(transmit_queue, (void**)&p);
196 if (err) {

646 sprintf_ssnprintf(foo, sizeof(foo), "%u", masscan->tcp_connection_timeout);
647 tcpcon_set_parameter( tcpcon,
648 "timeout",
649 strlen(foo),
650 foo);
651 }
652 if (masscan->tcp_hello_timeout) {
653 char foo[64];
654 sprintf_ssnprintf(foo, sizeof(foo), "%u", masscan->tcp_connection_timeout);
655 tcpcon_set_parameter( tcpcon,
656 "hello-timeout",
657 strlen(foo),
658 foo);
659 }
660
661 for (pay = masscan->tcp_payloads; pay; pay = pay->next) {
662 char name[64];
663 sprintf_ssnprintf(name, sizeof(name), "hello-string[%u]", pay->port);
664 tcpcon_set_parameter( tcpcon,
665 name,
666 strlen(pay->payload_base64),
667 pay->payload_base64);
668 }
669
670 }

672 /*
673 * In "offline" mode, we don't have any receive threads, so simply
674 * wait until transmitter thread is done then go to the end
675 */
676 if (masscan->is_offline) {
677 while (!is_rx_done)
678 pixie_usleep(10000);
679 parms->done_receiving = 1;
680 goto end;
681 }

683 /*
684 * Receive packets. This is where we catch any responses and print
685 * them to the terminal.
686 */
687 LOG(1, "THREAD: recv: starting main loop\n");
688 while (!is_rx_done) {
689 int status;
690 unsigned length;
691 unsigned secs;
692 unsigned usecs;
693 const unsigned char *px;
694 int err;
695 unsigned x;
696 struct PreprocessedInfo parsed;
697 unsigned ip_me;
698 unsigned port_me;
699 unsigned ip_them;
700 unsigned port_them;
701 unsigned seqno_me;
702 unsigned seqno_them;
703 unsigned cookie;
704
705 /*
706 * RECEIVE
707 *
708 * This is the boring part of actually receiving a packet
709 */
710 err = rawsock_recv_packet(
711 adapter,
712 &length,
713 &secs,
714 &usecs,
715 &px);
716
717 if (err != 0) {
718 if (tcpcon)
719 tcpcon_timeouts(tcpcon, (unsigned)time(0), 0);
720 continue;
721 }
722
723
724 /*
725 * Do any TCP event timeouts based on the current timestamp from
726 * the packet. For example, if the connection has been open for
727 * around 10 seconds, we'll close the connection. (--banners)
728 */
729 if (tcpcon) {
730 tcpcon_timeouts(tcpcon, secs, usecs);
731 }
732
733 if (length > 1514)
734 continue;
735
736 /*
737 * "Preprocess" the response packet. This means to go through and
738 * figure out where the TCP/IP headers are and the locations of
739 * some fields, like IP address and port numbers.
740 */
741 x = preprocess_frame(px, length, data_link, &parsed);
742 if (!x)
743 continue; /* corrupt packet */
744 ip_me = parsed.ip_dst[0]<<24 | parsed.ip_dst[1]<<16
745 | parsed.ip_dst[2]<< 8 | parsed.ip_dst[3]<<0;
746 ip_them = parsed.ip_src[0]<<24 | parsed.ip_src[1]<<16
747 | parsed.ip_src[2]<< 8 | parsed.ip_src[3]<<0;
748 port_me = parsed.port_dst;
749 port_them = parsed.port_src;
750 seqno_them = TCP_SEQNO(px, parsed.transport_offset)(px[parsed.transport_offset+4]<<24|px[parsed.transport_offset
+5]<<16|px[parsed.transport_offset+6]<<8|px[parsed
.transport_offset+7]);
751 seqno_me = TCP_ACKNO(px, parsed.transport_offset)(px[parsed.transport_offset+8]<<24|px[parsed.transport_offset
+9]<<16|px[parsed.transport_offset+10]<<8|px[parsed
.transport_offset+11]);
752
753
754 switch (parsed.ip_protocol) {
755 case 132: /* SCTP */
756 cookie = syn_cookie(ip_them, port_them | (Proto_SCTP<<16), ip_me, port_me, entropy) & 0xFFFFFFFF;
757 break;
758 default:
759 cookie = syn_cookie(ip_them, port_them, ip_me, port_me, entropy) & 0xFFFFFFFF;
760 }
761
762 /* verify: my IP address */
763 if (!is_my_ip(&parms->src, ip_me))
764 continue;
765
766 /*
767 * Handle non-TCP protocols
768 */
769 switch (parsed.found) {
770 case FOUND_ARP:
771 LOGip(2, ip_them, 0, "-> ARP [%u] \n", px[parsed.found_offset]);
772 switch (px[parsed.found_offset + 6]<<8 | px[parsed.found_offset+7]) {
773 case 1: /* request */
774 /* This function will transmit a "reply" to somebody's ARP request
775 * for our IP address (as part of our user-mode TCP/IP).
776 * Since we completely bypass the TCP/IP stack, we have to handle ARPs
777 * ourself, or the router will lose track of us.*/
778 arp_response( ip_me,
779 parms->adapter_mac,
780 px, length,
781 parms->packet_buffers,
782 parms->transmit_queue);
783 break;
784 case 2: /* response */
785 /* This is for "arp scan" mode, where we are ARPing targets rather
786 * than port scanning them */
787
788 /* If we aren't doing an ARP scan, then ignore ARP responses */
789 if (!masscan->is_arp)
790 break;
791
792 /* If this response isn't in our range, then ignore it */
793 if (!rangelist_is_contains(&masscan->targets, ip_them))
794 break;
795
796 /* Ignore duplicates */
797 if (dedup_is_duplicate(dedup, ip_them, 0, ip_me, 0))
798 continue;
799
800 /* ...everything good, so now report this response */
801 handle_arp(out, secs, px, length, &parsed);
802 break;
803 }
804 continue;
805 case FOUND_UDP:
806 case FOUND_DNS:
807 if (!is_nic_port(masscan, port_me))
808 continue;
809 if (parms->masscan->nmap.packet_trace)
810 packet_trace(stdoutstdout, parms->pt_start, px, length, 0);
811 handle_udp(out, secs, px, length, &parsed, entropy);
812 continue;
813 case FOUND_ICMP:
814 handle_icmp(out, secs, px, length, &parsed, entropy);
815 continue;
816 case FOUND_SCTP:
817 handle_sctp(out, secs, px, length, cookie, &parsed, entropy);
818 break;
819 case FOUND_TCP:
820 /* fall down to below */
821 break;
822 default:
823 continue;
824 }
825
826
827 /* verify: my port number */
828 if (!is_my_port(&parms->src, port_me))
829 continue;
830 if (parms->masscan->nmap.packet_trace)
831 packet_trace(stdoutstdout, parms->pt_start, px, length, 0);
832
833 /* Save raw packet in --pcap file */
834 if (pcapfile) {
835 pcapfile_writeframe(
836 pcapfile,
837 px,
838 length,
839 length,
840 secs,
841 usecs);
842 }
843
844 {
845 char buf[64];
846 LOGip(5, ip_them, port_them, "-> TCP ackno=0x%08x flags=0x%02x(%s)\n",
847 seqno_me,
848 TCP_FLAGS(px, parsed.transport_offset)(px[(parsed.transport_offset)+13]),
849 reason_string(TCP_FLAGS(px, parsed.transport_offset)(px[(parsed.transport_offset)+13]), buf, sizeof(buf)));
850 }
851
852 /* If recording --banners, create a new "TCP Control Block (TCB)" */
853 if (tcpcon) {
854 struct TCP_Control_Block *tcb;
855
856 /* does a TCB already exist for this connection? */
857 tcb = tcpcon_lookup_tcb(tcpcon,
858 ip_me, ip_them,
859 port_me, port_them);
860
861 if (TCP_IS_SYNACK(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x12) == 0x12)) {
862 if (cookie != seqno_me - 1) {
863 LOG(2, "%u.%u.%u.%u - bad cookie: ackno=0x%08x expected=0x%08x\n",
864 (ip_them>>24)&0xff, (ip_them>>16)&0xff, (ip_them>>8)&0xff, (ip_them>>0)&0xff,
865 seqno_me-1, cookie);
866 continue;
867 }
868
869 if (tcb == NULL((void*)0)) {
870 tcb = tcpcon_create_tcb(tcpcon,
871 ip_me, ip_them,
872 port_me, port_them,
873 seqno_me, seqno_them+1,
874 parsed.ip_ttl);
875 (*status_tcb_count)++;
876 }
877
878 tcpcon_handle(tcpcon, tcb, TCP_WHAT_SYNACK,
879 0, 0, secs, usecs, seqno_them+1);
880
881 } else if (tcb) {
882 /* If this is an ACK, then handle that first */
883 if (TCP_IS_ACK(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x10) == 0x10)) {
884 tcpcon_handle(tcpcon, tcb, TCP_WHAT_ACK,
885 0, seqno_me, secs, usecs, seqno_them);
886 }
887
888 /* If this contains payload, handle that */
889 if (parsed.app_length) {
890 tcpcon_handle(tcpcon, tcb, TCP_WHAT_DATA,
891 px + parsed.app_offset, parsed.app_length,
892 secs, usecs, seqno_them);
893 }
894
895 /* If this is a FIN, handle that. Note that ACK +
896 * payload + FIN can come together */
897 if (TCP_IS_FIN(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x1) == 0x1)
898 && !TCP_IS_RST(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x4) == 0x4)) {
899 tcpcon_handle(tcpcon, tcb, TCP_WHAT_FIN,
900 0, parsed.app_length, secs, usecs, seqno_them);
901 }
902
903 /* If this is a RST, then we'll be closing the connection */
904 if (TCP_IS_RST(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x4) == 0x4)) {
905 tcpcon_handle(tcpcon, tcb, TCP_WHAT_RST,
906 0, 0, secs, usecs, seqno_them);
907 }
908 } else if (TCP_IS_FIN(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x1) == 0x1)) {
909 /*
910 * NO TCB!
911 * This happens when we've sent a FIN, deleted our connection,
912 * but the other side didn't get the packet.
913 */
914 if (!TCP_IS_RST(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x4) == 0x4))
915 tcpcon_send_FIN(
916 tcpcon,
917 ip_me, ip_them,
918 port_me, port_them,
919 seqno_them, seqno_me);
920 }
921
922 }
923
924 if (TCP_IS_SYNACK(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x12) == 0x12)
925 || TCP_IS_RST(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x4) == 0x4)) {
926
927 /* figure out the status */
928 status = PortStatus_Unknown;
929 if (TCP_IS_SYNACK(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x12) == 0x12))
930 status = PortStatus_Open;
931 if (TCP_IS_RST(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x4) == 0x4)) {
932 status = PortStatus_Closed;
933 }
934
935 /* verify: syn-cookies */
936 if (cookie != seqno_me - 1) {
937 LOG(5, "%u.%u.%u.%u - bad cookie: ackno=0x%08x expected=0x%08x\n",
938 (ip_them>>24)&0xff, (ip_them>>16)&0xff,
939 (ip_them>>8)&0xff, (ip_them>>0)&0xff,
940 seqno_me-1, cookie);
941 continue;
942 }
943
944 /* verify: ignore duplicates */
945 if (dedup_is_duplicate(dedup, ip_them, port_them, ip_me, port_me))
946 continue;
947
948 /* keep statistics on number received */
949 if (TCP_IS_SYNACK(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x12) == 0x12))
950 (*status_synack_count)++;
951
952 /*
953 * This is where we do the output
954 */
955 output_report_status(
956 out,
957 global_now,
958 status,
959 ip_them,
960 6, /* ip proto = tcp */
961 port_them,
962 px[parsed.transport_offset + 13], /* tcp flags */
963 parsed.ip_ttl,
964 parsed.mac_src
965 );
966
967 /*
968 * Send RST so other side isn't left hanging (only doing this in
969 * complete stateless mode where we aren't tracking banners)
970 */
971 if (tcpcon == NULL((void*)0) && !masscan->is_noreset)
972 tcp_send_RST(
973 &parms->tmplset->pkts[Proto_TCP],
974 parms->packet_buffers,
975 parms->transmit_queue,
976 ip_them, ip_me,
977 port_them, port_me,
978 0, seqno_me);
979
980 }
981 }
982
983
984 LOG(1, "THREAD: recv: stopping thread #%u\n", parms->nic_index);
985
986 /*
987 * cleanup
988 */
989end:
990 if (tcpcon)
991 tcpcon_destroy_table(tcpcon);
992 dedup_destroy(dedup);
993 output_destroy(out);
994 if (pcapfile)
995 pcapfile_close(pcapfile);
996
997 for (;;) {
998 void *p;
999 int err;
1000 err = rte_ring_sc_dequeue(parms->packet_buffers, (void**)&p);
1001 if (err == 0)
1002 free(p);
1003 else
1004 break;
1005 }
1006
1007 /* Thread is about to exit */
1008 parms->done_receiving = 1;
1009
1010}

1040 * Called from main() to initiate the scan.
1041 * Launches the 'transmit_thread()' and 'receive_thread()' and waits for
1042 * them to exit.
1043 ***************************************************************************/

1045main_scan(struct Masscan *masscan)
1046{

1048 uint64_t count_ips;
1049 uint64_t count_ports;
1050 uint64_t range;
1051 unsigned index;
1052 unsigned *picker;
1053 time_t now = time(0);
1054 struct Status status;
1055 uint64_t min_index = UINT64_MAX(18446744073709551615UL);
1056 struct MassScript *script = NULL((void*)0);
1057
1058 memset(parms_array, 0, sizeof(parms_array));
1059
1060 /*
1061 * Script initialization
1062 */
1063 if (masscan->script.name) {
1064 unsigned i;
1065 unsigned is_error;
1066 script = script_lookup(masscan->script.name);
1067
1068 /* If no ports specified on command-line, grab default ports */
1069 is_error = 0;

1071 rangelist_parse_ports(&masscan->ports, script->ports, &is_error);

1076 r->begin = (r->begin&0xFFFF) | Templ_Script;
1077 r->end = (r->end & 0xFFFF) | Templ_Script;

1080
1081 /*
1082 * Initialize the task size
1083 */
1084 count_ips = rangelist_count(&masscan->targets);
1085 if (count_ips == 0) {
1086 LOG(0, "FAIL: target IP address list empty\n");
1087 LOG(0, " [hint] try something like \"--range 10.0.0.0/8\"\n");
1088 LOG(0, " [hint] try something like \"--range 192.168.0.100-192.168.0.200\"\n");
1089 return 1;
1090 }
1091 count_ports = rangelist_count(&masscan->ports);
1092 if (count_ports == 0) {
1093 LOG(0, "FAIL: no ports were specified\n");
1094 LOG(0, " [hint] try something like \"-p80,8000-9000\"\n");
1095 LOG(0, " [hint] try something like \"--ports 0-65535\"\n");
1096 return 1;
1097 }

1100 /*
1101 * If doing an ARP scan, then don't allow port scanning
1102 */
1103 if (rangelist_is_contains(&masscan->ports, Templ_ARP)) {
1104 if (masscan->ports.count != 1) {
1105 LOG(0, "FAIL: cannot arpscan and portscan at the same time\n");
1106 return 1;
1107 }
1108 }
1109
1110 /*
1111 * If the IP address range is very big, then require that that the
1112 * user apply an exclude range
1113 */

1115 LOG(0, "FAIL: range too big, need confirmation\n");
1116 LOG(0, " [hint] to prevent acccidents, at least one --exclude must be specified\n");
1117 LOG(0, " [hint] use \"--exclude 255.255.255.255\" as a simple confirmation\n");
1118 exit(1);

1122 * trim the nmap UDP payloads down to only those ports we are using. This
1123 * makes lookups faster at high packet rates.
1124 */

1128 * of walking large memory tables. When we scan the entire Internet
1129 * our --excludefile will chop up our pristine 0.0.0.0/0 range into

1163 parms->router_mac
1164 );

1171 LOG(0, " [hint] if it has no IP address, "
1172 "manually set with \"--adapter-ip 192.168.100.5\"\n");

1180 * scanning. Then, we adjust the template with additional features,
1181 * such as the IP address and so on.

1185 parms->tmplset,
1186 parms->adapter_mac,
1187 parms->router_mac,

1190 masscan->seed);

1199 masscan->nic[index].src.port.range = 1;

1206 * Set the "TTL" (IP time-to-live) of everything we send.
1207 */
1208 if (masscan->nmap.ttl)
1209 template_set_ttl(parms->tmplset, masscan->nmap.ttl);

1217 */

1223 */

1230 struct PacketBuffer *p;

1234 exit(1);

1246 * THIS IS WHERE THE PROGRAM STARTS SPEWING OUT PACKETS AT A HIGH
1247 * RATE OF SPEED.

1255 */

1273 masscan->ports.list->begin == Templ_ICMP_echo && \

1275 { /* ICMP only */

1277 LOG(0, "Initiating ICMP Echo Scan\n");
1278 LOG(0, "Scanning %u hosts\n",(unsigned)count_ips);

1281 {
1282 LOG(0, " -- forced options: -sS -Pn -n --randomize-hosts -v --send-eth\n");
1283 LOG(0, "Initiating SYN Stealth Scan\n");
1284 LOG(0, "Scanning %u hosts [%u port%s/host]\n",
1285 (unsigned)count_ips, (unsigned)count_ports, (count_ports==1)?"":"s");

1290 * Now wait for <ctrl-c> to be pressed OR for threads to exit
1291 */
1292 LOG(1, "THREAD: status: starting thread\n");

1294 status.is_infinite = masscan->is_infinite;
1295 while (!is_tx_done && masscan->output.is_status_updates) {
1296 unsigned i;

1443 }
