131/* Collect external relocations. */
132static int collect_reloc(BuildCtx *ctx, uint8_t *addr, int idx, int type)
133{
134 if (ctx->nreloc >= BUILD_MAX_RELOC200) {
135 fprintf(stderrstderr, "Error: too many relocations, increase BUILD_MAX_RELOC.\n");
136 exit(1);
137 }
138 if (relocmap[idx] < 0) {
139 relocmap[idx] = ctx->nrelocsym;
140 ctx->relocsym[ctx->nrelocsym] = sym_decorate(ctx, "", extnames[idx]);
141 ctx->nrelocsym++;
142 }
143 ctx->reloc[ctx->nreloc].ofs = (int32_t)(addr - ctx->code);
144 ctx->reloc[ctx->nreloc].sym = relocmap[idx];
145 ctx->reloc[ctx->nreloc].type = type;
146 ctx->nreloc++;
147#if LJ_TARGET_XBOX360
148 return (int)(ctx->code - addr) + 4; /* Encode symbol offset of .text. */
149#else
150 return 0; /* Encode symbol offset of 0. */
151#endif
152}

154/* Naive insertion sort. Performance doesn't matter here. */
155static void sym_insert(BuildCtx *ctx, int32_t ofs,
156 const char *prefix, const char *suffix)
157{
158 ptrdiff_t i = ctx->nsym++;
159 while (i > 0) {
160 if (ctx->sym[i-1].ofs <= ofs)
161 break;
162 ctx->sym[i] = ctx->sym[i-1];
163 i--;
164 }
165 ctx->sym[i].ofs = ofs;
166 ctx->sym[i].name = sym_decorate(ctx, prefix, suffix);
167}

169/* Build the machine code. */
170static int build_code(BuildCtx *ctx)
171{
172 int status;
173 int i;
174
175 /* Initialize DynASM structures. */
176 ctx->nglob = GLOB__MAX;
177 ctx->glob = (void **)malloc(ctx->nglob*sizeof(void *));
178 memset(ctx->glob, 0, ctx->nglob*sizeof(void *));
179 ctx->nreloc = 0;
180
181 ctx->globnames = globnames;
182 ctx->relocsym = (const char **)malloc(NRELOCSYM(sizeof(extnames)/sizeof(extnames[0])-1)*sizeof(const char *));
183 ctx->nrelocsym = 0;
184 for (i = 0; i < (int)NRELOCSYM(sizeof(extnames)/sizeof(extnames[0])-1); i++) relocmap[i] = -1;
185
186 ctx->dasm_ident = DASM_IDENT"DynASM 1.3.0";
187 ctx->dasm_arch = DASM_ARCH"x86";
188
189 dasm_init(Dstctx, DASM_MAXSECTION2);
190 dasm_setupglobal(Dstctx, ctx->glob, ctx->nglob);
191 dasm_setup(Dstctx, build_actionlist);
192
193 /* Call arch-specific backend to emit the code. */
194 ctx->npc = build_backend(ctx);
195
196 /* Finalize the code. */
197 (void)dasm_checkstep(Dstctx, -1);
198 if ((status = dasm_link(Dstctx, &ctx->codesz))) return status;
199 ctx->code = (uint8_t *)malloc(ctx->codesz);
200 if ((status = dasm_encode(Dstctx, (void *)ctx->code))) return status;
201
202 /* Allocate symbol table and bytecode offsets. */
203 ctx->beginsym = sym_decorate(ctx, "", LABEL_PREFIX"lj_" "vm_asm_begin");
204 ctx->sym = (BuildSym *)malloc((ctx->npc+ctx->nglob+1)*sizeof(BuildSym));
205 ctx->nsym = 0;
206 ctx->bc_ofs = (int32_t *)malloc(ctx->npc*sizeof(int32_t));
207
208 /* Collect the opcodes (PC labels). */
209 for (i = 0; i < ctx->npc; i++) {
210 int32_t ofs = dasm_getpclabel(Dstctx, i);
211 if (ofs < 0) return 0x22000000|i;
212 ctx->bc_ofs[i] = ofs;
213 if ((LJ_HASJIT1 ||
214 !(i == BC_JFORI || i == BC_JFORL || i == BC_JITERL || i == BC_JLOOP ||
215 i == BC_IFORL || i == BC_IITERL || i == BC_ILOOP)) &&
216 (LJ_HASFFI1 || i != BC_KCDATA))
217 sym_insert(ctx, ofs, LABEL_PREFIX_BC"lj_" "BC_", bc_names[i]);
218 }
219
220 /* Collect the globals (named labels). */
221 for (i = 0; i < ctx->nglob; i++) {
222 const char *gl = globnames[i];
223 int len = (int)strlen(gl);
224 if (!ctx->glob[i]) {
225 fprintf(stderrstderr, "Error: undefined global %s\n", gl);
226 exit(2);
227 }
228 /* Skip the _Z symbols. */
229 if (!(len >= 2 && gl[len-2] == '_' && gl[len-1] == 'Z'))
230 sym_insert(ctx, (int32_t)((uint8_t *)(ctx->glob[i]) - ctx->code),
231 LABEL_PREFIX"lj_", globnames[i]);
232 }
233
234 /* Close the address range. */
235 sym_insert(ctx, (int32_t)ctx->codesz, "", "");
236 ctx->nsym--;
237
238 dasm_free(Dstctx);
239
240 return 0;
241}

293static const char *lower(char *buf, const char *s)
294{
295 char *p = buf;
296 while (*s) {
297 *p++ = (*s >= 'A' && *s <= 'Z') ? *s+0x20 : *s;
298 s++;
299 }
300 *p = '\0';
301 return buf;
302}

307 int i;

383/* Parse the output mode name. */
384static BuildMode parsemode(const char *mode)
385{
386 int i;
387 for (i = 0; modenames[i]; i++)
388 if (!strcmp(mode, modenames[i]))
389 return (BuildMode)i;
390 usage();
391 return (BuildMode)-1;
392}
393
394/* Parse arguments. */
395static void parseargs(BuildCtx *ctx, char **argv)
396{
397 const char *a;
398 int i;
399 ctx->mode = (BuildMode)-1;
400 ctx->outname = "-";
401 for (i = 1; (a = argv[i]) != NULL((void*)0); i++) {
402 if (a[0] != '-')
403 break;
404 switch (a[1]) {
405 case '-':
406 if (a[2]) goto err;
407 i++;
408 goto ok;
409 case '\0':
410 goto ok;
411 case 'm':
412 i++;
413 if (a[2] || argv[i] == NULL((void*)0)) goto err;
414 ctx->mode = parsemode(argv[i]);
415 break;
416 case 'o':
417 i++;
418 if (a[2] || argv[i] == NULL((void*)0)) goto err;
419 ctx->outname = argv[i];
420 break;
421 default: err:
422 usage();
423 break;
424 }
425 }
426ok:
427 ctx->args = argv+i;
428 if (ctx->mode == (BuildMode)-1) goto err;
429}
430
431int main(int argc, char **argv)
432{
433 BuildCtx ctx_;
434 BuildCtx *ctx = &ctx_;
435 int status, binmode;
436
437 if (sizeof(void *) != 4*LJ_320+8*LJ_641) {
438 fprintf(stderrstderr,"Error: pointer size mismatch in cross-build.\n");
439 fprintf(stderrstderr,"Try: make HOST_CC=\"gcc -m32\" CROSS=...\n\n");
440 return 1;
441 }
442
443 UNUSED(argc)((void)(argc));
444 parseargs(ctx, argv);
445
446 if ((status = build_code(ctx))) {
447 fprintf(stderrstderr,"Error: DASM error %08x\n", status);
448 return 1;
449 }
450
451 switch (ctx->mode) {
452 case BUILD_peobj:
453 case BUILD_raw:
454 binmode = 1;
455 break;
456 default:
457 binmode = 0;
458 break;
459 }
460
461 if (ctx->outname[0] == '-' && ctx->outname[1] == '\0') {
462 ctx->fp = stdoutstdout;
463#if defined(_WIN32)
464 if (binmode)
465 _setmode(_fileno(stdoutstdout), _O_BINARY); /* Yuck. */
466#endif
467 } else if (!(ctx->fp = fopen(ctx->outname, binmode ? "wb" : "w"))) {
468 fprintf(stderrstderr, "Error: cannot open output file '%s': %s\n",
469 ctx->outname, strerror(errno(*__errno_location ())));
470 exit(1);
471 }
472
473 switch (ctx->mode) {
474 case BUILD_elfasm:
475 case BUILD_coffasm:
476 case BUILD_machasm:
477 emit_asm(ctx);
478 emit_asm_debug(ctx);
479 break;
480 case BUILD_peobj:
481 emit_peobj(ctx);
482 break;
483 case BUILD_raw:
484 emit_raw(ctx);
485 break;
486 case BUILD_bcdef:
487 emit_bcdef(ctx);
488 emit_lib(ctx);
489 break;
490 case BUILD_vmdef:
491 emit_vmdef(ctx);
492 emit_lib(ctx);
493 break;
494 case BUILD_ffdef:
495 case BUILD_libdef:
496 case BUILD_recdef:
497 emit_lib(ctx);
498 break;
499 case BUILD_folddef:
500 emit_fold(ctx);
501 break;
502 default:
503 break;
504 }
505
506 fflush(ctx->fp);
507 if (ferror(ctx->fp)) {
508 fprintf(stderrstderr, "Error: cannot write to output file: %s\n",
509 strerror(errno(*__errno_location ())));
510 exit(1);
511 }
512 fclose(ctx->fp);
513
514 return 0;
515}
