260/*
261 * According to the letter of standard difference between pointers
262 * is specified to be valid only within same object. This makes
263 * it formally challenging to determine if input and output buffers
264 * are not partially overlapping with standard pointer arithmetic.
265 */
266#ifdef PTRDIFF_Tsize_t
267# undef PTRDIFF_Tsize_t
268#endif
269#if defined(OPENSSL_SYS_VMS) && __INITIAL_POINTER_SIZE==64
270/*
271 * Then we have VMS that distinguishes itself by adhering to
272 * sizeof(size_t)==4 even in 64-bit builds, which means that
273 * difference between two pointers might be truncated to 32 bits.
274 * In the context one can even wonder how comparison for
275 * equality is implemented. To be on the safe side we adhere to
276 * PTRDIFF_T even for comparison for equality.
277 */
278# define PTRDIFF_Tsize_t uint64_t
279#else
280# define PTRDIFF_Tsize_t size_t
281#endif
282
283int is_partially_overlapping(const void *ptr1, const void *ptr2, int len)
284{
285 PTRDIFF_Tsize_t diff = (PTRDIFF_Tsize_t)ptr1-(PTRDIFF_Tsize_t)ptr2;
286 /*
287 * Check for partially overlapping buffers. [Binary logical
288 * operations are used instead of boolean to minimize number
289 * of conditional branches.]
290 */
291 int overlapped = (len > 0) & (diff != 0) & ((diff < (PTRDIFF_Tsize_t)len) |
292 (diff > (0 - (PTRDIFF_Tsize_t)len)));
293
294 return overlapped;
295}

458 EVPerr(EVP_F_EVP_DECRYPTUPDATE, EVP_R_INVALID_OPERATION)ERR_put_error(6,(166),(148),"../deps/openssl/openssl/crypto/evp/evp_enc.c"
,458);
459 return 0;
460 }
461
462 b = ctx->cipher->block_size;
463
464 if (EVP_CIPHER_CTX_test_flags(ctx, EVP_CIPH_FLAG_LENGTH_BITS0x2000))
465 cmpl = (cmpl + 7) / 8;
466
467 /*
468 * CCM mode needs to know about the case where inl == 0 - it means the
469 * plaintext/ciphertext length is 0
470 */
471 if (inl < 0
472 || (inl == 0
473 && EVP_CIPHER_mode(ctx->cipher)(EVP_CIPHER_flags(ctx->cipher) & 0xF0007) != EVP_CIPH_CCM_MODE0x7)) {
474 *outl = 0;
475 return inl == 0;
476 }
477
478 if (ctx->cipher->flags & EVP_CIPH_FLAG_CUSTOM_CIPHER0x100000) {
479 if (b == 1 && is_partially_overlapping(out, in, cmpl)) {
480 EVPerr(EVP_F_EVP_DECRYPTUPDATE, EVP_R_PARTIALLY_OVERLAPPING)ERR_put_error(6,(166),(162),"../deps/openssl/openssl/crypto/evp/evp_enc.c"
,480);
481 return 0;
482 }
483
484 fix_len = ctx->cipher->do_cipher(ctx, out, in, inl);
485 if (fix_len < 0) {
486 *outl = 0;
487 return 0;
488 } else
489 *outl = fix_len;
490 return 1;
491 }
492
493 if (ctx->flags & EVP_CIPH_NO_PADDING0x100)
494 return evp_EncryptDecryptUpdate(ctx, out, outl, in, inl);
495
496 OPENSSL_assert(b <= sizeof(ctx->final))(void)((b <= sizeof(ctx->final)) ? 0 : (OPENSSL_die("assertion failed: "
"b <= sizeof(ctx->final)", "../deps/openssl/openssl/crypto/evp/evp_enc.c"
, 496), 1));
497
498 if (ctx->final_used) {
499 /* see comment about PTRDIFF_T comparison above */
500 if (((PTRDIFF_Tsize_t)out == (PTRDIFF_Tsize_t)in)
501 || is_partially_overlapping(out, in, b)) {
502 EVPerr(EVP_F_EVP_DECRYPTUPDATE, EVP_R_PARTIALLY_OVERLAPPING)ERR_put_error(6,(166),(162),"../deps/openssl/openssl/crypto/evp/evp_enc.c"
,502);
503 return 0;
504 }
505 memcpy(out, ctx->final, b);
506 out += b;
507 fix_len = 1;
508 } else
509 fix_len = 0;
510
511 if (!evp_EncryptDecryptUpdate(ctx, out, outl, in, inl))
512 return 0;
513
514 /*
515 * if we have 'decrypted' a multiple of block size, make sure we have a
516 * copy of this last block
517 */
518 if (b > 1 && !ctx->buf_len) {
519 *outl -= b;
520 ctx->final_used = 1;
521 memcpy(ctx->final, &out[*outl], b);
522 } else
523 ctx->final_used = 0;
524
525 if (fix_len)
526 *outl += b;
527
528 return 1;
529}

538int EVP_DecryptFinal_ex(EVP_CIPHER_CTX *ctx, unsigned char *out, int *outl)
539{
540 int i, n;
541 unsigned int b;
542
543 /* Prevent accidental use of encryption context when decrypting */
544 if (ctx->encrypt) {
545 EVPerr(EVP_F_EVP_DECRYPTFINAL_EX, EVP_R_INVALID_OPERATION)ERR_put_error(6,(101),(148),"../deps/openssl/openssl/crypto/evp/evp_enc.c"
,545);
546 return 0;
547 }
548
549 *outl = 0;
550
551 if (ctx->cipher->flags & EVP_CIPH_FLAG_CUSTOM_CIPHER0x100000) {
552 i = ctx->cipher->do_cipher(ctx, out, NULL((void*)0), 0);
553 if (i < 0)
554 return 0;
555 else
556 *outl = i;
557 return 1;
558 }
559
560 b = ctx->cipher->block_size;
561 if (ctx->flags & EVP_CIPH_NO_PADDING0x100) {
562 if (ctx->buf_len) {
563 EVPerr(EVP_F_EVP_DECRYPTFINAL_EX,ERR_put_error(6,(101),(138),"../deps/openssl/openssl/crypto/evp/evp_enc.c"
,564)
564 EVP_R_DATA_NOT_MULTIPLE_OF_BLOCK_LENGTH)ERR_put_error(6,(101),(138),"../deps/openssl/openssl/crypto/evp/evp_enc.c"
,564);
565 return 0;
566 }
567 *outl = 0;
568 return 1;
569 }
570 if (b > 1) {
571 if (ctx->buf_len || !ctx->final_used) {
572 EVPerr(EVP_F_EVP_DECRYPTFINAL_EX, EVP_R_WRONG_FINAL_BLOCK_LENGTH)ERR_put_error(6,(101),(109),"../deps/openssl/openssl/crypto/evp/evp_enc.c"
,572);
573 return 0;
574 }
575 OPENSSL_assert(b <= sizeof(ctx->final))(void)((b <= sizeof(ctx->final)) ? 0 : (OPENSSL_die("assertion failed: "
"b <= sizeof(ctx->final)", "../deps/openssl/openssl/crypto/evp/evp_enc.c"
, 575), 1));
576
577 /*
578 * The following assumes that the ciphertext has been authenticated.
579 * Otherwise it provides a padding oracle.
580 */
581 n = ctx->final[b - 1];
582 if (n == 0 || n > (int)b) {
583 EVPerr(EVP_F_EVP_DECRYPTFINAL_EX, EVP_R_BAD_DECRYPT)ERR_put_error(6,(101),(100),"../deps/openssl/openssl/crypto/evp/evp_enc.c"
,583);
584 return 0;
585 }
586 for (i = 0; i < n; i++) {
587 if (ctx->final[--b] != n) {
588 EVPerr(EVP_F_EVP_DECRYPTFINAL_EX, EVP_R_BAD_DECRYPT)ERR_put_error(6,(101),(100),"../deps/openssl/openssl/crypto/evp/evp_enc.c"
,588);
589 return 0;
590 }
591 }
592 n = ctx->cipher->block_size - n;
593 for (i = 0; i < n; i++)
594 out[i] = ctx->final[i];
595 *outl = n;
596 } else
597 *outl = 0;
598 return 1;
599}
