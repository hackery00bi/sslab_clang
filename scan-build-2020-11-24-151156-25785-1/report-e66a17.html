518JSNativeContextSpecialization::InferHasInPrototypeChainResult
519JSNativeContextSpecialization::InferHasInPrototypeChain(
520 Node* receiver, Node* effect, HeapObjectRef const& prototype) {
521 ZoneHandleSet<Map> receiver_maps;
522 NodeProperties::InferReceiverMapsResult result =
523 NodeProperties::InferReceiverMapsUnsafe(broker(), receiver, effect,
524 &receiver_maps);
525 if (result == NodeProperties::kNoReceiverMaps) return kMayBeInPrototypeChain;
526
527 // Try to determine either that all of the {receiver_maps} have the given
528 // {prototype} in their chain, or that none do. If we can't tell, return
529 // kMayBeInPrototypeChain.
530 bool all = true;
531 bool none = true;
532 for (size_t i = 0; i < receiver_maps.size(); ++i) {
533 MapRef map(broker(), receiver_maps[i]);
534 if (result == NodeProperties::kUnreliableReceiverMaps && !map.is_stable()) {
535 return kMayBeInPrototypeChain;
536 }
537 while (true) {
538 if (IsSpecialReceiverInstanceType(map.instance_type())) {
539 return kMayBeInPrototypeChain;
540 }
541 if (!map.IsJSObjectMap()) {
542 all = false;
543 break;
544 }
545 if (should_disallow_heap_access() && !map.serialized_prototype()) {
546 TRACE_BROKER_MISSING(broker(), "prototype data for map " << map)do { if (broker()->tracing_enabled()) StdoutStream{} <<
broker()->Trace() << "Missing " << "prototype data for map "
<< map << " (" << "../deps/v8/src/compiler/js-native-context-specialization.cc"
<< ":" << 546 << ")" << std::endl; }
while (false);
547 return kMayBeInPrototypeChain;
548 }
549 if (map.prototype().equals(prototype)) {
550 none = false;
551 break;
552 }
553 map = map.prototype().map();
554 if (!map.is_stable()) return kMayBeInPrototypeChain;
555 if (map.oddball_type() == OddballType::kNull) {
556 all = false;
557 break;
558 }
559 }
560 }
561 DCHECK_IMPLIES(all, !none)((void) 0);
562 if (!all && !none) return kMayBeInPrototypeChain;
563
564 {
565 base::Optional<JSObjectRef> last_prototype;
566 if (all) {
567 // We don't need to protect the full chain if we found the prototype, we
568 // can stop at {prototype}. In fact we could stop at the one before
569 // {prototype} but since we're dealing with multiple receiver maps this
570 // might be a different object each time, so it's much simpler to include
571 // {prototype}. That does, however, mean that we must check {prototype}'s
572 // map stability.
573 if (!prototype.map().is_stable()) return kMayBeInPrototypeChain;
574 last_prototype = prototype.AsJSObject();
575 }
576 WhereToStart start = result == NodeProperties::kUnreliableReceiverMaps
577 ? kStartAtReceiver
578 : kStartAtPrototype;
579 dependencies()->DependOnStablePrototypeChains(receiver_maps, start,
580 last_prototype);
581 }
582
583 DCHECK_EQ(all, !none)((void) 0);
584 return all ? kIsInPrototypeChain : kIsNotInPrototypeChain;
585}

791// TODO(neis): Try to merge this with ReduceNamedAccess by introducing a new
792// PropertyAccessInfo kind for global accesses and using the existing mechanism
793// for building loads/stores.
794Reduction JSNativeContextSpecialization::ReduceGlobalAccess(
795 Node* node, Node* receiver, Node* value, NameRef const& name,
796 AccessMode access_mode, Node* key, PropertyCellRef const& property_cell) {
797 Node* effect = NodeProperties::GetEffectInput(node);
798 Node* control = NodeProperties::GetControlInput(node);
799
800 ObjectRef property_cell_value = property_cell.value();
801 if (property_cell_value.IsHeapObject() &&
802 property_cell_value.AsHeapObject().map().oddball_type() ==
803 OddballType::kHole) {
804 // The property cell is no longer valid.
805 return NoChange();
806 }
807
808 PropertyDetails property_details = property_cell.property_details();
809 PropertyCellType property_cell_type = property_details.cell_type();
810 DCHECK_EQ(kData, property_details.kind())((void) 0);
811
812 // We have additional constraints for stores.
813 if (access_mode == AccessMode::kStore) {
814 if (property_details.IsReadOnly()) {
815 // Don't even bother trying to lower stores to read-only data properties.
816 return NoChange();
817 } else if (property_cell_type == PropertyCellType::kUndefined) {
818 // There's no fast-path for dealing with undefined property cells.
819 return NoChange();
820 } else if (property_cell_type == PropertyCellType::kConstantType) {
821 // There's also no fast-path to store to a global cell which pretended
822 // to be stable, but is no longer stable now.
823 if (property_cell_value.IsHeapObject() &&
824 !property_cell_value.AsHeapObject().map().is_stable()) {
825 return NoChange();
826 }
827 }
828 } else if (access_mode == AccessMode::kHas) {
829 // has checks cannot follow the fast-path used by loads when these
830 // conditions hold.
831 if ((property_details.IsConfigurable() || !property_details.IsReadOnly()) &&
832 property_details.cell_type() != PropertyCellType::kConstant &&
833 property_details.cell_type() != PropertyCellType::kUndefined)
834 return NoChange();
835 }
836
837 // Ensure that {key} matches the specified {name} (if {key} is given).
838 if (key != nullptr) {
839 effect = BuildCheckEqualsName(name, key, effect, control);
840 }
841
842 // If we have a {receiver} to validate, we do so by checking that its map is
843 // the (target) global proxy's map. This guarantees that in fact the receiver
844 // is the global proxy.
845 if (receiver != nullptr) {
846 effect = graph()->NewNode(
847 simplified()->CheckMaps(
848 CheckMapsFlag::kNone,
849 ZoneHandleSet<Map>(
850 HeapObjectRef(broker(), global_proxy()).map().object())),
851 receiver, effect, control);
852 }
853
854 if (access_mode == AccessMode::kLoad || access_mode == AccessMode::kHas) {
855 // Load from non-configurable, read-only data property on the global
856 // object can be constant-folded, even without deoptimization support.
857 if (!property_details.IsConfigurable() && property_details.IsReadOnly()) {
858 value = access_mode == AccessMode::kHas
859 ? jsgraph()->TrueConstant()
860 : jsgraph()->Constant(property_cell_value);
861 } else {
862 // Record a code dependency on the cell if we can benefit from the
863 // additional feedback, or the global property is configurable (i.e.
864 // can be deleted or reconfigured to an accessor property).
865 if (property_details.cell_type() != PropertyCellType::kMutable ||
866 property_details.IsConfigurable()) {
867 dependencies()->DependOnGlobalProperty(property_cell);
868 }
869
870 // Load from constant/undefined global property can be constant-folded.
871 if (property_details.cell_type() == PropertyCellType::kConstant ||
872 property_details.cell_type() == PropertyCellType::kUndefined) {
873 value = access_mode == AccessMode::kHas
874 ? jsgraph()->TrueConstant()
875 : jsgraph()->Constant(property_cell_value);
876 DCHECK(!property_cell_value.IsHeapObject() ||((void) 0)
877 property_cell_value.AsHeapObject().map().oddball_type() !=((void) 0)
878 OddballType::kHole)((void) 0);
879 } else {
880 DCHECK_NE(AccessMode::kHas, access_mode)((void) 0);
881
882 // Load from constant type cell can benefit from type feedback.
883 MaybeHandle<Map> map;
884 Type property_cell_value_type = Type::NonInternal();
885 MachineRepresentation representation = MachineRepresentation::kTagged;
886 if (property_details.cell_type() == PropertyCellType::kConstantType) {
887 // Compute proper type based on the current value in the cell.
888 if (property_cell_value.IsSmi()) {
889 property_cell_value_type = Type::SignedSmall();
890 representation = MachineRepresentation::kTaggedSigned;
891 } else if (property_cell_value.IsHeapNumber()) {
892 property_cell_value_type = Type::Number();
893 representation = MachineRepresentation::kTaggedPointer;
894 } else {
895 MapRef property_cell_value_map =
896 property_cell_value.AsHeapObject().map();
897 property_cell_value_type = Type::For(property_cell_value_map);
898 representation = MachineRepresentation::kTaggedPointer;
899
900 // We can only use the property cell value map for map check
901 // elimination if it's stable, i.e. the HeapObject wasn't
902 // mutated without the cell state being updated.
903 if (property_cell_value_map.is_stable()) {
904 dependencies()->DependOnStableMap(property_cell_value_map);
905 map = property_cell_value_map.object();
906 }
907 }
908 }
909 value = effect = graph()->NewNode(
910 simplified()->LoadField(ForPropertyCellValue(
911 representation, property_cell_value_type, map, name)),
912 jsgraph()->Constant(property_cell), effect, control);
913 }
914 }
915 } else {
916 DCHECK_EQ(AccessMode::kStore, access_mode)((void) 0);
917 DCHECK(!property_details.IsReadOnly())((void) 0);
918 switch (property_details.cell_type()) {
919 case PropertyCellType::kUndefined: {
920 UNREACHABLE()V8_Fatal("unreachable code");
921 break;
922 }
923 case PropertyCellType::kConstant: {
924 // Record a code dependency on the cell, and just deoptimize if the new
925 // value doesn't match the previous value stored inside the cell.
926 dependencies()->DependOnGlobalProperty(property_cell);
927 Node* check =
928 graph()->NewNode(simplified()->ReferenceEqual(), value,
929 jsgraph()->Constant(property_cell_value));
930 effect = graph()->NewNode(
931 simplified()->CheckIf(DeoptimizeReason::kValueMismatch), check,
932 effect, control);
933 break;
934 }
935 case PropertyCellType::kConstantType: {
936 // Record a code dependency on the cell, and just deoptimize if the new
937 // values' type doesn't match the type of the previous value in the
938 // cell.
939 dependencies()->DependOnGlobalProperty(property_cell);
940 Type property_cell_value_type;
941 MachineRepresentation representation = MachineRepresentation::kTagged;
942 if (property_cell_value.IsHeapObject()) {
943 // We cannot do anything if the {property_cell_value}s map is no
944 // longer stable.
945 MapRef property_cell_value_map =
946 property_cell_value.AsHeapObject().map();
947 dependencies()->DependOnStableMap(property_cell_value_map);
948
949 // Check that the {value} is a HeapObject.
950 value = effect = graph()->NewNode(simplified()->CheckHeapObject(),
951 value, effect, control);
952
953 // Check {value} map against the {property_cell} map.
954 effect = graph()->NewNode(
955 simplified()->CheckMaps(
956 CheckMapsFlag::kNone,
957 ZoneHandleSet<Map>(property_cell_value_map.object())),
958 value, effect, control);
959 property_cell_value_type = Type::OtherInternal();
960 representation = MachineRepresentation::kTaggedPointer;
961 } else {
962 // Check that the {value} is a Smi.
963 value = effect = graph()->NewNode(
964 simplified()->CheckSmi(FeedbackSource()), value, effect, control);
965 property_cell_value_type = Type::SignedSmall();
966 representation = MachineRepresentation::kTaggedSigned;
967 }
968 effect = graph()->NewNode(simplified()->StoreField(ForPropertyCellValue(
969 representation, property_cell_value_type,
970 MaybeHandle<Map>(), name)),
971 jsgraph()->Constant(property_cell), value,
972 effect, control);
973 break;
974 }
975 case PropertyCellType::kMutable: {
976 // Record a code dependency on the cell, and just deoptimize if the
977 // property ever becomes read-only.
978 dependencies()->DependOnGlobalProperty(property_cell);
979 effect = graph()->NewNode(
980 simplified()->StoreField(ForPropertyCellValue(
981 MachineRepresentation::kTagged, Type::NonInternal(),
982 MaybeHandle<Map>(), name)),
983 jsgraph()->Constant(property_cell), value, effect, control);
984 break;
985 }
986 }
987 }
988
989 ReplaceWithValue(node, value, effect, control);
990 return Replace(value);
991}

3199 check, *control);
3200
3201 Node* if_true = graph()->NewNode(common()->IfTrue(), branch);
3202 // Do a real bounds check against {length}. This is in order to protect
3203 // against a potential typer bug leading to the elimination of the
3204 // NumberLessThan above.
3205 Node* etrue = index = graph()->NewNode(
3206 simplified()->CheckBounds(FeedbackSource(),
3207 CheckBoundsFlag::kConvertStringAndMinusZero |
3208 CheckBoundsFlag::kAbortOnOutOfBounds),
3209 index, length, *effect, if_true);
3210 Node* masked_index = graph()->NewNode(simplified()->PoisonIndex(), index);
3211 Node* vtrue = etrue =
3212 graph()->NewNode(simplified()->StringCharCodeAt(), receiver,
3213 masked_index, etrue, if_true);
3214 vtrue = graph()->NewNode(simplified()->StringFromSingleCharCode(), vtrue);
3215
3216 Node* if_false = graph()->NewNode(common()->IfFalse(), branch);
3217 Node* vfalse = jsgraph()->UndefinedConstant();
3218
3219 *control = graph()->NewNode(common()->Merge(2), if_true, if_false);
3220 *effect =
3221 graph()->NewNode(common()->EffectPhi(2), etrue, *effect, *control);
3222 return graph()->NewNode(common()->Phi(MachineRepresentation::kTagged, 2),
3223 vtrue, vfalse, *control);
3224 } else {
3225 // Ensure that {index} is less than {receiver} length.
3226 index = *effect = graph()->NewNode(
3227 simplified()->CheckBounds(FeedbackSource(),
3228 CheckBoundsFlag::kConvertStringAndMinusZero),
3229 index, length, *effect, *control);
3230
3231 Node* masked_index = graph()->NewNode(simplified()->PoisonIndex(), index);
3232
3233 // Return the character from the {receiver} as single character string.
3234 Node* value = *effect =
3235 graph()->NewNode(simplified()->StringCharCodeAt(), receiver,
3236 masked_index, *effect, *control);
3237 value = graph()->NewNode(simplified()->StringFromSingleCharCode(), value);

3242Node* JSNativeContextSpecialization::BuildExtendPropertiesBackingStore(
3243 const MapRef& map, Node* properties, Node* effect, Node* control) {
3244 // TODO(bmeurer/jkummerow): Property deletions can undo map transitions
3245 // while keeping the backing store around, meaning that even though the
3246 // map might believe that objects have no unused property fields, there
3247 // might actually be some. It would be nice to not create a new backing
3248 // store in that case (i.e. when properties->length() >= new_length).
3249 // However, introducing branches and Phi nodes here would make it more
3250 // difficult for escape analysis to get rid of the backing stores used
3251 // for intermediate states of chains of property additions. That makes
3252 // it unclear what the best approach is here.
3253 DCHECK_EQ(0, map.UnusedPropertyFields())((void) 0);
3254 // Compute the length of the old {properties} and the new properties.
3255 int length = map.NextFreePropertyIndex() - map.GetInObjectProperties();
3256 int new_length = length + JSObject::kFieldsAdded;
3257 // Collect the field values from the {properties}.
3258 ZoneVector<Node*> values(zone());
3259 values.reserve(new_length);
3260 for (int i = 0; i < length; ++i) {
3261 Node* value = effect = graph()->NewNode(
3262 simplified()->LoadField(AccessBuilder::ForFixedArraySlot(i)),
3263 properties, effect, control);
3264 values.push_back(value);
3265 }
3266 // Initialize the new fields to undefined.
3267 for (int i = 0; i < JSObject::kFieldsAdded; ++i) {
3268 values.push_back(jsgraph()->UndefinedConstant());
3269 }
3270
3271 // Compute new length and hash.
3272 Node* hash;
3273 if (length == 0) {
3274 hash = graph()->NewNode(
3275 common()->Select(MachineRepresentation::kTaggedSigned),
3276 graph()->NewNode(simplified()->ObjectIsSmi(), properties), properties,
3277 jsgraph()->SmiConstant(PropertyArray::kNoHashSentinel));
3278 hash = effect = graph()->NewNode(common()->TypeGuard(Type::SignedSmall()),
3279 hash, effect, control);
3280 hash =
3281 graph()->NewNode(simplified()->NumberShiftLeft(), hash,
3282 jsgraph()->Constant(PropertyArray::HashField::kShift));
3283 } else {
3284 hash = effect = graph()->NewNode(
3285 simplified()->LoadField(AccessBuilder::ForPropertyArrayLengthAndHash()),
3286 properties, effect, control);
3287 hash =
3288 graph()->NewNode(simplified()->NumberBitwiseAnd(), hash,
3289 jsgraph()->Constant(PropertyArray::HashField::kMask));
3290 }
3291 Node* new_length_and_hash = graph()->NewNode(
3292 simplified()->NumberBitwiseOr(), jsgraph()->Constant(new_length), hash);
3293 // TDOO(jarin): Fix the typer to infer tighter bound for NumberBitwiseOr.
3294 new_length_and_hash = effect =
3295 graph()->NewNode(common()->TypeGuard(Type::SignedSmall()),
3296 new_length_and_hash, effect, control);
3297
3298 // Allocate and initialize the new properties.
3299 AllocationBuilder a(jsgraph(), effect, control);
3300 a.Allocate(PropertyArray::SizeFor(new_length), AllocationType::kYoung,
3301 Type::OtherInternal());
3302 a.Store(AccessBuilder::ForMap(), jsgraph()->PropertyArrayMapConstant());
3303 a.Store(AccessBuilder::ForPropertyArrayLengthAndHash(), new_length_and_hash);
3304 for (int i = 0; i < new_length; ++i) {
3305 a.Store(AccessBuilder::ForFixedArraySlot(i), values[i]);
3306 }
3307 return a.Finish();
3308}

3345 NodeProperties::InferReceiverMapsUnsafe(broker(), receiver, effect,

3360 receiver_maps->push_back(maps[i]);

3363 }
