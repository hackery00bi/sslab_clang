1137std::unique_ptr<TransferData> JSTransferable::TransferOrClone(
1138 TransferMode mode) const {
1139 // Call `this[symbol]()` where `symbol` is `kClone` or `kTransfer`,
1140 // which should return an object with `data` and `deserializeInfo` properties;
1141 // `data` is written to the serializer later, and `deserializeInfo` is stored
1142 // on the `TransferData` instance as a string.
1143 HandleScope handle_scope(env()->isolate());
1144 Local<Context> context = env()->isolate()->GetCurrentContext();
1145 Local<Symbol> method_name = mode == TransferMode::kCloneable ?
1146 env()->messaging_clone_symbol() : env()->messaging_transfer_symbol();
1147
1148 Local<Value> method;
1149 if (!object()->Get(context, method_name).ToLocal(&method)) {
1150 return {};
1151 }
1152 if (method->IsFunction()) {
1153 Local<Value> result_v;
1154 if (!method.As<Function>()->Call(
1155 context, object(), 0, nullptr).ToLocal(&result_v)) {
1156 return {};
1157 }
1158
1159 if (result_v->IsObject()) {
1160 Local<Object> result = result_v.As<Object>();
1161 Local<Value> data;
1162 Local<Value> deserialize_info;
1163 if (!result->Get(context, env()->data_string()).ToLocal(&data) ||
1164 !result->Get(context, env()->deserialize_info_string())
1165 .ToLocal(&deserialize_info)) {
1166 return {};
1167 }
1168 Utf8Value deserialize_info_str(env()->isolate(), deserialize_info);
1169 if (*deserialize_info_str == nullptr) return {};
1170 return std::make_unique<Data>(
1171 *deserialize_info_str, Global<Value>(env()->isolate(), data));
1172 }
1173 }
1174
1175 if (mode == TransferMode::kTransferable)
1176 return TransferOrClone(TransferMode::kCloneable);
1177 else
1178 return {};
1179}
