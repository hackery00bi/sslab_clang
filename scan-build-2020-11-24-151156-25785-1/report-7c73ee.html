1490#if V8_TARGET_ARCH_X641
1491#if V8_OS_MACOSX
1492 // The Darwin kernel [as of macOS 10.12.5] does not clean up page
1493 // directory entries [PDE] created from mmap or mach_vm_allocate, even
1494 // after the region is destroyed. Using a virtual address space that is
1495 // too large causes a leak of about 1 wired [can never be paged out] page
1496 // per call to mmap(). The page is only reclaimed when the process is
1497 // killed. Confine the hint to a 32-bit section of the virtual address
1498 // space. See crbug.com/700928.

1500 result = reinterpret_cast<void*>(mmap_region_base_ + offset);
1501#endif // V8_OS_MACOSX
1502#endif // V8_TARGET_ARCH_X64
1503 return result;
1504 }
