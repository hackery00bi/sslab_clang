53template <typename T, IrOpcode::Value kOpcode>
54struct ValueMatcher : public NodeMatcher {
55 using ValueType = T;
56
57 explicit ValueMatcher(Node* node) : NodeMatcher(node) {
58 static_assert(kOpcode != IrOpcode::kFoldConstant, "unsupported opcode");
59 if (node->opcode() == IrOpcode::kFoldConstant) {
60 node = node->InputAt(1);
61 }
62 DCHECK_NE(node->opcode(), IrOpcode::kFoldConstant)((void) 0);
63 has_value_ = opcode() == kOpcode;
64 if (has_value_) {
65 value_ = OpParameter<T>(node->op());
66 }
67 }
68
69 bool HasValue() const { return has_value_; }
70 const T& Value() const {
71 DCHECK(HasValue())((void) 0);
72 return value_;
73 }
74
75 private:
76 T value_;
uninitialized field 'this->left_.ValueMatcher::value_'
uninitialized field 'this->right_.ValueMatcher::value_'
77 bool has_value_;
78};

501 bool matches() const { return matches_; }
502 Node* index() const { return index_; }
503 int scale() const { return scale_; }
504 Node* base() const { return base_; }

509 bool matches_;
510 Node* index_;
511 int scale_;
512 Node* base_;
513 Node* displacement_;

516 void Initialize(Node* node, AddressOptions options) {
517 // The BaseWithIndexAndDisplacementMatcher canonicalizes the order of
518 // displacements and scale factors that are used as inputs, so instead of
519 // enumerating all possible patterns by brute force, checking for node
520 // clusters using the following templates in the following order suffices to
521 // find all of the interesting cases (S = index * scale, B = base input, D =
522 // displacement input):
523 // (S + (B + D))
524 // (S + (B + B))
525 // (S + D)
526 // (S + B)
527 // ((S + D) + B)
528 // ((S + B) + D)
529 // ((B + D) + B)
530 // ((B + B) + D)
531 // (B + D)
532 // (B + B)
533 if (node->InputCount() < 2) return;
534 AddMatcher m(node, options & AddressOption::kAllowInputSwap);
535 Node* left = m.left().node();
536 Node* right = m.right().node();
537 Node* displacement = nullptr;
538 Node* base = nullptr;
539 Node* index = nullptr;
540 Node* scale_expression = nullptr;
541 bool power_of_two_plus_one = false;
542 DisplacementMode displacement_mode = kPositiveDisplacement;
543 int scale = 0;
544 if (m.HasIndexInput() && OwnedByAddressingOperand(left)) {
545 index = m.IndexInput();
546 scale = m.scale();
547 scale_expression = left;
548 power_of_two_plus_one = m.power_of_two_plus_one();
549 bool match_found = false;
550 if (right->opcode() == AddMatcher::kSubOpcode &&
551 OwnedByAddressingOperand(right)) {
552 AddMatcher right_matcher(right);
553 if (right_matcher.right().HasValue()) {
554 // (S + (B - D))
555 base = right_matcher.left().node();
556 displacement = right_matcher.right().node();
557 displacement_mode = kNegativeDisplacement;
558 match_found = true;
559 }
560 }
561 if (!match_found) {
562 if (right->opcode() == AddMatcher::kAddOpcode &&
563 OwnedByAddressingOperand(right)) {
564 AddMatcher right_matcher(right);
565 if (right_matcher.right().HasValue()) {
566 // (S + (B + D))
567 base = right_matcher.left().node();
568 displacement = right_matcher.right().node();
569 } else {
570 // (S + (B + B))
571 base = right;
572 }
573 } else if (m.right().HasValue()) {
574 // (S + D)
575 displacement = right;
576 } else {
577 // (S + B)
578 base = right;
579 }
580 }
581 } else {
582 bool match_found = false;
583 if (left->opcode() == AddMatcher::kSubOpcode &&
584 OwnedByAddressingOperand(left)) {
585 AddMatcher left_matcher(left);
586 Node* left_left = left_matcher.left().node();
587 Node* left_right = left_matcher.right().node();
588 if (left_matcher.right().HasValue()) {
589 if (left_matcher.HasIndexInput() && left_left->OwnedBy(left)) {
590 // ((S - D) + B)
591 index = left_matcher.IndexInput();
592 scale = left_matcher.scale();
593 scale_expression = left_left;
594 power_of_two_plus_one = left_matcher.power_of_two_plus_one();
595 displacement = left_right;
596 displacement_mode = kNegativeDisplacement;
597 base = right;
598 } else {
599 // ((B - D) + B)
600 index = left_left;
601 displacement = left_right;
602 displacement_mode = kNegativeDisplacement;
603 base = right;
604 }
605 match_found = true;
606 }
607 }
608 if (!match_found) {
609 if (left->opcode() == AddMatcher::kAddOpcode &&
610 OwnedByAddressingOperand(left)) {
611 AddMatcher left_matcher(left);
612 Node* left_left = left_matcher.left().node();
613 Node* left_right = left_matcher.right().node();
614 if (left_matcher.HasIndexInput() && left_left->OwnedBy(left)) {
615 if (left_matcher.right().HasValue()) {
616 // ((S + D) + B)
617 index = left_matcher.IndexInput();
618 scale = left_matcher.scale();
619 scale_expression = left_left;
620 power_of_two_plus_one = left_matcher.power_of_two_plus_one();
621 displacement = left_right;
622 base = right;
623 } else if (m.right().HasValue()) {
624 if (left->OwnedBy(node)) {
625 // ((S + B) + D)
626 index = left_matcher.IndexInput();
627 scale = left_matcher.scale();
628 scale_expression = left_left;
629 power_of_two_plus_one = left_matcher.power_of_two_plus_one();
630 base = left_right;
631 displacement = right;
632 } else {
633 // (B + D)
634 base = left;
635 displacement = right;
636 }
637 } else {
638 // (B + B)
639 index = left;
640 base = right;
641 }
642 } else {
643 if (left_matcher.right().HasValue()) {
644 // ((B + D) + B)
645 index = left_left;
646 displacement = left_right;
647 base = right;
648 } else if (m.right().HasValue()) {
649 if (left->OwnedBy(node)) {
650 // ((B + B) + D)
651 index = left_left;
652 base = left_right;
653 displacement = right;
654 } else {
655 // (B + D)
656 base = left;
657 displacement = right;
658 }
659 } else {
660 // (B + B)
661 index = left;
662 base = right;
663 }
664 }
665 } else {
666 if (m.right().HasValue()) {
667 // (B + D)
668 base = left;
669 displacement = right;
670 } else {
671 // (B + B)
672 base = left;
673 index = right;
674 }
675 }
676 }
677 }
678 int64_t value = 0;
679 if (displacement != nullptr) {
680 switch (displacement->opcode()) {
681 case IrOpcode::kInt32Constant: {
682 value = OpParameter<int32_t>(displacement->op());
683 break;
684 }
685 case IrOpcode::kInt64Constant: {
686 value = OpParameter<int64_t>(displacement->op());
687 break;
688 }
689 default:
690 UNREACHABLE()V8_Fatal("unreachable code");
691 break;
692 }
693 if (value == 0) {
694 displacement = nullptr;
695 }
696 }
697 if (power_of_two_plus_one) {
698 if (base != nullptr) {
699 // If the scale requires explicitly using the index as the base, but a
700 // base is already part of the match, then the (1 << N + 1) scale factor
701 // can't be folded into the match and the entire index * scale
702 // calculation must be computed separately.
703 index = scale_expression;
704 scale = 0;
705 } else {
706 base = index;
707 }
708 }
709 if (!(options & AddressOption::kAllowScale) && scale != 0) {
710 index = scale_expression;
711 scale = 0;
712 }
713 base_ = base;
714 displacement_ = displacement;
715 displacement_mode_ = displacement_mode;
716 index_ = index;
717 scale_ = scale;
718 matches_ = true;
719 }

722 for (auto use : node->use_edges()) {

727 case IrOpcode::kProtectedLoad:
728 case IrOpcode::kInt32Add:
729 case IrOpcode::kInt64Add:

736 // Otherwise it is used as an address and skipped.
737 break;
738 default:
739 // Non-addressing use found.
740 return false;
741 }
742 }
743 return true;
744 }

42// Representation selection and lowering of {Simplified} operators to machine
43// operators are interwined. We use a fixpoint calculation to compute both the
44// output representation and the best possible lowering for {Simplified} nodes.
45// Representation change insertion ensures that all values are in the correct
46// machine representation after this phase, as dictated by the machine
47// operators themselves.
48enum Phase {
49 // 1.) PROPAGATE: Traverse the graph from the end, pushing usage information
50 // backwards from uses to definitions, around cycles in phis, according
51 // to local rules for each operator.
52 // During this phase, the usage information for a node determines the best
53 // possible lowering for each operator so far, and that in turn determines
54 // the output representation.
55 // Therefore, to be correct, this phase must iterate to a fixpoint before
56 // the next phase can begin.

185bool CanOverflowSigned32(const Operator* op, Type left, Type right,
186 TypeCache const* type_cache, Zone* type_zone) {
187 // We assume the inputs are checked Signed32 (or known statically to be
188 // Signed32). Technically, the inputs could also be minus zero, which we treat
189 // as 0 for the purpose of this function.
190 if (left.Maybe(Type::MinusZero())) {
191 left = Type::Union(left, type_cache->kSingletonZero, type_zone);
192 }
193 if (right.Maybe(Type::MinusZero())) {
194 right = Type::Union(right, type_cache->kSingletonZero, type_zone);
195 }
196 left = Type::Intersect(left, Type::Signed32(), type_zone);
197 right = Type::Intersect(right, Type::Signed32(), type_zone);
198 if (left.IsNone() || right.IsNone()) return false;
199 switch (op->opcode()) {
200 case IrOpcode::kSpeculativeSafeIntegerAdd:
201 return (left.Max() + right.Max() > kMaxInt) ||
202 (left.Min() + right.Min() < kMinInt);
203
204 case IrOpcode::kSpeculativeSafeIntegerSubtract:
205 return (left.Max() - right.Min() > kMaxInt) ||
206 (left.Min() - right.Max() < kMinInt);
207
208 default:
209 UNREACHABLE()V8_Fatal("unreachable code");
210 }
211 return true;
212}

500 Type Weaken(Node* node, Type previous_type, Type current_type) {
501 // If the types have nothing to do with integers, return the types.
502 Type const integer = type_cache_->kInteger;
503 if (!previous_type.Maybe(integer)) {
504 return current_type;
505 }
506 DCHECK(current_type.Maybe(integer))((void) 0);
507
508 Type current_integer = Type::Intersect(current_type, integer, graph_zone());
509 DCHECK(!current_integer.IsNone())((void) 0);
510 Type previous_integer =
511 Type::Intersect(previous_type, integer, graph_zone());
512 DCHECK(!previous_integer.IsNone())((void) 0);
513
514 // Once we start weakening a node, we should always weaken.
515 if (!GetInfo(node)->weakened()) {
516 // Only weaken if there is range involved; we should converge quickly
517 // for all other types (the exception is a union of many constants,
518 // but we currently do not increase the number of constants in unions).
519 Type previous = previous_integer.GetRange();
520 Type current = current_integer.GetRange();
521 if (current.IsInvalid() || previous.IsInvalid()) {
522 return current_type;
523 }
524 // Range is involved => we are weakening.
525 GetInfo(node)->set_weakened();
526 }
527
528 return Type::Union(current_type,
529 op_typer_.WeakenRange(previous_integer, current_integer),
530 graph_zone());
531 }

533 // Generates a pre-order traversal of the nodes, starting with End.
534 void GenerateTraversal() {
535 ZoneStack<NodeState> stack(zone_);
536
537 stack.push({graph()->end(), 0});
538 GetInfo(graph()->end())->set_pushed();
539 while (!stack.empty()) {
540 NodeState& current = stack.top();
541 Node* node = current.node;
542
543 // If there is an unvisited input, push it and continue with that node.
544 bool pushed_unvisited = false;
545 while (current.input_index < node->InputCount()) {
546 Node* input = node->InputAt(current.input_index);
547 NodeInfo* input_info = GetInfo(input);
548 current.input_index++;
549 if (input_info->unvisited()) {
550 input_info->set_pushed();
551 stack.push({input, 0});
552 pushed_unvisited = true;
553 break;
554 } else if (input_info->pushed()) {
555 // Optimization for the Retype phase.
556 // If we had already pushed (and not visited) an input, it means that
557 // the current node will be visited in the Retype phase before one of
558 // its inputs. If this happens, the current node might need to be
559 // revisited.
560 MarkAsPossibleRevisit(node, input);
561 }
562 }
563
564 if (pushed_unvisited) continue;
565
566 stack.pop();
567 NodeInfo* info = GetInfo(node);
568 info->set_visited();
569
570 // Generate the traversal
571 traversal_nodes_.push_back(node);
572 }
573 }

584 // Tries to update the feedback type of the node, as well as setting its
585 // machine representation (in VisitNode). Returns true iff updating the
586 // feedback type is successful.
587 bool RetypeNode(Node* node) {
588 NodeInfo* info = GetInfo(node);
589 info->set_visited();
590 bool updated = UpdateFeedbackType(node);
591 TRACE(" visit #%d: %s\n", node->id(), node->op()->mnemonic());
592 VisitNode<RETYPE>(node, info->truncation(), nullptr);
593 TRACE(" ==> output %s\n", MachineReprToString(info->representation()));
594 return updated;

598 // change the truncation of one of our inputs (see EnqueueInput<PROPAGATE> for
599 // this). If we change the truncation of an already visited node, we will add
600 // it to the revisit queue.
601 void PropagateTruncation(Node* node) {
602 NodeInfo* info = GetInfo(node);
603 info->set_visited();
604 TRACE(" visit #%d: %s (trunc: %s)\n", node->id(), node->op()->mnemonic(),
605 info->truncation().description());
606 VisitNode<PROPAGATE>(node, info->truncation(), nullptr);
607 }

609 // Backward propagation of truncations to a fixpoint.
610 void RunPropagatePhase() {
611 TRACE("--{Propagate phase}--\n");
612 ResetNodeInfoState();
613 DCHECK(revisit_queue_.empty())((void) 0);
614
615 // Process nodes in reverse post order, with End as the root.
616 for (auto it = traversal_nodes_.crbegin(); it != traversal_nodes_.crend();
617 ++it) {
618 PropagateTruncation(*it);
619
620 while (!revisit_queue_.empty()) {
621 Node* node = revisit_queue_.front();
622 revisit_queue_.pop();
623 PropagateTruncation(node);
624 }
625 }
626 }
627
628 // Forward propagation of types from type feedback to a fixpoint.
629 void RunRetypePhase() {
630 TRACE("--{Retype phase}--\n");
631 ResetNodeInfoState();
632 DCHECK(revisit_queue_.empty())((void) 0);
633
634 for (auto it = traversal_nodes_.cbegin(); it != traversal_nodes_.cend();
635 ++it) {
636 Node* node = *it;
637 if (!RetypeNode(node)) continue;
638
639 auto revisit_it = might_need_revisit_.find(node);
640 if (revisit_it == might_need_revisit_.end()) continue;
641
642 for (Node* const user : revisit_it->second) {
643 PushNodeToRevisitIfVisited(user);
644 }
645
646 // Process the revisit queue.
647 while (!revisit_queue_.empty()) {
648 Node* revisit_node = revisit_queue_.front();
649 revisit_queue_.pop();
650 if (!RetypeNode(revisit_node)) continue;
651 // Here we need to check all uses since we can't easily know which
652 // nodes will need to be revisited due to having an input which was
653 // a revisited node.
654 for (Node* const user : revisit_node->uses()) {
655 PushNodeToRevisitIfVisited(user);
656 }
657 }
658 }
659 }
660
661 // Lowering and change insertion phase.
662 void RunLowerPhase(SimplifiedLowering* lowering) {
663 TRACE("--{Lower phase}--\n");
664 for (auto it = traversal_nodes_.cbegin(); it != traversal_nodes_.cend();
665 ++it) {
666 Node* node = *it;
667 NodeInfo* info = GetInfo(node);
668 TRACE(" visit #%d: %s\n", node->id(), node->op()->mnemonic());
669 // Reuse {VisitNode()} so the representation rules are in one place.
670 SourcePositionTable::Scope scope(
671 source_positions_, source_positions_->GetSourcePosition(node));
672 NodeOriginTable::Scope origin_scope(node_origins_, "simplified lowering",
673 node);
674 VisitNode<LOWER>(node, info->truncation(), lowering);
675 }
676
677 // Perform the final replacements.
678 for (NodeVector::iterator i = replacements_.begin();
679 i != replacements_.end(); ++i) {
680 Node* node = *i;

993 // Infer representation for phi-like nodes.
994 // The {node} parameter is only used to decide on the int64 representation.
995 // Once the type system supports an external pointer type, the {node}
996 // parameter can be removed.
997 MachineRepresentation GetOutputInfoForPhi(Node* node, Type type,
998 Truncation use) {
999 // Compute the representation.
1000 if (type.Is(Type::None())) {
1001 return MachineRepresentation::kNone;
1002 } else if (type.Is(Type::Signed32()) || type.Is(Type::Unsigned32())) {
1003 return MachineRepresentation::kWord32;
1004 } else if (type.Is(Type::NumberOrOddball()) && use.IsUsedAsWord32()) {
1005 return MachineRepresentation::kWord32;
1006 } else if (type.Is(Type::Boolean())) {
1007 return MachineRepresentation::kBit;
1008 } else if (type.Is(Type::NumberOrOddball()) &&
1009 use.TruncatesOddballAndBigIntToNumber()) {
1010 return MachineRepresentation::kFloat64;
1011 } else if (type.Is(Type::Union(Type::SignedSmall(), Type::NaN(), zone()))) {
1012 // TODO(turbofan): For Phis that return either NaN or some Smi, it's
1013 // beneficial to not go all the way to double, unless the uses are
1014 // double uses. For tagging that just means some potentially expensive
1015 // allocation code; we might want to do the same for -0 as well?
1016 return MachineRepresentation::kTagged;
1017 } else if (type.Is(Type::Number())) {
1018 return MachineRepresentation::kFloat64;
1019 } else if (type.Is(Type::BigInt()) && use.IsUsedAsWord64()) {
1020 return MachineRepresentation::kWord64;
1021 } else if (type.Is(Type::ExternalPointer()) ||
1022 type.Is(Type::SandboxedExternalPointer())) {
1023 return MachineType::PointerRepresentation();
1024 }
1025 return MachineRepresentation::kTagged;
1026 }

1331 WriteBarrierKind WriteBarrierKindFor(
1332 BaseTaggedness base_taggedness,
1333 MachineRepresentation field_representation, Type field_type,
1334 MachineRepresentation value_representation, Node* value) {
1335 if (base_taggedness == kTaggedBase &&
1336 CanBeTaggedPointer(field_representation)) {
1337 Type value_type = NodeProperties::GetType(value);
1338 if (value_representation == MachineRepresentation::kTaggedSigned) {
1339 // Write barriers are only for stores of heap objects.
1340 return kNoWriteBarrier;
1341 }
1342 if (field_type.Is(Type::BooleanOrNullOrUndefined()) ||
1343 value_type.Is(Type::BooleanOrNullOrUndefined())) {
1344 // Write barriers are not necessary when storing true, false, null or
1345 // undefined, because these special oddballs are always in the root set.
1346 return kNoWriteBarrier;
1347 }
1348 if (value_type.IsHeapConstant()) {
1349 RootIndex root_index;
1350 const RootsTable& roots_table = jsgraph_->isolate()->roots_table();
1351 if (roots_table.IsRootHandle(value_type.AsHeapConstant()->Value(),
1352 &root_index)) {
1353 if (RootsTable::IsImmortalImmovable(root_index)) {
1354 // Write barriers are unnecessary for immortal immovable roots.
1355 return kNoWriteBarrier;
1356 }
1357 }
1358 }
1359 if (field_representation == MachineRepresentation::kTaggedPointer ||
1360 value_representation == MachineRepresentation::kTaggedPointer) {
1361 // Write barriers for heap objects are cheaper.
1362 return kPointerWriteBarrier;
1363 }
1364 NumberMatcher m(value);
1365 if (m.HasValue()) {
1366 if (IsSmiDouble(m.Value())) {
1367 // Storing a smi doesn't need a write barrier.
1368 return kNoWriteBarrier;
1369 }
1370 // The NumberConstant will be represented as HeapNumber.
1371 return kPointerWriteBarrier;
1372 }
1373 return kFullWriteBarrier;
1374 }
1375 return kNoWriteBarrier;
1376 }
