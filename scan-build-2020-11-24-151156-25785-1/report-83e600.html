611void Http2Session::Close(uint32_t code, bool socket_closed) {
612 Debug(this, "closing session");
613
614 if (is_closing())
615 return;
616 set_closing();
617
618 // Stop reading on the i/o stream
619 if (stream_ != nullptr) {
620 set_reading_stopped();
621 stream_->ReadStop();
622 }
623
624 // If the socket is not closed, then attempt to send a closing GOAWAY
625 // frame. There is no guarantee that this GOAWAY will be received by
626 // the peer but the HTTP/2 spec recommends sending it anyway. We'll
627 // make a best effort.
628 if (!socket_closed) {
629 Debug(this, "terminating session with code %d", code);
630 CHECK_EQ(nghttp2_session_terminate_session(session_.get(), code), 0)do { if (__builtin_expect(!!(!((nghttp2_session_terminate_session

631 SendPendingData();
632 } else if (stream_ != nullptr) {
633 stream_->RemoveStreamListener(this);
634 }
635
636 set_destroyed();
637
638 // If we are writing we will get to make the callback in OnStreamAfterWrite.

701// Used as one of the Padding Strategy functions. Will attempt to ensure
702// that the total frame size, including header bytes, are 8-byte aligned.
703// If maxPayloadLen is smaller than the number of bytes necessary to align,
704// will return maxPayloadLen instead.
705ssize_t Http2Session::OnDWordAlignedPadding(size_t frameLen,
706 size_t maxPayloadLen) {
707 size_t r = (frameLen + 9) % 8;
708 if (r == 0) return frameLen; // If already a multiple of 8, return.
709
710 size_t pad = frameLen + (8 - r);

712 // If maxPayloadLen happens to be less than the calculated pad length,
713 // use the max instead, even tho this means the frame will not be
714 // aligned.
715 pad = std::min(maxPayloadLen, pad);
716 Debug(this, "using frame size padding: %d", pad);
717 return pad;
718}

1019// When nghttp2 receives a DATA frame, it will deliver the data payload to
1020// us in discrete chunks. We push these into a linked list stored in the

1036 // only a precaution at this point.

1053 // into those buffers.

1055 // know about the HTTP2 session associated with this stream, so they know

1078 // more data is being requested.

1089 } while (len != 0);

1096 Debug(session, "receive paused");
1097 return NGHTTP2_ERR_PAUSE;
1098 }

1100 return 0;

1368// Called by OnFrameReceived when a complete PING frame has been received.
1369void Http2Session::HandlePingFrame(const nghttp2_frame* frame) {
1370 Isolate* isolate = env()->isolate();
1371 HandleScope scope(isolate);
1372 Local<Context> context = env()->context();
1373 Context::Scope context_scope(context);
1374 Local<Value> arg;

1376 if (ack) {
1377 BaseObjectPtr<Http2Ping> ping = PopPing();
1378
1379 if (!ping) {
1380 // PING Ack is unsolicited. Treat as a connection error. The HTTP/2
1381 // spec does not require this, but there is no legitimate reason to
1382 // receive an unsolicited PING ack on a connection. Either the peer
1383 // is buggy or malicious, and we're not going to tolerate such
1384 // nonsense.

1396 arg = Buffer::Copy(
1397 env(),
1398 reinterpret_cast<const char*>(frame->ping.opaque_data),
1399 8).ToLocalChecked();
1400 MakeCallback(env()->http2session_on_ping_function(), 1, &arg);
1401}

1440void Http2Session::OnStreamAfterWrite(WriteWrap* w, int status) {

node::Assert(args); } while (0); } } while (0);
1444 set_write_in_progress(false);
1445
1446 // Inform all pending writes about their completion.
1447 ClearOutgoing(status);
1448

1559// Queue a given block of data for sending. This always creates a copy,
1560// so it is used for the cases in which nghttp2 requests sending of a
1561// small chunk of data.
1562void Http2Session::CopyDataIntoOutgoing(const uint8_t* src, size_t src_length) {
1563 size_t offset = outgoing_storage_.size();
1564 outgoing_storage_.resize(offset + src_length);
1565 memcpy(&outgoing_storage_[offset], src, src_length);
1566
1567 // Store with a base of `nullptr` initially, since future resizes

1569 // The correct base pointers will be set later, before writing to the

1576// Prompts nghttp2 to begin serializing it's pending data and pushes each
1577// chunk out to the i/o socket to be sent. This is a particularly hot method
1578// that will generally be called at least twice be event loop iteration.
1579// This is a potential performance optimization target later.
1580// Returns non-zero value if a write is already in progress.
1581uint8_t Http2Session::SendPendingData() {
1582 Debug(this, "sending pending data");
1583 // Do not attempt to send data on the socket if the destroying flag has
1584 // been set. That means everything is shutting down and the socket
1585 // will not be usable.
1586 if (is_destroyed())
1587 return 0;
1588 set_write_scheduled(false);
1589
1590 // SendPendingData should not be called recursively.
1591 if (is_sending())
1592 return 1;
1593 // This is cleared by ClearOutgoing().
1594 set_sending();
1595
1596 ssize_t src_length;
1597 const uint8_t* src;
1598
1599 CHECK(outgoing_buffers_.empty())do { if (__builtin_expect(!!(!(outgoing_buffers_.empty())), 0
)) { do { static const node::AssertionInfo args = { "../src/node_http2.cc"
":" "1599", "outgoing_buffers_.empty()", __PRETTY_FUNCTION__
}; node::Assert(args); } while (0); } } while (0);
1600 CHECK(outgoing_storage_.empty())do { if (__builtin_expect(!!(!(outgoing_storage_.empty())), 0
)) { do { static const node::AssertionInfo args = { "../src/node_http2.cc"
":" "1600", "outgoing_storage_.empty()", __PRETTY_FUNCTION__
}; node::Assert(args); } while (0); } } while (0);
1601
1602 // Part One: Gather data from nghttp2
1603
1604 while ((src_length = nghttp2_session_mem_send(session_.get(), &src)) > 0) {
1605 Debug(this, "nghttp2 has %d bytes to send", src_length);
1606 CopyDataIntoOutgoing(src, src_length);
1607 }
1608
1609 CHECK_NE(src_length, NGHTTP2_ERR_NOMEM)do { if (__builtin_expect(!!(!((src_length) != (NGHTTP2_ERR_NOMEM
))), 0)) { do { static const node::AssertionInfo args = { "../src/node_http2.cc"
":" "1609", "(src_length) != (NGHTTP2_ERR_NOMEM)", __PRETTY_FUNCTION__
}; node::Assert(args); } while (0); } } while (0);
1610
1611 if (stream_ == nullptr) {
1612 // It would seem nice to bail out earlier, but `nghttp2_session_mem_send()`
1613 // does take care of things like closing the individual streams after
1614 // a socket has been torn down, so we still need to call it.
1615 ClearOutgoing(UV_ECANCELED);
1616 return 0;
1617 }
1618
1619 // Part Two: Pass Data to the underlying stream
1620
1621 size_t count = outgoing_buffers_.size();
1622 if (count == 0) {
1623 ClearOutgoing(0);
1624 return 0;
1625 }
1626 MaybeStackBuffer<uv_buf_t, 32> bufs;
1627 bufs.AllocateSufficientStorage(count);
1628
1629 // Set the buffer base pointers for copied data that ended up in the
1630 // sessions's own storage since it might have shifted around during gathering.
1631 // (Those are marked by having .base == nullptr.)
1632 size_t offset = 0;
1633 size_t i = 0;
1634 for (const NgHttp2StreamWrite& write : outgoing_buffers_) {
1635 statistics_.data_sent += write.buf.len;
1636 if (write.buf.base == nullptr) {
1637 bufs[i++] = uv_buf_init(
1638 reinterpret_cast<char*>(outgoing_storage_.data() + offset),
1639 write.buf.len);
1640 offset += write.buf.len;
1641 } else {
1642 bufs[i++] = write.buf;
1643 }
1644 }
1645
1646 chunks_sent_since_last_write_++;
1647
1648 CHECK(!is_write_in_progress())do { if (__builtin_expect(!!(!(!is_write_in_progress())), 0))
{ do { static const node::AssertionInfo args = { "../src/node_http2.cc"
":" "1648", "!is_write_in_progress()", __PRETTY_FUNCTION__ }
; node::Assert(args); } while (0); } } while (0);
1649 set_write_in_progress();
1650 StreamWriteResult res = underlying_stream()->Write(*bufs, count);
1651 if (!res.async) {
1652 set_write_in_progress(false);
1653 ClearOutgoing(res.err);
1654 }
1655
1656 MaybeStopReading();
1657
1658 return 0;
1659}

1662// This callback is called from nghttp2 when it wants to send DATA frames for a
1663// given Http2Stream, when we set the `NGHTTP2_DATA_FLAG_NO_COPY` flag earlier
1664// in the Http2Stream::Provider::Stream::OnRead callback.
1665// We take the write information directly out of the stream's data queue.
1666int Http2Session::OnSendData(
1667 nghttp2_session* session_,
1668 nghttp2_frame* frame,
1669 const uint8_t* framehd,
1670 size_t length,
1671 nghttp2_data_source* source,
1672 void* user_data) {
1673 Http2Session* session = static_cast<Http2Session*>(user_data);
1674 BaseObjectPtr<Http2Stream> stream = session->FindStream(frame->hd.stream_id);
1675 if (!stream) return 0;
1676
1677 // Send the frame header + a byte that indicates padding length.
1678 session->CopyDataIntoOutgoing(framehd, 9);
1679 if (frame->data.padlen > 0) {
1680 uint8_t padding_byte = frame->data.padlen - 1;
1681 CHECK_EQ(padding_byte, frame->data.padlen - 1)do { if (__builtin_expect(!!(!((padding_byte) == (frame->data
.padlen - 1))), 0)) { do { static const node::AssertionInfo args
= { "../src/node_http2.cc" ":" "1681", "(padding_byte) == (frame->data.padlen - 1)"
, __PRETTY_FUNCTION__ }; node::Assert(args); } while (0); } }
while (0);
1682 session->CopyDataIntoOutgoing(&padding_byte, 1);
1683 }
1684
1685 Debug(session, "nghttp2 has %d bytes to send directly", length);
1686 while (length > 0) {
1687 // nghttp2 thinks that there is data available (length > 0), which means
1688 // we told it so, which means that we *should* have data available.
1689 CHECK(!stream->queue_.empty())do { if (__builtin_expect(!!(!(!stream->queue_.empty())), 0
)) { do { static const node::AssertionInfo args = { "../src/node_http2.cc"
":" "1689", "!stream->queue_.empty()", __PRETTY_FUNCTION__
}; node::Assert(args); } while (0); } } while (0);
1690
1691 NgHttp2StreamWrite& write = stream->queue_.front();
1692 if (write.buf.len <= length) {
1693 // This write does not suffice by itself, so we can consume it completely.
1694 length -= write.buf.len;
1695 session->PushOutgoingBuffer(std::move(write));
1696 stream->queue_.pop();
1697 continue;
1698 }
1699
1700 // Slice off `length` bytes of the first write in the queue.
1701 session->PushOutgoingBuffer(NgHttp2StreamWrite {
1702 uv_buf_init(write.buf.base, length)
1703 });
1704 write.buf.base += length;
1705 write.buf.len -= length;
1706 break;
1707 }
1708
1709 if (frame->data.padlen > 0) {
1710 // Send padding if that was requested.
1711 session->PushOutgoingBuffer(NgHttp2StreamWrite {
1712 uv_buf_init(const_cast<char*>(zero_bytes_256), frame->data.padlen - 1)
1713 });
1714 }
1715
1716 return 0;
1717}

1746// Callback used to receive inbound data from the i/o stream
1747void Http2Session::OnStreamRead(ssize_t nread, const uv_buf_t& buf_) {
1748 HandleScope handle_scope(env()->isolate());
1749 Context::Scope context_scope(env()->context());
1750 Http2Scope h2scope(this);
1751 CHECK_NOT_NULL(stream_)do { if (__builtin_expect(!!(!((stream_) != nullptr)), 0)) { do
{ static const node::AssertionInfo args = { "../src/node_http2.cc"
":" "1751", "(stream_) != nullptr", __PRETTY_FUNCTION__ }; node
::Assert(args); } while (0); } } while (0);
1752 Debug(this, "receiving %d bytes, offset %d", nread, stream_buf_offset_);
1753 AllocatedBuffer buf(env(), buf_);
1754
1755 // Only pass data on if nread > 0
1756 if (nread <= 0) {
1757 if (nread < 0) {
1758 PassReadErrorToPreviousListener(nread);
1759 }
1760 return;
1761 }
1762
1763 statistics_.data_received += nread;
1764
1765 if (LIKELY(stream_buf_offset_ == 0)__builtin_expect(!!(stream_buf_offset_ == 0), 1)) {
1766 // Shrink to the actual amount of used data.
1767 buf.Resize(nread);
1768 } else {
1769 // This is a very unlikely case, and should only happen if the ReadStart()
1770 // call in OnStreamAfterWrite() immediately provides data. If that does
1771 // happen, we concatenate the data we received with the already-stored
1772 // pending input data, slicing off the already processed part.
1773 size_t pending_len = stream_buf_.len - stream_buf_offset_;
1774 AllocatedBuffer new_buf =
1775 AllocatedBuffer::AllocateManaged(env(), pending_len + nread);
1776 memcpy(new_buf.data(), stream_buf_.base + stream_buf_offset_, pending_len);
1777 memcpy(new_buf.data() + pending_len, buf.data(), nread);
1778
1779 buf = std::move(new_buf);
1780 nread = buf.size();
1781 stream_buf_offset_ = 0;
1782 stream_buf_ab_.Reset();
1783
1784 // We have now fully processed the stream_buf_ input chunk (by moving the
1785 // remaining part into buf, which will be accounted for below).
1786 DecrementCurrentSessionMemory(stream_buf_.len);
1787 }
1788
1789 IncrementCurrentSessionMemory(nread);
1790
1791 // Remember the current buffer, so that OnDataChunkReceived knows the
1792 // offset of a DATA frame's data into the socket read buffer.
1793 stream_buf_ = uv_buf_init(buf.data(), static_cast<unsigned int>(nread));
1794
1795 Isolate* isolate = env()->isolate();
1796
1797 // Store this so we can create an ArrayBuffer for read data from it.
1798 // DATA frames will be emitted as slices of that ArrayBuffer to avoid having
1799 // to copy memory.
1800 stream_buf_allocation_ = std::move(buf);
1801
1802 ssize_t ret = ConsumeHTTP2Data();
1803
1804 if (UNLIKELY(ret < 0)__builtin_expect(!!(ret < 0), 0)) {
1805 Debug(this, "fatal error receiving data: %d", ret);
1806 Local<Value> arg = Integer::New(isolate, static_cast<int32_t>(ret));
1807 MakeCallback(env()->http2session_on_error_function(), 1, &arg);
1808 return;
1809 }
1810
1811 MaybeStopReading();
1812}

2321 BaseObjectPtr<AsyncWrap> finished =
2322 std::move(stream->queue_.front().req_wrap);
2323 stream->queue_.pop();

2334 // actually taking the buffers out of the queue.
