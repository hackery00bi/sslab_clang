53template <typename T, IrOpcode::Value kOpcode>
54struct ValueMatcher : public NodeMatcher {
55 using ValueType = T;
56
57 explicit ValueMatcher(Node* node) : NodeMatcher(node) {
58 static_assert(kOpcode != IrOpcode::kFoldConstant, "unsupported opcode");
59 if (node->opcode() == IrOpcode::kFoldConstant) {
60 node = node->InputAt(1);
61 }
62 DCHECK_NE(node->opcode(), IrOpcode::kFoldConstant)((void) 0);
63 has_value_ = opcode() == kOpcode;
64 if (has_value_) {
65 value_ = OpParameter<T>(node->op());
66 }
67 }
68
69 bool HasValue() const { return has_value_; }
70 const T& Value() const {
71 DCHECK(HasValue())((void) 0);
72 return value_;
73 }
74
75 private:
76 T value_;
uninitialized field 'this->left_.ValueMatcher::value_'
uninitialized field 'this->right_.ValueMatcher::value_'
77 bool has_value_;
78};

496 (node->op()->HasProperty(Operator::kCommutative)
497 ? AddressOption::kAllowInputSwap
498 : AddressOption::kAllowNone));

501 bool matches() const { return matches_; }
502 Node* index() const { return index_; }
503 int scale() const { return scale_; }
504 Node* base() const { return base_; }
505 Node* displacement() const { return displacement_; }

508 private:
509 bool matches_;
510 Node* index_;
511 int scale_;
512 Node* base_;
513 Node* displacement_;

516 void Initialize(Node* node, AddressOptions options) {
517 // The BaseWithIndexAndDisplacementMatcher canonicalizes the order of
518 // displacements and scale factors that are used as inputs, so instead of
519 // enumerating all possible patterns by brute force, checking for node
520 // clusters using the following templates in the following order suffices to
521 // find all of the interesting cases (S = index * scale, B = base input, D =
522 // displacement input):
523 // (S + (B + D))
524 // (S + (B + B))
525 // (S + D)
526 // (S + B)
527 // ((S + D) + B)
528 // ((S + B) + D)
529 // ((B + D) + B)
530 // ((B + B) + D)
531 // (B + D)
532 // (B + B)
533 if (node->InputCount() < 2) return;
534 AddMatcher m(node, options & AddressOption::kAllowInputSwap);
535 Node* left = m.left().node();
536 Node* right = m.right().node();
537 Node* displacement = nullptr;
538 Node* base = nullptr;
539 Node* index = nullptr;
540 Node* scale_expression = nullptr;
541 bool power_of_two_plus_one = false;
542 DisplacementMode displacement_mode = kPositiveDisplacement;
543 int scale = 0;
544 if (m.HasIndexInput() && OwnedByAddressingOperand(left)) {
545 index = m.IndexInput();
546 scale = m.scale();
547 scale_expression = left;
548 power_of_two_plus_one = m.power_of_two_plus_one();
549 bool match_found = false;
550 if (right->opcode() == AddMatcher::kSubOpcode &&
551 OwnedByAddressingOperand(right)) {
552 AddMatcher right_matcher(right);
553 if (right_matcher.right().HasValue()) {
554 // (S + (B - D))
555 base = right_matcher.left().node();
556 displacement = right_matcher.right().node();
557 displacement_mode = kNegativeDisplacement;
558 match_found = true;
559 }
560 }
561 if (!match_found) {
562 if (right->opcode() == AddMatcher::kAddOpcode &&
563 OwnedByAddressingOperand(right)) {
564 AddMatcher right_matcher(right);
565 if (right_matcher.right().HasValue()) {
566 // (S + (B + D))
567 base = right_matcher.left().node();
568 displacement = right_matcher.right().node();
569 } else {
570 // (S + (B + B))
571 base = right;
572 }
573 } else if (m.right().HasValue()) {
574 // (S + D)
575 displacement = right;
576 } else {
577 // (S + B)
578 base = right;
579 }
580 }
581 } else {
582 bool match_found = false;
583 if (left->opcode() == AddMatcher::kSubOpcode &&
584 OwnedByAddressingOperand(left)) {
585 AddMatcher left_matcher(left);
586 Node* left_left = left_matcher.left().node();
587 Node* left_right = left_matcher.right().node();
588 if (left_matcher.right().HasValue()) {
589 if (left_matcher.HasIndexInput() && left_left->OwnedBy(left)) {
590 // ((S - D) + B)
591 index = left_matcher.IndexInput();
592 scale = left_matcher.scale();
593 scale_expression = left_left;
594 power_of_two_plus_one = left_matcher.power_of_two_plus_one();
595 displacement = left_right;
596 displacement_mode = kNegativeDisplacement;
597 base = right;
598 } else {
599 // ((B - D) + B)
600 index = left_left;
601 displacement = left_right;
602 displacement_mode = kNegativeDisplacement;
603 base = right;
604 }
605 match_found = true;
606 }
607 }
608 if (!match_found) {
609 if (left->opcode() == AddMatcher::kAddOpcode &&
610 OwnedByAddressingOperand(left)) {
611 AddMatcher left_matcher(left);
612 Node* left_left = left_matcher.left().node();
613 Node* left_right = left_matcher.right().node();
614 if (left_matcher.HasIndexInput() && left_left->OwnedBy(left)) {
615 if (left_matcher.right().HasValue()) {
616 // ((S + D) + B)
617 index = left_matcher.IndexInput();
618 scale = left_matcher.scale();
619 scale_expression = left_left;
620 power_of_two_plus_one = left_matcher.power_of_two_plus_one();
621 displacement = left_right;
622 base = right;
623 } else if (m.right().HasValue()) {
624 if (left->OwnedBy(node)) {
625 // ((S + B) + D)
626 index = left_matcher.IndexInput();
627 scale = left_matcher.scale();
628 scale_expression = left_left;
629 power_of_two_plus_one = left_matcher.power_of_two_plus_one();
630 base = left_right;
631 displacement = right;
632 } else {
633 // (B + D)
634 base = left;
635 displacement = right;
636 }
637 } else {
638 // (B + B)
639 index = left;
640 base = right;
641 }
642 } else {
643 if (left_matcher.right().HasValue()) {
644 // ((B + D) + B)
645 index = left_left;
646 displacement = left_right;
647 base = right;
648 } else if (m.right().HasValue()) {
649 if (left->OwnedBy(node)) {
650 // ((B + B) + D)
651 index = left_left;
652 base = left_right;
653 displacement = right;
654 } else {
655 // (B + D)
656 base = left;
657 displacement = right;
658 }
659 } else {
660 // (B + B)
661 index = left;
662 base = right;
663 }
664 }
665 } else {
666 if (m.right().HasValue()) {
667 // (B + D)
668 base = left;
669 displacement = right;
670 } else {
671 // (B + B)
672 base = left;
673 index = right;
674 }
675 }
676 }
677 }
678 int64_t value = 0;
679 if (displacement != nullptr) {
680 switch (displacement->opcode()) {
681 case IrOpcode::kInt32Constant: {
682 value = OpParameter<int32_t>(displacement->op());
683 break;
684 }
685 case IrOpcode::kInt64Constant: {
686 value = OpParameter<int64_t>(displacement->op());
687 break;
688 }
689 default:
690 UNREACHABLE()V8_Fatal("unreachable code");
691 break;
692 }
693 if (value == 0) {
694 displacement = nullptr;
695 }
696 }
697 if (power_of_two_plus_one) {
698 if (base != nullptr) {
699 // If the scale requires explicitly using the index as the base, but a
700 // base is already part of the match, then the (1 << N + 1) scale factor
701 // can't be folded into the match and the entire index * scale
702 // calculation must be computed separately.
703 index = scale_expression;
704 scale = 0;
705 } else {
706 base = index;
707 }
708 }
709 if (!(options & AddressOption::kAllowScale) && scale != 0) {
710 index = scale_expression;
711 scale = 0;
712 }
713 base_ = base;
714 displacement_ = displacement;
715 displacement_mode_ = displacement_mode;
716 index_ = index;
717 scale_ = scale;
718 matches_ = true;
719 }

721 static bool OwnedByAddressingOperand(Node* node) {
722 for (auto use : node->use_edges()) {
723 Node* from = use.from();
724 switch (from->opcode()) {
725 case IrOpcode::kLoad:
726 case IrOpcode::kPoisonedLoad:
727 case IrOpcode::kProtectedLoad:
728 case IrOpcode::kInt32Add:
729 case IrOpcode::kInt64Add:
730 // Skip addressing uses.
731 break;
732 case IrOpcode::kStore:
733 case IrOpcode::kProtectedStore:
734 // If the stored value is this node, it is not an addressing use.
735 if (from->InputAt(2) == node) return false;
736 // Otherwise it is used as an address and skipped.
737 break;
738 default:
739 // Non-addressing use found.
740 return false;
741 }
742 }
743 return true;
744 }

231Node* MachineOperatorReducer::Int32Div(Node* dividend, int32_t divisor) {
232 DCHECK_NE(0, divisor)((void) 0);
233 DCHECK_NE(std::numeric_limits<int32_t>::min(), divisor)((void) 0);
234 base::MagicNumbersForDivision<uint32_t> const mag =
235 base::SignedDivisionByConstant(bit_cast<uint32_t>(divisor));
236 Node* quotient = graph()->NewNode(machine()->Int32MulHigh(), dividend,
237 Uint32Constant(mag.multiplier));
238 if (divisor > 0 && bit_cast<int32_t>(mag.multiplier) < 0) {
239 quotient = Int32Add(quotient, dividend);
240 } else if (divisor < 0 && bit_cast<int32_t>(mag.multiplier) > 0) {
241 quotient = Int32Sub(quotient, dividend);
242 }
243 return Int32Add(Word32Sar(quotient, mag.shift), Word32Shr(dividend, 31));
244}

246Node* MachineOperatorReducer::Uint32Div(Node* dividend, uint32_t divisor) {
247 DCHECK_LT(0u, divisor)((void) 0);
248 // If the divisor is even, we can avoid using the expensive fixup by shifting
249 // the dividend upfront.
250 unsigned const shift = base::bits::CountTrailingZeros(divisor);
251 dividend = Word32Shr(dividend, shift);
252 divisor >>= shift;
253 // Compute the magic number for the (shifted) divisor.
254 base::MagicNumbersForDivision<uint32_t> const mag =
255 base::UnsignedDivisionByConstant(divisor, shift);
256 Node* quotient = graph()->NewNode(machine()->Uint32MulHigh(), dividend,
257 Uint32Constant(mag.multiplier));
258 if (mag.add) {
259 DCHECK_LE(1u, mag.shift)((void) 0);
260 quotient = Word32Shr(
261 Int32Add(Word32Shr(Int32Sub(dividend, quotient), 1), quotient),
262 mag.shift - 1);
263 } else {
264 quotient = Word32Shr(quotient, mag.shift);
265 }
266 return quotient;
267}

1189 Uint32BinopMatcher m(value);
1190 if (m.right().HasValue() && ((rep == MachineRepresentation::kWord8 &&
1191 (m.right().Value() & 0xFF) == 0xFF) ||
1192 (rep == MachineRepresentation::kWord16 &&
1193 (m.right().Value() & 0xFFFF) == 0xFFFF))) {
1194 node->ReplaceInput(value_input, m.left().node());
1195 return Changed(node);
1196 }
1197 break;
1198 }
1199 case IrOpcode::kWord32Sar: {
1200 Int32BinopMatcher m(value);
1201 if (m.left().IsWord32Shl() && ((rep == MachineRepresentation::kWord8 &&
1202 m.right().IsInRange(1, 24)) ||
1203 (rep == MachineRepresentation::kWord16 &&
1204 m.right().IsInRange(1, 16)))) {
1205 Int32BinopMatcher mleft(m.left().node());
1206 if (mleft.right().Is(m.right().Value())) {
1207 node->ReplaceInput(value_input, mleft.left().node());
1208 return Changed(node);
1209 }
1210 }
1211 break;
1212 }
1213 default:
1214 break;
1215 }
1216 return NoChange();

1363Reduction MachineOperatorReducer::ReduceWord32Shl(Node* node) {
1364 DCHECK_EQ(IrOpcode::kWord32Shl, node->opcode())((void) 0);
1365 Int32BinopMatcher m(node);
1366 if (m.right().Is(0)) return Replace(m.left().node()); // x << 0 => x
1367 if (m.IsFoldable()) { // K << K => K (K stands for arbitrary constants)
1368 return ReplaceInt32(
1369 base::ShlWithWraparound(m.left().Value(), m.right().Value()));
1370 }
1371 if (m.right().IsInRange(1, 31)) {
1372 if (m.left().IsWord32Sar() || m.left().IsWord32Shr()) {
1373 Int32BinopMatcher mleft(m.left().node());
1374
1375 // If x >> K only shifted out zeros:
1376 // (x >> K) << L => x if K == L
1377 // (x >> K) << L => x >> (K-L) if K > L
1378 // (x >> K) << L => x << (L-K) if K < L
1379 // Since this is used for Smi untagging, we currently only need it for
1380 // signed shifts.
1381 if (mleft.op() == machine()->Word32SarShiftOutZeros() &&
1382 mleft.right().IsInRange(1, 31)) {
1383 Node* x = mleft.left().node();
1384 int k = mleft.right().Value();
1385 int l = m.right().Value();
1386 if (k == l) {
1387 return Replace(x);
1388 } else if (k > l) {
1389 node->ReplaceInput(0, x);
1390 node->ReplaceInput(1, Uint32Constant(k - l));
1391 NodeProperties::ChangeOp(node, machine()->Word32Sar());
1392 return Changed(node).FollowedBy(ReduceWord32Sar(node));
1393 } else {
1394 DCHECK(k < l)((void) 0);
1395 node->ReplaceInput(0, x);
1396 node->ReplaceInput(1, Uint32Constant(l - k));
1397 return Changed(node);
1398 }
1399 }
1400
1401 // (x >>> K) << K => x & ~(2^K - 1)
1402 // (x >> K) << K => x & ~(2^K - 1)
1403 if (mleft.right().Is(m.right().Value())) {
1404 node->ReplaceInput(0, mleft.left().node());
1405 node->ReplaceInput(1,
1406 Uint32Constant(std::numeric_limits<uint32_t>::max()
1407 << m.right().Value()));
1408 NodeProperties::ChangeOp(node, machine()->Word32And());
1409 return Changed(node).FollowedBy(ReduceWord32And(node));
1410 }
1411 }
1412 }
1413 return ReduceWord32Shifts(node);
1414}

1540template <typename WordNAdapter>
1541Reduction MachineOperatorReducer::ReduceWordNAnd(Node* node) {
1542 using A = WordNAdapter;
1543 A a(this);
1544
1545 typename A::IntNBinopMatcher m(node);
1546 if (m.right().Is(0)) return Replace(m.right().node()); // x & 0 => 0
1547 if (m.right().Is(-1)) return Replace(m.left().node()); // x & -1 => x
1548 if (m.left().IsComparison() && m.right().Is(1)) { // CMP & 1 => CMP
1549 return Replace(m.left().node());
1550 }
1551 if (m.IsFoldable()) { // K & K => K (K stands for arbitrary constants)
1552 return a.ReplaceIntN(m.left().Value() & m.right().Value());
1553 }
1554 if (m.LeftEqualsRight()) return Replace(m.left().node()); // x & x => x
1555 if (A::IsWordNAnd(m.left()) && m.right().HasValue()) {
1556 typename A::IntNBinopMatcher mleft(m.left().node());
1557 if (mleft.right().HasValue()) { // (x & K) & K => x & K
1558 node->ReplaceInput(0, mleft.left().node());
1559 node->ReplaceInput(
1560 1, a.IntNConstant(m.right().Value() & mleft.right().Value()));
1561 return Changed(node).FollowedBy(a.ReduceWordNAnd(node));
1562 }
1563 }
1564 if (m.right().IsNegativePowerOf2()) {
1565 typename A::intN_t const mask = m.right().Value();
1566 typename A::intN_t const neg_mask = base::NegateWithWraparound(mask);
1567 if (A::IsWordNShl(m.left())) {
1568 typename A::UintNBinopMatcher mleft(m.left().node());
1569 if (mleft.right().HasValue() &&
1570 (mleft.right().Value() & (A::WORD_SIZE - 1)) >=
1571 base::bits::CountTrailingZeros(mask)) {
1572 // (x << L) & (-1 << K) => x << L iff L >= K
1573 return Replace(mleft.node());
1574 }
1575 } else if (A::IsIntNAdd(m.left())) {
1576 typename A::IntNBinopMatcher mleft(m.left().node());
1577 if (mleft.right().HasValue() &&
1578 (mleft.right().Value() & mask) == mleft.right().Value()) {
1579 // (x + (K << L)) & (-1 << L) => (x & (-1 << L)) + (K << L)
1580 node->ReplaceInput(0,
1581 a.WordNAnd(mleft.left().node(), m.right().node()));
1582 node->ReplaceInput(1, mleft.right().node());
1583 NodeProperties::ChangeOp(node, a.IntNAdd(machine()));
1584 return Changed(node).FollowedBy(a.ReduceIntNAdd(node));
1585 }
1586 if (A::IsIntNMul(mleft.left())) {
1587 typename A::IntNBinopMatcher mleftleft(mleft.left().node());
1588 if (mleftleft.right().IsMultipleOf(neg_mask)) {
1589 // (y * (K << L) + x) & (-1 << L) => (x & (-1 << L)) + y * (K << L)
1590 node->ReplaceInput(
1591 0, a.WordNAnd(mleft.right().node(), m.right().node()));
1592 node->ReplaceInput(1, mleftleft.node());
1593 NodeProperties::ChangeOp(node, a.IntNAdd(machine()));
1594 return Changed(node).FollowedBy(a.ReduceIntNAdd(node));
1595 }
1596 }
1597 if (A::IsIntNMul(mleft.right())) {
1598 typename A::IntNBinopMatcher mleftright(mleft.right().node());
1599 if (mleftright.right().IsMultipleOf(neg_mask)) {
1600 // (x + y * (K << L)) & (-1 << L) => (x & (-1 << L)) + y * (K << L)
1601 node->ReplaceInput(0,
1602 a.WordNAnd(mleft.left().node(), m.right().node()));
1603 node->ReplaceInput(1, mleftright.node());
1604 NodeProperties::ChangeOp(node, a.IntNAdd(machine()));
1605 return Changed(node).FollowedBy(a.ReduceIntNAdd(node));
1606 }
1607 }
1608 if (A::IsWordNShl(mleft.left())) {
1609 typename A::IntNBinopMatcher mleftleft(mleft.left().node());
1610 if (mleftleft.right().Is(base::bits::CountTrailingZeros(mask))) {
1611 // (y << L + x) & (-1 << L) => (x & (-1 << L)) + y << L
1612 node->ReplaceInput(
1613 0, a.WordNAnd(mleft.right().node(), m.right().node()));
1614 node->ReplaceInput(1, mleftleft.node());
1615 NodeProperties::ChangeOp(node, a.IntNAdd(machine()));
1616 return Changed(node).FollowedBy(a.ReduceIntNAdd(node));
1617 }
1618 }
1619 if (A::IsWordNShl(mleft.right())) {
1620 typename A::IntNBinopMatcher mleftright(mleft.right().node());
1621 if (mleftright.right().Is(base::bits::CountTrailingZeros(mask))) {
1622 // (x + y << L) & (-1 << L) => (x & (-1 << L)) + y << L
1623 node->ReplaceInput(0,
1624 a.WordNAnd(mleft.left().node(), m.right().node()));
1625 node->ReplaceInput(1, mleftright.node());
1626 NodeProperties::ChangeOp(node, a.IntNAdd(machine()));
1627 return Changed(node).FollowedBy(a.ReduceIntNAdd(node));
1628 }
1629 }
1630 } else if (A::IsIntNMul(m.left())) {
1631 typename A::IntNBinopMatcher mleft(m.left().node());
1632 if (mleft.right().IsMultipleOf(neg_mask)) {
1633 // (x * (K << L)) & (-1 << L) => x * (K << L)
1634 return Replace(mleft.node());
1635 }
1636 }
1637 }

1643// Represents an operation of the form `(source & mask) == masked_value`.
1644struct BitfieldCheck {
1645 Node* source;
1646 uint32_t mask;
1647 uint32_t masked_value;
1648 bool truncate_from_64_bit;
1649
1650 static base::Optional<BitfieldCheck> Detect(Node* node) {
1651 // There are two patterns to check for here:
1652 // 1. Single-bit checks: `(val >> shift) & 1`, where:
1653 // - the shift may be omitted, and/or
1654 // - the result may be truncated from 64 to 32
1655 // 2. Equality checks: `(val & mask) == expected`, where:
1656 // - val may be truncated from 64 to 32 before masking (see
1657 // ReduceWord32EqualForConstantRhs)
1658 if (node->opcode() == IrOpcode::kWord32Equal) {
1659 Uint32BinopMatcher eq(node);
1660 if (eq.left().IsWord32And()) {
1661 Uint32BinopMatcher mand(eq.left().node());
1662 if (mand.right().HasValue()) {
1663 BitfieldCheck result{mand.left().node(), mand.right().Value(),
1664 eq.right().Value(), false};
1665 if (mand.left().IsTruncateInt64ToInt32()) {
1666 result.truncate_from_64_bit = true;
1667 result.source =
1668 NodeProperties::GetValueInput(mand.left().node(), 0);
1669 }
1670 return result;
1671 }
1672 }
1673 } else {
1674 if (node->opcode() == IrOpcode::kTruncateInt64ToInt32) {
1675 return TryDetectShiftAndMaskOneBit<Word64Adapter>(
1676 NodeProperties::GetValueInput(node, 0));
1677 } else {
1678 return TryDetectShiftAndMaskOneBit<Word32Adapter>(node);
1679 }
1680 }
1681 return {};
1682 }
1683
1684 base::Optional<BitfieldCheck> TryCombine(const BitfieldCheck& other) {
1685 if (source != other.source ||
1686 truncate_from_64_bit != other.truncate_from_64_bit)
1687 return {};
1688 uint32_t overlapping_bits = mask & other.mask;
1689 // It would be kind of strange to have any overlapping bits, but they can be
1690 // allowed as long as they don't require opposite values in the same
1691 // positions.
1692 if ((masked_value & overlapping_bits) !=
1693 (other.masked_value & overlapping_bits))
1694 return {};
1695 return BitfieldCheck{source, mask | other.mask,
1696 masked_value | other.masked_value,
1697 truncate_from_64_bit};
1698 }
1699
1700 private:
1701 template <typename WordNAdapter>
1702 static base::Optional<BitfieldCheck> TryDetectShiftAndMaskOneBit(Node* node) {
1703 // Look for the pattern `(val >> shift) & 1`. The shift may be omitted.
1704 if (WordNAdapter::IsWordNAnd(NodeMatcher(node))) {
1705 typename WordNAdapter::IntNBinopMatcher mand(node);
1706 if (mand.right().HasValue() && mand.right().Value() == 1) {
1707 if (WordNAdapter::IsWordNShr(mand.left()) ||
1708 WordNAdapter::IsWordNSar(mand.left())) {
1709 typename WordNAdapter::UintNBinopMatcher shift(mand.left().node());
1710 if (shift.right().HasValue() && shift.right().Value() < 32u) {
1711 uint32_t mask = 1 << shift.right().Value();
1712 return BitfieldCheck{shift.left().node(), mask, mask,
1713 WordNAdapter::WORD_SIZE == 64};
1714 }
1715 }
1716 return BitfieldCheck{mand.left().node(), 1, 1,
1717 WordNAdapter::WORD_SIZE == 64};
1718 }
1719 }
1720 return {};
1721 }
1722};

1726Reduction MachineOperatorReducer::ReduceWord32And(Node* node) {
1727 DCHECK_EQ(IrOpcode::kWord32And, node->opcode())((void) 0);
1728 Reduction reduction = ReduceWordNAnd<Word32Adapter>(node);
1729 if (reduction.Changed()) {
1730 return reduction;
1731 }
1732
1733 // Attempt to detect multiple bitfield checks from the same bitfield struct
1734 // and fold them into a single check.
1735 Int32BinopMatcher m(node);
1736 if (auto right_bitfield = BitfieldCheck::Detect(m.right().node())) {
1737 if (auto left_bitfield = BitfieldCheck::Detect(m.left().node())) {
1738 if (auto combined_bitfield = left_bitfield->TryCombine(*right_bitfield)) {
1739 Node* source = combined_bitfield->source;
1740 if (combined_bitfield->truncate_from_64_bit) {
1741 source = TruncateInt64ToInt32(source);
1742 }
1743 node->ReplaceInput(0, Word32And(source, combined_bitfield->mask));
1744 node->ReplaceInput(1, Int32Constant(combined_bitfield->masked_value));
1745 NodeProperties::ChangeOp(node, machine()->Word32Equal());
1746 return Changed(node).FollowedBy(ReduceWord32Equal(node));
1747 }
1748 }
1749 }
1750
1751 return NoChange();
1752}

1759Reduction MachineOperatorReducer::TryMatchWord32Ror(Node* node) {
1760 DCHECK(IrOpcode::kWord32Or == node->opcode() ||((void) 0)
1761 IrOpcode::kWord32Xor == node->opcode())((void) 0);
1762 Int32BinopMatcher m(node);
1763 Node* shl = nullptr;
1764 Node* shr = nullptr;
1765 // Recognize rotation, we are matching:
1766 // * x << y | x >>> (32 - y) => x ror (32 - y), i.e x rol y
1767 // * x << (32 - y) | x >>> y => x ror y
1768 // * x << y ^ x >>> (32 - y) => x ror (32 - y), i.e. x rol y
1769 // * x << (32 - y) ^ x >>> y => x ror y
1770 // as well as their commuted form.
1771 if (m.left().IsWord32Shl() && m.right().IsWord32Shr()) {
1772 shl = m.left().node();
1773 shr = m.right().node();
1774 } else if (m.left().IsWord32Shr() && m.right().IsWord32Shl()) {
1775 shl = m.right().node();
1776 shr = m.left().node();
1777 } else {
1778 return NoChange();
1779 }
1780
1781 Int32BinopMatcher mshl(shl);
1782 Int32BinopMatcher mshr(shr);
1783 if (mshl.left().node() != mshr.left().node()) return NoChange();
1784
1785 if (mshl.right().HasValue() && mshr.right().HasValue()) {
1786 // Case where y is a constant.
1787 if (mshl.right().Value() + mshr.right().Value() != 32) return NoChange();
1788 } else {
1789 Node* sub = nullptr;
1790 Node* y = nullptr;
1791 if (mshl.right().IsInt32Sub()) {
1792 sub = mshl.right().node();
1793 y = mshr.right().node();
1794 } else if (mshr.right().IsInt32Sub()) {
1795 sub = mshr.right().node();
1796 y = mshl.right().node();
1797 } else {
1798 return NoChange();
1799 }
1800
1801 Int32BinopMatcher msub(sub);
1802 if (!msub.left().Is(32) || msub.right().node() != y) return NoChange();
1803 }
1804
1805 node->ReplaceInput(0, mshl.left().node());
1806 node->ReplaceInput(1, mshr.right().node());
1807 NodeProperties::ChangeOp(node, machine()->Word32Ror());
1808 return Changed(node);
1809}

1811template <typename WordNAdapter>
1812Reduction MachineOperatorReducer::ReduceWordNOr(Node* node) {
1813 using A = WordNAdapter;
1814 A a(this);
1815
1816 typename A::IntNBinopMatcher m(node);
1817 if (m.right().Is(0)) return Replace(m.left().node()); // x | 0 => x
1818 if (m.right().Is(-1)) return Replace(m.right().node()); // x | -1 => -1
1819 if (m.IsFoldable()) { // K | K => K (K stands for arbitrary constants)
1820 return a.ReplaceIntN(m.left().Value() | m.right().Value());
1821 }
1822 if (m.LeftEqualsRight()) return Replace(m.left().node()); // x | x => x
1823
1824 // (x & K1) | K2 => x | K2 if K2 has ones for every zero bit in K1.
1825 // This case can be constructed by UpdateWord and UpdateWord32 in CSA.
1826 if (m.right().HasValue()) {
1827 if (A::IsWordNAnd(m.left())) {
1828 typename A::IntNBinopMatcher mand(m.left().node());
1829 if (mand.right().HasValue()) {
1830 if ((m.right().Value() | mand.right().Value()) == -1) {
1831 node->ReplaceInput(0, mand.left().node());
1832 return Changed(node);
1833 }
1834 }
1835 }
1836 }
1837
1838 return a.TryMatchWordNRor(node);
1839}

1882Reduction MachineOperatorReducer::ReduceWord32Equal(Node* node) {
1883 Int32BinopMatcher m(node);
1884 if (m.IsFoldable()) { // K == K => K (K stands for arbitrary constants)
1885 return ReplaceBool(m.left().Value() == m.right().Value());
1886 }
1887 if (m.left().IsInt32Sub() && m.right().Is(0)) { // x - y == 0 => x == y
1888 Int32BinopMatcher msub(m.left().node());
1889 node->ReplaceInput(0, msub.left().node());
1890 node->ReplaceInput(1, msub.right().node());
1891 return Changed(node);
1892 }
1893 // TODO(turbofan): fold HeapConstant, ExternalReference, pointer compares
1894 if (m.LeftEqualsRight()) return ReplaceBool(true); // x == x => true
1895 if (m.right().HasValue()) {
1896 base::Optional<std::pair<Node*, uint32_t>> replacements;
1897 if (m.left().IsTruncateInt64ToInt32()) {
1898 replacements = ReduceWord32EqualForConstantRhs<Word64Adapter>(
1899 NodeProperties::GetValueInput(m.left().node(), 0),
1900 static_cast<uint32_t>(m.right().Value()));
1901 } else {
1902 replacements = ReduceWord32EqualForConstantRhs<Word32Adapter>(
1903 m.left().node(), static_cast<uint32_t>(m.right().Value()));
1904 }
1905 if (replacements) {
1906 node->ReplaceInput(0, replacements->first);
1907 node->ReplaceInput(1, Uint32Constant(replacements->second));
1908 return Changed(node);
1909 }
1910 }
1911
1912 return NoChange();
1913}

1952Reduction MachineOperatorReducer::ReduceFloat64Compare(Node* node) {
1953 DCHECK(IrOpcode::kFloat64Equal == node->opcode() ||((void) 0)
1954 IrOpcode::kFloat64LessThan == node->opcode() ||((void) 0)
1955 IrOpcode::kFloat64LessThanOrEqual == node->opcode())((void) 0);
1956 Float64BinopMatcher m(node);
1957 if (m.IsFoldable()) {
1958 switch (node->opcode()) {
1959 case IrOpcode::kFloat64Equal:
1960 return ReplaceBool(m.left().Value() == m.right().Value());
1961 case IrOpcode::kFloat64LessThan:
1962 return ReplaceBool(m.left().Value() < m.right().Value());
1963 case IrOpcode::kFloat64LessThanOrEqual:
1964 return ReplaceBool(m.left().Value() <= m.right().Value());
1965 default:
1966 UNREACHABLE()V8_Fatal("unreachable code");
1967 }
1968 } else if ((m.left().IsChangeFloat32ToFloat64() &&
1969 m.right().IsChangeFloat32ToFloat64()) ||
1970 (m.left().IsChangeFloat32ToFloat64() &&
1971 IsFloat64RepresentableAsFloat32(m.right())) ||
1972 (IsFloat64RepresentableAsFloat32(m.left()) &&
1973 m.right().IsChangeFloat32ToFloat64())) {
1974 // As all Float32 values have an exact representation in Float64, comparing
1975 // two Float64 values both converted from Float32 is equivalent to comparing
1976 // the original Float32s, so we can ignore the conversions. We can also
1977 // reduce comparisons of converted Float64 values against constants that
1978 // can be represented exactly as Float32.
1979 switch (node->opcode()) {
1980 case IrOpcode::kFloat64Equal:
1981 NodeProperties::ChangeOp(node, machine()->Float32Equal());
1982 break;
1983 case IrOpcode::kFloat64LessThan:
1984 NodeProperties::ChangeOp(node, machine()->Float32LessThan());
1985 break;
1986 case IrOpcode::kFloat64LessThanOrEqual:
1987 NodeProperties::ChangeOp(node, machine()->Float32LessThanOrEqual());
1988 break;
1989 default:
1990 UNREACHABLE()V8_Fatal("unreachable code");
1991 }
1992 node->ReplaceInput(
1993 0, m.left().HasValue()
1994 ? Float32Constant(static_cast<float>(m.left().Value()))
1995 : m.left().InputAt(0));
1996 node->ReplaceInput(
1997 1, m.right().HasValue()
1998 ? Float32Constant(static_cast<float>(m.right().Value()))
1999 : m.right().InputAt(0));
2000 return Changed(node);
2001 }
2002 return NoChange();
2003}

2014Reduction MachineOperatorReducer::ReduceConditional(Node* node) {
2015 DCHECK(node->opcode() == IrOpcode::kBranch ||((void) 0)
2016 node->opcode() == IrOpcode::kDeoptimizeIf ||((void) 0)
2017 node->opcode() == IrOpcode::kDeoptimizeUnless ||((void) 0)
2018 node->opcode() == IrOpcode::kTrapIf ||((void) 0)
2019 node->opcode() == IrOpcode::kTrapUnless)((void) 0);
2020 // This reducer only applies operator reductions to the branch condition.
2021 // Reductions involving control flow happen elsewhere. Non-zero inputs are
2022 // considered true in all conditional ops.
2023 NodeMatcher condition(NodeProperties::GetValueInput(node, 0));
2024 if (condition.IsTruncateInt64ToInt32()) {
2025 if (auto replacement =
2026 ReduceConditionalN<Word64Adapter>(condition.node())) {
2027 NodeProperties::ReplaceValueInput(node, *replacement, 0);
2028 return Changed(node);
2029 }
2030 } else if (auto replacement = ReduceConditionalN<Word32Adapter>(node)) {
2031 NodeProperties::ReplaceValueInput(node, *replacement, 0);
2032 return Changed(node);
2033 }
2034 return NoChange();
2035}

2037template <typename WordNAdapter>
2038base::Optional<Node*> MachineOperatorReducer::ReduceConditionalN(Node* node) {
2039 NodeMatcher condition(NodeProperties::GetValueInput(node, 0));
2040 // Branch conditions are 32-bit comparisons against zero, so they are the
2041 // opposite of a 32-bit `x == 0` node. To avoid repetition, we can reuse logic
2042 // for Word32Equal: if `x == 0` can reduce to `y == 0`, then branch(x) can
2043 // reduce to branch(y).
2044 auto replacements =
2045 ReduceWord32EqualForConstantRhs<WordNAdapter>(condition.node(), 0);
2046 if (replacements && replacements->second == 0) return replacements->first;
2047 return {};
2048}

2050template <typename WordNAdapter>
2051base::Optional<std::pair<Node*, uint32_t>>
2052MachineOperatorReducer::ReduceWord32EqualForConstantRhs(Node* lhs,
2053 uint32_t rhs) {
2054 if (WordNAdapter::IsWordNAnd(NodeMatcher(lhs))) {
2055 typename WordNAdapter::UintNBinopMatcher mand(lhs);
2056 if ((WordNAdapter::IsWordNShr(mand.left()) ||
2057 WordNAdapter::IsWordNSar(mand.left())) &&
2058 mand.right().HasValue()) {
2059 typename WordNAdapter::UintNBinopMatcher mshift(mand.left().node());
2060 // ((x >> K1) & K2) == K3 => (x & (K2 << K1)) == (K3 << K1)
2061 if (mshift.right().HasValue()) {
2062 auto shift_bits = mshift.right().Value();
2063 auto mask = mand.right().Value();
2064 // Make sure that we won't shift data off the end, and that all of the
2065 // data ends up in the lower 32 bits for 64-bit mode.
2066 if (shift_bits <= base::bits::CountLeadingZeros(mask) &&
2067 shift_bits <= base::bits::CountLeadingZeros(rhs) &&
2068 mask << shift_bits <= std::numeric_limits<uint32_t>::max()) {
2069 Node* new_input = mshift.left().node();
2070 uint32_t new_mask = static_cast<uint32_t>(mask << shift_bits);
2071 uint32_t new_rhs = rhs << shift_bits;
2072 if (WordNAdapter::WORD_SIZE == 64) {
2073 // We can truncate before performing the And.
2074 new_input = TruncateInt64ToInt32(new_input);
2075 }
2076 return std::make_pair(Word32And(new_input, new_mask), new_rhs);
2077 }
2078 }
2079 }
2080 }
2081 return {};
2082}
