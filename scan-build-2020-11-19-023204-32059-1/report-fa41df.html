196 const AllocationStats& stats) {

198 DCHECK(pages_.empty())((void) 0);
199 DCHECK(!pages.empty())((void) 0);
200
201 // It's not possible to copy the AllocationStats directly as the new pages
202 // will be mapped to different addresses.
203 stats_.IncreaseCapacity(stats.Capacity());
204
205 v8::PageAllocator* page_allocator = GetPlatformPageAllocator();
206 DCHECK(page_allocator->CanAllocateSharedPages())((void) 0);
207
208 for (const ReadOnlyPage* page : pages) {
209 size_t size = RoundUp(page->size(), page_allocator->AllocatePageSize());
210 // 1. Allocate some new memory for a shared copy of the page and copy the
211 // original contents into it. Doesn't need to be V8 page aligned, since
212 // we'll never use it directly.
213 auto shared_memory = page_allocator->AllocateSharedPages(size, page);
214 void* ptr = shared_memory->GetMemory();
215 CHECK_NOT_NULL(ptr)do { if ((__builtin_expect(!!(!((ptr) != nullptr)), 0))) { V8_Fatal

216
217 // 2. Copy the contents of the original page into the shared page.
218 ReadOnlyPage* new_page = reinterpret_cast<ReadOnlyPage*>(ptr);
219
220 pages_.push_back(new_page);
221 shared_memory_.push_back(std::move(shared_memory));
222 // This is just CompressTagged but inlined so it will always compile.
223 Tagged_t compressed_address = CompressTagged(page->address());
224 page_offsets_.push_back(compressed_address);
225
226 // 3. Update the accounting stats so the allocated bytes are for the new
227 // shared page rather than the original.
228 stats_.IncreaseAllocatedBytes(page->allocated_bytes(), new_page);
229 }

231 InitializeRootsFrom(isolate);

410// Only iterates over a single chunk as the chunk iteration is done externally.
411class ReadOnlySpaceObjectIterator : public ObjectIterator {
412 public:
413 ReadOnlySpaceObjectIterator(Heap* heap, ReadOnlySpace* space,
414 BasicMemoryChunk* chunk)
415 : cur_addr_(kNullAddress), cur_end_(kNullAddress), space_(space) {}
416
417 // Advance to the next object, skipping free spaces and other fillers and
418 // skipping the special garbage section of which there is one per space.
419 // Returns nullptr when the iteration has ended.
420 HeapObject Next() override {
421 HeapObject next_obj = FromCurrentPage();
422 if (!next_obj.is_null()) return next_obj;
423 return HeapObject();
424 }
425
426 private:
427 HeapObject FromCurrentPage() {
428 while (cur_addr_ != cur_end_) {
429 if (cur_addr_ == space_->top() && cur_addr_ != space_->limit()) {
430 cur_addr_ = space_->limit();
431 continue;
432 }
433 HeapObject obj = HeapObject::FromAddress(cur_addr_);
434 const int obj_size = obj.Size();
435 cur_addr_ += obj_size;
436 DCHECK_LE(cur_addr_, cur_end_)((void) 0);
437 if (!obj.IsFreeSpaceOrFiller()) {
438 if (obj.IsCode()) {
439 DCHECK(Code::cast(obj).is_builtin())((void) 0);
440 DCHECK_CODEOBJECT_SIZE(obj_size, space_)((void) 0);
441 } else {
442 DCHECK_OBJECT_SIZE(obj_size)((void) 0);
443 }
444 return obj;
445 }
446 }
447 return HeapObject();
448 }
449
450 Address cur_addr_; // Current iteration point.
451 Address cur_end_; // End iteration point.
452 ReadOnlySpace* space_;
453};

481void ReadOnlySpace::Verify(Isolate* isolate) {
482 bool allocation_pointer_found_in_space = top_ == limit_;
483 VerifyReadOnlyPointersVisitor visitor(isolate->heap());
484
485 for (BasicMemoryChunk* page : pages_) {
486 if (ReadOnlyHeap::IsReadOnlySpaceShared()) {
487 CHECK_NULL(page->owner())do { if ((__builtin_expect(!!(!((page->owner()) == nullptr
)), 0))) { V8_Fatal("Check failed: %s.", "(page->owner()) == nullptr"
); } } while (false);
488 } else {
489 CHECK_EQ(page->owner(), this)do { bool _cmp = ::v8::base::CmpEQImpl< typename ::v8::base
::pass_value_or_ref<decltype(page->owner())>::type, typename
::v8::base::pass_value_or_ref<decltype(this)>::type>
((page->owner()), (this)); do { if ((__builtin_expect(!!(!
(_cmp)), 0))) { V8_Fatal("Check failed: %s.", "page->owner()"
" " "==" " " "this"); } } while (false); } while (false);
490 }
491
492 if (page == Page::FromAllocationAreaAddress(top_)) {
493 allocation_pointer_found_in_space = true;
494 }
495 ReadOnlySpaceObjectIterator it(isolate->heap(), this, page);
496 Address end_of_previous_object = page->area_start();
497 Address top = page->area_end();
498
499 for (HeapObject object = it.Next(); !object.is_null(); object = it.Next()) {
500 CHECK(end_of_previous_object <= object.address())do { if ((__builtin_expect(!!(!(end_of_previous_object <= object
.address())), 0))) { V8_Fatal("Check failed: %s.", "end_of_previous_object <= object.address()"
); } } while (false);
501
502 Map map = object.map();
503 CHECK(map.IsMap())do { if ((__builtin_expect(!!(!(map.IsMap())), 0))) { V8_Fatal
("Check failed: %s.", "map.IsMap()"); } } while (false);
504
505 // The object itself should look OK.
506 object.ObjectVerify(isolate);
507
508 // All the interior pointers should be contained in the heap.
509 int size = object.Size();
510 object.IterateBody(map, size, &visitor);
511 CHECK(object.address() + size <= top)do { if ((__builtin_expect(!!(!(object.address() + size <=
top)), 0))) { V8_Fatal("Check failed: %s.", "object.address() + size <= top"
); } } while (false);
512 end_of_previous_object = object.address() + size;
513
514 CHECK(!object.IsExternalString())do { if ((__builtin_expect(!!(!(!object.IsExternalString())),
0))) { V8_Fatal("Check failed: %s.", "!object.IsExternalString()"
); } } while (false);
515 CHECK(!object.IsJSArrayBuffer())do { if ((__builtin_expect(!!(!(!object.IsJSArrayBuffer())), 0
))) { V8_Fatal("Check failed: %s.", "!object.IsJSArrayBuffer()"
); } } while (false);
516 }
517 }
518 CHECK(allocation_pointer_found_in_space)do { if ((__builtin_expect(!!(!(allocation_pointer_found_in_space
)), 0))) { V8_Fatal("Check failed: %s.", "allocation_pointer_found_in_space"
); } } while (false);
519
520#ifdef DEBUG
521 VerifyCounters(isolate->heap());
522#endif
523}

561void ReadOnlySpace::FreeLinearAllocationArea() {
562 // Mark the old linear allocation area with a free space map so it can be
563 // skipped when scanning the heap.
564 if (top_ == kNullAddress) {
565 DCHECK_EQ(kNullAddress, limit_)((void) 0);
566 return;
567 }
568
569 // Clear the bits in the unused black area.
570 ReadOnlyPage* page = pages_.back();
571 heap()->incremental_marking()->marking_state()->bitmap(page)->ClearRange(
572 page->AddressToMarkbitIndex(top_), page->AddressToMarkbitIndex(limit_));
573
574 heap()->CreateFillerObjectAt(top_, static_cast<int>(limit_ - top_),
575 ClearRecordedSlots::kNo);
576
577 BasicMemoryChunk::UpdateHighWaterMark(top_);
578
579 top_ = kNullAddress;
580 limit_ = kNullAddress;
581}

583void ReadOnlySpace::EnsureSpaceForAllocation(int size_in_bytes) {
584 if (top_ + size_in_bytes <= limit_) {
585 return;
586 }
587
588 DCHECK_GE(size_in_bytes, 0)((void) 0);
589
590 FreeLinearAllocationArea();
591
592 BasicMemoryChunk* chunk =
593 heap()->memory_allocator()->AllocateReadOnlyPage(AreaSize(), this);
594 capacity_ += AreaSize();
595
596 accounting_stats_.IncreaseCapacity(chunk->area_size());
597 AccountCommitted(chunk->size());
598 CHECK_NOT_NULL(chunk)do { if ((__builtin_expect(!!(!((chunk) != nullptr)), 0))) { V8_Fatal
("Check failed: %s.", "(chunk) != nullptr"); } } while (false
);
599 pages_.push_back(static_cast<ReadOnlyPage*>(chunk));
600
601 heap()->CreateFillerObjectAt(chunk->area_start(),
602 static_cast<int>(chunk->area_size()),
603 ClearRecordedSlots::kNo);
604
605 top_ = chunk->area_start();
606 limit_ = chunk->area_end();
607 return;
608}

610HeapObject ReadOnlySpace::TryAllocateLinearlyAligned(
611 int size_in_bytes, AllocationAlignment alignment) {
612 Address current_top = top_;
613 int filler_size = Heap::GetFillToAlign(current_top, alignment);
614
615 Address new_top = current_top + filler_size + size_in_bytes;
616 if (new_top > limit_) return HeapObject();
617
618 // Allocation always occurs in the last chunk for RO_SPACE.
619 BasicMemoryChunk* chunk = pages_.back();
620 int allocated_size = filler_size + size_in_bytes;
621 accounting_stats_.IncreaseAllocatedBytes(allocated_size, chunk);
622 chunk->IncreaseAllocatedBytes(allocated_size);
623
624 top_ = new_top;
625 if (filler_size > 0) {
626 return Heap::PrecedeWithFiller(ReadOnlyRoots(heap()),
627 HeapObject::FromAddress(current_top),
628 filler_size);
629 }
630
631 return HeapObject::FromAddress(current_top);
632}

690size_t ReadOnlyPage::ShrinkToHighWaterMark() {
691 // Shrink pages to high water mark. The water mark points either to a filler
692 // or the area_end.
693 HeapObject filler = HeapObject::FromAddress(HighWaterMark());
694 if (filler.address() == area_end()) return 0;
695 CHECK(filler.IsFreeSpaceOrFiller())do { if ((__builtin_expect(!!(!(filler.IsFreeSpaceOrFiller())
), 0))) { V8_Fatal("Check failed: %s.", "filler.IsFreeSpaceOrFiller()"
); } } while (false);
696 DCHECK_EQ(filler.address() + filler.Size(), area_end())((void) 0);
697
698 size_t unused = RoundDown(static_cast<size_t>(area_end() - filler.address()),
699 MemoryAllocator::GetCommitPageSize());
700 if (unused > 0) {
701 DCHECK_EQ(0u, unused % MemoryAllocator::GetCommitPageSize())((void) 0);
702 if (FLAG_trace_gc_verbose) {
703 PrintIsolate(heap()->isolate(), "Shrinking page %p: end %p -> %p\n",
704 reinterpret_cast<void*>(this),
705 reinterpret_cast<void*>(area_end()),
706 reinterpret_cast<void*>(area_end() - unused));
707 }
708 heap()->CreateFillerObjectAt(
709 filler.address(),
710 static_cast<int>(area_end() - filler.address() - unused),
711 ClearRecordedSlots::kNo);
712 heap()->memory_allocator()->PartialFreeMemory(
713 this, address() + size() - unused, unused, area_end() - unused);
714 if (filler.address() != area_end()) {
715 CHECK(filler.IsFreeSpaceOrFiller())do { if ((__builtin_expect(!!(!(filler.IsFreeSpaceOrFiller())
), 0))) { V8_Fatal("Check failed: %s.", "filler.IsFreeSpaceOrFiller()"
); } } while (false);
716 CHECK_EQ(filler.address() + filler.Size(), area_end())do { bool _cmp = ::v8::base::CmpEQImpl< typename ::v8::base
::pass_value_or_ref<decltype(filler.address() + filler.Size
())>::type, typename ::v8::base::pass_value_or_ref<decltype
(area_end())>::type>((filler.address() + filler.Size())
, (area_end())); do { if ((__builtin_expect(!!(!(_cmp)), 0)))
{ V8_Fatal("Check failed: %s.", "filler.address() + filler.Size()"
" " "==" " " "area_end()"); } } while (false); } while (false
);
717 }
718 }
719 return unused;
720}

722void ReadOnlySpace::ShrinkPages() {
723 BasicMemoryChunk::UpdateHighWaterMark(top_);
724 heap()->CreateFillerObjectAt(top_, static_cast<int>(limit_ - top_),
725 ClearRecordedSlots::kNo);
726
727 for (ReadOnlyPage* chunk : pages_) {
728 DCHECK(chunk->IsFlagSet(Page::NEVER_EVACUATE))((void) 0);
729 size_t unused = chunk->ShrinkToHighWaterMark();
730 capacity_ -= unused;
731 accounting_stats_.DecreaseCapacity(static_cast<intptr_t>(unused));
732 AccountUncommitted(unused);
733 }
734 limit_ = pages_.back()->area_end();
735}
