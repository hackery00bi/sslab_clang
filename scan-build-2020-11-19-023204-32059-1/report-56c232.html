508// The minimum fraction digits is the number of '0' after '.' in the skeleton
509// The maximum fraction digits is the number of '#' after the above '0's plus
510// the minimum fraction digits.
511// For example, as skeleton .000#### rounding-mode-half-up
512// 123
513// 4567
514// Set The minimum as 3 and maximum as 7.
515bool JSNumberFormat::FractionDigitsFromSkeleton(
516 const icu::UnicodeString& skeleton, int32_t* minimum, int32_t* maximum) {
517 icu::UnicodeString search(".");
518 int32_t index = skeleton.indexOf(search);
519 if (index < 0) return false;
520 *minimum = 0;
521 index++; // skip the '.'
522 while (index < skeleton.length() && skeleton[index] == '0') {
523 (*minimum)++;
524 index++;
525 }
526 *maximum = *minimum;
527 while (index < skeleton.length() && skeleton[index] == '#') {
528 (*maximum)++;
529 index++;
530 }
531 return true;
532}
533
534// Return true if there are significant digits, false if not.
535// The minimum significant digits is the number of '@' in the skeleton
536// The maximum significant digits is the number of '#' after these '@'s plus
537// the minimum significant digits.
538// Ex: Skeleton as "@@@@@####### rounding-mode-half-up"
539// 12345
540// 6789012
541// Set The minimum as 5 and maximum as 12.
542bool JSNumberFormat::SignificantDigitsFromSkeleton(
543 const icu::UnicodeString& skeleton, int32_t* minimum, int32_t* maximum) {
544 icu::UnicodeString search("@");
545 int32_t index = skeleton.indexOf(search);
546 if (index < 0) return false;
547 *minimum = 1;
548 index++; // skip the first '@'
549 while (index < skeleton.length() && skeleton[index] == '@') {
550 (*minimum)++;
551 index++;
552 }
553 *maximum = *minimum;
554 while (index < skeleton.length() && skeleton[index] == '#') {
555 (*maximum)++;
556 index++;
557 }
558 return true;
559}

563// Ex: percent .### rounding-mode-half-up
564// Special case for "percent"
565// Ex: "measure-unit/length-kilometer per-measure-unit/duration-hour .###
566// rounding-mode-half-up" should return "kilometer-per-unit".
567// Ex: "measure-unit/duration-year .### rounding-mode-half-up" should return
568// "year".
569std::string UnitFromSkeleton(const icu::UnicodeString& skeleton) {
570 std::string str;
571 str = skeleton.toUTF8String<std::string>(str);
572 std::string search("measure-unit/");
573 size_t begin = str.find(search);
574 if (begin == str.npos) {
575 // Special case for "percent".
576 if (str.find("percent") != str.npos) {
577 return "percent";
578 }
579 return "";
580 }
581 // Skip the type (ex: "length").
582 // "measure-unit/length-kilometer per-measure-unit/duration-hour"
583 // b
584 begin = str.find("-", begin + search.size());
585 if (begin == str.npos) {
586 return "";
587 }
588 begin++; // Skip the '-'.
589 // Find the end of the subtype.
590 size_t end = str.find(" ", begin);
591 // "measure-unit/length-kilometer per-measure-unit/duration-hour"
592 // b e
593 if (end == str.npos) {
594 end = str.size();
595 return str.substr(begin, end - begin);
596 }
597 // "measure-unit/length-kilometer per-measure-unit/duration-hour"
598 // b e
599 // [result ]
600 std::string result = str.substr(begin, end - begin);
601 begin = end + 1;
602 // "measure-unit/length-kilometer per-measure-unit/duration-hour"
603 // [result ]eb
604 std::string search_per("per-measure-unit/");
605 begin = str.find(search_per, begin);
606 // "measure-unit/length-kilometer per-measure-unit/duration-hour"
607 // [result ]e b
608 if (begin == str.npos) {
609 return result;
610 }
611 // Skip the type (ex: "duration").
612 begin = str.find("-", begin + search_per.size());
613 // "measure-unit/length-kilometer per-measure-unit/duration-hour"
614 // [result ]e b
615 if (begin == str.npos) {
616 return result;
617 }
618 begin++; // Skip the '-'.
619 // "measure-unit/length-kilometer per-measure-unit/duration-hour"
620 // [result ]e b
621 end = str.find(" ", begin);
622 if (end == str.npos) {
623 end = str.size();
624 }
625 // "measure-unit/length-kilometer per-measure-unit/duration-hour"
626 // [result ] b e
627 return result + "-per-" + str.substr(begin, end - begin);
628}

646icu::number::LocalizedNumberFormatter
647JSNumberFormat::SetDigitOptionsToFormatter(
648 const icu::number::LocalizedNumberFormatter& icu_number_formatter,
649 const Intl::NumberFormatDigitOptions& digit_options) {
650 icu::number::LocalizedNumberFormatter result = icu_number_formatter;
651 if (digit_options.minimum_integer_digits > 1) {
652 result = result.integerWidth(icu::number::IntegerWidth::zeroFillTo(
653 digit_options.minimum_integer_digits));
654 }
655
656 // Value -1 of minimum_significant_digits represent the roundingtype is
657 // "compact-rounding".
658 if (digit_options.minimum_significant_digits < 0) {
659 return result;
660 }
661 icu::number::Precision precision =
662 (digit_options.minimum_significant_digits > 0)
663 ? icu::number::Precision::minMaxSignificantDigits(
664 digit_options.minimum_significant_digits,
665 digit_options.maximum_significant_digits)
666 : icu::number::Precision::minMaxFraction(
667 digit_options.minimum_fraction_digits,
668 digit_options.maximum_fraction_digits);
669
670 return result.precision(precision);
671}

1298bool cmp_NumberFormatSpan(const NumberFormatSpan& a,
1299 const NumberFormatSpan& b) {
1300 // Regions that start earlier should be encountered earlier.
1301 if (a.begin_pos < b.begin_pos) return true;
1302 if (a.begin_pos > b.begin_pos) return false;
1303 // For regions that start in the same place, regions that last longer should
1304 // be encountered earlier.
1305 if (a.end_pos < b.end_pos) return false;
1306 if (a.end_pos > b.end_pos) return true;
1307 // For regions that are exactly the same, one of them must be the "literal"
1308 // backdrop we added, which has a field_id of -1, so consider higher field_ids
1309 // to be later.
1310 return a.field_id < b.field_id;
1311}

1315// Flattens a list of possibly-overlapping "regions" to a list of
1316// non-overlapping "parts". At least one of the input regions must span the
1317// entire space of possible indexes. The regions parameter will sorted in-place
1318// according to some criteria; this is done for performance to avoid copying the
1319// input.
1320std::vector<NumberFormatSpan> FlattenRegionsToParts(
1321 std::vector<NumberFormatSpan>* regions) {
1322 // The intention of this algorithm is that it's used to translate ICU "fields"
1323 // to JavaScript "parts" of a formatted string. Each ICU field and JavaScript
1324 // part has an integer field_id, which corresponds to something like "grouping
1325 // separator", "fraction", or "percent sign", and has a begin and end
1326 // position. Here's a diagram of:
1327
1328 // var nf = new Intl.NumberFormat(['de'], {style:'currency',currency:'EUR'});
1329 // nf.formatToParts(123456.78);
1330
1331 // : 6
1332 // input regions: 0000000211 7
1333 // ('-' means -1): ------------
1334 // formatted string: "123.456,78"
1335 // output parts: 0006000211-7
1336
1337 // To illustrate the requirements of this algorithm, here's a contrived and
1338 // convoluted example of inputs and expected outputs:
1339
1340 // : 4
1341 // : 22 33 3
1342 // : 11111 22
1343 // input regions: 0000000 111
1344 // : ------------
1345 // formatted string: "abcdefghijkl"
1346 // output parts: 0221340--231
1347 // (The characters in the formatted string are irrelevant to this function.)
1348
1349 // We arrange the overlapping input regions like a mountain range where
1350 // smaller regions are "on top" of larger regions, and we output a birds-eye
1351 // view of the mountains, so that smaller regions take priority over larger
1352 // regions.
1353 std::sort(regions->begin(), regions->end(), cmp_NumberFormatSpan);
1354 std::vector<size_t> overlapping_region_index_stack;
1355 // At least one item in regions must be a region spanning the entire string.
1356 // Due to the sorting above, the first item in the vector will be one of them.
1357 overlapping_region_index_stack.push_back(0);
1358 NumberFormatSpan top_region = regions->at(0);
1359 size_t region_iterator = 1;
1360 int32_t entire_size = top_region.end_pos;
1361
1362 std::vector<NumberFormatSpan> out_parts;
1363
1364 // The "climber" is a cursor that advances from left to right climbing "up"
1365 // and "down" the mountains. Whenever the climber moves to the right, that
1366 // represents an item of output.
1367 int32_t climber = 0;
1368 while (climber < entire_size) {
1369 int32_t next_region_begin_pos;
1370 if (region_iterator < regions->size()) {
1371 next_region_begin_pos = regions->at(region_iterator).begin_pos;
1372 } else {
1373 // finish off the rest of the input by proceeding to the end.
1374 next_region_begin_pos = entire_size;
1375 }
1376
1377 if (climber < next_region_begin_pos) {
1378 while (top_region.end_pos < next_region_begin_pos) {
1379 if (climber < top_region.end_pos) {
1380 // step down
1381 out_parts.push_back(NumberFormatSpan(top_region.field_id, climber,
1382 top_region.end_pos));
1383 climber = top_region.end_pos;
1384 } else {
1385 // drop down
1386 }
1387 overlapping_region_index_stack.pop_back();
1388 top_region = regions->at(overlapping_region_index_stack.back());
1389 }
1390 if (climber < next_region_begin_pos) {
1391 // cross a plateau/mesa/valley
1392 out_parts.push_back(NumberFormatSpan(top_region.field_id, climber,
1393 next_region_begin_pos));
1394 climber = next_region_begin_pos;
1395 }
1396 }
1397 if (region_iterator < regions->size()) {
1398 overlapping_region_index_stack.push_back(region_iterator++);
1399 top_region = regions->at(overlapping_region_index_stack.back());
1400 }
1401 }
1402 return out_parts;
1403}
