495 ASN1_STRING *strtmp;
496 ASN1_OBJECT *otmp;
497 int utype;
498 const unsigned char *cont;
499 unsigned char c;
500 int len;
501 const ASN1_PRIMITIVE_FUNCS *pf;
502 pf = it->funcs;
503 if (pf && pf->prim_i2c)
504 return pf->prim_i2c(pval, cout, putype, it);
505
506 /* Should type be omitted? */
507 if ((it->itype != ASN1_ITYPE_PRIMITIVE0x0)
508 || (it->utype != V_ASN1_BOOLEAN1)) {
509 if (!*pval)
510 return -1;
511 }
512
513 if (it->itype == ASN1_ITYPE_MSTRING0x5) {
514 /* If MSTRING type set the underlying type */
515 strtmp = (ASN1_STRING *)*pval;
516 utype = strtmp->type;
517 *putype = utype;
518 } else if (it->utype == V_ASN1_ANY-4) {
519 /* If ANY set type and pointer to value */
520 ASN1_TYPE *typ;
521 typ = (ASN1_TYPE *)*pval;
522 utype = typ->type;
523 *putype = utype;
524 pval = &typ->value.asn1_value;
525 } else
526 utype = *putype;
527
528 switch (utype) {
529 case V_ASN1_OBJECT6:
530 otmp = (ASN1_OBJECT *)*pval;
531 cont = otmp->data;
532 len = otmp->length;
533 if (cont == NULL((void*)0) || len == 0)
534 return -1;
535 break;
536
537 case V_ASN1_NULL5:
538 cont = NULL((void*)0);
539 len = 0;
540 break;
541
542 case V_ASN1_BOOLEAN1:
543 tbool = (ASN1_BOOLEAN *)pval;
544 if (*tbool == -1)
545 return -1;
546 if (it->utype != V_ASN1_ANY-4) {
547 /*
548 * Default handling if value == size field then omit
549 */
550 if (*tbool && (it->size > 0))
551 return -1;
552 if (!*tbool && !it->size)
553 return -1;
554 }
555 c = (unsigned char)*tbool;
556 cont = &c;
557 len = 1;
558 break;
559
560 case V_ASN1_BIT_STRING3:
561 return i2c_ASN1_BIT_STRING((ASN1_BIT_STRING *)*pval,
562 cout ? &cout : NULL((void*)0));
563
564 case V_ASN1_INTEGER2:
565 case V_ASN1_ENUMERATED10:
566 /*
567 * These are all have the same content format as ASN1_INTEGER
568 */
569 return i2c_ASN1_INTEGER((ASN1_INTEGER *)*pval, cout ? &cout : NULL((void*)0));
570
571 case V_ASN1_OCTET_STRING4:
572 case V_ASN1_NUMERICSTRING18:
573 case V_ASN1_PRINTABLESTRING19:
574 case V_ASN1_T61STRING20:
575 case V_ASN1_VIDEOTEXSTRING21:
576 case V_ASN1_IA5STRING22:
577 case V_ASN1_UTCTIME23:
578 case V_ASN1_GENERALIZEDTIME24:
579 case V_ASN1_GRAPHICSTRING25:
580 case V_ASN1_VISIBLESTRING26:
581 case V_ASN1_GENERALSTRING27:
582 case V_ASN1_UNIVERSALSTRING28:
583 case V_ASN1_BMPSTRING30:
584 case V_ASN1_UTF8STRING12:
585 case V_ASN1_SEQUENCE16:
586 case V_ASN1_SET17:
587 default:
588 /* All based on ASN1_STRING and handled the same */
589 strtmp = (ASN1_STRING *)*pval;
590 /* Special handling for NDEF */
591 if ((it->size == ASN1_TFLG_NDEF(0x1<<11))
592 && (strtmp->flags & ASN1_STRING_FLAG_NDEF0x010)) {
593 if (cout) {
594 strtmp->data = cout;
595 strtmp->length = 0;
596 }
597 /* Special return code */
598 return -2;
599 }
600 cont = strtmp->data;
601 len = strtmp->length;
602
603 break;
604
605 }
606 if (cout && len)
607 memcpy(cout, cont, len);
608 return len;
609}
