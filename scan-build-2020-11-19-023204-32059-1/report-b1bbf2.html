342// NOTE: Call to this function could change both `buf`'s and `count`'s
343// values, shifting their base and decrementing their length. This is
344// required in order to skip the data that was successfully written via
345// uv_try_write().
346int LibuvStreamWrap::DoTryWrite(uv_buf_t** bufs, size_t* count) {
347 int err;
348 size_t written;
349 uv_buf_t* vbufs = *bufs;
350 size_t vcount = *count;
351
352 err = uv_try_write(stream(), vbufs, vcount);
353 if (err == UV_ENOSYS || err == UV_EAGAIN)
354 return 0;
355 if (err < 0)
356 return err;
357
358 // Slice off the buffers: skip all written buffers and slice the one that
359 // was partially written.
360 written = err;
361 for (; vcount > 0; vbufs++, vcount--) {
362 // Slice
363 if (vbufs[0].len > written) {
364 vbufs[0].base += written;
365 vbufs[0].len -= written;
366 written = 0;
Value stored to 'written' is never read
367 break;
368
369 // Discard
370 } else {
371 written -= vbufs[0].len;
372 }
373 }
374
375 *bufs = vbufs;
376 *count = vcount;
377
378 return 0;
379}
