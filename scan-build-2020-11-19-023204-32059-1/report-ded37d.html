180bool IsRSAPrivateKey(const unsigned char* data, size_t size) {
181 // Both RSAPrivateKey and RSAPublicKey structures start with a SEQUENCE.
182 size_t offset, len;
183 if (!IsASN1Sequence(data, size, &offset, &len))
184 return false;
185
186 // An RSAPrivateKey sequence always starts with a single-byte integer whose
187 // value is either 0 or 1, whereas an RSAPublicKey starts with the modulus
188 // (which is the product of two primes and therefore at least 4), so we can
189 // decide the type of the structure based on the first three bytes of the
190 // sequence.
