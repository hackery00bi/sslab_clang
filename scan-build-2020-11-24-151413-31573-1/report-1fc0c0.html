449// FastHexToBuffer()
450// FastHex64ToBuffer()
451// FastHex32ToBuffer()
452// FastTimeToBuffer()
453// These are intended for speed. FastIntToBuffer() assumes the
454// integer is non-negative. FastHexToBuffer() puts output in
455// hex rather than decimal. FastTimeToBuffer() puts the output

466// which may not be the beginning of the input buffer.

686// ----------------------------------------------------------------------
687// StrCat()
688// This merges the given strings or numbers, with no delimiter. This
689// is designed to be the fastest possible way to construct a string out
690// of a mix of raw C strings, strings, bool values,
691// and numeric values.
692//
693// Don't use this for user-visible strings. The localization process
694// works poorly on strings built up out of fragments.
695//
696// For clarity and performance, don't use StrCat when appending to a
697// string. In particular, avoid using any of these (anti-)patterns:
698// str.append(StrCat(...)
699// str += StrCat(...)
700// str = StrCat(str, ...)
701// where the last is the worse, with the potential to change a loop
702// from a linear time operation with O(1) dynamic allocations into a
703// quadratic time operation with O(n) dynamic allocations. StrAppend
704// is a better choice than any of the above, subject to the restriction
705// of StrAppend(&str, a, b, c, ...) that none of the a, b, c, ... may
706// be a reference into str.
707// ----------------------------------------------------------------------
708

768// ----------------------------------------------------------------------
769// Join()
770// These methods concatenate a range of components into a C++ string, using
771// the C-string "delim" as a separator between components.
772// ----------------------------------------------------------------------
773template <typename Iterator>
774void Join(Iterator start, Iterator end, const char* delim,
775 std::string* result) {
776 for (Iterator it = start; it != end; ++it) {
777 if (it != start) {
778 result->append(delim);
779 }
780 StrAppend(result, *it);
781 }
782}
783
784template <typename Range>
785std::string Join(const Range& components, const char* delim) {
786 std::string result;
787 Join(components.begin(), components.end(), delim, &result);
788 return result;
789}

892// CleanStringLineEndings()
893// Clean up a multi-line string to conform to Unix line endings.
894// Reads from src and appends to dst, so usually dst should be empty.
895//

897// be added automatically.
898//
899// Four different types of input are correctly handled:
900//
901// - Unix/Linux files: line ending is LF: pass through unchanged
902//

913//
914// Limitations:
915//
916// This does not do the right thing for CRCRLF files created by
917// broken programs that do another Unix->DOS conversion on files
918// that are already in CRLF format. For this, a two-pass approach
919// brute-force would be needed that
920//
921// (1) determines the presence of LF (first one is ok)
922// (2) if yes, removes any CR, else convert every CR to LF
