607 enum_type, use_lower_camel_for_enums, case_insensitive_enum_parsing,
608 ignore_unknown_values, &is_unknown_enum_value);
609 if (e.ok() && !is_unknown_enum_value) {
610 WireFormatLite::WriteEnum(field_number, e.value(), stream);
611 }
612 return e.status();
613}
614
615ProtoWriter* ProtoWriter::RenderPrimitiveField(
616 const google::protobuf::Field& field, const google::protobuf::Type& type,
617 const DataPiece& data) {
618 util::Status status;
619
620 // Pushing a ProtoElement and then pop it off at the end for 2 purposes:
621 // error location reporting and required field accounting.
622 //
623 // For proto3, since there is no required field tracking, we only need to
624 // push ProtoElement for error cases.
625 if (!element_->proto3()) {
626 element_.reset(new ProtoElement(element_.release(), &field, type, false));
627 }
628
629 switch (field.kind()) {
630 case google::protobuf::Field::TYPE_INT32: {
631 status = WriteInt32(field.number(), data, stream_.get());
632 break;
633 }
634 case google::protobuf::Field::TYPE_SFIXED32: {
635 status = WriteSFixed32(field.number(), data, stream_.get());
636 break;
637 }
638 case google::protobuf::Field::TYPE_SINT32: {
639 status = WriteSInt32(field.number(), data, stream_.get());
640 break;
641 }
642 case google::protobuf::Field::TYPE_FIXED32: {
643 status = WriteFixed32(field.number(), data, stream_.get());
644 break;
645 }
646 case google::protobuf::Field::TYPE_UINT32: {
647 status = WriteUInt32(field.number(), data, stream_.get());
648 break;
649 }
650 case google::protobuf::Field::TYPE_INT64: {
651 status = WriteInt64(field.number(), data, stream_.get());
652 break;
653 }
654 case google::protobuf::Field::TYPE_SFIXED64: {
655 status = WriteSFixed64(field.number(), data, stream_.get());
656 break;
657 }
658 case google::protobuf::Field::TYPE_SINT64: {
659 status = WriteSInt64(field.number(), data, stream_.get());
660 break;
661 }
662 case google::protobuf::Field::TYPE_FIXED64: {
663 status = WriteFixed64(field.number(), data, stream_.get());
664 break;
665 }
666 case google::protobuf::Field::TYPE_UINT64: {
667 status = WriteUInt64(field.number(), data, stream_.get());
668 break;
669 }
670 case google::protobuf::Field::TYPE_DOUBLE: {
671 status = WriteDouble(field.number(), data, stream_.get());
672 break;
673 }
674 case google::protobuf::Field::TYPE_FLOAT: {
675 status = WriteFloat(field.number(), data, stream_.get());
676 break;
677 }
678 case google::protobuf::Field::TYPE_BOOL: {
679 status = WriteBool(field.number(), data, stream_.get());
680 break;
681 }
682 case google::protobuf::Field::TYPE_BYTES: {
683 status = WriteBytes(field.number(), data, stream_.get());
684 break;
685 }
686 case google::protobuf::Field::TYPE_STRING: {
687 status = WriteString(field.number(), data, stream_.get());
688 break;
689 }
690 case google::protobuf::Field::TYPE_ENUM: {
691 status = WriteEnum(
692 field.number(), data, typeinfo_->GetEnumByTypeUrl(field.type_url()),
693 stream_.get(), use_lower_camel_for_enums_,
694 case_insensitive_enum_parsing_, ignore_unknown_enum_values_);
695 break;
696 }
697 default: // TYPE_GROUP, TYPE_MESSAGE, TYPE_UNKNOWN.
698 status = util::Status(util::error::INVALID_ARGUMENT,
699 data.ValueAsStringOrDefault(""));
700 }
701
702 if (!status.ok()) {
703 // Push a ProtoElement for location reporting purposes.
704 if (element_->proto3()) {
705 element_.reset(new ProtoElement(element_.release(), &field, type, false));
706 }
707 InvalidValue(field.type_url().empty()
708 ? google::protobuf::Field_Kind_Name(field.kind())
709 : field.type_url(),
710 status.message());
711 element_.reset(element()->pop());
712 return this;
713 }
714
715 if (!element_->proto3()) element_.reset(element()->pop());
716
717 return this;
718}

720const google::protobuf::Field* ProtoWriter::BeginNamed(StringPiece name,
721 bool is_list) {
722 if (invalid_depth_ > 0) {
723 ++invalid_depth_;
724 return nullptr;
725 }
726 const google::protobuf::Field* field = Lookup(name);
727 if (field == nullptr) {
728 ++invalid_depth_;
729 // InvalidName() already called in Lookup().
730 return nullptr;
731 }
732 if (is_list && !IsRepeated(*field)) {
733 ++invalid_depth_;
734 InvalidName(name, "Proto field is not repeating, cannot start list.");
735 return nullptr;
736 }
737 return field;
738}
739
740const google::protobuf::Field* ProtoWriter::Lookup(
741 StringPiece unnormalized_name) {
742 ProtoElement* e = element();
743 if (e == nullptr) {
744 InvalidName(unnormalized_name, "Root element must be a message.");
745 return nullptr;
746 }
747 if (unnormalized_name.empty()) {
748 // Objects in repeated field inherit the same field descriptor.
749 if (e->parent_field() == nullptr) {
750 InvalidName(unnormalized_name, "Proto fields must have a name.");
751 } else if (!IsRepeated(*e->parent_field())) {
752 InvalidName(unnormalized_name, "Proto fields must have a name.");
753 return nullptr;
754 }
755 return e->parent_field();
756 }
757 const google::protobuf::Field* field =
758 typeinfo_->FindField(&e->type(), unnormalized_name);
759 if (field == nullptr && !ignore_unknown_fields_) {
760 InvalidName(unnormalized_name, "Cannot find field.");
761 }
762 return field;
763}

765const google::protobuf::Type* ProtoWriter::LookupType(
766 const google::protobuf::Field* field) {
767 return ((field->kind() == google::protobuf::Field::TYPE_MESSAGE ||
768 field->kind() == google::protobuf::Field::TYPE_GROUP)
769 ? typeinfo_->GetTypeByTypeUrl(field->type_url())
770 : &element_->type());
771}

773void ProtoWriter::WriteRootMessage() {
774 GOOGLE_DCHECK(!done_)while(false) !(!(!done_)) ? (void)0 : ::google::protobuf::internal
::LogFinisher() = ::google::protobuf::internal::LogMessage( ::
google::protobuf::LOGLEVEL_FATAL, "google/protobuf/util/internal/proto_writer.cc"
, 774) << "CHECK failed: " "!done_" ": ";
775 int curr_pos = 0;
776 // Calls the destructor of CodedOutputStream to remove any uninitialized
777 // memory from the Cord before we read it.
778 stream_.reset(nullptr);
779 const void* data;
780 int length;
781 io::ArrayInputStream input_stream(buffer_.data(), buffer_.size());
782 while (input_stream.Next(&data, &length)) {
783 if (length == 0) continue;
784 int num_bytes = length;
785 // Write up to where we need to insert the size field.
786 // The number of bytes we may write is the smaller of:
787 // - the current fragment size
788 // - the distance to the next position where a size field needs to be
789 // inserted.
790 if (!size_insert_.empty() &&
791 size_insert_.front().pos - curr_pos < num_bytes) {
792 num_bytes = size_insert_.front().pos - curr_pos;
793 }
794 output_->Append(static_cast<const char*>(data), num_bytes);
795 if (num_bytes < length) {
796 input_stream.BackUp(length - num_bytes);
797 }
798 curr_pos += num_bytes;
799 // Insert the size field.
800 // size_insert_.front(): the next <index, size> pair to be written.
801 // size_insert_.front().pos: position of the size field.
802 // size_insert_.front().size: the size (integer) to be inserted.
803 if (!size_insert_.empty() && curr_pos == size_insert_.front().pos) {
804 // Varint32 occupies at most 10 bytes.
805 uint8 insert_buffer[10];
806 uint8* insert_buffer_pos = CodedOutputStream::WriteVarint32ToArray(
807 size_insert_.front().size, insert_buffer);
808 output_->Append(reinterpret_cast<const char*>(insert_buffer),
809 insert_buffer_pos - insert_buffer);
810 size_insert_.pop_front();
811 }
812 }
813 output_->Flush();
814 stream_.reset(new CodedOutputStream(&adapter_));
815 done_ = true;
816}
