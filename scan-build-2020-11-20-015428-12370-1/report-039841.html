2667 parse_func(ls, line);
2668 break;
2669 case TK_local:
2670 lj_lex_next(ls);
2671 parse_local(ls);
2672 break;
2673 case TK_return:
2674 parse_return(ls);
2675 return 1; /* Must be last. */
2676 case TK_break:
2677 lj_lex_next(ls);
2678 parse_break(ls);
2679 return !LJ_520; /* Must be last in Lua 5.1. */
2680#if LJ_520
2681 case ';':
2682 lj_lex_next(ls);
2683 break;
2684#endif
2685 case TK_label:
2686 parse_label(ls);
2687 break;
2688 case TK_goto:
2689 if (LJ_520 || lj_lex_lookahead(ls) == TK_name) {
2690 lj_lex_next(ls);
2691 parse_goto(ls);
2692 break;
2693 } /* else: fallthrough */
2694 default:
2695 parse_call_assign(ls);
2696 break;
2697 }
2698 return 0;
2699}
