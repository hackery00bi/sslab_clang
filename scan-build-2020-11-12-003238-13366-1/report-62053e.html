112/* Return zero if strings are the same, non-zero if they are not.
113 * The comparison is performed in a way that prevents an attacker to obtain
114 * information about the nature of the strings just monitoring the execution
115 * time of the function.
116 *
117 * Note that limiting the comparison length to strings up to 512 bytes we
118 * can avoid leaking any information about the password length and any
119 * possible branch misprediction related leak.

124 * a or b are fixed (our password) length, and the difference is only
125 * relative to the length of the user provided string, so no information
126 * leak is possible in the following two lines of code. */

128 unsigned int blen = strlen(b);

130 int diff = 0;

133 * Note that this will never pass the first test in practical circumstances

716/* Set user properties according to the string "op". The following
717 * is a description of what different strings will do:
718 *

1090 * the possible commands and just subtracting other single commands or
1091 * categories, or if, instead, the ACL was created just adding commands
1092 * and command categories from scratch, not allowing future commands by
1093 * default (loaded via modules). This is useful when rewriting the ACLs
1094 * with ACL SAVE. */
1095 if (nextid == USER_COMMAND_BITS_COUNT1024-1) nextid++;
1096 return thisid;
1097}

1113/* Check if the command is ready to be executed in the client 'c', already
1114 * referenced by c->cmd, and can be executed by this client according to the
1115 * ACLs associated to the client user c->user.
1116 *
1117 * If the user can execute the command ACL_OK is returned, otherwise
1118 * ACL_DENIED_CMD or ACL_DENIED_KEY is returned: the first in case the
1119 * command cannot be executed because the user is not allowed to run such
1120 * command, the second if the command is denied because the user is trying
1121 * to access keys that are not among the specified patterns. */
1122int ACLCheckCommandPerm(client *c, int *keyidxptr) {
1123 user *u = c->user;
1124 uint64_t id = c->cmd->id;
1125
1126 /* If there is no associated user, the connection can run anything. */
1127 if (u == NULL((void*)0)) return ACL_OK0;
1128
1129 /* Check if the user can execute this command. */
1130 if (!(u->flags & USER_FLAG_ALLCOMMANDS(1<<3)) &&
1131 c->cmd->proc != authCommand)
1132 {
1133 /* If the bit is not set we have to check further, in case the
1134 * command is allowed just with that specific subcommand. */
1135 if (ACLGetUserCommandBit(u,id) == 0) {
1136 /* Check if the subcommand matches. */
1137 if (c->argc < 2 ||
1138 u->allowed_subcommands == NULL((void*)0) ||
1139 u->allowed_subcommands[id] == NULL((void*)0))
1140 {
1141 return ACL_DENIED_CMD1;
1142 }
1143
1144 long subid = 0;
1145 while (1) {
1146 if (u->allowed_subcommands[id][subid] == NULL((void*)0))
1147 return ACL_DENIED_CMD1;
1148 if (!strcasecmp(c->argv[1]->ptr,
1149 u->allowed_subcommands[id][subid]))
1150 break; /* Subcommand match found. Stop here. */
1151 subid++;
1152 }
1153 }
1154 }
1155
1156 /* Check if the user can execute commands explicitly touching the keys
1157 * mentioned in the command arguments. */
1158 if (!(c->user->flags & USER_FLAG_ALLKEYS(1<<2)) &&
1159 (c->cmd->getkeys_proc || c->cmd->firstkey))
1160 {
1161 getKeysResult result = GETKEYS_RESULT_INIT{ {0}, ((void*)0), 0, 256 };
1162 int numkeys = getKeysFromCommand(c->cmd,c->argv,c->argc,&result);
1163 int *keyidx = result.keys;
1164 for (int j = 0; j < numkeys; j++) {
1165 listIter li;
1166 listNode *ln;
1167 listRewind(u->patterns,&li);
1168
1169 /* Test this key against every pattern. */
1170 int match = 0;
1171 while((ln = listNext(&li))) {
1172 sds pattern = listNodeValue(ln)((ln)->value);
1173 size_t plen = sdslen(pattern);
1174 int idx = keyidx[j];
1175 if (stringmatchlen(pattern,plen,c->argv[idx]->ptr,
1176 sdslen(c->argv[idx]->ptr),0))
1177 {
1178 match = 1;
1179 break;
1180 }
1181 }
1182 if (!match) {
1183 if (keyidxptr) *keyidxptr = keyidx[j];
1184 getKeysFreeResult(&result);
1185 return ACL_DENIED_KEY2;
1186 }
1187 }
1188 getKeysFreeResult(&result);
1189 }
1190
1191 /* If we survived all the above checks, the user can execute the
1192 * command. */
1193 return ACL_OK0;
1194}

1200/* Given an argument vector describing a user in the form:
1201 *
1202 * user <username> ... ACL rules and flags ...
1203 *
1204 * this function validates, and if the syntax is valid, appends
1205 * the user definition to a list for later loading.
1206 *
1207 * The rules are tested for validity and if there obvious syntax errors
1208 * the function returns C_ERR and does nothing, otherwise C_OK is returned
1209 * and the user is appended to the list.
1210 *
1211 * Note that this function cannot stop in case of commands that are not found
1212 * and, in that case, the error will be emitted later, because certain
1213 * commands may be defined later once modules are loaded.
1214 *
1215 * When an error is detected and C_ERR is returned, the function populates
1216 * by reference (if not set to NULL) the argc_err argument with the index
1217 * of the argv vector that caused the error. */
1218int ACLAppendUserForLoading(sds *argv, int argc, int *argc_err) {
1219 if (argc < 2 || strcasecmp(argv[0],"user")) {
1220 if (argc_err) *argc_err = 0;
1221 return C_ERR-1;
1222 }
1223
1224 /* Try to apply the user rules in a fake user to see if they
1225 * are actually valid. */
1226 user *fakeuser = ACLCreateUnlinkedUser();
1227
1228 for (int j = 2; j < argc; j++) {
1229 if (ACLSetUser(fakeuser,argv[j],sdslen(argv[j])) == C_ERR-1) {
1230 if (errno(*__errno_location ()) != ENOENT2) {
1231 ACLFreeUser(fakeuser);
1232 if (argc_err) *argc_err = j;
1233 return C_ERR-1;
1234 }
1235 }
1236 }
1237
1238 /* Rules look valid, let's append the user to the list. */
1239 sds *copy = zmalloc(sizeof(sds)*argc);
1240 for (int j = 1; j < argc; j++) copy[j-1] = sdsdup(argv[j]);
1241 copy[argc-1] = NULL((void*)0);
1242 listAddNodeTail(UsersToLoad,copy);
1243 ACLFreeUser(fakeuser);
1244 return C_OK0;
1245}

1293/* This function loads the ACL from the specified filename: every line
1294 * is validated and should be either empty or in the format used to specify
1295 * users in the redis.conf configuration or in the ACL file, that is:
1296 *
1297 * user <username> ... rules ...
1298 *
1299 * Note that this function considers comments starting with '#' as errors
1300 * because the ACL file is meant to be rewritten, and comments would be
1301 * lost after the rewrite. Yet empty lines are allowed to avoid being too
1302 * strict.
1303 *
1304 * One important part of implementing ACL LOAD, that uses this function, is
1305 * to avoid ending with broken rules if the ACL file is invalid for some
1306 * reason, so the function will attempt to validate the rules before loading
1307 * each user. For every line that will be found broken the function will
1308 * collect an error message.
1309 *
1310 * IMPORTANT: If there is at least a single error, nothing will be loaded
1311 * and the rules will remain exactly as they were.
1312 *
1313 * At the end of the process, if no errors were found in the whole file then
1314 * NULL is returned. Otherwise an SDS string describing in a single line
1315 * a description of all the issues found is returned. */
1316sds ACLLoadFromFile(const char *filename) {
1317 FILE *fp;
1318 char buf[1024];
1319
1320 /* Open the ACL file. */
1321 if ((fp = fopen(filename,"r")) == NULL((void*)0)) {
1322 sds errors = sdscatprintf(sdsempty(),
1323 "Error loading ACLs, opening file '%s': %s",
1324 filename, strerror(errno(*__errno_location ())));
1325 return errors;
1326 }
1327
1328 /* Load the whole file as a single string in memory. */
1329 sds acls = sdsempty();
1330 while(fgets(buf,sizeof(buf),fp) != NULL((void*)0))
1331 acls = sdscat(acls,buf);
1332 fclose(fp);
1333
1334 /* Split the file into lines and attempt to load each line. */
1335 int totlines;
1336 sds *lines, errors = sdsempty();
1337 lines = sdssplitlen(acls,strlen(acls),"\n",1,&totlines);
1338 sdsfree(acls);
1339
1340 /* We need a fake user to validate the rules before making changes
1341 * to the real user mentioned in the ACL line. */
1342 user *fakeuser = ACLCreateUnlinkedUser();
1343
1344 /* We do all the loading in a fresh instance of the Users radix tree,
1345 * so if there are errors loading the ACL file we can rollback to the
1346 * old version. */
1347 rax *old_users = Users;
1348 user *old_default_user = DefaultUser;
1349 Users = raxNew();
1350 ACLInitDefaultUser();
1351
1352 /* Load each line of the file. */
1353 for (int i = 0; i < totlines; i++) {
1354 sds *argv;
1355 int argc;
1356 int linenum = i+1;
1357
1358 lines[i] = sdstrim(lines[i]," \t\r\n");
1359
1360 /* Skip blank lines */
1361 if (lines[i][0] == '\0') continue;
1362
1363 /* Split into arguments */
1364 argv = sdssplitlen(lines[i],sdslen(lines[i])," ",1,&argc);
1365 if (argv == NULL((void*)0)) {
1366 errors = sdscatprintf(errors,
1367 "%s:%d: unbalanced quotes in acl line. ",
1368 server.acl_filename, linenum);
1369 continue;
1370 }
1371
1372 /* Skip this line if the resulting command vector is empty. */
1373 if (argc == 0) {
1374 sdsfreesplitres(argv,argc);
1375 continue;
1376 }
1377
1378 /* The line should start with the "user" keyword. */
1379 if (strcmp(argv[0],"user") || argc < 2) {
1380 errors = sdscatprintf(errors,
1381 "%s:%d should start with user keyword followed "
1382 "by the username. ", server.acl_filename,
1383 linenum);
1384 sdsfreesplitres(argv,argc);
1385 continue;
1386 }
1387
1388 /* Spaces are not allowed in usernames. */
1389 if (ACLStringHasSpaces(argv[1],sdslen(argv[1]))) {
1390 errors = sdscatprintf(errors,
1391 "'%s:%d: username '%s' contains invalid characters. ",
1392 server.acl_filename, linenum, argv[1]);
1393 sdsfreesplitres(argv,argc);
1394 continue;
1395 }
1396
1397 /* Try to process the line using the fake user to validate if
1398 * the rules are able to apply cleanly. At this stage we also
1399 * trim trailing spaces, so that we don't have to handle that
1400 * in ACLSetUser(). */
1401 ACLSetUser(fakeuser,"reset",-1);
1402 int j;
1403 for (j = 2; j < argc; j++) {
1404 argv[j] = sdstrim(argv[j],"\t\r\n");
1405 if (ACLSetUser(fakeuser,argv[j],sdslen(argv[j])) != C_OK0) {
1406 char *errmsg = ACLSetUserStringError();
1407 errors = sdscatprintf(errors,
1408 "%s:%d: %s. ",
1409 server.acl_filename, linenum, errmsg);
1410 continue;
1411 }
1412 }
1413
1414 /* Apply the rule to the new users set only if so far there
1415 * are no errors, otherwise it's useless since we are going
1416 * to discard the new users set anyway. */
1417 if (sdslen(errors) != 0) {
1418 sdsfreesplitres(argv,argc);
1419 continue;
1420 }
1421
1422 /* We can finally lookup the user and apply the rule. If the
1423 * user already exists we always reset it to start. */
1424 user *u = ACLCreateUser(argv[1],sdslen(argv[1]));
1425 if (!u) {
1426 u = ACLGetUserByName(argv[1],sdslen(argv[1]));
1427 serverAssert(u != NULL)((u != ((void*)0))?(void)0 : (_serverAssert("u != NULL","acl.c"
,1427),_exit(1)));
1428 ACLSetUser(u,"reset",-1);
1429 }
1430
1431 /* Note that the same rules already applied to the fake user, so
1432 * we just assert that everything goes well: it should. */
1433 for (j = 2; j < argc; j++)
1434 serverAssert(ACLSetUser(u,argv[j],sdslen(argv[j])) == C_OK)((ACLSetUser(u,argv[j],sdslen(argv[j])) == 0)?(void)0 : (_serverAssert
("ACLSetUser(u,argv[j],sdslen(argv[j])) == C_OK","acl.c",1434
),_exit(1)));
1435
1436 sdsfreesplitres(argv,argc);
1437 }
1438
1439 ACLFreeUser(fakeuser);
1440 sdsfreesplitres(lines,totlines);
1441 DefaultUser = old_default_user; /* This pointer must never change. */
1442
1443 /* Check if we found errors and react accordingly. */
1444 if (sdslen(errors) == 0) {
1445 /* The default user pointer is referenced in different places: instead
1446 * of replacing such occurrences it is much simpler to copy the new
1447 * default user configuration in the old one. */
1448 user *new = ACLGetUserByName("default",7);
1449 serverAssert(new != NULL)((new != ((void*)0))?(void)0 : (_serverAssert("new != NULL","acl.c"
,1449),_exit(1)));
1450 ACLCopyUser(DefaultUser,new);
1451 ACLFreeUser(new);
1452 raxInsert(Users,(unsigned char*)"default",7,DefaultUser,NULL((void*)0));
1453 raxRemove(old_users,(unsigned char*)"default",7,NULL((void*)0));
1454 ACLFreeUsersSet(old_users);
1455 sdsfree(errors);
1456 return NULL((void*)0);
1457 } else {
1458 ACLFreeUsersSet(Users);
1459 Users = old_users;
1460 errors = sdscat(errors,"WARNING: ACL errors detected, no change to the previously active ACL rules was performed");
1461 return errors;
1462 }
1463}

1465/* Generate a copy of the ACLs currently in memory in the specified filename.
1466 * Returns C_OK on success or C_ERR if there was an error during the I/O.
1467 * When C_ERR is returned a log is produced with hints about the issue. */
1468int ACLSaveToFile(const char *filename) {
1469 sds acl = sdsempty();
1470 int fd = -1;
1471 sds tmpfilename = NULL((void*)0);
1472 int retval = C_ERR-1;
1473
1474 /* Let's generate an SDS string containing the new version of the
1475 * ACL file. */
1476 raxIterator ri;
1477 raxStart(&ri,Users);
1478 raxSeek(&ri,"^",NULL((void*)0),0);
1479 while(raxNext(&ri)) {
1480 user *u = ri.data;
1481 /* Return information in the configuration file format. */
1482 sds user = sdsnew("user ");
1483 user = sdscatsds(user,u->name);
1484 user = sdscatlen(user," ",1);
1485 sds descr = ACLDescribeUser(u);
1486 user = sdscatsds(user,descr);
1487 sdsfree(descr);
1488 acl = sdscatsds(acl,user);
1489 acl = sdscatlen(acl,"\n",1);
1490 sdsfree(user);
1491 }
1492 raxStop(&ri);
1493
1494 /* Create a temp file with the new content. */
1495 tmpfilename = sdsnew(filename);
1496 tmpfilename = sdscatfmt(tmpfilename,".tmp-%i-%I",
1497 (int)getpid(),(int)mstime());
1498 if ((fd = open(tmpfilename,O_WRONLY01|O_CREAT0100,0644)) == -1) {
1499 serverLog(LL_WARNING3,"Opening temp ACL file for ACL SAVE: %s",
1500 strerror(errno(*__errno_location ())));
1501 goto cleanup;
1502 }
1503
1504 /* Write it. */
1505 if (write(fd,acl,sdslen(acl)) != (ssize_t)sdslen(acl)) {
1506 serverLog(LL_WARNING3,"Writing ACL file for ACL SAVE: %s",
1507 strerror(errno(*__errno_location ())));
1508 goto cleanup;
1509 }
1510 close(fd); fd = -1;
1511
1512 /* Let's replace the new file with the old one. */
1513 if (rename(tmpfilename,filename) == -1) {
1514 serverLog(LL_WARNING3,"Renaming ACL file for ACL SAVE: %s",
1515 strerror(errno(*__errno_location ())));
1516 goto cleanup;
1517 }
1518 sdsfree(tmpfilename); tmpfilename = NULL((void*)0);
1519 retval = C_OK0; /* If we reached this point, everything is fine. */
1520
1521cleanup:
1522 if (fd != -1) close(fd);
1523 if (tmpfilename) unlink(tmpfilename);
1524 sdsfree(tmpfilename);
1525 sdsfree(acl);
1526 return retval;
1527}

1566#define ACL_LOG_CTX_TOPLEVEL0 0
1567#define ACL_LOG_CTX_LUA1 1
1568#define ACL_LOG_CTX_MULTI2 2
1569#define ACL_LOG_GROUPING_MAX_TIME_DELTA60000 60000
1570
1571/* This structure defines an entry inside the ACL log. */
1572typedef struct ACLLogEntry {
1573 uint64_t count; /* Number of times this happened recently. */
1574 int reason; /* Reason for denying the command. ACL_DENIED_*. */
1575 int context; /* Toplevel, Lua or MULTI/EXEC? ACL_LOG_CTX_*. */
1576 sds object; /* The key name or command name. */
1577 sds username; /* User the client is authenticated with. */
1578 mstime_t ctime; /* Milliseconds time of last update to this entry. */
1579 sds cinfo; /* Client info (last client if updated). */
1580} ACLLogEntry;
1581
1582/* This function will check if ACL entries 'a' and 'b' are similar enough
1583 * that we should actually update the existing entry in our ACL log instead
1584 * of creating a new one. */
1585int ACLLogMatchEntry(ACLLogEntry *a, ACLLogEntry *b) {
1586 if (a->reason != b->reason) return 0;
1587 if (a->context != b->context) return 0;
1588 mstime_t delta = a->ctime - b->ctime;
1589 if (delta < 0) delta = -delta;
1590 if (delta > ACL_LOG_GROUPING_MAX_TIME_DELTA60000) return 0;
1591 if (sdscmp(a->object,b->object) != 0) return 0;
1592 if (sdscmp(a->username,b->username) != 0) return 0;
1593 return 1;
1594}

1597void ACLFreeLogEntry(void *leptr) {

1599 sdsfree(le->object);
1600 sdsfree(le->username);
1601 sdsfree(le->cinfo);

1605/* Adds a new entry in the ACL log, making sure to delete the old entry
1606 * if we reach the maximum length allowed for the log. This function attempts
1607 * to find similar entries in the current log in order to bump the counter of
1608 * the log entry instead of creating many entries for very similar ACL
1609 * rules issues.
1610 *
1611 * The keypos argument is only used when the reason is ACL_DENIED_KEY, since
1612 * it allows the function to log the key name that caused the problem.
1613 * Similarly the username is only passed when we failed to authenticate the
1614 * user with AUTH or HELLO, for the ACL_DENIED_AUTH reason. Otherwise
1615 * it will just be NULL.
1616 */
1617void addACLLogEntry(client *c, int reason, int keypos, sds username) {
1618 /* Create a new entry. */
1619 struct ACLLogEntry *le = zmalloc(sizeof(*le));
1620 le->count = 1;
1621 le->reason = reason;
1622 le->username = sdsdup(reason == ACL_DENIED_AUTH3 ? username : c->user->name);
1623 le->ctime = mstime();
1624
1625 switch(reason) {
1626 case ACL_DENIED_CMD1: le->object = sdsnew(c->cmd->name); break;
1627 case ACL_DENIED_KEY2: le->object = sdsnew(c->argv[keypos]->ptr); break;
1628 case ACL_DENIED_AUTH3: le->object = sdsnew(c->argv[0]->ptr); break;
1629 default: le->object = sdsempty();
1630 }
1631
1632 client *realclient = c;
1633 if (realclient->flags & CLIENT_LUA(1<<8)) realclient = server.lua_caller;
1634
1635 le->cinfo = catClientInfoString(sdsempty(),realclient);
1636 if (c->flags & CLIENT_MULTI(1<<3)) {
1637 le->context = ACL_LOG_CTX_MULTI2;
1638 } else if (c->flags & CLIENT_LUA(1<<8)) {
1639 le->context = ACL_LOG_CTX_LUA1;
1640 } else {
1641 le->context = ACL_LOG_CTX_TOPLEVEL0;
1642 }
1643
1644 /* Try to match this entry with past ones, to see if we can just
1645 * update an existing entry instead of creating a new one. */
1646 long toscan = 10; /* Do a limited work trying to find duplicated. */
1647 listIter li;
1648 listNode *ln;
1649 listRewind(ACLLog,&li);
1650 ACLLogEntry *match = NULL((void*)0);
1651 while (toscan-- && (ln = listNext(&li)) != NULL((void*)0)) {
1652 ACLLogEntry *current = listNodeValue(ln)((ln)->value);
1653 if (ACLLogMatchEntry(current,le)) {
1654 match = current;
1655 listDelNode(ACLLog,ln);
1656 listAddNodeHead(ACLLog,current);
1657 break;
1658 }
1659 }
1660
1661 /* If there is a match update the entry, otherwise add it as a
1662 * new one. */
1663 if (match) {
1664 /* We update a few fields of the existing entry and bump the
1665 * counter of events for this entry. */
1666 sdsfree(match->cinfo);
1667 match->cinfo = le->cinfo;
1668 match->ctime = le->ctime;
1669 match->count++;
1670
1671 /* Release the old entry. */
1672 le->cinfo = NULL((void*)0);
1673 ACLFreeLogEntry(le);
1674 } else {
1675 /* Add it to our list of entires. We'll have to trim the list
