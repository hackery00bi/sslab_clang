88static uint32_t
89decode(uint32_t* state, uint32_t* codep, uint32_t byte) {
90 uint32_t type = utf8d[byte];
91
92 *codep = (*state != UTF8_ACCEPT0) ?
93 (byte & 0x3fu) | (*codep << 6) :
94 (0xff >> type) & (byte);
95
96 *state = utf8d[256 + *state + type];
97 return *state;
98}
