611static int batch_option_callback(const struct option *opt,
612 const char *arg,
613 int unset)
614{
615 struct batch_options *bo = opt->value;
616
617 BUG_ON_OPT_NEG(unset)do { if ((unset)) BUG_fl("builtin/cat-file.c", 617, "option callback does not expect negation"
); } while (0);
618
619 if (bo->enabled) {
620 return error(_("only one batch option may be specified"))(error(_("only one batch option may be specified")), const_error
());
621 }
622
623 bo->enabled = 1;
624 bo->print_contents = !strcmp(opt->long_name, "batch");
625 bo->format = arg;
626
627 return 0;
628}
