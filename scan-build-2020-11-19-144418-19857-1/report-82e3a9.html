485static int fetch_with_import(struct transport *transport,
486 int nr_heads, struct ref **to_fetch)
487{
488 struct child_process fastimport;

501 if (posn->status & REF_STATUS_UPTODATE)

503

512 * remote-helpers that advertise the bidi-import capability are required to
513 * buffer the complete batch of import commands until this newline before
514 * sending data to fast-import.
515 * These helpers read back data from fast-import on their stdin, which could
516 * be mixed with import commands, otherwise.
517 */
518
519 if (finish_command(&fastimport))
520 die(_("error while running fast-import"));
521
522 /*
523 * The fast-import stream of a remote helper that advertises
524 * the "refspec" capability writes to the refs named after the
525 * right hand side of the first refspec matching each ref we
526 * were fetching.
527 *
528 * (If no "refspec" capability was specified, for historical
529 * reasons we default to the equivalent of *:*.)
530 *
531 * Store the result in to_fetch[i].old_sha1. Callers such
532 * as "git fetch" can use the value to write feedback to the
533 * terminal, populate FETCH_HEAD, and determine what new value
534 * should be written to peer_ref if the update is a
535 * fast-forward or this is a forced update.
536 */
537 for (i = 0; i < nr_heads; i++) {
538 char *private, *name;
539 posn = to_fetch[i];
540 if (posn->status & REF_STATUS_UPTODATE)
541 continue;
542 name = posn->symref ? posn->symref : posn->name;
543 if (data->rs.nr)
544 private = apply_refspecs(&data->rs, name);

546 private = xstrdup(name);

549 die(_("could not read ref %s"), private);
550 free(private);
551 }
552 }
553 strbuf_release(&buf);
554 return 0;

732static int push_update_ref_status(struct strbuf *buf,
733 struct push_update_ref_state *state,
734 struct ref *remote_refs)
735{
736 char *refname, *msg;
737 int status, forced = 0;
738
739 if (starts_with(buf->buf, "option ")) {
740 struct object_id old_oid, new_oid;
741 const char *key, *val;
742 char *p;
743
744 if (!state->hint || !(state->report || state->new_report))
745 die(_("'option' without a matching 'ok/error' directive"));
746 if (state->new_report) {
747 if (!state->hint->report) {
748 state->hint->report = xcalloc(1, sizeof(struct ref_push_report));
749 state->report = state->hint->report;
750 } else {
751 state->report = state->hint->report;
752 while (state->report->next)
753 state->report = state->report->next;
754 state->report->next = xcalloc(1, sizeof(struct ref_push_report));
755 state->report = state->report->next;
756 }
757 state->new_report = 0;
758 }
759 key = buf->buf + 7;
760 p = strchr(key, ' ');
761 if (p)
762 *p++ = '\0';
763 val = p;
764 if (!strcmp(key, "refname"))
765 state->report->ref_name = xstrdup_or_null(val);
766 else if (!strcmp(key, "old-oid") && val &&
767 !parse_oid_hex(val, &old_oid, &val))
768 state->report->old_oid = oiddup(&old_oid);
769 else if (!strcmp(key, "new-oid") && val &&
770 !parse_oid_hex(val, &new_oid, &val))
771 state->report->new_oid = oiddup(&new_oid);
772 else if (!strcmp(key, "forced-update"))
773 state->report->forced_update = 1;
774 /* Not update remote namespace again. */
775 return 1;
776 }
777
778 state->report = NULL((void*)0);
779 state->new_report = 0;
780
781 if (starts_with(buf->buf, "ok ")) {
782 status = REF_STATUS_OK;
783 refname = buf->buf + 3;
784 } else if (starts_with(buf->buf, "error ")) {
785 status = REF_STATUS_REMOTE_REJECT;
786 refname = buf->buf + 6;
787 } else
788 die(_("expected ok/error, helper said '%s'"), buf->buf);
789
790 msg = strchr(refname, ' ');
791 if (msg) {
792 struct strbuf msg_buf = STRBUF_INIT{ .alloc = 0, .len = 0, .buf = strbuf_slopbuf };
793 const char *end;
794
795 *msg++ = '\0';
796 if (!unquote_c_style(&msg_buf, msg, &end))
797 msg = strbuf_detach(&msg_buf, NULL((void*)0));
798 else
799 msg = xstrdup(msg);
800 strbuf_release(&msg_buf);
801
802 if (!strcmp(msg, "no match")) {
803 status = REF_STATUS_NONE;
804 FREE_AND_NULL(msg)do { free(msg); (msg) = ((void*)0); } while (0);
805 }
806 else if (!strcmp(msg, "up to date")) {
807 status = REF_STATUS_UPTODATE;
808 FREE_AND_NULL(msg)do { free(msg); (msg) = ((void*)0); } while (0);
809 }
810 else if (!strcmp(msg, "non-fast forward")) {
811 status = REF_STATUS_REJECT_NONFASTFORWARD;
812 FREE_AND_NULL(msg)do { free(msg); (msg) = ((void*)0); } while (0);
813 }
814 else if (!strcmp(msg, "already exists")) {
815 status = REF_STATUS_REJECT_ALREADY_EXISTS;
816 FREE_AND_NULL(msg)do { free(msg); (msg) = ((void*)0); } while (0);
817 }
818 else if (!strcmp(msg, "fetch first")) {
819 status = REF_STATUS_REJECT_FETCH_FIRST;
820 FREE_AND_NULL(msg)do { free(msg); (msg) = ((void*)0); } while (0);
821 }
822 else if (!strcmp(msg, "needs force")) {
823 status = REF_STATUS_REJECT_NEEDS_FORCE;
824 FREE_AND_NULL(msg)do { free(msg); (msg) = ((void*)0); } while (0);
825 }
826 else if (!strcmp(msg, "stale info")) {
827 status = REF_STATUS_REJECT_STALE;
828 FREE_AND_NULL(msg)do { free(msg); (msg) = ((void*)0); } while (0);
829 }
830 else if (!strcmp(msg, "forced update")) {
831 forced = 1;
832 FREE_AND_NULL(msg)do { free(msg); (msg) = ((void*)0); } while (0);
833 }
834 }
835
836 if (state->hint)
837 state->hint = find_ref_by_name(state->hint, refname);
838 if (!state->hint)
839 state->hint = find_ref_by_name(remote_refs, refname);
840 if (!state->hint) {
841 warning(_("helper reported unexpected status of %s"), refname);
842 return 1;
843 }
844
845 if (state->hint->status != REF_STATUS_NONE) {
846 /*
847 * Earlier, the ref was marked not to be pushed, so ignore the ref
848 * status reported by the remote helper if the latter is 'no match'.
849 */
850 if (status == REF_STATUS_NONE)
851 return 1;
852 }
853
854 if (status == REF_STATUS_OK)
855 state->new_report = 1;
856 state->hint->status = status;
857 state->hint->forced_update |= forced;
858 state->hint->remote_status = msg;
859 return !(status == REF_STATUS_OK);
860}

862static int push_update_refs_status(struct helper_data *data,
863 struct ref *remote_refs,
864 int flags)
865{
866 struct ref *ref;
867 struct ref_push_report *report;
868 struct strbuf buf = STRBUF_INIT{ .alloc = 0, .len = 0, .buf = strbuf_slopbuf };
869 struct push_update_ref_state state = { remote_refs, NULL((void*)0), 0 };
870
871 for (;;) {
872 if (recvline(data, &buf)) {
873 strbuf_release(&buf);
874 return 1;
875 }
876 if (!buf.len)
877 break;
878 push_update_ref_status(&buf, &state, remote_refs);
879 }
880 strbuf_release(&buf);
881
882 if (flags & TRANSPORT_PUSH_DRY_RUN(1<<2) || !data->rs.nr || data->no_private_update)
883 return 0;
884
885 /* propagate back the update to the remote namespace */
886 for (ref = remote_refs; ref; ref = ref->next) {
887 char *private;
888
889 if (ref->status != REF_STATUS_OK)
890 continue;
891
892 if (!ref->report) {
893 private = apply_refspecs(&data->rs, ref->name);
894 if (!private)
895 continue;
896 update_ref("update by helper", private, &(ref->new_oid),
897 NULL((void*)0), 0, 0);
898 free(private);
899 } else {
900 for (report = ref->report; report; report = report->next) {
901 private = apply_refspecs(&data->rs,
902 report->ref_name
903 ? report->ref_name
904 : ref->name);
905 if (!private)
906 continue;
907 update_ref("update by helper", private,
908 report->new_oid
909 ? report->new_oid
910 : &(ref->new_oid),
911 NULL((void*)0), 0, 0);
912 free(private);
913 }
914 }
915 }
916 return 0;
917}

945static int push_refs_with_push(struct transport *transport,
946 struct ref *remote_refs, int flags)
947{
948 int force_all = flags & TRANSPORT_PUSH_FORCE(1<<1);
949 int mirror = flags & TRANSPORT_PUSH_MIRROR(1<<3);
950 int atomic = flags & TRANSPORT_PUSH_ATOMIC(1<<13);
951 struct helper_data *data = transport->data;
952 struct strbuf buf = STRBUF_INIT{ .alloc = 0, .len = 0, .buf = strbuf_slopbuf };
953 struct ref *ref;
954 struct string_list cas_options = STRING_LIST_INIT_DUP{ ((void*)0), 0, 0, 1, ((void*)0) };
955 struct string_list_item *cas_option;
956
957 get_helper(transport);
958 if (!data->push)
959 return 1;
960
961 for (ref = remote_refs; ref; ref = ref->next) {
962 if (!ref->peer_ref && !mirror)
963 continue;
964
965 /* Check for statuses set by set_ref_status_for_push() */
966 switch (ref->status) {
967 case REF_STATUS_REJECT_NONFASTFORWARD:
968 case REF_STATUS_REJECT_STALE:
969 case REF_STATUS_REJECT_ALREADY_EXISTS:
970 if (atomic) {
971 reject_atomic_push(remote_refs, mirror);
972 string_list_clear(&cas_options, 0);
973 return 0;
974 } else
975 continue;
976 case REF_STATUS_UPTODATE:
977 continue;
978 default:
979 ; /* do nothing */
980 }
981
982 if (force_all)
983 ref->force = 1;
984
985 strbuf_addstr(&buf, "push ");
986 if (!ref->deletion) {
987 if (ref->force)
988 strbuf_addch(&buf, '+');
989 if (ref->peer_ref)
990 strbuf_addstr(&buf, ref->peer_ref->name);
991 else
992 strbuf_addstr(&buf, oid_to_hex(&ref->new_oid));
993 }
994 strbuf_addch(&buf, ':');
995 strbuf_addstr(&buf, ref->name);
996 strbuf_addch(&buf, '\n');
997
998 /*
999 * The "--force-with-lease" options without explicit
1000 * values to expect have already been expanded into
1001 * the ref->old_oid_expect[] field; we can ignore
1002 * transport->smart_options->cas altogether and instead
1003 * can enumerate them from the refs.
1004 */
1005 if (ref->expect_old_sha1) {
1006 struct strbuf cas = STRBUF_INIT{ .alloc = 0, .len = 0, .buf = strbuf_slopbuf };
1007 strbuf_addf(&cas, "%s:%s",
1008 ref->name, oid_to_hex(&ref->old_oid_expect));
1009 string_list_append_nodup(&cas_options,
1010 strbuf_detach(&cas, NULL((void*)0)));
1011 }
1012 }
1013 if (buf.len == 0) {
1014 string_list_clear(&cas_options, 0);
1015 return 0;
1016 }
1017
1018 for_each_string_list_item(cas_option, &cas_options)for (cas_option = (&cas_options)->items; cas_option &&
cas_option < (&cas_options)->items + (&cas_options
)->nr; ++cas_option)
1019 set_helper_option(transport, "cas", cas_option->string);
1020 set_common_push_options(transport, data->name, flags);
1021
1022 strbuf_addch(&buf, '\n');
1023 sendline(data, &buf);
1024 strbuf_release(&buf);
1025 string_list_clear(&cas_options, 0);
1026
1027 return push_update_refs_status(data, remote_refs, flags);
1028}

1167static struct ref *get_refs_list_using_list(struct transport *transport,
1168 int for_push)
1169{
1170 struct helper_data *data = transport->data;
1171 struct child_process *helper;
1172 struct ref *ret = NULL((void*)0);
1173 struct ref **tail = &ret;
1174 struct ref *posn;
1175 struct strbuf buf = STRBUF_INIT{ .alloc = 0, .len = 0, .buf = strbuf_slopbuf };
1176
1177 data->get_refs_list_called = 1;
1178 helper = get_helper(transport);
1179
1180 if (data->object_format) {
1181 write_str_in_full(helper->in, "option object-format\n");
1182 if (recvline(data, &buf) || strcmp(buf.buf, "ok"))
1183 exit(128)exit(trace2_cmd_exit_fl("transport-helper.c", 1183, (128)));
1184 }
1185
1186 if (data->push && for_push)
1187 write_str_in_full(helper->in, "list for-push\n");
1188 else
1189 write_str_in_full(helper->in, "list\n");
1190
1191 while (1) {
1192 char *eov, *eon;
1193 if (recvline(data, &buf))
1194 exit(128)exit(trace2_cmd_exit_fl("transport-helper.c", 1194, (128)));
1195
1196 if (!*buf.buf)
1197 break;
1198 else if (buf.buf[0] == ':') {
1199 const char *value;
1200 if (skip_prefix(buf.buf, ":object-format ", &value)) {
1201 int algo = hash_algo_by_name(value);
1202 if (algo == GIT_HASH_UNKNOWN0)
1203 die(_("unsupported object format '%s'"),
1204 value);
1205 transport->hash_algo = &hash_algos[algo];
1206 }
1207 continue;
1208 }
1209
1210 eov = strchr(buf.buf, ' ');
1211 if (!eov)
1212 die(_("malformed response in ref list: %s"), buf.buf);
1213 eon = strchr(eov + 1, ' ');
1214 *eov = '\0';
1215 if (eon)
1216 *eon = '\0';
1217 *tail = alloc_ref(eov + 1);
1218 if (buf.buf[0] == '@')
1219 (*tail)->symref = xstrdup(buf.buf + 1);
1220 else if (buf.buf[0] != '?')
1221 get_oid_hex_algop(buf.buf, &(*tail)->old_oid, transport->hash_algo);
1222 if (eon) {
1223 if (has_attribute(eon + 1, "unchanged")) {
1224 (*tail)->status |= REF_STATUS_UPTODATE;
1225 if (read_ref((*tail)->name, &(*tail)->old_oid) < 0)
1226 die(_("could not read ref %s"),
1227 (*tail)->name);
1228 }
1229 }
1230 tail = &((*tail)->next);
1231 }
1232 if (debug)
1233 fprintf(stderrstderr, "Debug: Read ref listing.\n");
1234 strbuf_release(&buf);
1235
1236 for (posn = ret; posn; posn = posn->next)
1237 resolve_remote_symref(posn, ret);
1238
1239 return ret;
1240}
