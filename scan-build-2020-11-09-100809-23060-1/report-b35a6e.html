594/*
595 * on successful case, it's up to the caller to close cmd->out
596 */
597static int do_reachable_revlist(struct child_process *cmd,
598 struct object_array *src,
599 struct object_array *reachable,
600 enum allow_uor allow_uor)
601{
602 static const char *argv[] = {
603 "rev-list", "--stdin", NULL((void*)0),
604 };
605 struct object *o;
606 FILE *cmd_in = NULL((void*)0);
607 int i;
608
609 cmd->argv = argv;
610 cmd->git_cmd = 1;
611 cmd->no_stderr = 1;
612 cmd->in = -1;
613 cmd->out = -1;
614
615 /*
616 * If the next rev-list --stdin encounters an unknown commit,
617 * it terminates, which will cause SIGPIPE in the write loop
618 * below.
619 */
620 sigchain_push(SIGPIPE13, SIG_IGN((__sighandler_t) 1));
621
622 if (start_command(cmd))
623 goto error;
624
625 cmd_in = xfdopen(cmd->in, "w");
626
627 for (i = get_max_object_index(); 0 < i; ) {
628 o = get_indexed_object(--i);
629 if (!o)
630 continue;
631 if (reachable && o->type == OBJ_COMMIT)
632 o->flags &= ~TMP_MARK(1u<<4);
633 if (!is_our_ref(o, allow_uor))
634 continue;
635 if (fprintf(cmd_in, "^%s\n", oid_to_hex(&o->oid)) < 0)
636 goto error;
637 }
638 for (i = 0; i < src->nr; i++) {
639 o = src->objects[i].item;
640 if (is_our_ref(o, allow_uor)) {
641 if (reachable)
642 add_object_array(o, NULL((void*)0), reachable);
643 continue;
644 }
645 if (reachable && o->type == OBJ_COMMIT)
646 o->flags |= TMP_MARK(1u<<4);
647 if (fprintf(cmd_in, "%s\n", oid_to_hex(&o->oid)) < 0)
648 goto error;
649 }
650 if (ferror(cmd_in) || fflush(cmd_in))
651 goto error;
652 fclose(cmd_in);
653 cmd->in = -1;
654 sigchain_pop(SIGPIPE13);
655
656 return 0;
657
658error:
659 sigchain_pop(SIGPIPE13);
660
661 if (cmd_in)
662 fclose(cmd_in);
663 if (cmd->out >= 0)
664 close(cmd->out);
665 return -1;
666}

668static int get_reachable_list(struct upload_pack_data *data,
669 struct object_array *reachable)
670{
671 struct child_process cmd = CHILD_PROCESS_INIT{ ((void*)0), { empty_strvec, 0, 0 }, { empty_strvec, 0, 0 } };
672 int i;
673 struct object *o;
674 char namebuf[GIT_MAX_HEXSZ(2 * 32) + 2]; /* ^ + hash + LF */
675 const unsigned hexsz = the_hash_algothe_repository->hash_algo->hexsz;
676
677 if (do_reachable_revlist(&cmd, &data->shallows, reachable,
678 data->allow_uor) < 0)
679 return -1;
680
681 while ((i = read_in_full(cmd.out, namebuf, hexsz + 1)) == hexsz + 1) {
Although the value stored to 'i' is used in the enclosing expression, the value is never actually read from 'i'
682 struct object_id oid;
683 const char *p;
684
685 if (parse_oid_hex(namebuf, &oid, &p) || *p != '\n')
686 break;
687
688 o = lookup_object(the_repository, &oid);
689 if (o && o->type == OBJ_COMMIT) {
690 o->flags &= ~TMP_MARK(1u<<4);
691 }
692 }
693 for (i = get_max_object_index(); 0 < i; i--) {
694 o = get_indexed_object(i - 1);
695 if (o && o->type == OBJ_COMMIT &&
696 (o->flags & TMP_MARK(1u<<4))) {
697 add_object_array(o, NULL((void*)0), reachable);
698 o->flags &= ~TMP_MARK(1u<<4);
699 }
700 }
701 close(cmd.out);
702
703 if (finish_command(&cmd))
704 return -1;
705
706 return 0;
707}

709static int has_unreachable(struct object_array *src, enum allow_uor allow_uor)
710{
711 struct child_process cmd = CHILD_PROCESS_INIT{ ((void*)0), { empty_strvec, 0, 0 }, { empty_strvec, 0, 0 } };
712 char buf[1];
713 int i;
714
715 if (do_reachable_revlist(&cmd, src, NULL((void*)0), allow_uor) < 0)
716 return 1;
717
718 /*
719 * The commits out of the rev-list are not ancestors of
720 * our ref.
721 */
722 i = read_in_full(cmd.out, buf, 1);
723 if (i)
724 goto error;
725 close(cmd.out);
726 cmd.out = -1;
727
728 /*
729 * rev-list may have died by encountering a bad commit
730 * in the history, in which case we do want to bail out
731 * even when it showed no commit.
732 */
733 if (finish_command(&cmd))
734 goto error;
735
736 /* All the non-tip ones are ancestors of what we advertised */
737 return 0;
738
739error:
740 if (cmd.out >= 0)
741 close(cmd.out);
742 return 1;
743}

774static void send_shallow(struct upload_pack_data *data,
775 struct commit_list *result)
776{
777 while (result) {
778 struct object *object = &result->item->object;
779 if (!(object->flags & (CLIENT_SHALLOW(1u << 18)|NOT_SHALLOW(1u << 17)))) {
780 packet_writer_write(&data->writer, "shallow %s",

784 }

789static void send_unshallow(struct upload_pack_data *data)
790{
791 int i;
792
793 for (i = 0; i < data->shallows.nr; i++) {
794 struct object *object = data->shallows.objects[i].item;
795 if (object->flags & NOT_SHALLOW(1u << 17)) {
796 struct commit_list *parents;
797 packet_writer_write(&data->writer, "unshallow %s",
798 oid_to_hex(&object->oid));
799 object->flags &= ~CLIENT_SHALLOW(1u << 18);
800 /*
801 * We want to _register_ "object" as shallow, but we
802 * also need to traverse object's parents to deepen a
803 * shallow clone. Unregister it for now so we can
804 * parse and add the parents to the want list, then
805 * re-register it.
806 */
807 unregister_shallow(&object->oid);
808 object->parsed = 0;
809 parse_commit_or_die((struct commit *)object);
810 parents = ((struct commit *)object)->parents;
811 while (parents) {
812 add_object_array(&parents->item->object,
813 NULL((void*)0), &data->want_obj);
814 parents = parents->next;
815 }
816 add_object_array(object, NULL((void*)0), &data->extra_edge_obj);
817 }
818 /* make sure commit traversal conforms to client */
819 register_shallow(the_repository, &object->oid);
820 }
821}

1048static void receive_needs(struct upload_pack_data *data,
1049 struct packet_reader *reader)
1050{
1051 int has_non_tip = 0;
1052
1053 data->shallow_nr = 0;
1054 for (;;) {
1055 struct object *o;
1056 const char *features;
1057 struct object_id oid_buf;
1058 const char *arg;
1059
1060 reset_timeout(data->timeout);
1061 if (packet_reader_read(reader) != PACKET_READ_NORMAL)
1062 break;
1063
1064 if (process_shallow(reader->line, &data->shallows))
1065 continue;
1066 if (process_deepen(reader->line, &data->depth))
1067 continue;
1068 if (process_deepen_since(reader->line, &data->deepen_since, &data->deepen_rev_list))
1069 continue;
1070 if (process_deepen_not(reader->line, &data->deepen_not, &data->deepen_rev_list))
1071 continue;
1072
1073 if (skip_prefix(reader->line, "filter ", &arg)) {
1074 if (!data->filter_capability_requested)
1075 die("git upload-pack: filtering capability not negotiated");
1076 list_objects_filter_die_if_populated(&data->filter_options);
1077 parse_list_objects_filter(&data->filter_options, arg);
1078 die_if_using_banned_filter(data);
1079 continue;
1080 }
1081
1082 if (!skip_prefix(reader->line, "want ", &arg) ||
1083 parse_oid_hex(arg, &oid_buf, &features))
1084 die("git upload-pack: protocol error, "
1085 "expected to get object ID, not '%s'", reader->line);
1086
1087 if (parse_feature_request(features, "deepen-relative"))
1088 data->deepen_relative = 1;
1089 if (parse_feature_request(features, "multi_ack_detailed"))
1090 data->multi_ack = MULTI_ACK_DETAILED;
1091 else if (parse_feature_request(features, "multi_ack"))
1092 data->multi_ack = MULTI_ACK;
1093 if (parse_feature_request(features, "no-done"))
1094 data->no_done = 1;
1095 if (parse_feature_request(features, "thin-pack"))
1096 data->use_thin_pack = 1;
1097 if (parse_feature_request(features, "ofs-delta"))
1098 data->use_ofs_delta = 1;
1099 if (parse_feature_request(features, "side-band-64k"))
1100 data->use_sideband = LARGE_PACKET_MAX65520;
1101 else if (parse_feature_request(features, "side-band"))
1102 data->use_sideband = DEFAULT_PACKET_MAX1000;
1103 if (parse_feature_request(features, "no-progress"))
1104 data->no_progress = 1;
1105 if (parse_feature_request(features, "include-tag"))
1106 data->use_include_tag = 1;
1107 if (data->allow_filter &&
1108 parse_feature_request(features, "filter"))
1109 data->filter_capability_requested = 1;
1110
1111 o = parse_object(the_repository, &oid_buf);
1112 if (!o) {
1113 packet_writer_error(&data->writer,
1114 "upload-pack: not our ref %s",
1115 oid_to_hex(&oid_buf));
1116 die("git upload-pack: not our ref %s",
1117 oid_to_hex(&oid_buf));
1118 }
1119 if (!(o->flags & WANTED(1u << 13))) {
1120 o->flags |= WANTED(1u << 13);
1121 if (!((data->allow_uor & ALLOW_ANY_SHA1) == ALLOW_ANY_SHA1
1122 || is_our_ref(o, data->allow_uor)))
1123 has_non_tip = 1;
1124 add_object_array(o, NULL((void*)0), &data->want_obj);
1125 }
1126 }
1127
1128 /*
1129 * We have sent all our refs already, and the other end
1130 * should have chosen out of them. When we are operating
1131 * in the stateless RPC mode, however, their choice may
1132 * have been based on the set of older refs advertised
1133 * by another process that handled the initial request.
1134 */
1135 if (has_non_tip)
1136 check_non_tip(data);
1137
1138 if (!data->use_sideband && data->daemon_mode)
1139 data->no_progress = 1;
1140
1141 if (data->depth == 0 && !data->deepen_rev_list && data->shallows.nr == 0)
1142 return;
1143
1144 if (send_shallow_list(data))
1145 packet_flush(1);
1146}

1350 }
1351 }

1361 struct object_id oid;
1362 struct object *o;

1433static void process_args(struct packet_reader *request,
1434 struct upload_pack_data *data)
1435{
1436 while (packet_reader_read(request) == PACKET_READ_NORMAL) {
1437 const char *arg = request->line;
1438 const char *p;
1439
1440 /* process want */
1441 if (parse_want(&data->writer, arg, &data->want_obj))
1442 continue;
1443 if (data->allow_ref_in_want &&
1444 parse_want_ref(&data->writer, arg, &data->wanted_refs,
1445 &data->want_obj))
1446 continue;
1447 /* process have line */
1448 if (parse_have(arg, &data->haves))
1449 continue;
1450
1451 /* process args like thin-pack */
1452 if (!strcmp(arg, "thin-pack")) {
1453 data->use_thin_pack = 1;
1454 continue;
1455 }
1456 if (!strcmp(arg, "ofs-delta")) {
1457 data->use_ofs_delta = 1;
1458 continue;
1459 }
1460 if (!strcmp(arg, "no-progress")) {
1461 data->no_progress = 1;
1462 continue;
1463 }
1464 if (!strcmp(arg, "include-tag")) {
1465 data->use_include_tag = 1;
1466 continue;
1467 }
1468 if (!strcmp(arg, "done")) {
1469 data->done = 1;
1470 continue;
1471 }
1472
1473 /* Shallow related arguments */
1474 if (process_shallow(arg, &data->shallows))
1475 continue;
1476 if (process_deepen(arg, &data->depth))
1477 continue;
1478 if (process_deepen_since(arg, &data->deepen_since,
1479 &data->deepen_rev_list))
1480 continue;
1481 if (process_deepen_not(arg, &data->deepen_not,
1482 &data->deepen_rev_list))
1483 continue;
1484 if (!strcmp(arg, "deepen-relative")) {
1485 data->deepen_relative = 1;
1486 continue;
1487 }
1488
1489 if (data->allow_filter && skip_prefix(arg, "filter ", &p)) {
1490 list_objects_filter_die_if_populated(&data->filter_options);
1491 parse_list_objects_filter(&data->filter_options, p);
1492 die_if_using_banned_filter(data);
1493 continue;
1494 }
1495
1496 if ((git_env_bool("GIT_TEST_SIDEBAND_ALL", 0) ||
1497 data->allow_sideband_all) &&
1498 !strcmp(arg, "sideband-all")) {
1499 data->writer.use_sideband = 1;
1500 continue;
1501 }
1502
1503 if (skip_prefix(arg, "packfile-uris ", &p)) {
1504 string_list_split(&data->uri_protocols, p, ',', -1);
1505 continue;
1506 }
1507
1508 /* ignore unknown lines maybe? */
1509 die("unexpected line: '%s'", arg);
1510 }
1511
1512 if (data->uri_protocols.nr && !data->writer.use_sideband)
1513 string_list_clear(&data->uri_protocols, 0);
1514
1515 if (request->status != PACKET_READ_FLUSH)
1516 die(_("expected flush after fetch arguments"));
1517}

1619enum fetch_state {
1620 FETCH_PROCESS_ARGS = 0,
1621 FETCH_SEND_ACKS,
1622 FETCH_SEND_PACK,
1623 FETCH_DONE,
1624};
1625
1626int upload_pack_v2(struct repository *r, struct strvec *keys,
1627 struct packet_reader *request)
1628{
1629 enum fetch_state state = FETCH_PROCESS_ARGS;
1630 struct upload_pack_data data;
1631
1632 clear_object_flags(ALL_FLAGS((1u << 11) | (1u << 12) | (1u << 13) | (1u
<< 14) | (1u << 16) | (1u << 17) | (1u <<
18) | (1u << 19)));
1633
1634 upload_pack_data_init(&data);
1635 data.use_sideband = LARGE_PACKET_MAX65520;
1636
1637 git_config(upload_pack_config, &data);
1638
1639 while (state != FETCH_DONE) {
1640 switch (state) {
1641 case FETCH_PROCESS_ARGS:
1642 process_args(request, &data);
1643
1644 if (!data.want_obj.nr) {
1645 /*
1646 * Request didn't contain any 'want' lines,
1647 * guess they didn't want anything.
1648 */
1649 state = FETCH_DONE;
1650 } else if (data.haves.nr) {
1651 /*
1652 * Request had 'have' lines, so lets ACK them.
1653 */
1654 state = FETCH_SEND_ACKS;
1655 } else {
1656 /*
1657 * Request had 'want's but no 'have's so we can
1658 * immedietly go to construct and send a pack.
1659 */
1660 state = FETCH_SEND_PACK;
1661 }
1662 break;
1663 case FETCH_SEND_ACKS:
1664 if (process_haves_and_send_acks(&data))
1665 state = FETCH_SEND_PACK;
1666 else
1667 state = FETCH_DONE;
1668 break;
1669 case FETCH_SEND_PACK:
1670 send_wanted_ref_info(&data);
1671 send_shallow_info(&data);
1672
1673 if (data.uri_protocols.nr) {
1674 create_pack_file(&data, &data.uri_protocols);
1675 } else {
1676 packet_writer_write(&data.writer, "packfile\n");
1677 create_pack_file(&data, NULL((void*)0));
1678 }
1679 state = FETCH_DONE;
1680 break;
1681 case FETCH_DONE:
1682 continue;
1683 }
1684 }
1685
1686 upload_pack_data_clear(&data);
1687 return 0;
1688}
