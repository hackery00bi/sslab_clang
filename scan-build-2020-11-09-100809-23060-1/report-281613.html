264 while (start < ptr && isdigit(ptr[-1])((sane_ctype[(unsigned char)(ptr[-1])] & (0x02)) != 0) && ptr[-1] != ':')
265 ptr--;
266 if (start < ptr && ptr[-1] == ':')
267 end = ptr - 1;
268 }
269
270 /*
271 * Find last component. To remain backwards compatible we
272 * also regard colons as path separators, such that
273 * cloning a repository 'foo:bar.git' would result in a
274 * directory 'bar' being guessed.
275 */
276 ptr = end;
277 while (start < ptr && !is_dir_sepgit_is_dir_sep(ptr[-1]) && ptr[-1] != ':')
278 ptr--;
279 start = ptr;
280
281 /*
282 * Strip .{bundle,git}.
283 */
284 len = end - start;
285 strip_suffix_mem(start, &len, is_bundle ? ".bundle" : ".git");
286
287 if (!len || (len == 1 && *start == '/'))
288 die(_("No directory name could be guessed.\n"
289 "Please specify a directory on the command line"));
290
291 if (is_bare)
292 dir = xstrfmt("%.*s.git", (int)len, start);
293 else
294 dir = xstrndup(start, len);
295 /*
296 * Replace sequences of 'control' characters and whitespace
297 * with one ascii space, remove leading and trailing spaces.
298 */
299 if (*dir) {
300 char *out = dir;
301 int prev_space = 1 /* strip leading whitespace */;
302 for (end = dir; *end; ++end) {
303 char ch = *end;
304 if ((unsigned char)ch < '\x20')
305 ch = '\x20';
306 if (isspace(ch)((sane_ctype[(unsigned char)(ch)] & (0x01)) != 0)) {
307 if (prev_space)
308 continue;
309 prev_space = 1;
310 } else
311 prev_space = 0;
312 *out++ = ch;
313 }
314 *out = '\0';
315 if (out > dir && prev_space)
316 out[-1] = '\0';
317 }
318 return dir;
319}

323 char *end = dir + strlen(dir);
324
325 while (dir < end - 1 && is_dir_sepgit_is_dir_sep(end[-1]))
326 end--;
327 *end = '\0';

365static void copy_alternates(struct strbuf *src, const char *src_repo)
366{
367 /*
368 * Read from the source objects/info/alternates file
369 * and copy the entries to corresponding file in the
370 * destination repository with add_to_alternates_file().
371 * Both src and dst have "$path/objects/info/alternates".
372 *
373 * Instead of copying bit-for-bit from the original,
374 * we need to append to existing one so that the already
375 * created entry via "clone -s" is not lost, and also
376 * to turn entries with paths relative to the original
377 * absolute, so that they can be used in the new repository.
378 */
379 FILE *in = xfopen(src->buf, "r");
380 struct strbuf line = STRBUF_INIT{ .alloc = 0, .len = 0, .buf = strbuf_slopbuf };
381
382 while (strbuf_getline(&line, in) != EOF(-1)) {
383 char *abs_path;
384 if (!line.len || line.buf[0] == '#')
385 continue;
386 if (is_absolute_path(line.buf)) {
387 add_to_alternates_file(line.buf);
388 continue;
389 }
390 abs_path = mkpathdup("%s/objects/%s", src_repo, line.buf);
391 if (!normalize_path_copy(abs_path, abs_path))
392 add_to_alternates_file(abs_path);
393 else
394 warning("skipping invalid relative alternate: %s/%s",
395 src_repo, line.buf);
396 free(abs_path);
397 }
398 strbuf_release(&line);
399 fclose(in);
400}

417static void copy_or_link_directory(struct strbuf *src, struct strbuf *dest,
418 const char *src_repo)
419{
420 int src_len, dest_len;
421 struct dir_iterator *iter;
422 int iter_status;
423 unsigned int flags;
424 struct strbuf realpath = STRBUF_INIT{ .alloc = 0, .len = 0, .buf = strbuf_slopbuf };
425
426 mkdir_if_missing(dest->buf, 0777);
427
428 flags = DIR_ITERATOR_PEDANTIC(1 << 0) | DIR_ITERATOR_FOLLOW_SYMLINKS(1 << 1);
429 iter = dir_iterator_begin(src->buf, flags);
430
431 if (!iter)
432 die_errno(_("failed to start iterator over '%s'"), src->buf);
433
434 strbuf_addch(src, '/');
435 src_len = src->len;
436 strbuf_addch(dest, '/');
437 dest_len = dest->len;
438
439 while ((iter_status = dir_iterator_advance(iter)) == ITER_OK0) {
440 strbuf_setlen(src, src_len);
441 strbuf_addstr(src, iter->relative_path);
442 strbuf_setlen(dest, dest_len);
443 strbuf_addstr(dest, iter->relative_path);
444
445 if (S_ISDIR(iter->st.st_mode)((((iter->st.st_mode)) & 0170000) == (0040000))) {
446 mkdir_if_missing(dest->buf, 0777);
447 continue;
448 }
449
450 /* Files that cannot be copied bit-for-bit... */
451 if (!fspathcmp(iter->relative_path, "info/alternates")) {
452 copy_alternates(src, src_repo);
453 continue;
454 }
455
456 if (unlink(dest->buf) && errno(*__errno_location ()) != ENOENT2)
457 die_errno(_("failed to unlink '%s'"), dest->buf);
458 if (!option_no_hardlinks) {
459 strbuf_realpath(&realpath, src->buf, 1);
460 if (!link(realpath.buf, dest->buf))
461 continue;
462 if (option_local > 0)
463 die_errno(_("failed to create link '%s'"), dest->buf);
464 option_no_hardlinks = 1;
465 }
466 if (copy_file_with_time(dest->buf, src->buf, 0666))
467 die_errno(_("failed to copy file to '%s'"), dest->buf);
468 }
469
470 if (iter_status != ITER_DONE-1) {
471 strbuf_setlen(src, src_len);
472 die(_("failed to iterate over '%s'"), src->buf);
473 }
474
475 strbuf_release(&realpath);
476}
