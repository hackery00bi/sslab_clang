27/****************************************************************************
28 * This skips over a name field while parsing the packet. If the name
29 * is just a two-byte compression field likce 0xc0 0x1a, then it'll skip
30 * those two bytes. However, when it does the skip, it does validate
31 * the name. Thus, if it's a compressed name, it'll follow the compression
32 * links to validate things like long names and infinite recursion.
33 ****************************************************************************/
34static unsigned
35dns_name_skip_validate(const unsigned char *px, unsigned offset, unsigned length, unsigned name_length)
36{
37 unsigned ERROR = length + 1;
38 unsigned result = offset + 2;
39 unsigned recursion = 0;
40
41 /* 'for all labels' */
42 for (;;) {
43 unsigned len;
44
45 /* validate: the eventual uncompressed name will be less than 255 */
46 if (name_length >= 255)
47 return ERROR;
48
49 /* validate: haven't gone off end of packet */
50 if (offset >= length)
51 return ERROR;
52
53 /* grab length of next label */
54 len = px[offset];
55
56 /* Do two types of processing, either a compression code or a
57 * original label. Note that we can alternate back and forth
58 * between these two states. */
59 if (len & 0xC0) {
60 /* validate: top 2 bits are 11*/
61 if ((len & 0xC0) != 0xC0)
62 return ERROR;
63
64 /* validate: enough bytes left for 2 byte compression field */
65 if (offset + 1 >= length)
66 return ERROR;
67
68 /* follow the compression pointer to the next location */
69 offset = (px[offset]&0x3F)<<8 | px[offset+1];
70
71 /* validate: follow a max of 4 links */
72 if (++recursion > 4)
73 return ERROR;
74 } else {
75 /* we have a normal label */
76 recursion = 0;
77
78 /* If the label-length is zero, then that meaans we've reached
79 * the end of the name */
80 if (len == 0) {
81 return result; /* end of domain name */
82 }
83
84 /* There are more labels to come, therefore skip this and go
85 * to the next one */
86 name_length += len + 1;
87 offset += len + 1;
88 }
89 }
90}

92/****************************************************************************
93 * Just skip the name, without validating whether it's valid or not. This
94 * is for re-parsing the packet usually, after we've validated that all
95 * the names are ok.
96 ****************************************************************************/
97unsigned
98dns_name_skip(const unsigned char px[], unsigned offset, unsigned max)
99{
100 unsigned name_length = 0;
101
102 /* Loop through all labels
103 * NOTE: the only way this loops around is in the case of a normal
104 * label. All other conditions cause a 'return' from this function */
105 for (;;) {
106 if (name_length >= 255)
107 return max + 1;
108
109 if (offset >= max)
110 return max + 1;
111
112 switch (px[offset]>>6) {
113 case 0:
114 /* uncompressed label */
115 if (px[offset] == 0) {
116 return offset+1; /* end of domain name */
117 } else {
118 name_length += px[offset] + 1;
119 offset += px[offset] + 1;
120 continue;
121 }
122 break;
123 case 3:
124 /* 0xc0 = compressed name */

131 case 1:
132 return max + 1;

325/***************************************************************************
326 * Set the "syn-cookie" style information so that we can validate replies
327 * match a valid request. We don't hold "state" on the requests, so this
328 * becomes a hash of the port/IP information.
329 * DNS has a two-byte "transaction id" field, so we can't use the full
330 * cookie, just the lower two bytes of it.
331 * Below in "handle_dns", we validate that the cookie is correct.
332 ***************************************************************************/
333unsigned
334dns_set_cookie(unsigned char *px, size_t length, uint64_t cookie)
335{
336 if (length > 2) {
337 px[0] = (unsigned char)(cookie >> 8);
338 px[1] = (unsigned char)(cookie >> 0);
339 return cookie & 0xFFFF;
340 } else
341 return 0;
342}

344/***************************************************************************
345 * Process a DNS packet received in response to UDP probes to port 53.
346 * This function has three main tasks:
347 * - parse the DNS protocol, and make sure it's valid DNS.
348 * - make sure that the DNS response matches a valid request using
349 * the "syn-cookie" approach.
350 * - parse the "version.bind" response and report it as the version
351 * string for the banner.
352 ***************************************************************************/
353unsigned
354handle_dns(struct Output *out, time_t timestamp,
355 const unsigned char *px, unsigned length,
356 struct PreprocessedInfo *parsed,
357 uint64_t entropy)
358{
359 unsigned ip_them;
360 unsigned ip_me;
361 unsigned port_them = parsed->port_src;
362 unsigned port_me = parsed->port_dst;
363 struct DNS_Incoming dns[1];
364 unsigned offset;
365 uint64_t seqno;
366
367 ip_them = parsed->ip_src[0]<<24 | parsed->ip_src[1]<<16
368 | parsed->ip_src[2]<< 8 | parsed->ip_src[3]<<0;
369 ip_me = parsed->ip_dst[0]<<24 | parsed->ip_dst[1]<<16
370 | parsed->ip_dst[2]<< 8 | parsed->ip_dst[3]<<0;
371
372 seqno = (unsigned)syn_cookie(ip_them, port_them | Templ_UDP, ip_me, port_me, entropy);
373
374 proto_dns_parse(dns, px, parsed->app_offset, parsed->app_offset + parsed->app_length);
375
376 if ((seqno & 0xFFFF) != dns->id)
377 return 1;
378
379 if (dns->qr != 1)
380 return 0;
381 if (dns->rcode != 0)
382 return 0;
383 if (dns->qdcount != 1)
384 return 0;
385 if (dns->ancount < 1)
386 return 0;
387 if (dns->rr_count < 2)
388 return 0;
389
390
391 offset = dns->rr_offset[1];
392 offset = dns_name_skip(px, offset, length);
393 if (offset + 10 >= length)
394 return 0;
395
396 {
397 unsigned type = px[offset+0]<<8 | px[offset+1];
398 unsigned xclass = px[offset+2]<<8 | px[offset+3];
399 unsigned rrlen = px[offset+8]<<8 | px[offset+9];

404 if (type != 0x10 || xclass != 3)

407 offset += 11;

415 }
