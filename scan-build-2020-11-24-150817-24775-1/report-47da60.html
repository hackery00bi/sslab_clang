88 * Standard CRC32 calculation.

90static unsigned
91crc_calc(const unsigned char *px, unsigned length)
92{
93 unsigned i;
94 unsigned crc;
95
96 crc = (unsigned)~0;
97 for (i = 0; i < length; i++) {
98 crc = crc32_table[(crc ^ px[i]) & 0xff] ^ (crc >> 8);
99 }
100 crc = ~crc;
101
102 return crc;
103}

105/***************************************************************************
106 * Zero-Access encrypts packets with a simple encryption of starting
107 * with the key "ftp2" XORed with each 4-byte word, then rotated left
108 * by one bit after every XOR. Because it's XOR, encryption is also
109 * decryption.
110 ***************************************************************************/
111static unsigned
112zadecrypt(const unsigned char *src, size_t src_len, unsigned char *dst, size_t dst_len)
113{
114 unsigned key;
115 size_t i;
116
117 key = 'f'<<24 | 't'<<16 | 'p'<<8 | '2'<<0;
118
119 for (i=0; i<dst_len && i<src_len; i+=4) {
120 dst[i+0] = src[i+0] ^ (unsigned char)(key>> 0);
121 dst[i+1] = src[i+1] ^ (unsigned char)(key>> 8);
122 dst[i+2] = src[i+2] ^ (unsigned char)(key>>16);
123 dst[i+3] = src[i+3] ^ (unsigned char)(key>>24);
124
125 key = key<<1 | key>>31;
126 }
127
128 return (unsigned)src_len;
129}
