167/***************************************************************************
168 * The receive thread doesn't transmit packets. Instead, it queues them
169 * up on the transmit thread. Every so often, the transmit thread needs
170 * to flush this transmit queue and send everything.
171 *
172 * This is an inherent design issue trying to send things as batches rather
173 * than individually. It increases latency, but increases performance. We
174 * don't really care about latency.
175 ***************************************************************************/
176static void
177flush_packets(struct Adapter *adapter,
178 PACKET_QUEUE *packet_buffers,
179 PACKET_QUEUE *transmit_queue,
180 uint64_t *packets_sent,
181 uint64_t *batchsize)
182{
183 /*
184 * Send a batch of queued packets
185 */
186 for ( ; (*batchsize); (*batchsize)--) {
187 int err;
188 struct PacketBuffer *p;
189
190 /*
191 * Get the next packet from the transmit queue. This packet was
192 * put there by a receive thread, and will contain things like
193 * an ACK or an HTTP request
194 */
195 err = rte_ring_sc_dequeue(transmit_queue, (void**)&p);
196 if (err) {
197 break; /* queue is empty, nothing to send */
198 }
199
200
201 /*
202 * Actually send the packet
203 */
204 rawsock_send_packet(adapter, p->px, (unsigned)p->length, 1);
205
206 /*
207 * Now that we are done with the packet, put it on the free list
208 * of buffers that the transmit thread can reuse
209 */
210 for (err=1; err; ) {
211 err = rte_ring_sp_enqueue(packet_buffers, p);
212 if (err) {
213 LOG(0, "transmit queue full (should be impossible)\n");
214 pixie_usleep(10000);
215 }
216 }
217
218
219 /*
220 * Remember that we sent a packet, which will be used in
221 * throttling.
222 */
223 (*packets_sent)++;
224 }
225
226}

250/***************************************************************************
251 * This thread spews packets as fast as it can
252 *
253 * THIS IS WHERE ALL THE EXCITEMENT HAPPENS!!!!
254 * 90% of CPU cycles are in the function.
255 *
256 ***************************************************************************/
257static void
258transmit_thread(void *v) /*aka. scanning_thread() */
259{
260 struct ThreadPair *parms = (struct ThreadPair *)v;
261 uint64_t i;
262 uint64_t start;
263 uint64_t end;
264 const struct Masscan *masscan = parms->masscan;
265 unsigned retries = masscan->retries;
266 unsigned rate = (unsigned)masscan->max_rate;
267 unsigned r = retries + 1;
268 uint64_t range;
269 struct BlackRock blackrock;
270 uint64_t count_ips = rangelist_count(&masscan->targets);
271 struct Throttler *throttler = parms->throttler;
272 struct TemplateSet pkt_template = templ_copy(parms->tmplset);
273 unsigned *picker = parms->picker;
274 struct Adapter *adapter = parms->adapter;
275 uint64_t packets_sent = 0;
276 unsigned increment = (masscan->shard.of-1) + masscan->nic_count;
277 unsigned src_ip;
278 unsigned src_ip_mask;
279 unsigned src_port;
280 unsigned src_port_mask;
281 uint64_t seed = masscan->seed;
282 uint64_t repeats = 0; /* --infinite repeats */
283 uint64_t *status_syn_count;
284 uint64_t entropy = masscan->seed;
285
286 LOG(1, "THREAD: xmit: starting thread #%u\n", parms->nic_index);
287
288 /* export a pointer to this variable outside this threads so
289 * that the 'status' system can print the rate of syns we are
290 * sending */
291 status_syn_count = (uint64_t*)malloc(sizeof(uint64_t));
292 *status_syn_count = 0;
293 parms->total_syns = status_syn_count;
294
295
296 /* Normally, we have just one source address. In special cases, though
297 * we can have multiple. */
298 get_sources(masscan, parms->nic_index,
299 &src_ip, &src_ip_mask,
300 &src_port, &src_port_mask);
301
302
303 /* "THROTTLER" rate-limits how fast we transmit, set with the
304 * --max-rate parameter */
305 throttler_start(throttler, masscan->max_rate/masscan->nic_count);
306
307infinite:
308
309 /* Create the shuffler/randomizer. This creates the 'range' variable,
310 * which is simply the number of IP addresses times the number of
311 * ports */
312 range = rangelist_count(&masscan->targets)
313 * rangelist_count(&masscan->ports);
314 blackrock_init(&blackrock, range, seed, masscan->blackrock_rounds);
315
316 /* Calculate the 'start' and 'end' of a scan. One reason to do this is
317 * to support --shard, so that multiple machines can co-operate on
318 * the same scan. Another reason to do this is so that we can bleed
319 * a little bit past the end when we have --retries. Yet another
320 * thing to do here is deal with multiple network adapters, which
321 * is essentially the same logic as shards. */
322 start = masscan->resume.index + (masscan->shard.one-1) + parms->nic_index;
323 end = range;
324 if (masscan->resume.count && end > start + masscan->resume.count)
325 end = start + masscan->resume.count;
326 end += retries * rate;
327
328
329 /* -----------------
330 * the main loop
331 * -----------------*/
332 LOG(3, "THREAD: xmit: starting main loop: [%llu..%llu]\n", start, end);
333 for (i=start; i<end; ) {
334 uint64_t batch_size;
335
336 /*
337 * Do a batch of many packets at a time. That because per-packet
338 * throttling is expensive at 10-million pps, so we reduce the
339 * per-packet cost by doing batches. At slower rates, the batch
340 * size will always be one. (--max-rate)
341 */
342 batch_size = throttler_next_batch(throttler, packets_sent);
343
344 /*
345 * Transmit packets from other thread, when doing --banners. This
346 * takes priority over sending SYN packets. If there is so much
347 * activity grabbing banners that we cannot transmit more SYN packets,
348 * then "batch_size" will get decremented to zero, and we won't be
349 * able to transmit SYN packets.
350 */
351 flush_packets(adapter, parms->packet_buffers, parms->transmit_queue,
352 &packets_sent, &batch_size);
353
354
355 /*
356 * Transmit a bunch of packets. At any rate slower than 100,000
357 * packets/second, the 'batch_size' is likely to be 1
358 */
359 while (batch_size && i < end) {
360 uint64_t xXx;
361 unsigned ip_them;
362 unsigned port_them;
363 unsigned ip_me;
364 unsigned port_me;
365 uint64_t cookie;
366
367
368 /*
369 * RANDOMIZE THE TARGET:
370 * This is kinda a tricky bit that picks a random IP and port
371 * number in order to scan. We monotonically increment the
372 * index 'i' from [0..range]. We then shuffle (randomly transmog)
373 * that index into some other, but unique/1-to-1, number in the
374 * same range. That way we visit all targets, but in a random
375 * order. Then, once we've shuffled the index, we "pick" the
376 * IP address and port that the index refers to.
377 */
378 xXx = (i + (r--) * rate);
379 if (rate > range)
380 xXx %= range;
381 else
382 while (xXx >= range)
383 xXx -= range;
384 xXx = blackrock_shuffle(&blackrock, xXx);
385 ip_them = rangelist_pick2(&masscan->targets, xXx % count_ips, picker);
386 port_them = rangelist_pick(&masscan->ports, xXx / count_ips);
387
388 /*
389 * SYN-COOKIE LOGIC
390 * Figure out the source IP/port, and the SYN cookie
391 */
392 if (src_ip_mask > 1 || src_port_mask > 1) {
393 uint64_t ck = syn_cookie((unsigned)(i+repeats),
394 (unsigned)((i+repeats)>>32),
395 (unsigned)xXx, (unsigned)(xXx>>32),
396 entropy);
397 port_me = src_port + (ck & src_port_mask);
398 ip_me = src_ip + ((ck>>16) & src_ip_mask);
399 } else {
400 ip_me = src_ip;
401 port_me = src_port;
402 }
403 cookie = syn_cookie(ip_them, port_them, ip_me, port_me, entropy);
404
405 /*
406 * SEND THE PROBE
407 * This is sorta the entire point of the program, but little
408 * exciting happens here. The thing to note that this may
409 * be a "raw" transmit that bypasses the kernel, meaning
410 * we can call this function millions of times a second.
411 */
412 rawsock_send_probe(
413 adapter,
414 ip_them, port_them,
415 ip_me, port_me,
416 (unsigned)cookie,
417 !batch_size, /* flush queue on last packet in batch */
418 &pkt_template
419 );
420 batch_size--;
421 packets_sent++;
422 (*status_syn_count)++;
423
424 /*
425 * SEQUENTIALLY INCREMENT THROUGH THE RANGE
426 * Yea, I know this is a puny 'i++' here, but it's a core feature
427 * of the system that is linearly increments through the range,
428 * but produces from that a shuffled sequence of targets (as
429 * described above). Because we are linearly incrementing this
430 * number, we can do lots of creative stuff, like doing clever
431 * retransmits and sharding.
432 */
433 if (r == 0) {
434 i += increment; /* <------ increment by 1 normally, more with shards/nics */
435 r = retries + 1;
436 }
437
438 } /* end of batch */
439
440
441 /* save our current location for resuming, if the user pressed
442 * <ctrl-c> to exit early */
443 parms->my_index = i;
444
445 /* If the user pressed <ctrl-c>, then we need to exit. but, in case
446 * the user wants to --resume the scan later, we save the current
447 * state in a file */
448 if (is_tx_done) {
449 break;
450 }
451 }
452
453 /*
454 * --infinite
455 * For load testing, go around and do this again
456 */
457 if (masscan->is_infinite && !is_tx_done) {
458 seed++;
459 repeats++;
460 goto infinite;
461 }
462
463 /*
464 * Flush any untransmitted packets. High-speed mechanisms like Windows
465 * "sendq" and Linux's "PF_RING" queue packets and transmit many together,
466 * so there may be some packets that we've queueud but not yet transmitted.
467 * This call makes sure they are transmitted.
468 */
469 rawsock_flush(adapter);
470
471 /*
472 * Wait until the receive thread realizes the scan is over
473 */
474 LOG(1, "THREAD: xmit done, waiting for receive thread to realize this\n");
475
476 /*
477 * We are done transmitting. However, response packets will take several
478 * seconds to arrive. Therefore, sit in short loop waiting for those
479 * packets to arrive. Pressing <ctrl-c> a second time will exit this
480 * prematurely.
481 */
482 while (!is_rx_done) {
483 unsigned k;
484 uint64_t batch_size;
485
486 for (k=0; k<1000; k++) {
487 /*
488 * Only send a few packets at a time, throttled according to the max
489 * --max-rate set by the user
490 */
491 batch_size = throttler_next_batch(throttler, packets_sent);
492
493
494 /* Transmit packets from the receive thread */
495 flush_packets( adapter,
496 parms->packet_buffers,
497 parms->transmit_queue,
498 &packets_sent,
499 &batch_size);
500
501 /* Make sure they've actually been transmitted, not just queued up for
502 * transmit */
503 rawsock_flush(adapter);
504
505 pixie_usleep(100);
506 }
507 }
508
509 /* Thread is about to exit */
510 parms->done_transmitting = 1;
511 LOG(1, "THREAD: xmit: stopping thread #%u\n", parms->nic_index);
512}

905 tcpcon_handle(tcpcon, tcb, TCP_WHAT_RST,

907 }
908 } else if (TCP_IS_FIN(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x1) == 0x1)) {

910 * NO TCB!

912 * but the other side didn't get the packet.

915 tcpcon_send_FIN(
916 tcpcon,
917 ip_me, ip_them,

922 }

933 }

937 LOG(5, "%u.%u.%u.%u - bad cookie: ackno=0x%08x expected=0x%08x\n",
938 (ip_them>>24)&0xff, (ip_them>>16)&0xff,
939 (ip_them>>8)&0xff, (ip_them>>0)&0xff,
940 seqno_me-1, cookie);
941 continue;
942 }
943
944 /* verify: ignore duplicates */
945 if (dedup_is_duplicate(dedup, ip_them, port_them, ip_me, port_me))
946 continue;
947
948 /* keep statistics on number received */
949 if (TCP_IS_SYNACK(px, parsed.transport_offset)(((px[(parsed.transport_offset)+13]) & 0x12) == 0x12))
950 (*status_synack_count)++;
951
952 /*
953 * This is where we do the output
954 */
955 output_report_status(
956 out,
957 global_now,
958 status,
959 ip_them,
960 6, /* ip proto = tcp */
961 port_them,
962 px[parsed.transport_offset + 13], /* tcp flags */
963 parsed.ip_ttl,
964 parsed.mac_src
965 );
966
967 /*
968 * Send RST so other side isn't left hanging (only doing this in
969 * complete stateless mode where we aren't tracking banners)
970 */
971 if (tcpcon == NULL((void*)0) && !masscan->is_noreset)
972 tcp_send_RST(
973 &parms->tmplset->pkts[Proto_TCP],
974 parms->packet_buffers,
975 parms->transmit_queue,
976 ip_them, ip_me,
977 port_them, port_me,
978 0, seqno_me);
979
980 }
981 }
982
983
984 LOG(1, "THREAD: recv: stopping thread #%u\n", parms->nic_index);
985
986 /*
987 * cleanup
988 */
989end:
990 if (tcpcon)
991 tcpcon_destroy_table(tcpcon);
992 dedup_destroy(dedup);
993 output_destroy(out);
994 if (pcapfile)
995 pcapfile_close(pcapfile);

998 void *p;
999 int err;
1000 err = rte_ring_sc_dequeue(parms->packet_buffers, (void**)&p);
1001 if (err == 0)
1002 free(p);
1003 else
1004 break;
1005 }

1007 /* Thread is about to exit */
1008 parms->done_receiving = 1;

1311 rate += parms->throttler->current_rate;
1312
1313 if (parms->total_tcbs)
1314 total_tcbs += *parms->total_tcbs;
1315 if (parms->total_synacks)
1316 total_synacks += *parms->total_synacks;
1317 if (parms->total_syns)
1318 total_syns += *parms->total_syns;
1319 }
1320
1321 if (min_index >= range && !masscan->is_infinite) {
1322 /* Note: This is how we can tell the scan has ended */
1323 is_tx_done = 1;
1324 }
1325
1326 /*
1327 * update screen about once per second with statistics,
1328 * namely packets/second.
1329 */
1330 if (masscan->output.is_status_updates)
1331 status_print(&status, min_index, range, rate,
1332 total_tcbs, total_synacks, total_syns,
1333 0);
1334
1335 /* Sleep for almost a second */
1336 pixie_mssleep(750);
1337 }
1338
1339 /*
1340 * If we haven't completed the scan, then save the resume
1341 * information.
1342 */
1343 if (min_index < count_ips * count_ports) {
1344 masscan->resume.index = min_index;
1345
1346 /* Write current settings to "paused.conf" so that the scan can be restarted */
1347 masscan_save_state(masscan);
1348 }
1349
1350
1351
1352 /*
1353 * Now wait for all threads to exit
1354 */
1355 now = time(0);
1356 for (;;) {
1357 unsigned transmit_count = 0;
1358 unsigned receive_count = 0;
1359 unsigned i;
1360 double rate = 0;
1361 uint64_t total_tcbs = 0;
1362 uint64_t total_synacks = 0;
1363 uint64_t total_syns = 0;
1364
1365
1366 /* Find the minimum index of all the threads */
1367 min_index = UINT64_MAX(18446744073709551615UL);
1368 for (i=0; i<masscan->nic_count; i++) {
1369 struct ThreadPair *parms = &parms_array[i];
1370
1371 if (min_index > parms->my_index)
1372 min_index = parms->my_index;
1373
1374 rate += parms->throttler->current_rate;
1375
1376 if (parms->total_tcbs)
1377 total_tcbs += *parms->total_tcbs;
1378 if (parms->total_synacks)
1379 total_synacks += *parms->total_synacks;
1380 if (parms->total_syns)
1381 total_syns += *parms->total_syns;
1382 }
1383
1384
1385
1386 if (time(0) - now >= masscan->wait)
1387 is_rx_done = 1;
1388
1389 if (masscan->output.is_status_updates) {
1390 status_print(&status, min_index, range, rate,
1391 total_tcbs, total_synacks, total_syns,
1392 masscan->wait - (time(0) - now));
1393
1394 for (i=0; i<masscan->nic_count; i++) {
1395 struct ThreadPair *parms = &parms_array[i];
1396
1397 transmit_count += parms->done_transmitting;
1398 receive_count += parms->done_receiving;
1399
1400 }
1401
1402 pixie_mssleep(250);
1403
1404 if (transmit_count < masscan->nic_count)
1405 continue;
1406 is_tx_done = 1;
1407 is_rx_done = 1;
1408 if (receive_count < masscan->nic_count)
1409 continue;
1410
1411 } else {
1412 /* [AFL-fuzz]
1413 * Join the threads, which doesn't allow us to print out
1414 * status messages, but allows us to exit cleaningly without
1415 * any waiting */
1416 for (i=0; i<masscan->nic_count; i++) {
1417 struct ThreadPair *parms = &parms_array[i];
1418
1419 pixie_thread_join(parms->thread_handle_xmit);
1420 parms->thread_handle_xmit = 0;
1421 pixie_thread_join(parms->thread_handle_recv);
1422 parms->thread_handle_recv = 0;
1423 }
1424 is_tx_done = 1;
1425 is_rx_done = 1;
1426 }
1427
1428 break;
1429 }
1430
1431 LOG(1, "THREAD: status: stopping thread\n");
1432
1433 /*
1434 * Now cleanup everything
1435 */
1436 status_finish(&status);
1437 rangelist_pick2_destroy(picker);
1438
1439 if (!masscan->output.is_status_updates) {
1440 uint64_t usec_now = pixie_gettime();
1441
1442 printf("%u milliseconds ellapsed\n", (unsigned)((usec_now - usec_start)/1000));
1443 }
1444 return 0;
1445}
