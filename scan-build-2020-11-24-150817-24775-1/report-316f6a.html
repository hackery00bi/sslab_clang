285{
286 if (adapter == 0)
287 return 0;

292 }
293

296 int err = PF_RING_ERROR_NO_TX_SLOT_AVAILABLE-4;
297

299 err = PFRING.send(adapter->ring, packet, length, (unsigned char)flush);
300 }
301 if (err < 0)

303 return err;
304 }

307 if (adapter->sendq) {

309 struct pcap_pkthdr hdr;
310 hdr.len = length;

312

314 if (err) {

709 adapter_name, err);
710 } else
711 LOG(2, "pfring:'%s': direction success\n", adapter_name);
712
713 /*
714 * Activate
715 *
716 * PF_RING requires a separate activation step.
717 */
718 LOG(2, "pfring:'%s': activating\n", adapter_name);
719 err = PFRING.enable_ring(adapter->ring);
720 if (err != 0) {
721 LOG(0, "pfring: '%s': ENABLE ERROR: %s\n",
722 adapter_name, strerror_x(errno(*__errno_location ())));
723 PFRING.close(adapter->ring);
724 adapter->ring = 0;
725 return 0;
726 } else
727 LOG(1, "pfring:'%s': successfully enabled\n", adapter_name);
728
729 return adapter;
730 }
731
732
733 /*----------------------------------------------------------------
734 * PORTABILITY: LIBPCAP
735 *
736 * This is the stanard that should work everywhere.
737 *----------------------------------------------------------------*/
738 {
739 LOG(1, "pcap: %s\n", PCAP.lib_version());
740 LOG(2, "pcap:'%s': opening...\n", adapter_name);
741
742 if (memcmp(adapter_name, "file:", 5) == 0) {
743 LOG(1, "pcap: file: %s\n", adapter_name+5);
744 is_pcap_file = 1;
745
746 adapter->pcap = PCAP.open_offline(
747 adapter_name+5, /* interface name */
748 errbuf);
749 } else {
750 adapter->pcap = PCAP.open_live(
751 adapter_name, /* interface name */
752 65536, /* max packet size */
753 8, /* promiscuous mode */
754 1000, /* read timeout in milliseconds */
755 errbuf);
756 }
757
758 if (adapter->pcap == NULL((void*)0)) {
759 LOG(0, "FAIL: %s\n", errbuf);
760 if (strstr(errbuf, "perm")) {
761 LOG(0, " [hint] need to sudo or run as root or something\n");
762 LOG(0, " [hint] I've got some local priv escalation "
763 "0days that might work\n");
764 }
765#if defined(__APPLE__)
766 if (strcmp(adapter_name, "vmnet1") == 0) {
767 LOG(0, " [hint] VMware on Macintosh doesn't support masscan\n");
768 }
769#endif
770
771 return 0;
772 } else
773 LOG(1, "pcap:'%s': successfully opened\n", adapter_name);
774
775
776 /* Figure out the link-type. We suport Ethernet and IP */
777 {
778 int dl = PCAP.datalink(adapter->pcap);
779 switch (dl) {
780 case 1: /* Ethernet */
781 adapter->link_type = dl;
782 break;
783 case 12: /* IP Raw */
784 adapter->link_type = dl;
785 break;
786 default:
787 LOG(0, "unknown data link type: %u(%s)\n",
788 dl, PCAP.datalink_val_to_name(dl));
789 break;
790
791 }
792 }
793
794#if 0
795 /* Set any BPF filters the user might've set */
796 if (bpf_filter) {
797 int err;
798 struct bpf_program prog;
799
800 err = pcap_compile(
801 adapter->pcap,
802 &prog, /* object code, output of compile */
803 bpf_filter, /* source code */
804 1, /* optimize to go fast */
805 0);
806
807 if (err) {
808 pcap_perror(adapter->pcap, "pcap_compile()");
809 exit(1);
810 }
811
812 err = pcap_setfilter(adapter->pcap, &prog);
813 if (err < 0) {
814 pcap_perror(adapter->pcap, "pcap_setfilter");
815 exit(1);
816 }
817 }
818#endif
819
820
821 }
822
823 /*----------------------------------------------------------------
824 * PORTABILITY: WINDOWS
825 *
826 * The transmit rate on Windows is really slow, like 40-kpps.
827 * The speed can be increased by using the "sendqueue" feature
828 * to roughly 300-kpps.
829 *----------------------------------------------------------------*/
830 adapter->sendq = 0;
831#if defined(WIN32)
832 if (is_sendq)
833 adapter->sendq = PCAP.sendqueue_alloc(SENDQ_SIZE65536 * 8);
834#endif
835
836
837 return adapter;
838}

842/***************************************************************************
843 * for testing when two Windows adapters have the same name. Sometimes
844 * the \Device\NPF_ string is prepended, sometimes not.
845 ***************************************************************************/
846int
847rawsock_is_adapter_names_equal(const char *lhs, const char *rhs)
848{
849 if (memcmp(lhs, "\\Device\\NPF_", 12) == 0)
850 lhs += 12;
851 if (memcmp(rhs, "\\Device\\NPF_", 12) == 0)
852 rhs += 12;
853 return strcmp(lhs, rhs) == 0;

857/***************************************************************************
858 * Runs some tests when the "--debug if" option is given on the
859 * command-line. This is useful to figure out why the interface you
860 * are accessing doesn't work.
861 ***************************************************************************/
862int
863rawsock_selftest_if(const char *ifname)
864{
865 int err;
866 unsigned ipv4 = 0;
867 unsigned router_ipv4 = 0;
868 unsigned char mac[6] = {0,0,0,0,0,0};
869 struct Adapter *adapter;
870 char ifname2[246];
871
872 if (ifname == NULL((void*)0) || ifname[0] == 0) {
873 err = rawsock_get_default_interface(ifname2, sizeof(ifname2));
874 if (err) {
875 fprintf(stderrstderr, "get-default-if: returned err %d\n", err);
876 return -1;
877 }
878 ifname = ifname2;
879 }
880
881 /* Name */
882 printf("if = %s\n", ifname);
883
884 /* IP address */
885 ipv4 = rawsock_get_adapter_ip(ifname);
886 if (ipv4 == 0) {
887 fprintf(stderrstderr, "get-ip: returned err\n");
888 } else {
889 printf("ip = %u.%u.%u.%u\n",
890 (unsigned char)(ipv4>>24),
891 (unsigned char)(ipv4>>16),
892 (unsigned char)(ipv4>>8),
893 (unsigned char)(ipv4>>0));
894 }
895
896 /* MAC address */
897 err = rawsock_get_adapter_mac(ifname, mac);
898 if (err) {
899 fprintf(stderrstderr, "get-adapter-mac: returned err=%d\n", err);
900 } else {
901 printf("mac = %02x-%02x-%02x-%02x-%02x-%02x\n",
902 mac[0], mac[1], mac[2], mac[3], mac[4], mac[5]);
903 }
904
905 /* Gateway IP */
906 err = rawsock_get_default_gateway(ifname, &router_ipv4);
907 if (err) {
908 fprintf(stderrstderr, "get-default-gateway: returned err=%d\n", err);
909 } else {
910 unsigned char router_mac[6];
911
912 printf("gateway = %u.%u.%u.%u\n",
913 (unsigned char)(router_ipv4>>24),
914 (unsigned char)(router_ipv4>>16),
915 (unsigned char)(router_ipv4>>8),
916 (unsigned char)(router_ipv4>>0));
917
918
919 adapter = rawsock_init_adapter(ifname, 0, 0, 0, 0, 0, 0, 0);
920 if (adapter == 0) {
921 printf("adapter[%s]: failed\n", ifname);
922 return -1;
923 } else {
924 printf("pcap = opened\n");
925 }
926
927 memset(router_mac, 0, 6);
928 arp_resolve_sync(
929 adapter,
930 ipv4,
931 mac,
932 router_ipv4,
933 router_mac);
934
935 if (memcmp(router_mac, "\0\0\0\0\0\0", 6) != 0) {
936 printf("gateway = %02x-%02x-%02x-%02x-%02x-%02x\n",
937 router_mac[0],
938 router_mac[1],
939 router_mac[2],
940 router_mac[3],
941 router_mac[4],
942 router_mac[5]
943 );
944 } else {
945 printf("gateway = [failed to ARP address]\n");
946 }
947 rawsock_close_adapter(adapter);
948 }
949
950 return 0;
951}
