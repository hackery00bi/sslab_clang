163static int ends_with_optional_cr(const char *l, long s, long i)
164{
165 int complete = s && l[s-1] == '\n';
166
167 if (complete)
168 s--;
169 if (s == i)

171 // do not ignore CR at the end of an incomplete line

177int xdl_recmatch(const char *l1, long s1, const char *l2, long s2, long flags)
178{
179 int i1, i2;
180
181 if (s1 == s2 && !memcmp(l1, l2, s1))
182 return 1;
183 if (!(flags & XDF_WHITESPACE_FLAGS((1 << 1) | (1 << 2) | (1 << 3) | (1 <<
4))))
184 return 0;
185
186 i1 = 0;
187 i2 = 0;
188
189 /*
190 * -w matches everything that matches with -b, and -b in turn
191 * matches everything that matches with --ignore-space-at-eol,
192 * which in turn matches everything that matches with --ignore-cr-at-eol.
193 *
194 * Each flavor of ignoring needs different logic to skip whitespaces
195 * while we have both sides to compare.
196 */
197 if (flags & XDF_IGNORE_WHITESPACE(1 << 1)) {
198 goto skip_ws;
199 while (i1 < s1 && i2 < s2) {
200 if (l1[i1++] != l2[i2++])
201 return 0;
202 skip_ws:
203 while (i1 < s1 && XDL_ISSPACE(l1[i1])(((*__ctype_b_loc ())[(int) (((unsigned char)(l1[i1])))] &

204 i1++;
205 while (i2 < s2 && XDL_ISSPACE(l2[i2])(((*__ctype_b_loc ())[(int) (((unsigned char)(l2[i2])))] &
(unsigned short int) _ISspace)))
206 i2++;
207 }
208 } else if (flags & XDF_IGNORE_WHITESPACE_CHANGE(1 << 2)) {
209 while (i1 < s1 && i2 < s2) {
210 if (XDL_ISSPACE(l1[i1])(((*__ctype_b_loc ())[(int) (((unsigned char)(l1[i1])))] &
(unsigned short int) _ISspace)) && XDL_ISSPACE(l2[i2])(((*__ctype_b_loc ())[(int) (((unsigned char)(l2[i2])))] &
(unsigned short int) _ISspace))) {
211 // Skip matching spaces and try again
212 while (i1 < s1 && XDL_ISSPACE(l1[i1])(((*__ctype_b_loc ())[(int) (((unsigned char)(l1[i1])))] &
(unsigned short int) _ISspace)))
213 i1++;

215 i2++;
216 continue;
217 }
218 if (l1[i1++] != l2[i2++])
219 return 0;
220 }
221 } else if (flags & XDF_IGNORE_WHITESPACE_AT_EOL(1 << 3)) {
222 while (i1 < s1 && i2 < s2 && l1[i1] == l2[i2]) {
223 i1++;
224 i2++;
225 }
226 } else if (flags & XDF_IGNORE_CR_AT_EOL(1 << 4)) {
227 // Find the first difference and see how the line ends
228 while (i1 < s1 && i2 < s2 && l1[i1] == l2[i2]) {

239 * ignore-whitespace-at-eol case may break out of the loop
240 * while there still are characters remaining on both lines.
241 */

252 }
