709/*
710 * Set the index of the currently selected item. The menu will scroll when
711 * necessary. When "n" is out of range don't scroll.
712 * This may be repeated when the preview window is used:
713 * "repeat" == 0: open preview window normally
714 * "repeat" == 1: open preview window but don't set the size
715 * "repeat" == 2: don't open preview window
716 * Returns TRUE when the window was resized and the location of the popup menu
717 * must be recomputed.
718 */
719 static int
720pum_set_selected(int n, int repeat UNUSED__attribute__((unused)))
721{
722 int resized = FALSE0;
723 int context = pum_height / 2;
724#ifdef FEAT_QUICKFIX
725 int prev_selected = pum_selected;
726#endif
727#if defined(FEAT_PROP_POPUP) && defined(FEAT_QUICKFIX)
728 int has_info = FALSE0;
729#endif
730
731 pum_selected = n;
732
733 if (pum_selected >= 0 && pum_selected < pum_size)
734 {
735 if (pum_first > pum_selected - 4)
736 {
737 // scroll down; when we did a jump it's probably a PageUp then
738 // scroll a whole page
739 if (pum_first > pum_selected - 2)
740 {
741 pum_first -= pum_height - 2;
742 if (pum_first < 0)
743 pum_first = 0;
744 else if (pum_first > pum_selected)
745 pum_first = pum_selected;
746 }
747 else
748 pum_first = pum_selected;
749 }
750 else if (pum_first < pum_selected - pum_height + 5)
751 {
752 // scroll up; when we did a jump it's probably a PageDown then
753 // scroll a whole page
754 if (pum_first < pum_selected - pum_height + 1 + 2)
755 {
756 pum_first += pum_height - 2;
757 if (pum_first < pum_selected - pum_height + 1)
758 pum_first = pum_selected - pum_height + 1;
759 }
760 else
761 pum_first = pum_selected - pum_height + 1;
762 }
763
764 // Give a few lines of context when possible.
765 if (context > 3)
766 context = 3;
767 if (pum_height > 2)
768 {
769 if (pum_first > pum_selected - context)
770 {
771 // scroll down
772 pum_first = pum_selected - context;
773 if (pum_first < 0)
774 pum_first = 0;
775 }
776 else if (pum_first < pum_selected + context - pum_height + 1)
777 {
778 // scroll up
779 pum_first = pum_selected + context - pum_height + 1;
780 }
781 }
782 // adjust for the number of lines displayed
783 if (pum_first > pum_size - pum_height)
784 pum_first = pum_size - pum_height;
785
786#if defined(FEAT_QUICKFIX)
787 /*
788 * Show extra info in the preview window if there is something and
789 * 'completeopt' contains "preview" or "popup" or "popuphidden".
790 * Skip this when tried twice already.
791 * Skip this also when there is not much room.
792 * NOTE: Be very careful not to sync undo!
793 */
794 if (pum_array[pum_selected].pum_info != NULL((void*)0)
795 && Rows > 10
796 && repeat <= 1
797 && vim_strchr(p_cot, 'p') != NULL((void*)0))
798 {
799 win_T *curwin_save = curwin;
800 tabpage_T *curtab_save = curtab;
801 int res = OK1;
802# ifdef FEAT_PROP_POPUP
803 use_popup_T use_popup;
804# else
805# define use_popup USEPOPUP_NONE
806# endif
807# ifdef FEAT_PROP_POPUP
808 has_info = TRUE1;
809 if (strstr((char *)p_cot, "popuphidden") != NULL((void*)0))
810 use_popup = USEPOPUP_HIDDEN;
811 else if (strstr((char *)p_cot, "popup") != NULL((void*)0))
812 use_popup = USEPOPUP_NORMAL;
813 else
814 use_popup = USEPOPUP_NONE;
815 if (use_popup != USEPOPUP_NONE)
816 // don't use WinEnter or WinLeave autocommands for the info
817 // popup
818 block_autocmds();
819# endif
820 // Open a preview window and set "curwin" to it.
821 // 3 lines by default, prefer 'previewheight' if set and smaller.
822 g_do_tagpreview = 3;
823 if (p_pvh > 0 && p_pvh < g_do_tagpreview)
824 g_do_tagpreview = p_pvh;
825 ++RedrawingDisabled;
826 // Prevent undo sync here, if an autocommand syncs undo weird
827 // things can happen to the undo tree.
828 ++no_u_sync;
829 resized = prepare_tagpreview(FALSE0, FALSE0, use_popup);
830 --no_u_sync;
831 --RedrawingDisabled;
832 g_do_tagpreview = 0;
833
834 if (curwin->w_p_pvww_onebuf_opt.wo_pvw
835# ifdef FEAT_PROP_POPUP
836 || (curwin->w_popup_flags & POPF_INFO0x80)
837# endif
838 )
839 {
840 if (!resized
841 && curbuf->b_nwindows == 1
842 && curbuf->b_fname == NULL((void*)0)
843 && bt_nofile(curbuf)
844 && curbuf->b_p_bh[0] == 'w')
845 {
846 // Already a "wipeout" buffer, make it empty.
847 while (!BUFEMPTY()(curbuf->b_ml.ml_line_count == 1 && *ml_get((linenr_T
)1) == '\000'))
848 ml_delete((linenr_T)1);
849 }
850 else
851 {
852 // Don't want to sync undo in the current buffer.
853 ++no_u_sync;
854 res = do_ecmd(0, NULL((void*)0), NULL((void*)0), NULL((void*)0), ECMD_ONE(linenr_T)1, 0, NULL((void*)0));
855 --no_u_sync;
856 if (res == OK1)
857 {
858 // Edit a new, empty buffer. Set options for a "wipeout"
859 // buffer.
860 set_option_value((char_u *)"swf", 0L, NULL((void*)0), OPT_LOCAL0x04);
861 set_option_value((char_u *)"bt", 0L,
862 (char_u *)"nofile", OPT_LOCAL0x04);
863 set_option_value((char_u *)"bh", 0L,
864 (char_u *)"wipe", OPT_LOCAL0x04);
865 set_option_value((char_u *)"diff", 0L,
866 NULL((void*)0), OPT_LOCAL0x04);
867 }
868 }
869 if (res == OK1)
870 {
871 char_u *p, *e;
872 linenr_T lnum = 0;
873
874 for (p = pum_array[pum_selected].pum_info; *p != NUL'\000'; )
875 {
876 e = vim_strchr(p, '\n');
877 if (e == NULL((void*)0))
878 {
879 ml_append(lnum++, p, 0, FALSE0);
880 break;
881 }
882 else
883 {
884 *e = NUL'\000';
885 ml_append(lnum++, p, (int)(e - p + 1), FALSE0);
886 *e = '\n';
887 p = e + 1;
888 }
889 }
890 // delete the empty last line
891 ml_delete(curbuf->b_ml.ml_line_count);
892
893 // Increase the height of the preview window to show the
894 // text, but no more than 'previewheight' lines.
895 if (repeat == 0 && use_popup == USEPOPUP_NONE)
896 {
897 if (lnum > p_pvh)
898 lnum = p_pvh;
899 if (curwin->w_height < lnum)
900 {
901 win_setheight((int)lnum);
902 resized = TRUE1;
903 }
904 }
905
906 curbuf->b_changed = 0;
907 curbuf->b_p_ma = FALSE0;
908 if (pum_selected != prev_selected)
909 {
910# ifdef FEAT_PROP_POPUP
911 curwin->w_firstline = 1;
912# endif
913 curwin->w_topline = 1;
914 }
915 else if (curwin->w_topline > curbuf->b_ml.ml_line_count)
916 curwin->w_topline = curbuf->b_ml.ml_line_count;
917 curwin->w_cursor.lnum = curwin->w_topline;
918 curwin->w_cursor.col = 0;
919# ifdef FEAT_PROP_POPUP
920 if (use_popup != USEPOPUP_NONE)
921 {
922 pum_position_info_popup(curwin);
923 if (win_valid(curwin_save))
924 redraw_win_later(curwin_save, SOME_VALID35);
925 }
926# endif
927 if ((curwin != curwin_save && win_valid(curwin_save))
928 || (curtab != curtab_save
929 && valid_tabpage(curtab_save)))
930 {
931 if (curtab != curtab_save && valid_tabpage(curtab_save))
932 goto_tabpage_tp(curtab_save, FALSE0, FALSE0);
933
934 // When the first completion is done and the preview
935 // window is not resized, skip the preview window's
936 // status line redrawing.
937 if (ins_compl_active() && !resized)
938 curwin->w_redr_status = FALSE0;
939
940 // Return cursor to where we were
941 validate_cursor();
942 redraw_later(SOME_VALID35);
943
944 // When the preview window was resized we need to
945 // update the view on the buffer. Only go back to
946 // the window when needed, otherwise it will always be
947 // redrawn.
948 if (resized && win_valid(curwin_save))
949 {
950 ++no_u_sync;
951 win_enter(curwin_save, TRUE1);
952 --no_u_sync;
953 update_topline();
954 }
955
956 // Update the screen before drawing the popup menu.
957 // Enable updating the status lines.
958 pum_pretend_not_visible = TRUE1;
959 // But don't draw text at the new popup menu position,
960 // it causes flicker. When resizing we need to draw
961 // anyway, the position may change later.
962 pum_will_redraw = !resized;
963 update_screen(0);
964 pum_pretend_not_visible = FALSE0;
965 pum_will_redraw = FALSE0;
966
967 if (!resized && win_valid(curwin_save))
968 {
969# ifdef FEAT_PROP_POPUP
970 win_T *wp = curwin;
971# endif
972 ++no_u_sync;
973 win_enter(curwin_save, TRUE1);
974 --no_u_sync;
975# ifdef FEAT_PROP_POPUP
976 if (use_popup == USEPOPUP_HIDDEN && win_valid(wp))
977 popup_hide(wp);
978# endif
979 }
980
981 // May need to update the screen again when there are
982 // autocommands involved.
983 pum_pretend_not_visible = TRUE1;
984 pum_will_redraw = !resized;
985 update_screen(0);
986 pum_pretend_not_visible = FALSE0;
987 pum_will_redraw = FALSE0;
988 call_update_screen = FALSE0;
989 }
990 }
991 }
992# if defined(FEAT_PROP_POPUP) && defined(FEAT_QUICKFIX)
993 if (WIN_IS_POPUP(curwin)((curwin)->w_popup_flags != 0))
994 // can't keep focus in a popup window
995 win_enter(firstwin, TRUE1);
996# endif
997# ifdef FEAT_PROP_POPUP
998 if (use_popup != USEPOPUP_NONE)
999 unblock_autocmds();
1000# endif
1001 }
1002#endif
1003 }
1004#if defined(FEAT_PROP_POPUP) && defined(FEAT_QUICKFIX)
1005 if (!has_info)
1006 // hide any popup info window
1007 popup_hide_info();
1008#endif
1009
1010 return resized;
1011}

1156#if defined(FEAT_BEVAL_TERM) || defined(PROTO)
1157static pumitem_T *balloon_array = NULL((void*)0);
1158static int balloon_arraysize;
1159
1160# define BALLOON_MIN_WIDTH50 50
1161# define BALLOON_MIN_HEIGHT10 10
1162
1163typedef struct {
1164 char_u *start;
1165 int bytelen;
1166 int cells;
1167 int indent;
1168} balpart_T;
1169
1170/*
1171 * Split a string into parts to display in the balloon.
1172 * Aimed at output from gdb. Attempts to split at white space, preserve quoted
1173 * strings and make a struct look good.
1174 * Resulting array is stored in "array" and returns the size of the array.
1175 */
1176 int
1177split_message(char_u *mesg, pumitem_T **array)
1178{
1179 garray_T ga;
1180 char_u *p;
1181 balpart_T *item;
1182 int quoted = FALSE0;
1183 int height;
1184 int line;
1185 int item_idx;
1186 int indent = 0;
1187 int max_cells = 0;
1188 int max_height = Rows / 2 - 1;
1189 int long_item_count = 0;
1190 int split_long_items = FALSE0;
1191
1192 ga_init2(&ga, sizeof(balpart_T), 20);
1193 p = mesg;
1194
1195 while (*p != NUL'\000')
1196 {
1197 if (ga_grow(&ga, 1) == FAIL0)
1198 goto failed;
1199 item = ((balpart_T *)ga.ga_data) + ga.ga_len;
1200 item->start = p;
1201 item->indent = indent;
1202 item->cells = indent * 2;
1203 ++ga.ga_len;
1204 while (*p != NUL'\000')
1205 {
1206 if (*p == '"')
1207 quoted = !quoted;
1208 else if (*p == '\n')
1209 break;
1210 else if (*p == '\\' && p[1] != NUL'\000')
1211 ++p;
1212 else if (!quoted)
1213 {
1214 if ((*p == ',' && p[1] == ' ') || *p == '{' || *p == '}')
1215 {
1216 // Looks like a good point to break.
1217 if (*p == '{')
1218 ++indent;
1219 else if (*p == '}' && indent > 0)
1220 --indent;
1221 ++item->cells;
1222 p = skipwhite(p + 1);
1223 break;
1224 }
1225 }
1226 item->cells += ptr2cells(p);
1227 p += mb_ptr2len(p);
1228 }
1229 item->bytelen = p - item->start;
1230 if (*p == '\n')
1231 ++p;
1232 if (item->cells > max_cells)
1233 max_cells = item->cells;
1234 long_item_count += (item->cells - 1) / BALLOON_MIN_WIDTH50;
1235 }
1236
1237 height = 2 + ga.ga_len;
1238
1239 // If there are long items and the height is below the limit: split lines
1240 if (long_item_count > 0 && height + long_item_count <= max_height)
1241 {
1242 split_long_items = TRUE1;
1243 height += long_item_count;
1244 }
1245
1246 // Limit to half the window height, it has to fit above or below the mouse
1247 // position.
1248 if (height > max_height)
1249 height = max_height;
1250 *array = ALLOC_CLEAR_MULT(pumitem_T, height)(pumitem_T *)alloc_clear(sizeof(pumitem_T) * (height));
1251 if (*array == NULL((void*)0))
1252 goto failed;
1253
1254 // Add an empty line above and below, looks better.
1255 (*array)->pum_text = vim_strsave((char_u *)"");
1256 (*array + height - 1)->pum_text = vim_strsave((char_u *)"");
1257
1258 for (line = 1, item_idx = 0; line < height - 1; ++item_idx)
1259 {
1260 int skip;
1261 int thislen;
1262 int copylen;
1263 int ind;
1264 int cells;
1265
1266 item = ((balpart_T *)ga.ga_data) + item_idx;
1267 if (item->bytelen == 0)
1268 (*array)[line++].pum_text = vim_strsave((char_u *)"");
1269 else
1270 for (skip = 0; skip < item->bytelen; skip += thislen)
1271 {
1272 if (split_long_items && item->cells >= BALLOON_MIN_WIDTH50)
1273 {
1274 cells = item->indent * 2;
1275 for (p = item->start + skip;
1276 p < item->start + item->bytelen;
1277 p += mb_ptr2len(p))
1278 if ((cells += ptr2cells(p)) > BALLOON_MIN_WIDTH50)
1279 break;
1280 thislen = p - (item->start + skip);
1281 }
1282 else
1283 thislen = item->bytelen;
1284
1285 // put indent at the start
1286 p = alloc(thislen + item->indent * 2 + 1);
1287 if (p == NULL((void*)0))
1288 {
1289 for (line = 0; line <= height - 1; ++line)
1290 vim_free((*array)[line].pum_text);
1291 vim_free(*array);
1292 goto failed;
1293 }
1294 for (ind = 0; ind < item->indent * 2; ++ind)
1295 p[ind] = ' ';
1296
1297 // exclude spaces at the end of the string
1298 for (copylen = thislen; copylen > 0; --copylen)
1299 if (item->start[skip + copylen - 1] != ' ')
1300 break;
1301
1302 vim_strncpy(p + ind, item->start + skip, copylen);
1303 (*array)[line].pum_text = p;
1304 item->indent = 0; // wrapped line has no indent
1305 ++line;
1306 }
1307 }
1308 ga_clear(&ga);
1309 return height;
1310
1311failed:
1312 ga_clear(&ga);
1313 return 0;
1314}

1423/*
1424 * Open the terminal version of the popup menu and don't return until it is
1425 * closed.
1426 */
1427 void
1428pum_show_popupmenu(vimmenu_T *menu)
1429{
1430 vimmenu_T *mp;
1431 int idx = 0;
1432 pumitem_T *array;
1433# ifdef FEAT_BEVAL_TERM
1434 int save_bevalterm = p_bevalterm;
1435# endif
1436 int mode;
1437
1438 pum_undisplay();
1439 pum_size = 0;
1440 mode = get_menu_mode_flag();
1441
1442 FOR_ALL_CHILD_MENUS(menu, mp)for ((mp) = (menu)->children; (mp) != ((void*)0); (mp) = (
mp)->next)
1443 if (menu_is_separator(mp->dname)
1444 || (mp->modes & mp->enabled & mode))
1445 ++pum_size;
1446
1447 // When there are only Terminal mode menus, using "popup Edit" results in
1448 // pum_size being zero.
1449 if (pum_size <= 0)
1450 {
1451 emsg(e_menuothermode);
1452 return;
1453 }
1454
1455 array = ALLOC_CLEAR_MULT(pumitem_T, pum_size)(pumitem_T *)alloc_clear(sizeof(pumitem_T) * (pum_size));
1456 if (array == NULL((void*)0))
1457 return;
1458
1459 FOR_ALL_CHILD_MENUS(menu, mp)for ((mp) = (menu)->children; (mp) != ((void*)0); (mp) = (
mp)->next)
1460 if (menu_is_separator(mp->dname))
1461 array[idx++].pum_text = (char_u *)"";
1462 else if (mp->modes & mp->enabled & mode)
1463 array[idx++].pum_text = mp->dname;
1464
1465 pum_array = array;
1466 pum_compute_size();
1467 pum_scrollbar = 0;
1468 pum_height = pum_size;
1469 pum_position_at_mouse(20);
1470
1471 pum_selected = -1;
1472 pum_first = 0;
1473# ifdef FEAT_BEVAL_TERM
1474 p_bevalterm = TRUE1; // track mouse movement
1475 mch_setmouse(TRUE1);
1476# endif
1477
1478 for (;;)
1479 {
1480 int c;
1481
1482 pum_redraw();
1483 setcursor_mayforce(TRUE1);
1484 out_flush();
1485
1486 c = vgetc();
1487
1488 // Bail out when typing Esc, CTRL-C or some callback closed the popup
1489 // menu.
1490 if (c == ESC'\033' || c == Ctrl_C3 || pum_array == NULL((void*)0))
1491 break;
1492 else if (c == CAR'\015' || c == NL'\012')
1493 {
1494 // enter: select current item, if any, and close
1495 pum_execute_menu(menu, mode);
1496 break;
1497 }
1498 else if (c == 'k' || c == K_UP(-(('k') + ((int)('u') << 8))) || c == K_MOUSEUP(-((253) + ((int)(KE_MOUSEUP) << 8))))
1499 {
1500 // cursor up: select previous item
1501 while (pum_selected > 0)
1502 {
1503 --pum_selected;
1504 if (*array[pum_selected].pum_text != NUL'\000')
1505 break;
1506 }
1507 }
1508 else if (c == 'j' || c == K_DOWN(-(('k') + ((int)('d') << 8))) || c == K_MOUSEDOWN(-((253) + ((int)(KE_MOUSEDOWN) << 8))))
1509 {
1510 // cursor down: select next item
1511 while (pum_selected < pum_size - 1)
1512 {
1513 ++pum_selected;
1514 if (*array[pum_selected].pum_text != NUL'\000')
1515 break;
1516 }
1517 }
1518 else if (c == K_RIGHTMOUSE(-((253) + ((int)(KE_RIGHTMOUSE) << 8))))
1519 {
1520 // Right mouse down: reposition the menu.
1521 vungetc(c);
1522 break;
1523 }
1524 else if (c == K_LEFTDRAG(-((253) + ((int)(KE_LEFTDRAG) << 8))) || c == K_RIGHTDRAG(-((253) + ((int)(KE_RIGHTDRAG) << 8))) || c == K_MOUSEMOVE(-((253) + ((int)(KE_MOUSEMOVE) << 8))))
1525 {
1526 // mouse moved: select item in the mouse row
1527 pum_select_mouse_pos();
1528 }
1529 else if (c == K_LEFTMOUSE(-((253) + ((int)(KE_LEFTMOUSE) << 8))) || c == K_LEFTMOUSE_NM(-((253) + ((int)(KE_LEFTMOUSE_NM) << 8))) || c == K_RIGHTRELEASE(-((253) + ((int)(KE_RIGHTRELEASE) << 8))))
1530 {
1531 // left mouse click: select clicked item, if any, and close;
1532 // right mouse release: select clicked item, close if any
1533 pum_select_mouse_pos();
1534 if (pum_selected >= 0)
1535 {
1536 pum_execute_menu(menu, mode);
1537 break;
1538 }
1539 if (c == K_LEFTMOUSE(-((253) + ((int)(KE_LEFTMOUSE) << 8))) || c == K_LEFTMOUSE_NM(-((253) + ((int)(KE_LEFTMOUSE_NM) << 8))))
1540 break;
1541 }
1542 }
1543
1544 vim_free(array);
1545 pum_undisplay();
1546# ifdef FEAT_BEVAL_TERM
1547 p_bevalterm = save_bevalterm;
1548 mch_setmouse(TRUE1);
1549# endif
1550}
