862/*
863 * Common code for dbg_find_breakpoint() and has_profiling().
864 */
865 static linenr_T
866debuggy_find(
867 int file, // TRUE for a file, FALSE for a function
868 char_u *fname, // file or function name
869 linenr_T after, // after this line number
870 garray_T *gap, // either &dbg_breakp or &prof_ga
871 int *fp) // if not NULL: return forceit
872{
873 struct debuggy *bp;
874 int i;
875 linenr_T lnum = 0;
876 char_u *name = fname;
877 int prev_got_int;
878
879 // Return quickly when there are no breakpoints.
880 if (gap->ga_len == 0)
881 return (linenr_T)0;
882
883 // Replace K_SNR in function name with "<SNR>".
884 if (!file && fname[0] == K_SPECIAL(0x80))
885 {
886 name = alloc(STRLEN(fname)strlen((char *)(fname)) + 3);
887 if (name == NULL((void*)0))
888 name = fname;
889 else
890 {
891 STRCPY(name, "<SNR>")strcpy((char *)(name), (char *)("<SNR>"));
892 STRCPY(name + 5, fname + 3)strcpy((char *)(name + 5), (char *)(fname + 3));
Call to function 'strcpy' is insecure as it does not provide bounding of the memory buffer. Replace unbounded copy functions with analogous functions that support length arguments such as 'strlcpy'. CWE-119
893 }
894 }
895
896 for (i = 0; i < gap->ga_len; ++i)
897 {
898 // Skip entries that are not useful or are for a line that is beyond
899 // an already found breakpoint.
900 bp = &DEBUGGY(gap, i)(((struct debuggy *)gap->ga_data)[i]);
901 if (((bp->dbg_type == DBG_FILE2) == file &&
902 bp->dbg_type != DBG_EXPR3 && (
903#ifdef FEAT_PROFILE
904 gap == &prof_ga ||
905#endif
906 (bp->dbg_lnum > after && (lnum == 0 || bp->dbg_lnum < lnum)))))
907 {
908 // Save the value of got_int and reset it. We don't want a
909 // previous interruption cancel matching, only hitting CTRL-C
910 // while matching should abort it.
911 prev_got_int = got_int;
912 got_int = FALSE0;
913 if (vim_regexec_prog(&bp->dbg_prog, FALSE0, name, (colnr_T)0))
914 {
915 lnum = bp->dbg_lnum;
916 if (fp != NULL((void*)0))
917 *fp = bp->dbg_forceit;
918 }
919 got_int |= prev_got_int;
920 }
921#ifdef FEAT_EVAL
922 else if (bp->dbg_type == DBG_EXPR3)
923 {
924 typval_T *tv;
925 int line = FALSE0;
926
927 prev_got_int = got_int;
928 got_int = FALSE0;
929

939 }
940 else

943 && tv->vval.v_number == FALSE0)
944 {
945 typval_T *v;
946
947 line = TRUE1;

949 // Need to evaluate again, typval_compare() overwrites
950 // "tv".
951 v = eval_expr(bp->dbg_name, NULL((void*)0));
952 debug_newval = typval_tostring(v);
953 free_tv(bp->dbg_val);
954 bp->dbg_val = v;
955 }
956 free_tv(tv);
957 }
958 }
959 else if (bp->dbg_val != NULL((void*)0))
960 {
961 debug_oldval = typval_tostring(bp->dbg_val);
962 debug_newval = typval_tostring(NULL((void*)0));
963 free_tv(bp->dbg_val);
964 bp->dbg_val = NULL((void*)0);
965 line = TRUE1;
966 }
967
968 if (line)
969 {
970 lnum = after > 0 ? after : 1;
971 break;
972 }
973
974 got_int |= prev_got_int;
975 }
976#endif
977 }
978 if (name != fname)
979 vim_free(name);
980
981 return lnum;
982}
