869/*

871 * - hitting <ESC> twice means: abandon command line.
872 * - wildcard expansion is only done when the 'wildchar' key is really
873 * typed, not when it comes from a macro
874 * Returns CMDLINE_CHANGED if command line is changed or CMDLINE_NOT_CHANGED.

877cmdline_wildchar_complete(

969 if (wim_flags[wim_index] & WIM_LONGEST0x02)
970 nextwild(xp, WILD_LONGEST7, options, escape);

989}
