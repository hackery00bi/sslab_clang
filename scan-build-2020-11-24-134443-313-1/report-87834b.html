1335/*
1336 * Check for a <> code in a user command.
1337 * "code" points to the '<'. "len" the length of the <> (inclusive).
1338 * "buf" is where the result is to be added.
1339 * "split_buf" points to a buffer used for splitting, caller should free it.
1340 * "split_len" is the length of what "split_buf" contains.
1341 * Returns the length of the replacement, which has been added to "buf".
1342 * Returns -1 if there was no match, and only the "<" has been copied.
1343 */
1344 static size_t
1345uc_check_code(
1346 char_u *code,
1347 size_t len,
1348 char_u *buf,
1349 ucmd_T *cmd, // the user command we're expanding
1350 exarg_T *eap, // ex arguments
1351 char_u **split_buf,
1352 size_t *split_len)
1353{
1354 size_t result = 0;
1355 char_u *p = code + 1;
1356 size_t l = len - 2;
1357 int quote = 0;
1358 enum {
1359 ct_ARGS,
1360 ct_BANG,
1361 ct_COUNT,
1362 ct_LINE1,
1363 ct_LINE2,
1364 ct_RANGE,
1365 ct_MODS,
1366 ct_REGISTER,
1367 ct_LT,
1368 ct_NONE
1369 } type = ct_NONE;
1370
1371 if ((vim_strchr((char_u *)"qQfF", *p) != NULL((void*)0)) && p[1] == '-')
1372 {
1373 quote = (*p == 'q' || *p == 'Q') ? 1 : 2;
1374 p += 2;
1375 l -= 2;
1376 }
1377
1378 ++l;
1379 if (l <= 1)
1380 type = ct_NONE;
1381 else if (STRNICMP(p, "args>", l)strncasecmp((char *)(p), (char *)("args>"), (size_t)(l)) == 0)
1382 type = ct_ARGS;
1383 else if (STRNICMP(p, "bang>", l)strncasecmp((char *)(p), (char *)("bang>"), (size_t)(l)) == 0)
1384 type = ct_BANG;
1385 else if (STRNICMP(p, "count>", l)strncasecmp((char *)(p), (char *)("count>"), (size_t)(l)) == 0)
1386 type = ct_COUNT;
1387 else if (STRNICMP(p, "line1>", l)strncasecmp((char *)(p), (char *)("line1>"), (size_t)(l)) == 0)
1388 type = ct_LINE1;
1389 else if (STRNICMP(p, "line2>", l)strncasecmp((char *)(p), (char *)("line2>"), (size_t)(l)) == 0)
1390 type = ct_LINE2;
1391 else if (STRNICMP(p, "range>", l)strncasecmp((char *)(p), (char *)("range>"), (size_t)(l)) == 0)
1392 type = ct_RANGE;
1393 else if (STRNICMP(p, "lt>", l)strncasecmp((char *)(p), (char *)("lt>"), (size_t)(l)) == 0)
1394 type = ct_LT;
1395 else if (STRNICMP(p, "reg>", l)strncasecmp((char *)(p), (char *)("reg>"), (size_t)(l)) == 0 || STRNICMP(p, "register>", l)strncasecmp((char *)(p), (char *)("register>"), (size_t)(l
)) == 0)
1396 type = ct_REGISTER;
1397 else if (STRNICMP(p, "mods>", l)strncasecmp((char *)(p), (char *)("mods>"), (size_t)(l)) == 0)
1398 type = ct_MODS;
1399
1400 switch (type)
1401 {
1402 case ct_ARGS:
1403 // Simple case first
1404 if (*eap->arg == NUL'\000')
1405 {
1406 if (quote == 1)
1407 {
1408 result = 2;
1409 if (buf != NULL((void*)0))
1410 STRCPY(buf, "''")strcpy((char *)(buf), (char *)("''"));
1411 }
1412 else
1413 result = 0;
1414 break;
1415 }
1416
1417 // When specified there is a single argument don't split it.
1418 // Works for ":Cmd %" when % is "a b c".
1419 if ((eap->argt & EX_NOSPC0x010) && quote == 2)
1420 quote = 1;
1421
1422 switch (quote)
1423 {
1424 case 0: // No quoting, no splitting
1425 result = STRLEN(eap->arg)strlen((char *)(eap->arg));
1426 if (buf != NULL((void*)0))
1427 STRCPY(buf, eap->arg)strcpy((char *)(buf), (char *)(eap->arg));
1428 break;
1429 case 1: // Quote, but don't split
1430 result = STRLEN(eap->arg)strlen((char *)(eap->arg)) + 2;
1431 for (p = eap->arg; *p; ++p)
1432 {
1433 if (enc_dbcs != 0 && (*mb_ptr2len)(p) == 2)
1434 // DBCS can contain \ in a trail byte, skip the
1435 // double-byte character.
1436 ++p;
1437 else
1438 if (*p == '\\' || *p == '"')
1439 ++result;
1440 }
1441
1442 if (buf != NULL((void*)0))
1443 {
1444 *buf++ = '"';
1445 for (p = eap->arg; *p; ++p)
1446 {
1447 if (enc_dbcs != 0 && (*mb_ptr2len)(p) == 2)
1448 // DBCS can contain \ in a trail byte, copy the
1449 // double-byte character to avoid escaping.
1450 *buf++ = *p++;
1451 else
1452 if (*p == '\\' || *p == '"')
1453 *buf++ = '\\';
1454 *buf++ = *p;
1455 }
1456 *buf = '"';
1457 }
1458
1459 break;
1460 case 2: // Quote and split (<f-args>)
1461 // This is hard, so only do it once, and cache the result
1462 if (*split_buf == NULL((void*)0))
1463 *split_buf = uc_split_args(eap->arg, split_len);
1464
1465 result = *split_len;
1466 if (buf != NULL((void*)0) && result != 0)
1467 STRCPY(buf, *split_buf)strcpy((char *)(buf), (char *)(*split_buf));
1468
1469 break;
1470 }
1471 break;
1472
1473 case ct_BANG:
1474 result = eap->forceit ? 1 : 0;
1475 if (quote)
1476 result += 2;
1477 if (buf != NULL((void*)0))
1478 {
1479 if (quote)
1480 *buf++ = '"';
1481 if (eap->forceit)
1482 *buf++ = '!';
1483 if (quote)
1484 *buf = '"';
1485 }
1486 break;
1487
1488 case ct_LINE1:
1489 case ct_LINE2:
1490 case ct_RANGE:
1491 case ct_COUNT:
1492 {
1493 char num_buf[20];
1494 long num = (type == ct_LINE1) ? eap->line1 :
1495 (type == ct_LINE2) ? eap->line2 :
1496 (type == ct_RANGE) ? eap->addr_count :
1497 (eap->addr_count > 0) ? eap->line2 : cmd->uc_def;
1498 size_t num_len;
1499
1500 sprintf(num_buf, "%ld", num)__builtin___sprintf_chk (num_buf, 1 - 1, __builtin_object_size
(num_buf, 1 > 1), "%ld", num);
1501 num_len = STRLEN(num_buf)strlen((char *)(num_buf));
1502 result = num_len;
1503
1504 if (quote)
1505 result += 2;
1506
1507 if (buf != NULL((void*)0))
1508 {
1509 if (quote)
1510 *buf++ = '"';
1511 STRCPY(buf, num_buf)strcpy((char *)(buf), (char *)(num_buf));
1512 buf += num_len;
1513 if (quote)
1514 *buf = '"';
1515 }
1516
1517 break;
1518 }
1519
1520 case ct_MODS:
1521 {
1522 result = produce_cmdmods(buf, &cmdmod, quote);
1523 break;
1524 }
1525
1526 case ct_REGISTER:
1527 result = eap->regname ? 1 : 0;
1528 if (quote)
1529 result += 2;
1530 if (buf != NULL((void*)0))
1531 {
1532 if (quote)
1533 *buf++ = '\'';
1534 if (eap->regname)
1535 *buf++ = eap->regname;
1536 if (quote)
1537 *buf = '\'';
1538 }
1539 break;
1540
1541 case ct_LT:
1542 result = 1;
1543 if (buf != NULL((void*)0))
1544 *buf = '<';
1545 break;
1546
1547 default:
1548 // Not recognized: just copy the '<' and return -1.
1549 result = (size_t)-1;
1550 if (buf != NULL((void*)0))
1551 *buf = '<';
1552 break;
1553 }
1554
1555 return result;
1556}
