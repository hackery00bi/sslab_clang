533 raxStart(&si->ri,s->rax);
534 if (!rev) {
535 if (start && (start->ms || start->seq)) {
536 raxSeek(&si->ri,"<=",(unsigned char*)si->start_key,
537 sizeof(si->start_key));
538 if (raxEOF(&si->ri)) raxSeek(&si->ri,"^",NULL((void*)0),0);
539 } else {
540 raxSeek(&si->ri,"^",NULL((void*)0),0);
541 }
542 } else {
543 if (end && (end->ms || end->seq)) {
544 raxSeek(&si->ri,"<=",(unsigned char*)si->end_key,
545 sizeof(si->end_key));
546 if (raxEOF(&si->ri)) raxSeek(&si->ri,"$",NULL((void*)0),0);
547 } else {
548 raxSeek(&si->ri,"$",NULL((void*)0),0);
549 }
550 }
551 si->stream = s;
552 si->lp = NULL((void*)0); /* There is no current listpack right now. */
553 si->lp_ele = NULL((void*)0); /* Current listpack cursor. */
554 si->rev = rev; /* Direction, if non-zero reversed, from end to start. */
555}

557/* Return 1 and store the current item ID at 'id' if there are still
558 * elements within the iteration range, otherwise return 0 in order to
559 * signal the iteration terminated. */
560int streamIteratorGetID(streamIterator *si, streamID *id, int64_t *numfields) {
561 while(1) { /* Will stop when element > stop_key or end of radix tree. */
562 /* If the current listpack is set to NULL, this is the start of the
563 * iteration or the previous listpack was completely iterated.
564 * Go to the next node. */
565 if (si->lp == NULL((void*)0) || si->lp_ele == NULL((void*)0)) {
566 if (!si->rev && !raxNext(&si->ri)) return 0;
567 else if (si->rev && !raxPrev(&si->ri)) return 0;
568 serverAssert(si->ri.key_len == sizeof(streamID))((si->ri.key_len == sizeof(streamID))?(void)0 : (_serverAssert
("si->ri.key_len == sizeof(streamID)","t_stream.c",568),_exit
(1)));
569 /* Get the master ID. */
570 streamDecodeID(si->ri.key,&si->master_id);
571 /* Get the master fields count. */
572 si->lp = si->ri.data;
573 si->lp_ele = lpFirst(si->lp); /* Seek items count */
574 si->lp_ele = lpNext(si->lp,si->lp_ele); /* Seek deleted count. */
575 si->lp_ele = lpNext(si->lp,si->lp_ele); /* Seek num fields. */
576 si->master_fields_count = lpGetInteger(si->lp_ele);
577 si->lp_ele = lpNext(si->lp,si->lp_ele); /* Seek first field. */
578 si->master_fields_start = si->lp_ele;
579 /* We are now pointing to the first field of the master entry.
580 * We need to seek either the first or the last entry depending
581 * on the direction of the iteration. */
582 if (!si->rev) {
583 /* If we are iterating in normal order, skip the master fields
584 * to seek the first actual entry. */
585 for (uint64_t i = 0; i < si->master_fields_count; i++)
586 si->lp_ele = lpNext(si->lp,si->lp_ele);
587 } else {
588 /* If we are iterating in reverse direction, just seek the
589 * last part of the last entry in the listpack (that is, the
590 * fields count). */
591 si->lp_ele = lpLast(si->lp);
592 }
593 } else if (si->rev) {
594 /* If we are iterating in the reverse order, and this is not
595 * the first entry emitted for this listpack, then we already
596 * emitted the current entry, and have to go back to the previous
597 * one. */
598 int lp_count = lpGetInteger(si->lp_ele);
599 while(lp_count--) si->lp_ele = lpPrev(si->lp,si->lp_ele);
600 /* Seek lp-count of prev entry. */
601 si->lp_ele = lpPrev(si->lp,si->lp_ele);
602 }
603
604 /* For every radix tree node, iterate the corresponding listpack,
605 * returning elements when they are within range. */
606 while(1) {
607 if (!si->rev) {
608 /* If we are going forward, skip the previous entry
609 * lp-count field (or in case of the master entry, the zero
610 * term field) */
611 si->lp_ele = lpNext(si->lp,si->lp_ele);
612 if (si->lp_ele == NULL((void*)0)) break;
613 } else {
614 /* If we are going backward, read the number of elements this
615 * entry is composed of, and jump backward N times to seek
616 * its start. */
617 int64_t lp_count = lpGetInteger(si->lp_ele);
618 if (lp_count == 0) { /* We reached the master entry. */
619 si->lp = NULL((void*)0);
620 si->lp_ele = NULL((void*)0);
621 break;
622 }
623 while(lp_count--) si->lp_ele = lpPrev(si->lp,si->lp_ele);
624 }
625
626 /* Get the flags entry. */
627 si->lp_flags = si->lp_ele;
628 int flags = lpGetInteger(si->lp_ele);
629 si->lp_ele = lpNext(si->lp,si->lp_ele); /* Seek ID. */
630
631 /* Get the ID: it is encoded as difference between the master
632 * ID and this entry ID. */
633 *id = si->master_id;
634 id->ms += lpGetInteger(si->lp_ele);
635 si->lp_ele = lpNext(si->lp,si->lp_ele);
636 id->seq += lpGetInteger(si->lp_ele);
637 si->lp_ele = lpNext(si->lp,si->lp_ele);
638 unsigned char buf[sizeof(streamID)];
639 streamEncodeID(buf,id);
640
641 /* The number of entries is here or not depending on the
642 * flags. */
643 if (flags & STREAM_ITEM_FLAG_SAMEFIELDS(1<<1)) {
644 *numfields = si->master_fields_count;
645 } else {
646 *numfields = lpGetInteger(si->lp_ele);
647 si->lp_ele = lpNext(si->lp,si->lp_ele);
648 }
649
650 /* If current >= start, and the entry is not marked as
651 * deleted, emit it. */
652 if (!si->rev) {
653 if (memcmp(buf,si->start_key,sizeof(streamID)) >= 0 &&
654 !(flags & STREAM_ITEM_FLAG_DELETED(1<<0)))
655 {
656 if (memcmp(buf,si->end_key,sizeof(streamID)) > 0)
657 return 0; /* We are already out of range. */
658 si->entry_flags = flags;
659 if (flags & STREAM_ITEM_FLAG_SAMEFIELDS(1<<1))
660 si->master_fields_ptr = si->master_fields_start;
661 return 1; /* Valid item returned. */
662 }
663 } else {
664 if (memcmp(buf,si->end_key,sizeof(streamID)) <= 0 &&
665 !(flags & STREAM_ITEM_FLAG_DELETED(1<<0)))
666 {
667 if (memcmp(buf,si->start_key,sizeof(streamID)) < 0)
668 return 0; /* We are already out of range. */
669 si->entry_flags = flags;
670 if (flags & STREAM_ITEM_FLAG_SAMEFIELDS(1<<1))
671 si->master_fields_ptr = si->master_fields_start;
672 return 1; /* Valid item returned. */
673 }
674 }
675
676 /* If we do not emit, we have to discard if we are going
677 * forward, or seek the previous entry if we are going
678 * backward. */
679 if (!si->rev) {
680 int64_t to_discard = (flags & STREAM_ITEM_FLAG_SAMEFIELDS(1<<1)) ?
681 *numfields : *numfields*2;
682 for (int64_t i = 0; i < to_discard; i++)
683 si->lp_ele = lpNext(si->lp,si->lp_ele);
684 } else {
685 int64_t prev_times = 4; /* flag + id ms + id seq + one more to
686 go back to the previous entry "count"
687 field. */
688 /* If the entry was not flagged SAMEFIELD we also read the
689 * number of fields, so go back one more. */
690 if (!(flags & STREAM_ITEM_FLAG_SAMEFIELDS(1<<1))) prev_times++;
691 while(prev_times--) si->lp_ele = lpPrev(si->lp,si->lp_ele);
692 }
693 }
694
695 /* End of listpack reached. Try the next/prev radix tree node. */
696 }
697}

699/* Get the field and value of the current item we are iterating. This should
700 * be called immediately after streamIteratorGetID(), and for each field
701 * according to the number of fields returned by streamIteratorGetID().
702 * The function populates the field and value pointers and the corresponding
703 * lengths by reference, that are valid until the next iterator call, assuming
704 * no one touches the stream meanwhile. */
705void streamIteratorGetField(streamIterator *si, unsigned char **fieldptr, unsigned char **valueptr, int64_t *fieldlen, int64_t *valuelen) {
706 if (si->entry_flags & STREAM_ITEM_FLAG_SAMEFIELDS(1<<1)) {
707 *fieldptr = lpGet(si->master_fields_ptr,fieldlen,si->field_buf);
708 si->master_fields_ptr = lpNext(si->lp,si->master_fields_ptr);
709 } else {
710 *fieldptr = lpGet(si->lp_ele,fieldlen,si->field_buf);
711 si->lp_ele = lpNext(si->lp,si->lp_ele);
712 }
713 *valueptr = lpGet(si->lp_ele,valuelen,si->value_buf);
714 si->lp_ele = lpNext(si->lp,si->lp_ele);
715}
716
717/* Remove the current entry from the stream: can be called after the
718 * GetID() API or after any GetField() call, however we need to iterate
719 * a valid entry while calling this function. Moreover the function
720 * requires the entry ID we are currently iterating, that was previously
721 * returned by GetID().
722 *
723 * Note that after calling this function, next calls to GetField() can't
724 * be performed: the entry is now deleted. Instead the iterator will
725 * automatically re-seek to the next entry, so the caller should continue
726 * with GetID(). */
727void streamIteratorRemoveEntry(streamIterator *si, streamID *current) {
728 unsigned char *lp = si->lp;
729 int64_t aux;
730
731 /* We do not really delete the entry here. Instead we mark it as
732 * deleted flagging it, and also incrementing the count of the
733 * deleted entries in the listpack header.
734 *
735 * We start flagging: */
736 int flags = lpGetInteger(si->lp_flags);
737 flags |= STREAM_ITEM_FLAG_DELETED(1<<0);
738 lp = lpReplaceInteger(lp,&si->lp_flags,flags);
739
740 /* Change the valid/deleted entries count in the master entry. */
741 unsigned char *p = lpFirst(lp);
742 aux = lpGetInteger(p);
743
744 if (aux == 1) {
745 /* If this is the last element in the listpack, we can remove the whole
746 * node. */
747 lpFree(lp);
748 raxRemove(si->stream->rax,si->ri.key,si->ri.key_len,NULL((void*)0));
749 } else {
750 /* In the base case we alter the counters of valid/deleted entries. */
751 lp = lpReplaceInteger(lp,&p,aux-1);
752 p = lpNext(lp,p); /* Seek deleted field. */
753 aux = lpGetInteger(p);
754 lp = lpReplaceInteger(lp,&p,aux+1);
755
756 /* Update the listpack with the new pointer. */
757 if (si->lp != lp)
758 raxInsert(si->stream->rax,si->ri.key,si->ri.key_len,lp,NULL((void*)0));
759 }
760
761 /* Update the number of entries counter. */
762 si->stream->length--;
763
764 /* Re-seek the iterator to fix the now messed up state. */
765 streamID start, end;
766 if (si->rev) {
767 streamDecodeID(si->start_key,&start);
768 end = *current;
769 } else {
770 start = *current;
771 streamDecodeID(si->end_key,&end);
772 }
773 streamIteratorStop(si);
774 streamIteratorStart(si,si->stream,&start,&end,si->rev);
775
776 /* TODO: perform a garbage collection here if the ration between
777 * deleted and valid goes over a certain limit. */
778}

921/* Send the stream items in the specified range to the client 'c'. The range
922 * the client will receive is between start and end inclusive, if 'count' is
923 * non zero, no more than 'count' elements are sent.
924 *
925 * The 'end' pointer can be NULL to mean that we want all the elements from
926 * 'start' till the end of the stream. If 'rev' is non zero, elements are
927 * produced in reversed order from end to start.
928 *
929 * The function returns the number of entries emitted.
930 *
931 * If group and consumer are not NULL, the function performs additional work:
932 * 1. It updates the last delivered ID in the group in case we are
933 * sending IDs greater than the current last ID.
934 * 2. If the requested IDs are already assigned to some other consumer, the
935 * function will not return it to the client.
936 * 3. An entry in the pending list will be created for every entry delivered
937 * for the first time to this consumer.
938 *
939 * The behavior may be modified passing non-zero flags:
940 *
941 * STREAM_RWR_NOACK: Do not create PEL entries, that is, the point "3" above
942 * is not performed.
943 * STREAM_RWR_RAWENTRIES: Do not emit array boundaries, but just the entries,
944 * and return the number of entries emitted as usually.
945 * This is used when the function is just used in order
946 * to emit data and there is some higher level logic.
947 *
948 * The final argument 'spi' (stream propagation info pointer) is a structure
949 * filled with information needed to propagate the command execution to AOF
950 * and slaves, in the case a consumer group was passed: we need to generate
951 * XCLAIM commands to create the pending list into AOF/slaves in that case.
952 *
953 * If 'spi' is set to NULL no propagation will happen even if the group was
954 * given, but currently such a feature is never used by the code base that
955 * will always pass 'spi' and propagate when a group is passed.
956 *
957 * Note that this function is recursive in certain cases. When it's called
958 * with a non NULL group and consumer argument, it may call
959 * streamReplyWithRangeFromConsumerPEL() in order to get entries from the
960 * consumer pending entries list. However such a function will then call
961 * streamReplyWithRange() in order to emit single entries (found in the
962 * PEL by ID) to the client. This is the use case for the STREAM_RWR_RAWENTRIES
963 * flag.
964 */
965#define STREAM_RWR_NOACK(1<<0) (1<<0) /* Do not create entries in the PEL. */
966#define STREAM_RWR_RAWENTRIES(1<<1) (1<<1) /* Do not emit protocol for array
967 boundaries, just the entries. */
968#define STREAM_RWR_HISTORY(1<<2) (1<<2) /* Only serve consumer local PEL. */
969size_t streamReplyWithRange(client *c, stream *s, streamID *start, streamID *end, size_t count, int rev, streamCG *group, streamConsumer *consumer, int flags, streamPropInfo *spi) {
970 void *arraylen_ptr = NULL((void*)0);
971 size_t arraylen = 0;
972 streamIterator si;
973 int64_t numfields;
974 streamID id;
975 int propagate_last_id = 0;
976 int noack = flags & STREAM_RWR_NOACK(1<<0);
977
978 /* If the client is asking for some history, we serve it using a
979 * different function, so that we return entries *solely* from its
980 * own PEL. This ensures each consumer will always and only see
981 * the history of messages delivered to it and not yet confirmed
982 * as delivered. */
983 if (group && (flags & STREAM_RWR_HISTORY(1<<2))) {
984 return streamReplyWithRangeFromConsumerPEL(c,s,start,end,count,
985 consumer);
986 }
987
988 if (!(flags & STREAM_RWR_RAWENTRIES(1<<1)))
989 arraylen_ptr = addReplyDeferredLen(c);
990 streamIteratorStart(&si,s,start,end,rev);
991 while(streamIteratorGetID(&si,&id,&numfields)) {
992 /* Update the group last_id if needed. */
993 if (group && streamCompareID(&id,&group->last_id) > 0) {
994 group->last_id = id;
995 /* Group last ID should be propagated only if NOACK was
996 * specified, otherwise the last id will be included
997 * in the propagation of XCLAIM itself. */
998 if (noack) propagate_last_id = 1;
999 }
1000
1001 /* Emit a two elements array for each item. The first is
1002 * the ID, the second is an array of field-value pairs. */
1003 addReplyArrayLen(c,2);
1004 addReplyStreamID(c,&id);
1005
1006 addReplyArrayLen(c,numfields*2);
1007
1008 /* Emit the field-value pairs. */
1009 while(numfields--) {
1010 unsigned char *key, *value;
1011 int64_t key_len, value_len;
1012 streamIteratorGetField(&si,&key,&value,&key_len,&value_len);
1013 addReplyBulkCBuffer(c,key,key_len);
1014 addReplyBulkCBuffer(c,value,value_len);
1015 }
1016
1017 /* If a group is passed, we need to create an entry in the
1018 * PEL (pending entries list) of this group *and* this consumer.
1019 *
1020 * Note that we cannot be sure about the fact the message is not
1021 * already owned by another consumer, because the admin is able
1022 * to change the consumer group last delivered ID using the
1023 * XGROUP SETID command. So if we find that there is already
1024 * a NACK for the entry, we need to associate it to the new
1025 * consumer. */
1026 if (group && !noack) {
1027 unsigned char buf[sizeof(streamID)];
1028 streamEncodeID(buf,&id);
1029
1030 /* Try to add a new NACK. Most of the time this will work and
1031 * will not require extra lookups. We'll fix the problem later
1032 * if we find that there is already a entry for this ID. */
1033 streamNACK *nack = streamCreateNACK(consumer);
1034 int group_inserted =
1035 raxTryInsert(group->pel,buf,sizeof(buf),nack,NULL((void*)0));
1036 int consumer_inserted =
1037 raxTryInsert(consumer->pel,buf,sizeof(buf),nack,NULL((void*)0));
1038
1039 /* Now we can check if the entry was already busy, and
1040 * in that case reassign the entry to the new consumer,
1041 * or update it if the consumer is the same as before. */
1042 if (group_inserted == 0) {
1043 streamFreeNACK(nack);
1044 nack = raxFind(group->pel,buf,sizeof(buf));
1045 serverAssert(nack != raxNotFound)((nack != raxNotFound)?(void)0 : (_serverAssert("nack != raxNotFound"
,"t_stream.c",1045),_exit(1)));
1046 raxRemove(nack->consumer->pel,buf,sizeof(buf),NULL((void*)0));
1047 /* Update the consumer and NACK metadata. */
1048 nack->consumer = consumer;
1049 nack->delivery_time = mstime();
1050 nack->delivery_count = 1;
1051 /* Add the entry in the new consumer local PEL. */
1052 raxInsert(consumer->pel,buf,sizeof(buf),nack,NULL((void*)0));
1053 } else if (group_inserted == 1 && consumer_inserted == 0) {
1054 serverPanic("NACK half-created. Should not be possible.")_serverPanic("t_stream.c",1054,"NACK half-created. Should not be possible."
),_exit(1);
1055 }
1056
1057 /* Propagate as XCLAIM. */
1058 if (spi) {
1059 robj *idarg = createObjectFromStreamID(&id);
1060 streamPropagateXCLAIM(c,spi->keyname,group,spi->groupname,idarg,nack);
1061 decrRefCount(idarg);
1062 }
1063 }
1064
1065 arraylen++;
1066 if (count && count == arraylen) break;
1067 }
1068
1069 if (spi && propagate_last_id)
1070 streamPropagateGroupID(c,spi->keyname,group,spi->groupname);
1071
1072 streamIteratorStop(&si);
1073 if (arraylen_ptr) setDeferredArrayLen(c,arraylen_ptr,arraylen);
1074 return arraylen;
1075}

1611 if (noack) flags |= STREAM_RWR_NOACK(1<<0);
1612 if (serve_history) flags |= STREAM_RWR_HISTORY(1<<2);
1613 streamReplyWithRange(c,s,&start,NULL((void*)0),count,0,
1614 groups ? groups[i] : NULL((void*)0),
1615 consumer, flags, &spi);
1616 if (groups) server.dirty++;
1617 }
1618 }
1619
1620 /* We replied synchronously! Set the top array len and return to caller. */
1621 if (arraylen) {
1622 if (c->resp == 2)
1623 setDeferredArrayLen(c,arraylen_ptr,arraylen);
1624 else
1625 setDeferredMapLen(c,arraylen_ptr,arraylen);
1626 goto cleanup;
1627 }
1628
1629 /* Block if needed. */
1630 if (timeout != -1) {
1631 /* If we are inside a MULTI/EXEC and the list is empty the only thing
1632 * we can do is treating it as a timeout (even with timeout 0). */
1633 if (c->flags & CLIENT_MULTI(1<<3)) {
1634 addReplyNullArray(c);
1635 goto cleanup;
1636 }
1637 blockForKeys(c, BLOCKED_STREAM4, c->argv+streams_arg, streams_count,
1638 timeout, NULL((void*)0), NULL((void*)0), ids);
1639 /* If no COUNT is given and we block, set a relatively small count:
1640 * in case the ID provided is too low, we do not want the server to
1641 * block just to serve this client a huge stream of messages. */
1642 c->bpop.xread_count = count ? count : XREAD_BLOCKED_DEFAULT_COUNT1000;
1643
1644 /* If this is a XREADGROUP + GROUP we need to remember for which
1645 * group and consumer name we are blocking, so later when one of the
1646 * keys receive more data, we can call streamReplyWithRange() passing
1647 * the right arguments. */
1648 if (groupname) {
1649 incrRefCount(groupname);
1650 incrRefCount(consumername);
1651 c->bpop.xread_group = groupname;
1652 c->bpop.xread_consumer = consumername;
1653 c->bpop.xread_group_noack = noack;
1654 } else {
1655 c->bpop.xread_group = NULL((void*)0);
1656 c->bpop.xread_consumer = NULL((void*)0);
1657 }
1658 goto cleanup;
1659 }
1660
1661 /* No BLOCK option, nor any stream we can serve. Reply as with a
1662 * timeout happened. */
1663 addReplyNullArray(c);
1664 /* Continue to cleanup... */
1665
1666cleanup: /* Cleanup. */
1667
1668 /* The command is propagated (in the READGROUP form) as a side effect
1669 * of calling lower level APIs. So stop any implicit propagation. */
1670 preventCommandPropagation(c);
1671 if (ids != static_ids) zfree(ids);
1672 zfree(groups);
1673}

1740/* Lookup the consumer with the specified name in the group 'cg': if the
1741 * consumer does not exist it is created unless SLC_NOCREAT flag was specified.
1742 * Its last seen time is updated unless SLC_NOREFRESH flag was specified. */
1743streamConsumer *streamLookupConsumer(streamCG *cg, sds name, int flags, int *created) {
1744 if (created) *created = 0;
1745 int create = !(flags & SLC_NOCREAT(1<<0));
1746 int refresh = !(flags & SLC_NOREFRESH(1<<1));
1747 streamConsumer *consumer = raxFind(cg->consumers,(unsigned char*)name,
1748 sdslen(name));

1754 raxInsert(cg->consumers,(unsigned char*)name,sdslen(name),

1756 consumer->seen_time = mstime();
1757 if (created) *created = 1;
1758 } else if (refresh)
1759 consumer->seen_time = mstime();
1760 return consumer;

1763/* Delete the consumer specified in the consumer group 'cg'. The consumer
1764 * may have pending messages: they are removed from the PEL, and the number
1765 * of pending messages "lost" is returned. */
1766uint64_t streamDelConsumer(streamCG *cg, sds name) {
1767 streamConsumer *consumer =
1768 streamLookupConsumer(cg,name,SLC_NOCREAT(1<<0)|SLC_NOREFRESH(1<<1),NULL((void*)0));
1769 if (consumer == NULL((void*)0)) return 0;
1770
1771 uint64_t retval = raxSize(consumer->pel);
1772
1773 /* Iterate all the consumer pending messages, deleting every corresponding
1774 * entry from the global entry. */
1775 raxIterator ri;
1776 raxStart(&ri,consumer->pel);
1777 raxSeek(&ri,"^",NULL((void*)0),0);
1778 while(raxNext(&ri)) {
1779 streamNACK *nack = ri.data;
1780 raxRemove(cg->pel,ri.key,ri.key_len,NULL((void*)0));
1781 streamFreeNACK(nack);
1782 }
1783 raxStop(&ri);
1784
1785 /* Deallocate the consumer. */
1786 raxRemove(cg->consumers,(unsigned char*)name,sdslen(name),NULL((void*)0));
1787 streamFreeConsumer(consumer);
1788 return retval;
1789}

1974/* XACK <key> <group> <id> <id> ... <id>
1975 *
1976 * Acknowledge a message as processed. In practical terms we just check the
1977 * pendine entries list (PEL) of the group, and delete the PEL entry both from
1978 * the group and the consumer (pending messages are referenced in both places).
1979 *
1980 * Return value of the command is the number of messages successfully
1981 * acknowledged, that is, the IDs we were actually able to resolve in the PEL.
1982 */
1983void xackCommand(client *c) {
1984 streamCG *group = NULL((void*)0);
1985 robj *o = lookupKeyRead(c->db,c->argv[1]);
1986 if (o) {
1987 if (checkType(c,o,OBJ_STREAM6)) return; /* Type error. */
1988 group = streamLookupCG(o->ptr,c->argv[2]->ptr);
1989 }
1990
1991 /* No key or group? Nothing to ack. */
1992 if (o == NULL((void*)0) || group == NULL((void*)0)) {
1993 addReply(c,shared.czero);
1994 return;
1995 }
1996
1997 /* Start parsing the IDs, so that we abort ASAP if there is a syntax
1998 * error: the return value of this command cannot be an error in case
1999 * the client successfully acknowledged some messages, so it should be
2000 * executed in a "all or nothing" fashion. */
2001 streamID static_ids[STREAMID_STATIC_VECTOR_LEN8];
2002 streamID *ids = static_ids;
2003 int id_count = c->argc-3;
2004 if (id_count > STREAMID_STATIC_VECTOR_LEN8)
2005 ids = zmalloc(sizeof(streamID)*id_count);
2006 for (int j = 3; j < c->argc; j++) {
2007 if (streamParseStrictIDOrReply(c,c->argv[j],&ids[j-3],0) != C_OK0) goto cleanup;
2008 }
2009
2010 int acknowledged = 0;
2011 for (int j = 3; j < c->argc; j++) {
2012 unsigned char buf[sizeof(streamID)];
2013 streamEncodeID(buf,&ids[j-3]);
2014
2015 /* Lookup the ID in the group PEL: it will have a reference to the
2016 * NACK structure that will have a reference to the consumer, so that
2017 * we are able to remove the entry from both PELs. */
2018 streamNACK *nack = raxFind(group->pel,buf,sizeof(buf));
2019 if (nack != raxNotFound) {
2020 raxRemove(group->pel,buf,sizeof(buf),NULL((void*)0));
2021 raxRemove(nack->consumer->pel,buf,sizeof(buf),NULL((void*)0));
2022 streamFreeNACK(nack);
2023 acknowledged++;
2024 server.dirty++;
2025 }
2026 }
2027 addReplyLongLong(c,acknowledged);
2028cleanup:
2029 if (ids != static_ids) zfree(ids);
2030}

2032/* XPENDING <key> <group> [<start> <stop> <count> [<consumer>]]
2033 *
2034 * If start and stop are omitted, the command just outputs information about
2035 * the amount of pending messages for the key/group pair, together with
2036 * the minimum and maximum ID of pending messages.
2037 *
2038 * If start and stop are provided instead, the pending messages are returned
2039 * with informations about the current owner, number of deliveries and last
2040 * delivery time and so forth. */
2041void xpendingCommand(client *c) {
2042 int justinfo = c->argc == 3; /* Without the range just outputs general
2043 informations about the PEL. */
2044 robj *key = c->argv[1];
2045 robj *groupname = c->argv[2];
2046 robj *consumername = (c->argc == 7) ? c->argv[6] : NULL((void*)0);
2047 streamID startid, endid;
2048 long long count;
2049
2050 /* Start and stop, and the consumer, can be omitted. */
2051 if (c->argc != 3 && c->argc != 6 && c->argc != 7) {
2052 addReply(c,shared.syntaxerr);
2053 return;
2054 }
2055
2056 /* Parse start/end/count arguments ASAP if needed, in order to report
2057 * syntax errors before any other error. */
2058 if (c->argc >= 6) {
2059 if (getLongLongFromObjectOrReply(c,c->argv[5],&count,NULL((void*)0)) == C_ERR-1)
2060 return;
2061 if (count < 0) count = 0;
2062 if (streamParseIDOrReply(c,c->argv[3],&startid,0) == C_ERR-1)
2063 return;
2064 if (streamParseIDOrReply(c,c->argv[4],&endid,UINT64_MAX(18446744073709551615UL)) == C_ERR-1)
2065 return;
2066 }
2067
2068 /* Lookup the key and the group inside the stream. */
2069 robj *o = lookupKeyRead(c->db,c->argv[1]);
2070 streamCG *group;
2071
2072 if (checkType(c,o,OBJ_STREAM6)) return;

2074 (group = streamLookupCG(o->ptr,groupname->ptr)) == NULL((void*)0))

2077 "group '%s'",
2078 (char*)key->ptr,(char*)groupname->ptr);
2079 return;
2080 }
2081
2082 /* XPENDING <key> <group> variant. */
2083 if (justinfo) {
2084 addReplyArrayLen(c,4);
2085 /* Total number of messages in the PEL. */
2086 addReplyLongLong(c,raxSize(group->pel));
2087 /* First and last IDs. */

2112 size_t arraylen = 0;

2117 addReplyBulkCBuffer(c,ri.key,ri.key_len);
2118 addReplyBulkLongLong(c,raxSize(consumer->pel));
2119 arraylen++;
2120 }
2121 setDeferredArrayLen(c,arraylen_ptr,arraylen);
2122 raxStop(&ri);
2123 }
2124 }

2126 else {

2132 NULL((void*)0));

2135 * just return an empty array. */

2137 addReplyArrayLen(c,0);
2138 return;
2139 }
2140 }

2143 unsigned char startkey[sizeof(streamID)];
2144 unsigned char endkey[sizeof(streamID)];
2145 raxIterator ri;

2160 addReplyArrayLen(c,4);
2161

2163 streamID id;

2166

2168 addReplyBulkCBuffer(c,nack->consumer->name,
2169 sdslen(nack->consumer->name));
2170
2171 /* Milliseconds elapsed since last delivery. */
2172 mstime_t elapsed = now - nack->delivery_time;
2173 if (elapsed < 0) elapsed = 0;
2174 addReplyLongLong(c,elapsed);
2175
2176 /* Number of deliveries. */
2177 addReplyLongLong(c,nack->delivery_count);
2178 }
2179 raxStop(&ri);
2180 setDeferredArrayLen(c,arraylen_ptr,arraylen);

2184/* XCLAIM <key> <group> <consumer> <min-idle-time> <ID-1> <ID-2>
2185 * [IDLE <milliseconds>] [TIME <mstime>] [RETRYCOUNT <count>]
2186 * [FORCE] [JUSTID]
2187 *
2188 * Gets ownership of one or multiple messages in the Pending Entries List
2189 * of a given stream consumer group.
2190 *
2191 * If the message ID (among the specified ones) exists, and its idle
2192 * time greater or equal to <min-idle-time>, then the message new owner
2193 * becomes the specified <consumer>. If the minimum idle time specified
2194 * is zero, messages are claimed regardless of their idle time.
2195 *
2196 * All the messages that cannot be found inside the pending entries list
2197 * are ignored, but in case the FORCE option is used. In that case we
2198 * create the NACK (representing a not yet acknowledged message) entry in
2199 * the consumer group PEL.
2200 *
2201 * This command creates the consumer as side effect if it does not yet
2202 * exists. Moreover the command reset the idle time of the message to 0,
2203 * even if by using the IDLE or TIME options, the user can control the
2204 * new idle time.
2205 *
2206 * The options at the end can be used in order to specify more attributes
2207 * to set in the representation of the pending message:
2208 *
2209 * 1. IDLE <ms>:
2210 * Set the idle time (last time it was delivered) of the message.
2211 * If IDLE is not specified, an IDLE of 0 is assumed, that is,
2212 * the time count is reset because the message has now a new
2213 * owner trying to process it.
2214 *
2215 * 2. TIME <ms-unix-time>:
2216 * This is the same as IDLE but instead of a relative amount of
2217 * milliseconds, it sets the idle time to a specific unix time
2218 * (in milliseconds). This is useful in order to rewrite the AOF
2219 * file generating XCLAIM commands.
2220 *
2221 * 3. RETRYCOUNT <count>:
2222 * Set the retry counter to the specified value. This counter is
2223 * incremented every time a message is delivered again. Normally
2224 * XCLAIM does not alter this counter, which is just served to clients
2225 * when the XPENDING command is called: this way clients can detect
2226 * anomalies, like messages that are never processed for some reason
2227 * after a big number of delivery attempts.
2228 *
2229 * 4. FORCE:
2230 * Creates the pending message entry in the PEL even if certain
2231 * specified IDs are not already in the PEL assigned to a different
2232 * client. However the message must be exist in the stream, otherwise
2233 * the IDs of non existing messages are ignored.
2234 *
2235 * 5. JUSTID:
2236 * Return just an array of IDs of messages successfully claimed,
2237 * without returning the actual message.
2238 *
2239 * 6. LASTID <id>:
2240 * Update the consumer group last ID with the specified ID if the
2241 * current last ID is smaller than the provided one.
2242 * This is used for replication / AOF, so that when we read from a
2243 * consumer group, the XCLAIM that gets propagated to give ownership
2244 * to the consumer, is also used in order to update the group current
2245 * ID.
2246 *
2247 * The command returns an array of messages that the user
2248 * successfully claimed, so that the caller is able to understand
2249 * what messages it is now in charge of. */
2250void xclaimCommand(client *c) {
2251 streamCG *group = NULL((void*)0);
2252 robj *o = lookupKeyRead(c->db,c->argv[1]);
2253 long long minidle; /* Minimum idle time argument. */
2254 long long retrycount = -1; /* -1 means RETRYCOUNT option not given. */
2255 mstime_t deliverytime = -1; /* -1 means IDLE/TIME options not given. */
2256 int force = 0;
2257 int justid = 0;
2258
2259 if (o) {
2260 if (checkType(c,o,OBJ_STREAM6)) return; /* Type error. */
2261 group = streamLookupCG(o->ptr,c->argv[2]->ptr);
2262 }
2263
2264 /* No key or group? Send an error given that the group creation
2265 * is mandatory. */
2266 if (o == NULL((void*)0) || group == NULL((void*)0)) {
2267 addReplyErrorFormat(c,"-NOGROUP No such key '%s' or "
2268 "consumer group '%s'", (char*)c->argv[1]->ptr,
2269 (char*)c->argv[2]->ptr);
2270 return;
2271 }
2272
2273 if (getLongLongFromObjectOrReply(c,c->argv[4],&minidle,
2274 "Invalid min-idle-time argument for XCLAIM")
2275 != C_OK0) return;
2276 if (minidle < 0) minidle = 0;
2277
2278 /* Start parsing the IDs, so that we abort ASAP if there is a syntax
2279 * error: the return value of this command cannot be an error in case
2280 * the client successfully claimed some message, so it should be
2281 * executed in a "all or nothing" fashion. */
2282 int j;
2283 streamID static_ids[STREAMID_STATIC_VECTOR_LEN8];
2284 streamID *ids = static_ids;
2285 int id_count = c->argc-5;
2286 if (id_count > STREAMID_STATIC_VECTOR_LEN8)
2287 ids = zmalloc(sizeof(streamID)*id_count);
2288 for (j = 5; j < c->argc; j++) {
2289 if (streamParseStrictIDOrReply(NULL((void*)0),c->argv[j],&ids[j-5],0) != C_OK0) break;
2290 }
2291 int last_id_arg = j-1; /* Next time we iterate the IDs we now the range. */
2292
2293 /* If we stopped because some IDs cannot be parsed, perhaps they
2294 * are trailing options. */
2295 mstime_t now = mstime();
2296 streamID last_id = {0,0};
2297 int propagate_last_id = 0;
2298 for (; j < c->argc; j++) {
2299 int moreargs = (c->argc-1) - j; /* Number of additional arguments. */
2300 char *opt = c->argv[j]->ptr;
2301 if (!strcasecmp(opt,"FORCE")) {
2302 force = 1;
2303 } else if (!strcasecmp(opt,"JUSTID")) {
2304 justid = 1;
2305 } else if (!strcasecmp(opt,"IDLE") && moreargs) {
2306 j++;
2307 if (getLongLongFromObjectOrReply(c,c->argv[j],&deliverytime,
2308 "Invalid IDLE option argument for XCLAIM")
2309 != C_OK0) goto cleanup;
2310 deliverytime = now - deliverytime;
2311 } else if (!strcasecmp(opt,"TIME") && moreargs) {
2312 j++;
2313 if (getLongLongFromObjectOrReply(c,c->argv[j],&deliverytime,
2314 "Invalid TIME option argument for XCLAIM")
2315 != C_OK0) goto cleanup;
2316 } else if (!strcasecmp(opt,"RETRYCOUNT") && moreargs) {
2317 j++;
2318 if (getLongLongFromObjectOrReply(c,c->argv[j],&retrycount,
2319 "Invalid RETRYCOUNT option argument for XCLAIM")
2320 != C_OK0) goto cleanup;
2321 } else if (!strcasecmp(opt,"LASTID") && moreargs) {
2322 j++;
2323 if (streamParseStrictIDOrReply(c,c->argv[j],&last_id,0) != C_OK0) goto cleanup;
2324 } else {
2325 addReplyErrorFormat(c,"Unrecognized XCLAIM option '%s'",opt);
2326 goto cleanup;
2327 }
2328 }
2329

2331 group->last_id = last_id;
2332 propagate_last_id = 1;
2333 }
2334
2335 if (deliverytime != -1) {
2336 /* If a delivery time was passed, either with IDLE or TIME, we
2337 * do some sanity check on it, and set the deliverytime to now
2338 * (which is a sane choice usually) if the value is bogus.
2339 * To raise an error here is not wise because clients may compute
2340 * the idle time doing some math starting from their local time,
2341 * and this is not a good excuse to fail in case, for instance,
2342 * the computer time is a bit in the future from our POV. */
2343 if (deliverytime < 0 || deliverytime > now) deliverytime = now;
2344 } else {
2345 /* If no IDLE/TIME option was passed, we want the last delivery
2346 * time to be now, so that the idle time of the message will be
2347 * zero. */
2348 deliverytime = now;
2349 }
2350
2351 /* Do the actual claiming. */
2352 streamConsumer *consumer = NULL((void*)0);
2353 void *arraylenptr = addReplyDeferredLen(c);
2354 size_t arraylen = 0;
2355 for (int j = 5; j <= last_id_arg; j++) {
2356 streamID id = ids[j-5];

2359
2360 /* Lookup the ID in the group PEL. */
2361 streamNACK *nack = raxFind(group->pel,buf,sizeof(buf));
2362
2363 /* If FORCE is passed, let's check if at least the entry
2364 * exists in the Stream. In such case, we'll crate a new
2365 * entry in the PEL from scratch, so that XCLAIM can also
2366 * be used to create entries in the PEL. Useful for AOF
2367 * and replication of consumer groups. */
2368 if (force && nack == raxNotFound) {
2369 streamIterator myiterator;
2370 streamIteratorStart(&myiterator,o->ptr,&id,&id,0);
2371 int64_t numfields;
2372 int found = 0;
2373 streamID item_id;
2374 if (streamIteratorGetID(&myiterator,&item_id,&numfields)) found = 1;
2375 streamIteratorStop(&myiterator);
2376
2377 /* Item must exist for us to create a NACK for it. */
2378 if (!found) continue;
2379
2380 /* Create the NACK. */
2381 nack = streamCreateNACK(NULL((void*)0));
2382 raxInsert(group->pel,buf,sizeof(buf),nack,NULL((void*)0));
2383 }
2384
2385 if (nack != raxNotFound) {
2386 /* We need to check if the minimum idle time requested
2387 * by the caller is satisfied by this entry.
2388 *
2389 * Note that the nack could be created by FORCE, in this
2390 * case there was no pre-existing entry and minidle should
2391 * be ignored, but in that case nick->consumer is NULL. */
2392 if (nack->consumer && minidle) {
2393 mstime_t this_idle = now - nack->delivery_time;
2394 if (this_idle < minidle) continue;
2395 }
2396 /* Remove the entry from the old consumer.
2397 * Note that nack->consumer is NULL if we created the
2398 * NACK above because of the FORCE option. */
2399 if (nack->consumer)
2400 raxRemove(nack->consumer->pel,buf,sizeof(buf),NULL((void*)0));
2401 /* Update the consumer and idle time. */
2402 if (consumer == NULL((void*)0))
2403 consumer = streamLookupConsumer(group,c->argv[3]->ptr,SLC_NONE0,NULL((void*)0));
2404 nack->consumer = consumer;
2405 nack->delivery_time = deliverytime;
2406 /* Set the delivery attempts counter if given, otherwise
2407 * autoincrement unless JUSTID option provided */
2408 if (retrycount >= 0) {
2409 nack->delivery_count = retrycount;
2410 } else if (!justid) {
2411 nack->delivery_count++;
2412 }
2413 /* Add the entry in the new consumer local PEL. */
2414 raxInsert(consumer->pel,buf,sizeof(buf),nack,NULL((void*)0));
2415 /* Send the reply for this entry. */
2416 if (justid) {
2417 addReplyStreamID(c,&id);
2418 } else {
2419 size_t emitted = streamReplyWithRange(c,o->ptr,&id,&id,1,0,
2420 NULL((void*)0),NULL((void*)0),STREAM_RWR_RAWENTRIES(1<<1),NULL((void*)0));
2421 if (!emitted) addReplyNull(c);
2422 }
2423 arraylen++;
2424
2425 /* Propagate this change. */
2426 streamPropagateXCLAIM(c,c->argv[1],group,c->argv[2],c->argv[j],nack);
2427 propagate_last_id = 0; /* Will be propagated by XCLAIM itself. */
2428 server.dirty++;
2429 }
2430 }
2431 if (propagate_last_id) {
2432 streamPropagateGroupID(c,c->argv[1],group,c->argv[2]);
2433 server.dirty++;
2434 }
2435 setDeferredArrayLen(c,arraylenptr,arraylen);
2436 preventCommandPropagation(c);
2437cleanup:
2438 if (ids != static_ids) zfree(ids);
2439}

2483/* General form: XTRIM <key> [... options ...]
2484 *
2485 * List of options:
2486 *
2487 * MAXLEN [~|=] <count> -- Trim so that the stream will be capped at
2488 * the specified length. Use ~ before the
2489 * count in order to demand approximated trimming
2490 * (like XADD MAXLEN option).
2491 */
2492
2493#define TRIM_STRATEGY_NONE0 0
2494#define TRIM_STRATEGY_MAXLEN1 1
2495void xtrimCommand(client *c) {
2496 robj *o;
2497
2498 /* If the key does not exist, we are ok returning zero, that is, the
2499 * number of elements removed from the stream. */
2500 if ((o = lookupKeyWriteOrReply(c,c->argv[1],shared.czero)) == NULL((void*)0)
2501 || checkType(c,o,OBJ_STREAM6)) return;
2502 stream *s = o->ptr;
2503
2504 /* Argument parsing. */
2505 int trim_strategy = TRIM_STRATEGY_NONE0;
2506 long long maxlen = -1; /* If left to -1 no trimming is performed. */
2507 int approx_maxlen = 0; /* If 1 only delete whole radix tree nodes, so
2508 the maxium length is not applied verbatim. */
2509 int maxlen_arg_idx = 0; /* Index of the count in MAXLEN, for rewriting. */
2510
2511 /* Parse options. */
2512 int i = 2; /* Start of options. */
2513 for (; i < c->argc; i++) {
2514 int moreargs = (c->argc-1) - i; /* Number of additional arguments. */
2515 char *opt = c->argv[i]->ptr;
2516 if (!strcasecmp(opt,"maxlen") && moreargs) {
2517 approx_maxlen = 0;
2518 trim_strategy = TRIM_STRATEGY_MAXLEN1;
2519 char *next = c->argv[i+1]->ptr;
2520 /* Check for the form MAXLEN ~ <count>. */
2521 if (moreargs >= 2 && next[0] == '~' && next[1] == '\0') {
2522 approx_maxlen = 1;
2523 i++;
2524 } else if (moreargs >= 2 && next[0] == '=' && next[1] == '\0') {
2525 i++;
2526 }
2527 if (getLongLongFromObjectOrReply(c,c->argv[i+1],&maxlen,NULL((void*)0))
2528 != C_OK0) return;
2529
2530 if (maxlen < 0) {
2531 addReplyError(c,"The MAXLEN argument must be >= 0.");
2532 return;
2533 }
2534 i++;
2535 maxlen_arg_idx = i;
2536 } else {
2537 addReply(c,shared.syntaxerr);
2538 return;
2539 }
2540 }
2541
2542 /* Perform the trimming. */
2543 int64_t deleted = 0;
2544 if (trim_strategy == TRIM_STRATEGY_MAXLEN1) {
2545 deleted = streamTrimByLength(s,maxlen,approx_maxlen);
2546 } else {
2547 addReplyError(c,"XTRIM called without an option to trim the stream");
2548 return;
2549 }
2550
2551 /* Propagate the write if needed. */
2552 if (deleted) {
2553 signalModifiedKey(c,c->db,c->argv[1]);
2554 notifyKeyspaceEvent(NOTIFY_STREAM(1<<10),"xtrim",c->argv[1],c->db->id);
2555 server.dirty += deleted;
2556 if (approx_maxlen) streamRewriteApproxMaxlen(c,s,maxlen_arg_idx);
2557 }
2558 addReplyLongLong(c,deleted);
2559}
